\tractatehead{מסכת שקלים}{shekalim}

\perek[157]{בְּאֶחָד}{בַּאֲדָר מַשְׁמִיעִין עַל הַשְּׁקָלִים\hdot
וְעַל הַכִּלְאַיִם\hdot
בַּחֲמִשָּׁה עָשָׂר בּוֹ\hdot
קוֹרִין אֶת הַמְּגִילָה בַּכְרָכִים\hdot
וּמְתַקְּנִין אֶת הַדְּרָכִים\hdot
וְאֶת הָרְחוֹבוֹת\hdot
וְאֶת מִקְוְאוֹת הַמַּיִם\hdot
וְעוֹשִׂין כָּל צָרְכֵי הָרַבִּים\hdot
וּמְצַיְּנִין אֶת הַקְּבָרוֹת\hdot
וְיוֹצְאִין אַף עַל הַכִּלְאַיִם׃}

\halacha[1152]{אָמַר רִבִּי יְהוּדָה בָּרִאשׁוֹנָה הָיוּ עוֹקְרִין וּמַשְׁלִיכִין לִפְנֵיהֶם\hdot
מִשֶּׁרַבּוּ עוֹבְרֵי עֲבֵירָה\hdot
הָיוּ עוֹקְרִין וּמַשְׁלִיכִין עַל הַדְּרָכִים\hdot
הִתְקִינוּ שֶׁיְּהוּ מַפְקִירִין כָּל הַשָּׂדֶה כֻּלָּהּ׃}

\halacha[744]{בַּחֲמִשָּׁה עָשָׂר בּוֹ\hdot
שֻׁלְחָנוֹת הָיוּ יוֹשְׁבִין בַּמְּדִינָה\hdot
בְּעֶשְׂרִים וַחֲמִשָּׁה יָֽשְׁבוּ בַמִּקְדָּשׁ\hdot
מִשֶּׁיָּֽשְׁבוּ בַמִּקְדָּשׁ\hdot
הִתְחִילוּ לְמַשְׁכֵּן\hdot
אֶת מִי מְמַשְׁכְּנִין\hdot
לְוִיִּם וְיִשְׂרָאֵלִים\hdot
גֵּרִים וַעֲבָדִים מְשוּחְרָרִים\hdot
אֲבָל לֹא נָשִׁים\hdot
וַעֲבָדִים\hdot
וּקְטַנִּים\hdot
כָּל קָטָן שֶׁהִתְחִיל אָבִיו לִשְׁקוֹל עַל יָדוֹ\hdot
שׁוּב אֵינוֹ פוֹסֵק\hdot
וְאֵין מְמַשְׁכְּנִין אֶת הַכֹּהֲנִים\hdot
מִפְּנֵי דַרְכֵי שָׁלוֹם׃}

\halacha[327]{אָמַר רִבִּי יְהוּדָה הֵעִיד בֶּן בּוּכְרִי בְיַבְנֶה\hdot
שֶׁכָּל כֹּהֵן שֶׁשּׁוֹקֵל אֵינוֹ חוֹטֵא\hdot
אָמַר לוֹ רַבָּן יוֹחָנָן בֶּן זַכָּאי לֹא כִי\hdot
אֶלָּא כָּל כֹּהֵן שֶׁאֵינוֹ שׁוֹקֵל חוֹטֵא\hdot
אֶלָּא שֶׁהַכֹּהֲנִים דּוֹרְשִׁים מִקְרָא זֶה לְעַצְמָן\hdot
וְכָל מִנְחַת כֹּהֵן כָּלִיל תִּהְיֶה לֹא תֵאָכֵל\hdot
הוֹאִיל וְהָעֹמֶר וּשְׁתֵּי הַלֶּחֶם וְלֶחֶם הַפָּנִים שֶׁלָּנוּ\hdot
הֵיאַךְ נֶאֱכָלִים׃}

\halacha[597]{אַף עַל פִּי שֶׁאָֽמְרוּ\hdot
אֵין מְמַשְׁכְּנִין נָשִׁים וַעֲבָדִים קְטַנִּים\hdot
אִם שָֽׁקְלוּ מְקַבְּלִין מִיָּדָן\hdot
הַגּוֹי וְהַכּוּתִי שֶׁשָּֽׁקְלוּ\hdot
אֵין מְקַבְּלִין מִיָּדָן\hdot
וְאֵין מְקַבְּלִין מִיָּדָם קִנֵּי זָבִים וְקִנֵּי זָבוֹת וְקִנֵּי יוֹלְדוֹת\hdot
וְחַטָּאוֹת וַאֲשָׁמוֹת\hdot
אֲבָל נְדָרִים וּנְדָבוֹת מְקַבְּלִין מִיָּדָן\hdot
זֶה הַכְּלָל\hdot
כָּל שֶׁנִּדָּר וְנִדָּב מְקַבְּלִין מִיָּדָן\hdot
וְכָל שְׁאֵינוֹ לֹא נִדָּר וְלֹא נִדָּב אֵין מְקַבְּלִין מִיָּדָן\hdot
וְכֵן הוּא מְפוֹרָשׁ עַל יְדֵי עֶזְרָא\hdot
(שֶׁנֶּאֱמַר)
לֹא לָכֶם וְלָנוּ לִבְנוֹת בַּיִת לֵאלֹהֵינוּ׃}

\halacha{וְאֵלּוּ שֶׁחַיָּבִין בְּקְָלְבּוֹן לְוִיִּם וְיִשְׂרָאֵלִים\hdot
וְגֵרִים וַעֲבָדִים מְשׁוּחֲרָרִים\hdot
אֲבָל לֹא (כֹהֲנִים) נָשִׁים וַעֲבָדִים וּקְטַנִּים\hdot
הַשּׁוֹקֵל עַל יְדֵי כֹּהֵן\hdot
עַל יְדֵי אִשָּׁה\hdot
עַל יְדֵי עֶבֶד\hdot
עַל יְדֵי קָטָן\hdot
פָּטוּר\hdot
וְאִם שָׁקַל עַל יָדוֹ\hdot
וְעַל יַד חֲבֵירוֹ חַיָּיב בְּקָלְבּוֹן אֶחָד\hdot
רִבִּי מֵאִיר אוֹמֵר שְׁנֵי קָלְבָּנוֹת\hdot
הַנּוֹתֵן סֶלַע וְנוֹטֵל שֶׁקֶל\hdot
חַיָּיב שְׁנֵי קָלְבָּנוֹת׃}

\halacha[1009]{הַשּׁוֹקֵל עַל יְדֵי עָנִי\hdot
וְעַל יְדֵי שְׁכֵנוֹ\hdot
וְעַל יַד בֶּן עִירוֹ פָּטוּר\hdot
וְאִם הִלְווּם חַיָּיב\hdot
הָאַחִין הַשֻּׁתָּפִין שֶׁחַיָּיבִים בְּקָלְבּוֹן\hdot
פְּטוּרִין מִמַּעְשַׂר בְּהֵמָה\hdot
וְשֶׁחֲיָּיבִין בְּמַעְשַׂר בְּהֵמָה\hdot
פְּטוּרִין מִן הַקָּלְבּוֹן\hdot
וְכַמָּה הוּא קָלְבּוֹן\hdot
מְעָה כֶסֶף\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים חֲצִי׃}

\perek[1888]{מְצָֽרְפִין}{שְׁקָלִים לְדַרְכּוֹנוֹת\hdot
מִפְּנֵי מַשְׁאוּי הַדֶּרֶךְ\hdot
כְּשֵׁם שֶׁהָיוּ שׁוֹפָרוֹת בַּמִּקְדָּשׁ\hdot
כַּךְ הָיוּ שׁוֹפָרוֹת בַּמְּדִינָה\hdot
בְּנֵי הָעִיר שֶׁשָּֽׁלְחוּ אֶת שִׁקְלֵיהֶן\hdot
נִגְנָבוּ אוֹ שֶׁנֶּאֱבָדוּ\hdot
אִם נִתְרְמָה (הַתְּרוּמָה)\hdot
נִשְׁבָּעִין לַגִּזְבָּרִים\hdot
וְאִם לָאוּ\hdot
נִשְׁבָּעִין לִבְנֵי הָעִיר\hdot
וּבְנֵי הָעִיר שׁוֹקְלִין תַּחְתֵּיהֶן\hdot
נִמְצָאוּ אוֹ שֶׁהֶחְזִירוּם הַגַּנָּבִים\hdot
אֵלּוּ וָאֵלּוּ שְׁקָלִין\hdot
וְאֵין עוֹלִין לָהֶן לְשָׁנָה הַבָּאָה׃}

\halacha[153]{הַנּוֹתֵן שִׁקְלוֹ לַחֲבֵירוֹ לִשְׁקוֹל עַל יָדוֹ\hdot
וּשְׁקָלוֹ עַל יְדֵי עַצְמוֹ\hdot
אִם נִתְרְמָה תְרוּמָה מָעַל\hdot
הַשּׁוֹקֵל שִׁקְלוֹ מִמָּעוֹת הֶקְדֶּשׁ\hdot
אִם נִתְרְמָה תְרוּמָה וְקָֽרְבָה הַבְּהֵמָה\hdot
מָעַל\hdot
מִדְּמֵי מַעֲשֵׂר שֵׁנִי וּמִדְּמֵי שְׁבִיעִית\hdot
יֹאכַל כְּנֶגְדָּן׃}
