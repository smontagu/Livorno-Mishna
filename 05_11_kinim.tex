\tractatehead{מסכת קנים}{kinim}

\perek{חַטַּאת}{הָעוֹף נַעֲשֵׂית לְמַטָּן\hdot
וְחַטַּאת בְּהֵמָה לְמַעְלָה\hdot
עוֹלַת הָעוֹף נַעֲשֵׂית לְמַעְלָה\hdot
וְעוֹלַת בְּהֵמָה לְמַטָּה\hdot
אִם שִׁנָּה בַּזֶּה וּבַזֶּה פָּסוּל\hdot
סֵדֶר קִנִּים כַּךְ הוּא הַחוֹבָה\hdot
אֶחָד חַטָּאת וְאֶחָד עוֹלָה\hdot
בַּנְּדָרִים וּנְדָבוֹת כֻּלָּן עוֹלוֹת\hdot
אֵיזֶהוּ נֶדֶר\hdot
הָאוֹמֵר הֲרֵי עָלַי עוֹלָה\hdot
וְאֵיזֶהוּ נְדָבָה\hdot
הָאוֹמֵר הֲרֵי זוֹ עוֹלָה\hdot
מַה בֵּין נְדָרִים לַנְּדָבוֹת\hdot
אֶלָּא שֶׁהַנְּדָרִים מֵתוּ אוֹ נִגְנָבוּ חַיָּבִים בְּאַחֲרָיוּתָם\hdot
וּנְדָבוֹת מֵתוּ אוֹ נִגְנָבוּ אֵין חַיָּבִים בְּאַחֲרָיוּתָם׃}

\halacha{חַטָּאת שֶׁנִּתְעָֽרְבָה בָעוֹלָה וְעוֹלָה בַחַטָּאת\hdot
אֲפִילוּ אַחַת בְּרִבּוֹא יָמוּתוּ כֻלָּם\hdot
חַטָּאת שֶׁנִּתְעָֽרְבָה בַחוֹבָה\hdot
אֵין כָּשֵׁר אֶלָּא כְּמִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה\hdot
וְכֵן עוֹלָה שֶׁנִּתְעָֽרְבָה בַחוֹבָה\hdot
אֵין כָּשֵׁר אֶלָּא כְּמִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה\hdot
בֵּין הַחוֹבָה מְרֻבָּה\hdot
וְהַנְּדָבָה מְמוּעֶטֶת\hdot
בֵּין שֶׁהַנְּדָבָה מְרֻבָּה\hdot
וְהַחוֹבָה מְמוּעֶטֶת\hdot
בֵּין שֶׁשְּׁתֵּיהֶן שָׁווֹת׃}

\halacha{בַּמֶּה דְבָרִים אֲמוּרִים בַּחוֹבָה וּבַנְּדָבָה\hdot
אֲבָל בַּחוֹבָה שְׁנִּתְעָֽרְבוּ זוֹ בַזּוֹ\hdot
אַחַת לְזוֹ וְאַחַת לְזוֹ\hdot
שְׁתַּיִם לְזוֹ וּשְׁתַּיִם לְזוֹ\hdot
שָׁלֹשׁ לְזוֹ וְשָׁלֹשׁ לְזוֹ\hdot
מֶחְצָה כָּשֵׁר וּמֶחְצָה פָּסוּל\hdot
אַחַת לְזוֹ וּשְׁתַּיִם לְזוֹ\hdot
שָׁלֹשׁ לְזוֹ וְעֶשֶׂר לְזוֹ וּמֵאָה לְזוֹ\hdot
הַמּוּעָט כָּשֵׁר\hdot
בֵּין מִשֵּׁם אֶחָד\hdot
בֵּין מִשְּׁנֵי שֵׁמוֹת\hdot
בֵּין מֵאִשָּׁה אַחַת\hdot
בִּין מִשְּׁתֵּי נָשִׁים׃}

\halacha{כֵּיצַד מִשֵּׁם אֶחָד\hdot
לֵידָה וְלֵידָה\hdot
זִיבָה וְזִיבָה\hdot
מִשֵּׁם אֶחָד\hdot
מִשְּׁנֵי שֵׁמוֹת\hdot
לֵידָה וְזִיבָה\hdot
כֵּיצַד\hdot
שְׁתֵּי נָשִׁים עַל זוֹ לֵידָה\hdot
וְעַל זוֹ לֵידָה\hdot
עַל זוֹ זִיבָה וְעַל זוֹ זִיבָה\hdot
מִשֵּׁם אֶחָד\hdot
מִשְּׁנֵי שֵׁמוֹת\hdot
עַל זוֹ לֵידָה\hdot
וְעַל זוֹ זִיבָה\hdot
רִבִּי יוֹסֵי אוֹמֵר שְׁתֵּי נָשִׁים שֶׁלָּֽקְחוּ קִנֵּיהֶן בְּעֵרוּב\hdot
אוֹ שֶׁנָּֽתְנוּ דְּמֵי קִנֵּיהֶן לַכֹּהֵן\hdot
לְאֵיזוֹ שֶׁיִּרְצֶה כֹהֵן יַקְרִיב חַטָּאת\hdot
וּלְאֵיזֶה שֶׁיִּרְצֶה כֹהֵן יַקְרִיב עוֹלָה\hdot
בֵּין מִשֵּׁם אֶחָד\hdot
בֵּין מִשְּׁנֵי שֵׁמוֹת׃}

\perek{קֵן}{סְתוּמָה שֶׁפָּרַח מִמֶּנָּה גּוֹזָל לַאֲוֵיר\hdot
אוֹ שֶׁפָּרַח בֵּין הַמֵּתוֹת\hdot
אוֹ שֶׁמֵּת אֶחָד מֵהֶן\hdot
יִקַּח זוּג לַשֵּׁנִי\hdot
פָּרַח לְבֵין הַקְּרֵיבוֹת פָּסוּל\hdot
וּפוֹסֵל אַחֵר כְּנֶגְדּוֹ\hdot
שֶׁהַגּוֹזָל הַפּוֹרֵחַ\hdot
פָּסוּל\hdot
וּפוֹסֵל אַחֵר כְּנֶגְדּוֹ׃}

\halacha{כֵּיצַד\hdot
שְׁתֵּי נָשִׁים לְזוֹ שְׁתֵּי קִנִּים וּלְזוֹ שְׁתֵּי קִנֵּים\hdot
פָרַח מִזּוֹ לְזוֹ\hdot
פּוֹסֵל אֶחָד בַּהֲלִיכָתוֹ\hdot
חָזַר פּוֹסֵל אֶחָד בַּחֲזִירָתוֹ\hdot
פָּרַח וְחָזַר\hdot
פָּרַח וְחָזַר\hdot
לֹא הִפְסִיד כְּלוּם\hdot
שֶׁאֲפִילוּ הֵן מְעוּרָבוֹת\hdot
אֵין פָּחוֹת מִשְּׁתַּיִם׃}
