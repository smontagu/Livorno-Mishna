\tractatehead{מסכת שקלים}{shekalim}

\perek[157]{בְּאֶחָד}{בַּאֲדָר מַשְׁמִיעִין עַל הַשְּׁקָלִים\hdot
וְעַל הַכִּלְאַיִם\hdot
בַּחֲמִשָּׁה עָשָׂר בּוֹ\hdot
קוֹרִין אֶת הַמְּגִילָה בַּכְרָכִים\hdot
וּמְתַקְּנִין אֶת הַדְּרָכִים\hdot
וְאֶת הָרְחוֹבוֹת\hdot
וְאֶת מִקְוְאוֹת הַמַּיִם\hdot
וְעוֹשִׂין כָּל צָרְכֵי הָרַבִּים\hdot
וּמְצַיְּנִין אֶת הַקְּבָרוֹת\hdot
וְיוֹצְאִין אַף עַל הַכִּלְאַיִם׃}
