\tractatehead{מסכת מדות}{midot}

\perek{בִּשְׁלֹשָׁה}{מְקוֹמוֹת הַכֹּהֲנִים שׁוֹמְרִים בְּבֵית הַמִּקְדָּשׁ\hdot
בְּבֵית אַבְטִינָס\hdot
וּבְבֵית הַנִּצּוֹץ\hdot
וּבְבֵית הַמּוֹקֵד\hdot
וְהַלְוִיִּם בְּעֶשְׂרִים וְאֶחָד מָקוֹם\hdot
חֲמִשָּׁה עַל חֲמִשָּׁה שַׁעֲרֵי הַר הַבַּיִת\hdot
אַרְבָּעָה עַל אַרְבַּע פִּנּוֹתָיו מִתּוֹכוֹ\hdot
חֲמִשָּׁה עַל חֲמִשָּׁה שַׁעֲרֵי הָעֲזָרָה\hdot
אַרְבָּעָה עַל אַרְבַּע פִּנֹתֶיהָ מִבַּחוּץ\hdot
וְאֶחָד בְּלִשְׁכַּת הַקָּרְבָּן\hdot
וְאֶחָד בְּלִשְׁכַּת הַפָּרֹכֶת\hdot
וְאֶחָד לְאַחוֹרֵי בֵית הַכַּפֹּרֶת׃}

\halacha{אִישׁ הַר הַבַּיִת הָיָה מְחַזֵּר עַל כָּל מִשְׁמָר וּמִשְׁמָר\hdot
וַאֲבוּקוֹת דּוֹלְקוֹת לְפָנָיו\hdot
וְכָל מִשְׁמָר שֶׁאֵינוֹ עוֹמֵד\hdot
אוֹמֵר לוֹ אִישׁ הַר הַבַּיִת שָׁלוֹם עָלֶיךָ\hdot
נִכָּר שֶׁהוּא יָשֵׁן\hdot
חוֹבְטוֹ בְמַקְלוֹ\hdot
וּרְשׁוּת הָיָה לוֹ לִשְׂרוֹף אֶת כְּסוּתוֹ\hdot
וְהֵם אוֹמְרִים מַה קּוֹל בַּעֲזָרָה\hdot
קוֹל בֶּן לֵוִי לוֹקֶה וּבְגָדָיו נִשְׂרָפִין\hdot
שֶׁיָּשֵׁן לוֹ עַל מִשְׁמָרוֹ\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר פַּעַם אַחַת מָֽצְאוּ אֶת אֲחִי אִמָּה יָשֵׁן\hdot
וְשָֽׂרְפוּ אֶת כְּסוּתוֹ׃}

\halacha{חֲמִשָּׁה שְׁעָרִים הָיוּ לְהַר הַבַּיִת\hdot
שְׁנֵי שַׁעֲרֵי חֻלְדָּה מִן הַדָּרוֹם\hdot
מְשַׁמְּשִׁין כְּנִיסָה וִיצִיאָה\hdot
קִיפוֹנוֹס מִן הַמַּעֲרָב (מְשַׁמֵּשׁ כְּנִיסָה וִיצִיאָה)\hdot
טֶדִי מִן הַצָּפוֹן לֹא הָיָה מְשַׁמֵּשׁ כְּלוּם\hdot
שַׁעַר הַמִּזְרָחִי עָלָיו שׁוּשָׁן הַבִּירָה צוּרָה\hdot
שֶׁבּוֹ כֹּהֵן גָּדוֹל הַשּׂוֹרֵף אֶת הַפָּרָה\hdot
וְכָל מְסַעֲדֶיהָ\hdot
יוֹצְאִין לְהַר הַמִּשְׁחָה׃}

\halacha{שִׁבְעָה שְׁעָרִים הָיוּ בָעֲזָרָה\hdot
שְׁלֹשָׁה בַצָּפוֹן וּשְׁלֹשָׁה בַדָּרוֹם\hdot
וְאֶחָד בַּמִּזְרָח\hdot
שֶׁבַּדָּרוֹם שַׁעַר הַדֶּלֶק\hdot
שֵׁנִי לוֹ שַׁעַר הַבְּכוֹרוֹת\hdot
שְׁלִישִׁי לוֹ שַׁעַר הַמַּיִם\hdot
שֶׁבַּמִּזְרָח שַׁעַר נִיקָנוֹר\hdot
וּשְׁתֵּי לְשָׁכוֹת הָיוּ לוֹ\hdot
אַחַת מִימִינוֹ וְאַחַת מִשְּׂמֹאלוֹ\hdot
אַחַת לִשְׁכַּת פִּינְחָס הַמַּלְבִּישׁ\hdot
וְאַחַת לִשְׁכַּת עוֹשֵׂי חֲבִיתִין׃}

\halacha{וְשֶׁבַּצָּפוֹן שַׁעַר הַנִּצּוֹץ\hdot
וּכְמִין אַכְסַדְרָה הָיָה\hdot
וַעֲלִיָּה בְּנוּיָה עַל גַּבָּיו\hdot
שֶׁהַכֹּהֲנִים שׁוֹמְרִים מִלְּמַעְלָן\hdot
וְהַלְוִיִּם מִלְּמַטָּן\hdot
וּפֶתַח הָיָה לוֹ לַחֵיל\hdot
שֵׁנִי לוֹ שַׁעַר הַקָּרְבָּן\hdot
שְׁלִישִׁי לוֹ בֵּית הַמּוֹקֵד׃}

\halacha{אַרְבַּע לְשָׁכוֹת הָיוּ בְּבֵית הַמּוֹקֵד\hdot
כְּקִטּוֹנוֹת פְּתוּחוֹת לִטְרַקְלִין\hdot
שְׁתַּיִם בַּקֹּדֶשׁ וּשְׁתַּיִם בַּחוֹל\hdot
וְרָאשֵׁי פִּסְפְּסִין מַבְדִּילִין בֵּין קֹדֶשׁ לְחוֹל\hdot
וּמַה הָיוּ מְשַׁמְּשׁוֹת\hdot
מַעֲרָבִית דְּרוֹמִית\hdot
הִיא הָֽיְתָה לִשְׁכַּת (טְלָאֵי) קָרְבָּן\hdot
דְּרוֹמִית מִזְרָחִית הִיא הָֽיְתָה לִשְׁכַּת (עוֹשֵׂי) לֶחֶם הַפָּנִים\hdot
מִזְרָחִית צְפוֹנִית בָּהּ גָּֽנְזוּ בְנֵי חַשְׁמוֹנָאי אֶת אַבְנֵי הַמִּזְבֵּחַ\hdot
שֶׁשִּׁקְּצוּם מַלְכֵי יָוָן\hdot
צְפוֹנִית מַעֲרָבִית\hdot
בָּהּ יוֹרְדִים לְבֵית הַטְּבִילָה׃}

\halacha{שְׁנֵי שְׁעָרִים הָיוּ לְבֵית הַמּוֹקֵד\hdot
אֶחָד פָּתוּחַ לַחֵיל\hdot
וְאֶחָד פָּתוּחַ לָעֲזָרָה\hdot
אָמַר רִבִּי יְהוּדָה זֶה שֶׁהָיָה פָּתוּחַ לָעֲזָרָה\hdot
פִּשְׁפֵּשׁ קָטָן הָיָה לוֹ\hdot
שֶׁבּוֹ נִכְנָסִין לִבְלוֹשׁ אֶת הָעֲזָרָה׃}

\halacha{בֵּית הַמּוֹקֵד כִּיפָה וּבַיִת גָּדוֹל הָיָה\hdot
מֻקָּף רוֹבְדִים שֶׁל אֶבֶן\hdot
וְזִקְנֵי בֵית אָב ישֵׁנִים שָׁם\hdot
וּמַפְתְּחוֹת הָעֲזָרָה בְּיָדָם\hdot
וּפִרְחֵי כְהוּנָה אִישׁ כְּסָתוֹ בָאָרֶץ\hdot
וּמָקוֹם הָיָה שָׁם אַמָּה עַל אַמָּה\hdot
וְטַבְלָא שֶׁל שַׁיִשׁ\hdot
וְטַבַּעַתַ הָֽיְתָה קְבוּעָה בָהּ\hdot
וְשַׁלְשֶׁלֶת שֶׁהַמַּפְתְּחוֹת הָיוּ תְלוּיוֹת בָּהּ\hdot
הִגִּיעַ זְמַן הַנְּעִילָה הִגְבִּיהַּ אֶת הַטַּבְלָה בַּטַּבַּעַת\hdot
וְנָטַל אֶת הַמַּפְתְּחוֹת מִן הַשַּׁלְשֶׁלֶת\hdot
וְנָעַל הַכֹּהֵן מִבִּפְנִים\hdot
וּבֶן לֵוִי יוֹשֵׁב לוֹ מִבַּחוּץ\hdot
גָּמַר מִלִּנְעוֹל\hdot
הֶחֱזִיר הַמַּפְתְּחוֹת לַשַּׁלְשֶׁלֶת\hdot
וְאֶת הַטַּבְלָא לִמְקוֹמָהּ\hdot
נָתַן כְּסֻתוֹ עָלֶיהָ\hdot
יָשֵׁן לוֹ\hdot
אִירַע קֶרִי בְּאַחַד מֵהֶם\hdot
יוֹצֵא וְהוֹלֵךְ לוֹ\hdot
בַּמְּסִיבָה הַהוֹלֶכֶת תַּחַת הַבִּירָה\hdot
וְהַנֵּרוֹת דּוֹלְקִים מִכָּאן וּמִכָּאן\hdot
עַד שֶׁהוּא מַגִּיעַ לְבֵית הַטְּבִילָה\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר בַּמְּסִיבָה הַהוֹלֶכֶת תַּחַת הַחֵיל\hdot
יוֹצֵא וְהוֹלֵךְ לוֹ בַטֶּדִי׃}

\perek{הַר}{הַבַּיִת הָיָה חָמֵשׁ מֵאוֹת אַמָּה עַל חָמֵשׁ מֵאוֹת אַמָּה\hdot
רֻבּוֹ מִן הַדָּרוֹם\hdot
שֵׁנִי לוֹ מִן הַמִּזְרָח\hdot
שְׁלִישִׁי לוֹ מִן הַצָּפוֹן\hdot
מִעוּטוֹ מִן הַמַּעֲרָב\hdot
מָקוֹם שֶׁהָיָה רוֹב מִדָּתוֹ\hdot
שָׁם הָיָה רוֹב תַּשׁמִישׁוֹ\hdot
׃}

\halacha{כָּל הַנִּכְנָסִין לְהַר הַבַּיִת\hdot
נִכְנָסִין דֶּרֶךְ יָמִין\hdot
וּמַקִּיפִין וְיוֹצְאִין דֶּרֶךְ שְׂמֹאל\hdot
חוּץ מִמִּי שֶׁאִרְעוֹ דָבָר\hdot
שֶׁהוּא מַקִּיף לִשְׂמֹאל\hdot
מַה לְּךָ מַקִּיף לִשְׂמֹאל\hdot
שֶׁאֲנִי אָבֵל\hdot
הַשּׁוֹכֵן בַּבַּיִת הַזֶּה יְנַחֲמֶךָ\hdot
שֶׁאֲנִי מְנֻדֶּה\hdot
הַשּׁוֹכֵן בַּבַּיִת הַזֶּה\hdot
יִתֵּן בְּלִבָּם וִיקָֽרְבוּךָ\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
אָמַר לוֹ רִבִּי יוֹסֵי עֲשִׂיתֶן כְּאִלּוּ עָֽבְרוּ עָלָיו אֶת הַדִּין\hdot
אֶלָּא הַשּׁוֹכֵן בַּבַּיִת הַזֶּה יִתֵּן בְּלִבָּךְ\hdot
וְתִשְׁמַע דִּבְרֵי חֲבֵרֶיךָ וִיקָֽרְבוּךָ׃}

\halacha{לִפְנִים מִמֶּנּוּ סוּרָג גָּבוֹהַּ עֲשָׂרָה טְפָחִים\hdot
וּשְׁלֹשׁ עֶשְׂרֵה פְרָצוֹת הָיוּ בוֹ\hdot
שֶׁפְּרָצוּם מַלְכֵי יָוָן\hdot
חָֽזְרוּ וְּגָדרוּם\hdot
וְגָֽזְרוּ כְּנֶגְדָּן שָׁלֹשׁ עֶשְׂרֵה הִשְׁתַּחֲוָיוֹת\hdot
לִפְנִים מִמֶּנּוּ הַחֵיל\hdot
עֶשֶׂר אַמּוֹת\hdot
וּשְׁתֵּים עֶשְׂרֵה מַעֲלוֹת הָיוּ שָׁם\hdot
רוּם מַעֲלָה חֲצִי אַמָּה\hdot
וְשִׁלְחָהּ חֲצִי אַמָּה\hdot
כָּל הַמַּעֲלוֹת שֶׁהָיוּ שָׁם\hdot
רוּם מַעֲלָה חֲצִי אַמָּה\hdot
וְשִׁלְחָהּ חֲצִי אַמָּה\hdot
חוּץ מִשֶּׁל אוּלָם\hdot
כָּל הַפְּתָחִים (וְהַשְּׁעָרִים) שֶׁהָיוּ שָׁם\hdot
גָּבְהָן עֶשְׂרִים אַמָּה\hdot
וְרָחְבָּן עֶשֶׂר אַמּוֹת\hdot
חוּץ מִשֶּׁל אוּלָם\hdot
כָּל הַפְּתָחִים שֶׁהָיוּ שָׁם\hdot
הָיוּ לָהֶן דְּלָתוֹת\hdot
חוּץ מִשֶּׁל אוּלָם\hdot
כָּל הַשְּׁעָרִים שֶׁהָיוּ שָׁם\hdot
הָיוּ לָהֶם שְׁקוּפוֹת חוּץ מִשַּׁעַר טֶדִי\hdot
שֶׁהָיוּ שָׁם שְׁתֵּי אֲבָנִים מֻטּוֹת זוֹ עַל גַּבֵּי זוֹ\hdot
כָּל הַשְּׁעָרִים שֶׁהָיוּ שָׁם\hdot
נִשְׁתַּנּוּ לִהְיוֹת שֶׁל זָהָב\hdot
חוּץ מִשַּׁעַר נִקָּנוֹר מִפְּנֵי שֶׁנַּעֲשָׂה בָהֵן נֵס\hdot
וְיֵשׁ אוֹמְרִים מִפְּנֵי שֶׁנְּחֻשְׁתָּן מַצְהִיב׃}

\halacha{כָּל הַכּוֹתָלִים שֶׁהָיוּ שָׁם\hdot
הָיוּ גְבוֹהִים חוּץ מִכֹּתֶל הַמִּזְרָחִי שֶׁהַכֹּהֵן הַשּׂוֹרֵף אֶת הַפָּרָה עוֹמֵד בְּרֹאשׁ הַר הַמִּשְׁחָה\hdot
וּמִתְכַּוֵּין וְרוֹאֶה בְּפִתְחוֹ שֶׁל הֵיכָל\hdot
בְּשַׁעַת הַזָּיַת הַדָּם׃}

\halacha{עֲזָרַת הַנָּשִׁים הָֽיְתָה אֹרֶךְ מֵאָה וּשְׁלֹשִׁים וְחָמֵשׁ\hdot
עַל רֹחַב מֵאָה וּשְׁלֹשִׁים וְחָמֵשׁ\hdot
וְאַרְבַּע לְשָׁכוֹת הָיוּ בְּאַרְבַּע מִקְצִעוֹתֶיהָ שֶׁל אַרְבָּעִים אַרְבָּעִים אַמָּה\hdot
וְלֹא הָיוּ מְקוֹרוֹת\hdot
וְכַךְ הֵם עֲתִידִים לִהְיוֹת שֶׁנֶּאֱמַר\hdot
וַיּוֹצִיאֵנִי אֶל הֶחָצֵר הַחִיצוֹנָה וַיַּעֲבִירֵנִי אֶל אַרְבַּעַת מִקְצוֹעֵי הֶחָצֵר וְהִנֵּה חָצֵר בִּמִקְצוֹעַ הֶחָצֵר\hdot
חָצֵר בִּמִקְצוֹעַ הֶחָצֵר\hdot
בְּאַרְבַּע מִקְצוֹעוֹת הֶחָצֵר חֲצֵרוֹת קְטוּרוֹת\hdot
וְאֵין קְטוּרוֹת אֶלָּא שֶׁאֵינָן מְקוֹרוֹת\hdot
וּמָה הָיוּ מְשַׁמְּשׁוֹת\hdot
דְּרוֹמִית מִזְרָחִית הִיא הָֽיְתָה לִשְׁכַּת הַנְּזִירִים\hdot
שֶׁשָּׁם הַנְּזִירִים מְבַשְּׁלִין אֶת שַׁלְמֵיהֶן\hdot
וּמְגַלְּחִין אֶת שְׂעָרָן\hdot
וּמְשַׁלְּחִין תַּחַת הַדּוּד\hdot
מִזְחָיִת צְפוֹנִית הִיא הָֽיְתָה לִשְׁכַּת דִּיר הָעֵצִים\hdot
שֶׁשָּׁם הַכֹּהֲנִים בַּעֲלֵי מוּמִין מַתְלִיעִים הָעֵצִים\hdot
וְכָל עֵץ שֶׁנִּמְצָא בוֹ תוֹלַעַת\hdot
פָּסוּל מֵעַל גַּבֵּי הַמִּזְבֵּחַ\hdot
צְפוֹנִית מַעֲרָבִית הִיא הָֽיְתָה לִשְׁכַּת מְצוֹרָעִים\hdot
מַעֲרָבִית דּרוֹמִית\hdot
אָמַר רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב שָׁכַחְתִּי מַה הַֽיְתָה מְשַׁמֶּשֶׁת\hdot
אַבָּא שָׁאוּל אוֹמֵר שָׁם הָיוּ נוֹתְנִין יַיִן וְשֶׁמֶן\hdot
וְהִיא הָֽיְתָה נִקְרֵאת לִשְׁכַּת בֵּית שְׁמַנְיָה\hdot
וַחֲלָקָה הָֽיְתָה בָרִאשׁוֹנָה\hdot
וְהִקִּיפוּהָ כְּצוֹצְרָה\hdot
שֶׁהַנָּשִׁים רוֹאוֹת מִלְּמַעְלָן\hdot
וְהָאֲנָשִׁים מִלְּמַטָּן\hdot
כְּדֵי שֶׁלֹּא יְהוּ מְעוּרָבִין\hdot
וַחֲמֵשׁ עֶשְׂרֵה מַעֲלוֹת עוֹלוֹת מִתּוֹכָהּ\hdot
לַעֲזָרַת יִשְׂרָאֵל\hdot
כְּנֶגֶד חֲמֵשׁ עֶשְׂרֵה שִׁיר הַמַּעֲלוֹת שֶׁבַּתְּהִלִּים\hdot
שֶׁעֲלֵיהֶן הַלְוִיִּם עוֹמְדִים בַּשִּׁיר\hdot
לֹא הָיוּ טְרוּטוֹת\hdot
אֶלָּא מֻקָּפוֹת כַּחֲצִי גֹרֶן עֲגוּלָה׃}

\halacha{וּלְשָׁכוֹת הָיוּ תַּחַת עֲזָרַת יִשְׂרָאֵל\hdot
וּפְתוּחוֹת לַעֲזָרַת הַנָּשִׁים\hdot
שֶׁשָּׁם הַלְוִיִּם נוֹתְנִים כִּנּוֹרוֹת וּנְבָלִים וּמְצִלְתַּיִם\hdot
וְכָל כְּלֵי שִׁיר\hdot
עֲזָרַת יִשְׂרָאֵל הָֽיְתָה אֹרֶך מֵאָה וּשְׁלֹשִׁים וְחָמֵשׁ\hdot
עַל רֹחַב אַחַת עֶשְׂרֵה\hdot
וְכֵן עַזֲרָת כֹּהֲנִים\hdot
הָֽיְתָה אֹרֶך מֵאָה וּשְׁלֹשִׁים וְחָמֵשׁ\hdot
עַל רֹחַב אַחַת עֶשְׂרֵה\hdot
וְרָאשֵׁי פִסְפְסִין מַבְדִּילִין\hdot
בֵּין עֲזָרַת יִשְׂרָאֵל לַעֲזָרַת כֹּהֲנִים\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר מַעֲלָה הָֽיְתָה שָׁם\hdot
וּגְבוֹהָה אָמָּה\hdot
וְהַדּוּכָן נָתוּן עָלֶיהָ\hdot
וּבָהּ שָׁלֹשׁ מַעֲלוֹת שֶׁל חֲצִי חֲצִי אַמָּה\hdot
נִמְצֵאת עֲזָרַת הַכֹּהֲנִים גְּבוֹהָה מֵעֲזָרַת יִשְׂרָאֵל\hdot
שְׁתֵּי אַמּוֹת וּמֶחְצָה\hdot
כָּל הָעֲזָרָה הָֽיְתָה אֹרֶךְ מֵאָה וּשְׁמוֹנִים וָשֶׁבַע\hdot
עַל רֹחַב מֵאָה וּשְׁלֹשִׁים וְחָמֵשׁ\hdot
וְשָׁלֹשׁ עֶשְׂרֵה הִשְׁתַּחֲוָיוֹת הָיוּ שָׁם\hdot
אַבָּא יוֹסֵי בֶן חָנֵן אוֹמֵר כְּנֶגֶד שְׁלֹשָׁה עָשָׂר שְׁעָרִים\hdot
(שְׁעָרִים דְּרוֹמִיִּים סְמוּכִים לַמַּעֲרָב\hdot)
שַׁעַר הָעֶלְיוֹן\hdot
שַׁעַר הַדֶּלֶק\hdot
שַׁעַר הַבְּכוֹרוֹת\hdot
שַׁעַר הַמַּיִם\hdot
וְלָמָּה נִקְרָא שְׁמוֹ שַׁעַר הַמַּיִם\hdot
שֶׁבּוֹ מַכְנִיסִין צְלוֹחִית שֶׁל מַיִם שֶׁל נִסּוּךְ בֶּחָג\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר וּבוֹ הַמַּים מְפַכִּים\hdot
וַעֲתִידִין לִהְיוֹת יוֹצְאִין מִתַּחַת מִפְתַּן הַבַּיִת\hdot
וּלְעֻמָתָן בַּצָּפוֹן סְמוּכִין לַמַּעֲרָב\hdot
שַׁעַר יְכָנְיָה\hdot
שַׁעַר הַקָּרְבָּן\hdot
שַׁעַר הַנָּשִׁים\hdot
שַׁעַר הַשִּׁיר\hdot
וְלָמָּה נִקְרָא שְׁמוֹ שַׁעַר יְכָנְיָה\hdot
שֶׁבּוֹ יָצָא יְכָנְיָה בְּגָלוּתוֹ\hdot
שֶׁבַּמִּזְרָח שַׁעַר נִקָּנוֹר\hdot
וּשְׁנֵי פִשְׁפְּשִׁים הָיוּ לוֹ\hdot
אֶחָד בִּימִינוֹ וְאֶחָד בִּשְׂמֹאלוֹ\hdot
וּשְׁנַיִם בַּמַּעֲרָב לֹא הָיָה לָהֶם שֵׁם׃}
