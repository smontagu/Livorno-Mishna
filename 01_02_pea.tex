\tractatehead{מסכת פאה}{pea}

\perek{אֵלּוּ}{דְבָרִים שֶׁאֵין לָהֶם שִׁיעוּר\hdot
הַפֵּאָה\hdot
וְהַבִּכּוּרִים\hdot
וְהָרֵאָיוֹן\hdot
וּגְמִילוּת חֲסָדִים\hdot
וְתַלְמוּד תּוֹרָה\hdot
אֵלּוּ דְבָרִים שֶׁאָדָם אוֹכֵל פֵּירוֹתֵיהֶן בָּעוֹלָם הַזֶּה\hdot
וְהַקֶּרֶן קַיֶּמֶת לוֹ לְעוֹלָם הַבָּא\hdot
כִּבּוּד אָב וָאֵם\hdot
וּגְמִילוּת חֲסָדִים\hdot
וַהֲבָאַת שָׁלוֹם בֵּין אָדָם לַחֲבֵירוֹ\hdot
וְתַלְמוּד תּוֹרָה כְּנֶגֶד כֻּלָּם׃}

\halacha{אֵין פּוֹחֲתִין לַפֵּאָה מִשִּׁשִּׁים\hdot
וְאַף עַל פִּי שֶׁאָֽמְרוּ\hdot
אֵין לַפֵּאָה שִׁיעוּר\hdot
הַכֹּל לְפִי גֹדֶל הַשָּׂדֶה\hdot
וּלְפִי רוֹב הָעֲנִיִּים\hdot
וּלְפִי רוֹב הָעֲנָוָה׃}

\halacha{נוֹתְנִין פֵּאָה מִתְּחִלַּת הַשָּׂדֶה וּמֵאֶמְצָעָהּ\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר וּבִלְבַד שֶׁיִּתֵּן בַּסּוֹף כַּשִּׁיעוּר\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם שִׁיֵּיר קֶלַח אֶחַד\hdot
סוֹמֵךְ לוֹ מִשּׁוּם פֵּאָה\hdot
וְאִם לָאוּ\hdot
אֵינוֹ נוֹתֵן אֶלָּא מִשּׁוּם הֶפְקֵר׃}

\halacha{כְּלָל אָֽמְרוּ בַּפֵּאָה\hdot
כָּל שֶׁהוּא אוֹכֶל\hdot
וְנִשְׁמָר\hdot
וְגִדּוּלָיו מִן הָאָרֶץ\hdot
וּלְקִיטָתוֹ כְּאֶחָד\hdot
וּמַכְנִיסוֹ לְקִיּוּם\hdot
חַיָּיב בַּפֵּאָה\hdot
וְהַתְּבוּאָה וְהַקָּטְנִיּוֹת בַּכְּלָל הַזֶּה׃}

\halacha{(וּבָאִילָן) הָאוֹג וְהַחֲרוּבִין וְהָאֱגוֹזִים וְהַשְּׁקֵדִים
וְהַגְּפָנִים וְהָרִמּוֹנִים וְהַזֵּיתִים וְהַתְּמָרִים חַיָּיבִין בַּפֵּאָה׃}

\halacha{לְעוֹלָם הוּא נוֹתֵן מִשּׁוּם פֵּאָה\hdot
וּפָטוּר מִן הַמַּעַשְׂרוֹת עַד שֶׁיִּמְרַח\hdot
וְנוֹתֵן מִשּׁוּם הֶפְקֵר\hdot
וּפָטוּר מִן הַמַּעַשְׂרוֹת עַד שֶׁיִּמְרַח\hdot
וּמַאֲכִיל לַבְּהֵמָה וְלַחַיָּה וְלָעוֹפוֹת\hdot
וּפָטוּר מִן הַמַּעַשְׂרוֹת עַד שֶׁיִּמְרַח\hdot
וְנוֹטֵל מִן הַגּוֹרֶן וְזוֹרֵעַ\hdot
וּפָטוּר מִן הַמַּעַשְׂרוֹת עַד שֶׁיִּמְרַח\hdot
דִּבְרֵי רִבִּי עֲקִיבָא\hdot
כֹּהֵן וְלֵוִי שֶׁלָּֽקְחוּ אֶת הַגּוֹרֶן\hdot
הַמַּעַשְׂרוֹת שֶׁלָּהֶם עַד שֶׁיּמְרָחוּ\hdot
הַמַּקְדִּישׁ וּפוֹדֶה חַיָּיב בַּמַּעַשְׂרוֹת\hdot
עַד שֶׁיִּמְרַח הַגִּזְבָּר׃}

\perek{וְאֵלּוּ}{מַפְסִיקִין לַפֵּאָה\hdot
הַנַּחַל וְהַשְּׁלוֹלִית\hdot
וְדֶרֶךְ הַיָּחִיד\hdot
וְדֶרֶךְ הָרַבִּים\hdot
וּשְׁבִיל הָרַבִּים\hdot
וּשְׁבִיל הַיָּחִיד הַקָּבוּעַ בִּימוֹת הַחַמָּה וּבִימוֹת הַגְּשָׁמִים\hdot
וְהַבּוֹר וְהַנִּיר וְזֶרַע אַחֵר\hdot
וְהַקּוֹצֵר לְשַׁחַת מַפְסִיק\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים אֵינוֹ מַפְסִיק אֶלָּא אִם כֵּן חָרָשׁ׃}

\halacha{אַמַּת הַמַּיִם שֶׁאֵינָהּ יְכוֹלָה לְהִקָּצֵר כְּאֶחַת\hdot
רִבִּי יְהוּדָה אוֹמֵר מַפְסֶקֶת\hdot
וְכָל הֶהָרִים אֲשֶׁר בַמַּעְדֵּר יֵעָֽדְרוּן\hdot
אַף עַל פִּי שֶׁאֵין הַבַּקָּר יָכוֹל לַעֲבוֹר בְּכֵלָיו\hdot
הוּא נוֹתֵן פֵּאָה לַכֹּל׃}

\halacha{הַכֹּל מַפְסִיק לַזְּרָעִים\hdot
וְאֵינוֹ מַפְסִיק לָאִילָן אֶלָּא גָדֵר\hdot
וְאִם הָיָה שֵׂעָר כּוֹתֵשׁ אֵינוֹ מַפְסִיק\hdot
אֶלָּא נוֹתֵן פֵּאָה לַכֹּל׃}

\halacha{וְלַחֲרוּבִין\hdot
כָּל הָרוֹאִין זֶה אֶת זֶה\hdot
אָמַר רַבָּן גַּמְלִיאֵל נוֹהֲגִין הָיוּ בֵית אַבָּא
שֶׁהָיוּ נוֹתְנִין פֵּאָה אַחַת לַזֵּיתִים שֶׁהָיוּ לָהֶם בְּכָל רוּחַ\hdot
וְלַחֲרוּבִין כָּל הָרוֹאִין זֶה אֶת זֶה\hdot
רִבִּי אֱלִיעֶזֶר בְּרִבִּי צָדוֹק אוֹמֵר מִשְּׁמוֹ\hdot
אַף לַחֲרוּבִין שֶׁהָיוּ לָהֶם בְּכָל הָעִיר׃}

\halacha{הַזּוֹרֵעַ אֶת שָׂדֵהוּ מִין אֶחָד\hdot
אַף עַל פִּי שֶׁהוּא עוֹשֵׂהוּ שְׁתֵּי גְּרָנוֹת\hdot
נוֹתֵן פֵּאָה אֶחָת\hdot
זְרָעָהּ שְׁנֵי מִינִין\hdot
אַף עַל פִּי שׁעֲשָׂאָן גּוֹרֶן אַחַת נוֹתֵן שְׁתֵּי פֵאוֹת\hdot
הַזּוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִטִּין\hdot
עֲשָׂאָן גוֹרֵן אַחַת נוֹתֵן פֵּאָה אֶחַת\hdot
שְׁתֵּי גְרָנוֹת נוֹתֵן שְׁתֵּי פֵאוֹת׃}

\halacha{מַעֲשֵׂה שֶׁזָּרַע רִבִּי שִׁמְעוֹן אִישׁ הַמִּצְפָּה (וּבָא) לִפְנֵי רַבָּן גַּמְלִיאֵל וְעָלוּ לְלִשְׁכַּת הַגָּזִית וְשָׁאֲלוּ\hdot
אָמַר נָחוּם הַלַּבְלָר\hdot
מְקֻבַּל אֲנִי מֵרִבִּי מִיאֲשָׁא שֶׁקִּבֵּל מֵאַבָּא\hdot
שֶׁקִּבֵּל מִן הַזּוּגוֹת שֶׁקִּבְּלוּ מִן הַנְּבִיאִים\hdot
הֲלָכָה לְמֹשֶׁה מִסִּינַי\hdot
בַּזּוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִיטִּין\hdot
אִם עֲשָׂאָן גּוֹרֶן אַחַת נוֹתֵן פֵּאָה אֶחָת\hdot
שְׁתֵּי גְרָנוֹת נוֹתֵן שְׁתֵּי פֵאוֹת׃}

\halacha{שָׂדֶה שֶׁקְּצָרוּהָ גּוֹיִם\hdot
קְצָרוּהָ לִסְטִים\hdot
קִרְסְמוּהָ נְמָלִים\hdot
שִׁבְּרַתָּהּ הָרוּחַ אוֹ בְהֵמָה\hdot
פְּטוּרָה\hdot
קָצַר חֶצְיָהּ וְקָֽצְרוּ לִסְטִים חֶצְיָהּ\hdot
פְּטוּרָה\hdot
שֶׁחוֹבַת הַפֵּאָה בַּקָּמָה׃}

\halacha{קָֽצְרוּ לִסְטִים חֶצְיָהּ\hdot
וְקָצַר הוּא חֶצְיָהּ\hdot
נוֹתֵן פֵּאָה מִמָּה שֶׁקָּצַר\hdot
קָצַר חֶצְיָהּ וּמָכַר חֶצְיָהּ\hdot
הַלּוֹקֵחַ נוֹתֵן פֵּאָה לַכֹּל\hdot
קָצַר חֶצְיָהּ וְהִקְדִּישׁ חֶצְיָהּ\hdot
הַפּוֹדֶה מִיַּד הַגִּזְבָּר\hdot
הוּא נוֹתֵן פֵּאָה לַכֹּל׃}

\perek{מַלְבֵּנוֹת}{הַתְּבוּאָה שֶׁבֵּין הַזֵּיתִים\hdot
בֵּית שַׁמָּאי אוֹמְרִים פֵּאָה מִכָּל אַחַת וְאֶחָת\hdot
וּבֵית הִלֵּל אוֹמְרִים מֵאַחַת עַל הַכֹּל\hdot
וּמוֹדִים שֶׁאִם הָיוּ רָאשֵׁי שׁוּרוֹת מְעוּרָבִין\hdot
שֶׁהוּא נוֹתֵן פֵּאָה מֵאַחַת עַל הַכֹּל׃}

\halacha{הַמְּנַמֵּר אֶת שָׂדֵהוּ וְשִׁיֵּיר קְלָחִים לַחִים\hdot
רִבִּי עֲקִיבָא אוֹמֵר נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד\hdot
וַחֲכָמִים אוֹמְרִים מֵאֶחָד עַל הַכֹּל\hdot
וּמוֹדִים חֲכָמִים לְרִבִּי עֲקִיבָא בְּזוֹרֵעַ שֶׁבֶת אוֹ חַרְדָּל בִּשְׁנַיִם וּבִשְׁלֹשָׁה מְקוֹמוֹת\hdot
שֶׁהוּא נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד׃}

\halacha{הַמַּחֲלִיק בְּצָלים לַחִים לַשּׁוּק\hdot
וּמְקַיֵּים יְבֵשִׁים לַגּוֹרֶן\hdot
נוֹתֵן פֵּאָה לְאֵלּוּ לְעַצְמָן\hdot
וּלְאֵלּוּ לְעַצְמָן\hdot
וְכֵן בָּאֲפוּנִין\hdot
וְכֵן בַּכֶּרֶם\hdot
הַמַּדֵּל נוֹתֵן מִן הַמְּשׁוּאָר עַל מַה שֶׁשִּׁיֵּר\hdot
וְהַמַּחֲלִיק מֵאַחַת יַד נוֹתֵן מִן הַמְּשׁוּאָר עַל הַכֹּל׃}

\halacha{הָאִמָּהוֹת שֶׁל בְּצָלִים חַיָּיבוֹת בַּפֵּאָה\hdot
וְרִבִּי יוֹסֵי פּוֹטֵר\hdot
מַלְּבְּנוֹת הַבְּצָלִים שֶׁבֵּין הַיָּרָק\hdot
רִבִּי יוֹסֵי אוֹמֵר פֵּאָה מִכָּל אַחַת וְאֶחָת\hdot
וַחֲכָמִים אוֹמְרִים מֵאַחַת עַל הַכֹּל׃}

\halacha{הָאַחִין שֶׁחָֽלְקוּ נוֹתְנִין שְׁתֵּי פֵאוֹת\hdot
חָֽזְרוּ וְנִשְׁתַּתְּפוּ\hdot
נוֹתְנִין פֵּאָה אֶחַת\hdot
שְׁנַיִם שֶׁלָֽקְחוּ אֶת הָאִילָן נוֹתְנִין פֵּאָה אֶחָת\hdot
לָקַח זֶה צְפוֹנוֹ וְזֶה דְּרוֹמוֹ\hdot
זֶה נוֹתֵן פֵּאָה לְעַצְמוֹ\hdot
וְזֶה נוֹתֵן פֵּאָה לְעַצְמוֹ\hdot
הַמּוֹכֵר קִלְחֵי אִילָן בְּתוֹךְ שָׂדֶהוּ\hdot
נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד\hdot
אָמַר רִבִּי יְהוּדָה אֵימָתַי\hdot
בִּזְמָן שֶׁלֹּא שִׁיֵּיר בַּעַל הַשָּׂדֶה\hdot
אֲבַל אִם שִׁיֵּיר בַּעַל הַשָׂדֶה\hdot
הוּא נוֹתֵן פֵּאָה לַכֹּל׃}

\halacha{רִבִּי אֱלִיעֶזֶר אוֹמֵר קַרְקַע בֵּית רוֹבַע חַיָּיב בַּפֵּאָה\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר הָעוֹשֶׂה סָאתַיִם\hdot
רִבִּי טַרְפוֹן אוֹמֵר שִׁשָּׁה עַל שִׁשָּׁה טְפָחִים\hdot
רִבִּי יְהוּדָה בֶּן בְּתֵירָא אוֹמֵר כְּדֵי לִקְצוֹר וְלִשְׁנוֹת\hdot
וַהֲלָכָה כִּדְבָרָיו\hdot
