\tractatehead{מסכת עדויות}{eduyot}

\perek{שַׁמָּאי}{אוֹמֵר\hdot
כָּל הַנָּשִׁים דַּיָּן שַׁעְתָּן\hdot
וְהִלֵּל אוֹמֵר מִפְּקִידָה לִפְקִידָה\hdot
אֲפִילוּ לְיָמִים הַרְבֶּה\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא מֵעֵת לְעֵת מְמָעֶטֶת עַל יָד מִפְּקִידָה לִפְקִידָה\hdot
וּמִפְּקִידָה לִפְקִידָה מְמָעֶטֶת עַל יָד מֵעֵת לְעֵת\hdot
כָּל אִשָּׁה שֶׁיֵּשׁ לָהּ וֶסֶת דַּיָּהּ שַׁעְתָּהּ\hdot
הַמְּשַׁמֶּשֶׁת בְּעֵדִים הֲרֵי זוֹ כִפְקִידָה\hdot
וּמְמָעֶטֶת עַל יָד מֵעֵת לְעֵת\hdot
וְעַל יָד מִפְּקִידָה לִפְקִידָה׃}

\halacha{שַׁמָּאי אוֹמֵר מִקַּב לַחַלָּה\hdot
וְהִלֵּל אוֹמֵר מִקַּבַּיִם\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא קַב וּמֶחְצָה חַיָּבִים בַּחַלָּה\hdot
וּמִשֶּׁהִגְדִּילוּ הַמִּדּוֹת אָֽמְרוּ\hdot
חֲמֵשֶׁת רְבִיעִים חַיָּבִין\hdot
רִבִּי יוֹסֵי אוֹמֵר חֲמִשָּׁה פְּטוּרִין\hdot
חֲמִשָּׁה וְעוֹד חַיָּבִין׃}

\halacha{הִלֵּל אוֹמֵר מְלֹא הִין מַיִם שְׁאוּבִין פּוֹסְלִין אֶת הַמִּקְוֶה\hdot
אֶלָּא שֶׁאָדָם חַיָּב לוֹמַר בִּלְשׁוֹן רַבּוֹ\hdot
וְשַׁמָּאי אוֹמֵר תִּשְׁעָה קַבִּין\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא עַד שֶׁבָּאוּ שְׁנֵי גַרְדִּיִּים מִשַּׁעַר הָאַשְׁפּוֹת שֶׁבִּירוּשָׁלַיִם\hdot
וְהֵעִידוּ מִשּׁוּם שְׁמַעְיָה וְאַבְטַלְיוֹן שֶׁשְּׁלֹשָׁה לוֹגִים מַיִם שְׁאוּבִין\hdot
פּוֹסְלִין אֶת הַמִּקְוֶה\hdot
וְקִיְּמוּ חֲכָמִים אֶת דִּבְרֵיהֶם׃}

\halacha{וְלָמָּה מַזְכִּירִין אֶת דִּבְרֵי שַׁמָּאי וְהִלֵּל לְבַטָּלָה\hdot
לְלַמֵּד לְדוֹרוֹת הַבָּאִים\hdot
שֶׁלֹּא יְהֵא אָדָם עוֹמֵד עַל דְּבָרָיו\hdot
שֶׁהֲרֵי אָבוֹת הָעוֹלָם\hdot
לֹא עָֽמְדוּ עַל דִּבְרֵיהֶם׃}
