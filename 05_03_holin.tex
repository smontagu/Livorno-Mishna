\tractatehead{מסכת חולין}{holin}

\perek{הַכֹּל}{שׁוֹחֲטִין וּשְׁחִיטָתָן כְּשֵׁרָה\hdot
חוּץ מֵחֵרֵשׁ שׁוֹטֶה וְקָטָן שֶׁמֶא יְקַלְקְלו בִשְׁחִיטָתָן\hdot
וְכֻלָּן שֶׁשָּׁחֲטוּ\hdot
וְאַחֵרִים רוֹאִין אוֹתָן\hdot
שְׁחִיטָתָן כְּשֵׁרָה\hdot
שְׁחִיטַת נָכְרִי נְבֵילָה\hdot
וּמְטַמְּאָה בְמַשָּׁא\hdot
הַשּׁוֹחֵט בַּלַּיְלָה\hdot
וְכֵן הַסּוֹמֵא שֶׁשָּׁחַט\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
הַשּׁוֹחֵט בְּשַׁבָּת וּבְיוֹם הַכִּפּוּרִים\hdot
אַף עַל פִּי שְׁמִּתַחַיֵּב בְּנַפְשׁוֹ\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה׃}

\halacha{הַשּׁוֹחֵט בְּמַגַּל יָד\hdot
בַּצּוּר וּבַקָּנֶה שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
הַכֹּל שׁוֹחֲטִין\hdot
וּלְעוֹלָם שׁוֹחֲטִין\hdot
וּבַכֹּל שׁוֹחֲטִין\hdot
חוּץ מִמַּגַּל קָצִיר וְהַמְּגֵירָה וְהַשִּׁנַּיִם וְהַצִּפֹּרֶן\hdot
מִפְּנֵי שֶׁהֵן חוֹנְקִין\hdot
הַשּׁוֹחֵט בְּמַגַּל קָצִיר בְּדֶרֶךְ הֲלִיכָתָהּ\hdot
בֵּית שַׁמָּאי פּוֹסְלִין\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot
וְאִם הֶחֱלִיקוּ שִׁנֶּיהָ\hdot
הֲרֵי הִיא כַּסַּכִּין׃}

\halacha{הַשּׁוֹחֵט מִתּוֹךְ הַטַּבַּעַת וְשִׁיֵּר בָּהּ מְלֹא הַחוּט עַל פְּנֵי כֻלָּהּ\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
רִבִּי יוֹסֵי בַּר יְהוּדָה אוֹמֵר מְלֹא הַחוּט עַל פְּנֵי רֻבָּהּ׃}

\halacha{הַשּׁוֹחֵט מִן הַצְּדָדִין\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
הַמּוֹלֵק מִן הַצְּדָדִין\hdot
מְלִיקָתוֹ פְּסוּלָה\hdot
הַשּׁוֹחֵט מִן הָעֹרֶף\hdot
שְׁחִיטָתוֹ פְּסוּלָה\hdot
הַמּוֹלֵק מִן הַעֹרֶף מְלִיקָתוֹ כְּשֵׁרָה\hdot
הַשּׁוֹחֵט מִן הַצַּוָּאר\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
הַמּוֹלֵק מִן הַצַּוָאר\hdot
מְלִיקָתוֹ פְּסוּלָה\hdot
שֶׁכָּל הָעֹרֶף כָּשֵׁר לַמְּלִיקָה\hdot
וְכָל הַצַּוָּאר כָּשֵׁר לַשְּׁחִיטָה\hdot
נִמְצָא כָּשֵׁר בַּשְּׁחִיטָה\hdot
פָּסוּל בַּמְּלִיקָה\hdot
כָּשֵׁר בַּמְּלִיקָה\hdot
פָּסוּל בַּשְּׁחִיטָה׃}

\halacha{כָּשֵׁר בַּתּוֹרִין\hdot
פָּסוּל בִּבְנֵי יוֹנָה\hdot
כָּשֵׁר בִּבְנֵי יוֹנָה\hdot
פָּסוּל בַּתּוֹרִין\hdot
תְּחִלַּת הַצִּיהוּב בְּזֶה וּבְזֶה פָּסוּל׃}

\halacha{כָּשֵׁר בַּפָּרָה פָּסוּל בָּעֲגָלָה\hdot
כָּשֵׁר בָּעֲגָלָה פָּסוּל בַּפָּרָה\hdot
כָּשֵׁר בַּכֹּהֲנִים פָּסוּל בַּלְוִיִּם\hdot
כָּשֵׁר בַּלְוִיּם פָּסוּל בַּכֹּהֲנִים\hdot
טָהוֹר בִּכְלֵי חֶרֶשׂ טָמֵא בְּכָל הַכֵּלִים\hdot
טָהוֹר בְּכָל הַכֵּלִים טָמֵא בִּכְלֵי חֶרֶשׂ\hdot
טָהוֹר בִּכְלֵי עֵץ טָמֵא בִּכְלֵי מַתְּכוֹת\hdot
טָהוֹר בִּכְלֵי מַתְּכוֹת טָמֵא בִּכְלֵי עֵץ\hdot
הַחַיָּב בַּשְּׁקֵדִים הַמָּרִים פָּטוּר בַּמְּתוּקִים\hdot
הַחַיָּב בַּמְּתוּקִים פָּטוּר בַּמָּרִים׃}

\halacha{הַתֶּמֶד עַד שֶׁלֹּא הֶחְמִיץ\hdot
אֵינוֹ נִקָּח בְּכֶסֶף מַעֲשֵׂר\hdot
וּפוֹסֵל אֶת הַמִּקְוֶה\hdot
מִשֶּׁהֶחְמִיץ נִקָּח בּכֶסֶף מַעֲשֵׂר\hdot
וְאֵינוֹ פוֹסֵל אֶת הַמִּקְוֶה\hdot
הָאַחִים הַשֻּׁתָּפִים כְּשֶׁחַיָּבִים בַּקָּלְבּוֹן\hdot
פְּטוּרִין מִמַּעְשַׂר בְּהֵמָה\hdot
כְּשֶׁחַיָּבִין בְּמַעְשַׂר בְּהֵמָה\hdot
פְּטוּרִין מִן הַקָּלְבּוֹן\hdot
כָּל מָּקוֹם שֶׁיֵּשׁ מֶכֶר אֵין קְנָס\hdot
וְכָל מָקוֹם שֶׁיֵּשׁ קְנָס אֵין מֶכֶר\hdot
כָּל מָקוֹם שֶׁיֵּשׁ מִיאוּן אֵין חֲלִיצָה\hdot
וְכָל מָקוֹם שֶׁיֵּשׁ חֲלִיצָה אֵין מִיאוּן\hdot
כָּל מָקוֹם שֶׁיֵּשׁ תְּקִיעָה אֵין הַבְדָּלָה\hdot
וְכָל מָקוֹם שֶׁיֵּשׁ הַבְדָּלָה\hdot
אֵין תְּקִיעָה\hdot
יוֹם טוֹב שֶׁחַל לִהְיוֹת בְּעֶרֶב שַׁבָּת\hdot
תּוֹקְעִין וְלֹא מַבְדִּילִין\hdot
בְּמוֹצָאֵי שַׁבָּת מַבְדִּילִין וְלֹא תוֹקְעִין\hdot
כֵּיצַד מַבְדִּילִין\hdot
הַמַּבְדִּיל בֵּין קֹדֶשׁ לְקֹדֶשׁ\hdot
רִבִּי דוֹסָא אוֹמֵר בֵּין קֹדֶשׁ חָמוּר לְקֹדֶשׁ הַקַּל׃}

\perek{הַשּׁוֹחֵט}{אֶחָד בָּעוֹף וּשְׁנַיִם בַּבְּהֵמָה\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
וְרֻבּוֹ שֶׁל אֶחָד כָּמוֹהוֹ\hdot
רִבִּי יְהוּדָה אוֹמֵר עַד שֶׁיִּשְׁחוֹט אֶת הַוָּרִידִין\hdot
חֲצִי אֶחָד בָּעוֹף וְאֶחָד וָחֵצִי בַבְּהֵמָה\hdot
שְׁחִיטָתוֹ פְּסוּלָה\hdot
רוֹב אֶחָד בָּעוֹף וְרוֹב שְׁנַיִם בַּבְּהֵמָה\hdot
שְׁחִיטָתוֹ כְּשֵׁרָה׃}

\halacha{הַשּׁוֹחֵט שְׁנֵי רָאשִׁין כְּאֶחָד שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
שְׁנַיִם אוֹחֲזִין בַּסַּכִּין וְשׁוֹחֲטִין\hdot
אֲפִילוּ אֶחָד לְמַעְלָה וְאֶחָד לְמַטָּה\hdot
שְׁחִיטָתָן כְּשֵׁרָה׃}

\halacha{הִתִּיז אֶת הָרֹאשׁ בְּבַת אַחַת פְּסוּלָה\hdot
הָיָה שׁוֹחֵט וְהִתִּיז אֶת הָרֹאשׁ בְּבַת אֶחָת\hdot
אִם יֵשׁ בַּסַּכִּין מְלֹא צַוָּאר כּשְׁרֵהָ\hdot
הָיָה שׁוֹחֵט וְהִתִּיז שְׁנֵי רָאשִׁין בְּבַת אֶחָת\hdot
אִם יֵשׁ בַּסַּכִּין מְלֹא צַוָּאר אֶחָד כְּשְׁרָה\hdot
בַּמֶּה דְּבָרִים אֲמוּרִים בִּזְמָן שֶׁהוֹלִיךְ וְלֹא הֵבִיא\hdot
אוֹ הֵבִיא וְלֹא הוֹלִיךְ\hdot
אֲבָל אִם הוֹלִיךְ וְהֵבִיא\hdot
אֲפִילוּ כָּל שֶׁהוּא\hdot
אֲפִילוּ בְּאִיזְמֵל כְּשֵׁרָה\hdot
נָֽפְלָה סַכִּין וּשְׁחָטָהּ\hdot
אַף עַל פִּי שֶׁשְּׁחֲטָה כְּדַרְכָּהּ פְּסוּלָה\hdot
שֶׁנֶּאֱמַר וְזָבַחְתָּ וְאָכַלְתָּ\hdot
מַה שֶּׁאַתָּה זוֹבֵחַ אַתָּה אוֹכֵל\hdot
נָֽפְלָה סַכִּין וְהִגְבִּיהָהּ\hdot
נָֽפְלוּ כֵלָיו וְהִגְבִּיהָן\hdot
הִשְׁחִיז אֶת הַסַּכִּין וְעַף\hdot
וּבָא חֲבֵרוֹ וְשָׁחַט\hdot
אִם שָׁהָה כְּדֵי שְׁחִיטָה פְּסוּלָה\hdot
רִבְּי שִׁמְעוֹן אוֹמֵר אִם שָׁהָה כְּדֵי בִקּוּר׃}

\halacha{שָׁחַט אֶת הַוֶּשֶׁט וּפָסַק אֶת הַגַּרְגֶּרֶת\hdot
אוֹ שָׁחַט אֶת הַגַּרְגֶּרֶת\hdot
וּפָסַק אֶת הַוֶּשֶׁט\hdot
אוֹ שֶׁשָּׁחַט אֶחָד מֵהֶן\hdot
וְהִמְתִּין לָהּ עַד שֶׁמֵּתָה\hdot
אוֹ שֶׁהֶחֱלִיד אֶת הַסַּכִּין תַּחַת הַשֵּׁנִי וּפְסָקוֹ\hdot
רִבִּי יְשֵׁבָב אוֹמֵר נְבֵלָה\hdot
רִבִּי עֲקִיבָא אוֹמֵר טְרֵפָה\hdot
כְּלָל אָמַר רִבִּי יְשֵׁבָב מִשּׁוּם רִבִּי יְהוֹשֻׁעַ\hdot
כָּל שֶׁנִּפְסְלָה בִּשְׁחִיטָתָהּ נְבֵלָה\hdot
כָּל שֶׁשְּׁחִיטָתָהּ כָּרָאוּי וְדָבָר אַחֵר גָּרַם לָהּ לִפָּסֵל טְרֵפָה\hdot
וְהוֹדָה לוֹ רִבִּי עֲקִיבָא׃}

\halacha{הַשּׁוֹחֵט בְּהֵמָה חַיָּה וְעוֹף \hdot
וְלֹא יָצָא מֵהֶן דָּם כְּשֵׁרִים\hdot
וְנֶאֱכָלִים בְּיָדַיִם מְסוּאֲבוֹת\hdot
לְפִי שֶׁלֹּא הֻכְשְׁרוּ בְדָם\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר הֻכְשְׁרוּ בַשְּׁחִיטָה׃}

\halacha{הַשּׁוֹחֵט אֶת הַמְּסֻכֶּנֶת\hdot
רַבָּן (שִׁמְעוֹן בֶּן) גַּמְלִיאֵל אוֹמֵר עַד שֶׁתְּפַרְכֵּס בַּיָּד וּבָרֶגֶל\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר דַּיָּהּ אִם זִנְּקָה\hdot
אָמַר רִבִּי שִׁמְעוֹן (אף) הַשּׁוֹחֵט בַּלַּיְלָה וּלְמָחָר הִשְׁכִּים\hdot
וּמָצָא כֳּתָלִים מְלֵאִים דָּם כְּשֵׁרָה שֶׁזִּנְּקָה\hdot
וּכְמִדַּת רִבִּי אֱלִיעֶזֶר\hdot
וַחֲכָמִים אוֹמְרִים עַד שֶׁתְּפַרְכֵּס בַּיָּד וּבָרֶגֶל\hdot
אוֹ עַד שֶׁתְּכַשְׁכֵּשׁ בִּזְנָבָהּ\hdot
אֶחָד בְּהֵמָה דַקָּה\hdot
וְאֶחָד בְּהֵמָה גַסָּה\hdot
בְּהֵמָה דַקָּה שֶׁפָּֽשְׁטָה יָדָהּ וְלֹא הֶחֱזְירָהּ פְּסוּלָה\hdot
שֶׁאֵינָהּ אֶלָּא הוֹצָאַת נֶפֶשׁ בִּלְבָד\hdot
בַּמֶּה דְבָרִים אֲמוּרִים שֶׁהָֽיְתָה בְחֶזְקַת מְסֻכֶּנֶת\hdot
אֲבָל אִם הָֽיְתָה בְחֶזְקַת בְּרִיאָה\hdot
אֲפִילוּ אֵין בָּהּ אֶחָד מִכָּל הַסִּימָנִים הַלָּלוּ כְּשֵׁרָה׃}

\halacha{הַשׁוֹחֵט לַנָּכְרִי שְׁחִיטָתוֹ כְּשֵׁרָה\hdot
וְרִבִּי אֱלִיעֶזֶר פּוֹסֵל\hdot

וּמַה בִּמְקוֹם שֶׁהַמַּחֲשָׁבָה פוֹסֶלֶת בַּמְּקֻדָּשִׁים\hdot
