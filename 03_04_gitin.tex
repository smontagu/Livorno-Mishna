\tractatehead{מסכת גיטין}{gitin}

\perek{הַמֵּבִיא}{גֵט מִמְּדִינַת הַיָּם\hdot
צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר אַף הַמֵּבִיא  מִן הָרֶקֶם וּמִן הֶחָגָר\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אֲפִילוּ מִכְּפַר לוּדִים לְלוּד\hdot
וַחֲכָמִים אוֹמְרִים אֵינוֹ צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַב וּבְפָנַי נֶחְתַּם\hdot
אֶלָּא הַמֵּבִיא מִמְּדִינַת הַיָּם\hdot
וְהַמּוֹלִיךְ וְהַמֵּבִיא מִמְּדִינָה לִמְדִינָה בִּמְדִינַת הַיָּם\hdot
צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר אֲפִילוּ מֵהַגְמוֹנִיָּא לְהַגְמוֹנִיָּא׃}

\halacha{רִבִּי יְהוּדָה אוֹמֵר מֵרֶקֶם לְמִזְרַח\hdot
וְרֶקֶם כְּמִזְרַח\hdot
מֵאַשְׁקְלוֹן לְדָרוֹם\hdot
וְאַשְׁקְלוֹן כְּדָרוֹם\hdot
מֵעַכּוֹ לְצָפוֹן\hdot
וְעַכּוֹ כְּצָפוֹן\hdot
רִבִּי מֵאִיר אוֹמֵר עַכּוֹ כְּאֶרֶץ יִשְׂרָאֵל לַגִּטִּין׃}

\halacha{הַמֵּבִיא גֵט בְּאֶרֶץ יִשְׂרָאֵל\hdot
אֵינוֹ צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם\hdot
אִם יֵשׁ עָלָיו עוֹרְרִים\hdot
יִתְקַיֵּם בּחוֹתָמָיו\hdot
הַמֵּבִיא גֵט מִמְּדִינַת הַיָּם\hdot
וְאֵינוֹ יָכוֹל לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם\hdot
אִם יֵשׁ עָלָיו עֵדִים יִתְקַיֵּם בְּחוֹתָמָיו׃}

\halacha{אֶחָד גִּטֵּי נָשִׁים\hdot
וְאֶחָד שִׁחְרוּרֵי עֲבָדִים\hdot
שָׁוִין לַמּוֹלִיךְ וְלַמֵּבִיא\hdot
וְזוֹ אַחַת מִן הַדְּרָכִים שֶׁשָּׁווּ גִּטֵּי נָשִׁים לְשִׁחְרוּרֵי עֵדִים׃}

\halacha{כָּל גֵּט שֶׁיֵּשׁ עָלָיו עֵד כּוּתִי פָּסוּל\hdot
חוּץ מִגִּטֵּי נָשִׁים וְשִׁחְרוּרֵי עֲבָדִים\hdot
מַעֲשֶׂה שֶׁהֵבִיאוּ לִפְנֵי רַבָּן גַּמְלִיאֵל לִכְפַר עוֹתְנָאי גֵּט אִשָּׁה\hdot
וְהָיוּ עֵדָיו עֵדֵי כוּתִים וְהִכְשִׁיר\hdot
כָּל הַשְּׁטָרוֹת הָעוֹלִים בְּעַרְכָאוֹת שֶׁל נָכְרִים\hdot
אַף עַל פִּי שֶׁחוֹתָמֵיהֶם נָכְרִים\hdot
כְּשֵׁרִים\hdot
חוּץ מִגִּטֵּי נָשִׁים וְשִׁחְרוּרֵי עֲבַדִים\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אַף אֵלּוּ כְּשֵׁרִים\hdot
לֹא הֻזְכְּרוּ אֶלָּא בִּזְמָן שֶׁנַּעֲשׂוּ בְּהֶדְיוֹט׃}

\halacha{הָאוֹמֵר תְּנוּ גֵט זֶה לְאִשְׁתִּי וּשְׁטַר שִׁחְרוּר זֶה לְעַבְדִּי\hdot
אִם רָצָה לַחֲזוֹר בִּשְׁנֵיהֶן יַחֲזוֹר דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים בְּגִטֵּי נָשִׁים\hdot
אֲבָל לֹא בְשִׁחְרוּרֵי עֲבָדִים\hdot
לְפִי שֶׁזָּכִין לְאָדָם שֶׁלֹּא בְפָנָיו\hdot
וְאֵין חָבִין לוֹ אֶלָּא בְפָנָיו\hdot
שֶׁאִם יִרְצֶה שֶׁלֹּא לָזוּן אֶת עַבְדּוֹ רַשָּׁאי\hdot
וְשֶׁלֹּא לָזוּן אֶת אִשְׁתּוֹ אֵינוֹ רַשָּׁאי\hdot
אָמַר לָהֶם וַהֲרֵי הוּא פוֹסֵל אֶת עַבְדּוֹ מִן הַתְּרוּמָה\hdot
כְּשֵׁם שֶׁהוּא פוֹסֵל (אֶת) אִשְׁתּוֹ\hdot
אָֽמְרוּ לוֹ מִפְּנֵי (שֶׁהוּא) קִנְיָנוֹ\hdot
הָאוֹמֵר תְּנוּ גֵט זֶה לְאִשְׁתִּי\hdot
וּשְׁטַר שִׁחְרוּר זֶה לְעַבְדִּי וּמֵת\hdot
לֹא יִתְּנוּ לְאַחַר מִיתָה\hdot
תְּנוּ מָנֶה לְאִישׁ פְּלוֹנִי וּמֵת\hdot
יִתְּנוּ לְאַחַר מִיתָה׃}

\perek{הַמֵּבִיא}{גֵט מִמְּדִינַת הַיָּם\hdot
וְאָמַר בְּפָנַי נִכְתַּב\hdot
אֲבָל לֹא בְּפָנַי נֶחְתַּם\hdot
בְּפָנַי נֶחְתַּם אֲבָל לֹא בְּפָנַי נִכְתַּב\hdot
בְּפָנַי נִכְתַּב כֻּלּוֹ וּבְפָנַי נֶחְתַּם חֶצְיוֹ\hdot
בְּפָנַי נִכְתַּב חֶצְיוֹ\hdot
וּבְפָנַי נֶחְתַּם כֻּלּוֹ\hdot
פָּסוּל\hdot
אֶחָד אוֹמֵר בְּפָנַי נִכְתַּב\hdot
וְאֶחָד אוֹמֵר בְּפָנַי נֶחְתַּם\hdot
פָּסוּל\hdot
שְׁנַיִם אוֹמְרִים בְּפָנֵינוּ נִכְתַּב\hdot
וְאֶחָד אוֹמֵר בְּפָנַי נֶחְתַּם\hdot
פָּסוּל\hdot
וְרִבִּי יְהוּדָה מַכְשִׁיר\hdot
אֶחָד אוֹמֵר בְּפָנַי נִכְתַּב\hdot
וּשְׁנַיִם אוֹמְרִים בְּפָנֵינוּ נֶחְתַּם\hdot
כָּשֵׁר׃}

\halacha{נִכְתַּב בַּיּוֹם וְנֶחְתַּם בַּיּוֹם\hdot
בַּלַּיְלָה וְנֶחְתַּם בַּלַּיְלָה\hdot
בַּלַּיְלָה וְנֶחְתַּם בַּיּוֹם\hdot
כָּשֵׁר\hdot
בַּיּוֹם וְנֶחְתַּם בַּלַיְלָה\hdot
פָּסוּל\hdot
רִבְּי שִׁמְעוֹן מַכְשִׁיר שֶׁהָיָה רִבִּי שִׁמְעוֹן אוֹמֵר כָּל הַגִּטִּין שֶׁנִּכְתְּבוּ בַיּוֹם\hdot
וְנֶחְתְּמוּ בַלַּיְלָה פְּסוּלִין\hdot
חוּץ מִגִּטֵּי נָשִׁים׃}

\halacha{בַּכֹּל כּוֹתְבִין\hdot
בַּדְּיוֹ בַּסַּם בַּסִּקְרָא וּבְקוֹמוֹס וּבְקַנְקַנְתּוֹס וּבְכָל דָּבָר שֶׁהוּא שֶׁל קַיָּמָא\hdot
אֵין כּוֹתְבִין לֹא בְמַשְׁקִין וְלֹא בְמֵי פֵירוֹת וְלֹא בְכָל דָּבָר שֶׁאֵינוֹ שֶׁל קַיָּמָא\hdot
עַל הַכֹּל כּוֹתְבִין\hdot
עַל הֶעָלֶה שֶׁל זַיִת\hdot
וְעַל הַקֶּרֶן שֶׁל פָּרָה וְנוֹתֵן לָהּ אֶת הַפָּרָה\hdot
עַל יָד שֶׁל עֶבֶד\hdot
וְנוֹתֵן לָהּ אֶת הָעֶבֶד\hdot
רִבִּי יוֹסֵי הַגָּלִילִי אוֹמֵר אֵין כּוֹתְבִין לֹא עַל דָּבָר שֶׁיֵּשׁ בּוֹ רוּחַ חַיִּים וְלֹא עַל הָאוֹכָלִים׃}

\halacha{אֵין כּוֹתְבִין בַּמְּחֻבָּר לַקַּרְקַע\hdot
כְּתָבוֹ בַּמְּחֻבָּר תְּלָשׁוֹ וַחֲתָמוֹ וּנְתָנוֹ לָהּ כָּשֵׁר\hdot
רִבִּי יְהוּדָה פּוֹסֵל עַד שֶׁתְּהֵא כְתִיבָתוֹ וַחֲתִימָתוֹ בַּתָּלוּשׁ\hdot
רִבִּי יְהוּדָה בֶּן בְּתֵירָא אוֹמֵר אֵין כּוֹתְבִין לֹא עַל הַנְּיָר הַּמָחוּק\hdot
וְלֹא עַל הַדִּפְתְּרָא מִפְּנֵי שֶׁהוּא יָכוֹל לְהִזְדַּיֵּף\hdot
וַחֲכָמִים מַכְשִׁירִין׃}

\halacha{הַכֹּל כְּשֵׁרִין לִכְתּוֹב אֶת הַגֵּט\hdot
אֲפִילוּ חֵרֵשׁ שׁוֹטֶה וְקָטָן\hdot
הָאִשָּׁה כוֹתֶבֶת אֶת גִּטָּהּ\hdot
וְהָאִישׁ כּוֹתֵב אֶת שׁוֹבָרוֹ\hdot
שֶׁאֵין קִיּוּם הַגֵּט אֶלָּא בְחוֹתָמָיו\hdot
הַכֹּל כְּשֵׁרִין לְהָבִיא אֶת הַגֵּט\hdot
חוּץ מֵחֵרֵשׁ שׁוֹטֶה וְקָטָן וְסוֹמֵא וְנָכְרִי׃}

\halacha{קִבֵּל הַקָּטָן וְהִגְדִּיל\hdot
חֵרֵשׁ וְנִתְפַּקַּח\hdot
סוֹמֵא וְנִתְפַּתַּח\hdot
שׁוֹטֶה וְנִשְׁתַּפָּה\hdot
נָכְרִי וְנִתְגַּיֵּר פָּסוּל\hdot
אֲבָל פִּקֵּחַ וְנִתְחָרַשׁ\hdot
וְחָזַר וְנִתְפַּקַּח\hdot
פָּתוּחַ וְנִסְתַּמָּא וְחָזַר וְנִתְפַּתַּח\hdot
שָׁפוּי וְנִשְׁתַּטָּה וְחָזַר וְנִשְׁתַּפָּה כָּשֵׁר\hdot
זֶה הַכְּלָל כָּל שֶׁתְּחִלָּתוֹ וְסוֹפוֹ בְּדַעַת\hdot
כָּשֵׁר׃}

\halacha{אַף הַנָּשִׁים שֶׁאֵינָן נֶאֱמָנוֹת לוֹמַר מֵת בַּעלָהּ\hdot
נֶאֱמָנוֹת לְהָבִיא אֶת גִּטָּהּ\hdot
חֲמוֹתָהּ וּבַת חֲמוֹתָהּ\hdot
וְצָרָתָהּ וִיבִמְתָּהּ וּבַת בַּעְלָהּ\hdot
מַה בֵּין גֵּט לְמִיתָה\hdot
שֶׁהַכְּתָב מוֹכִיחַ\hdot
הָאִשָּׁה עַצְמָהּ מְבִיאָה את גִּטָּהּ\hdot
וּבִלְבָד שֶׁהִיא צְרִיכָה לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתָּם}

\perek{כָּל}{גֵּט שֶׁנִּכְתַּב שֶׁלֹּא לְשׁוּם אִשָּׁה פָּסוּל\hdot
כֵּיצַד\hdot
הָיָה עוֹבֵר בַּשּׁוּק\hdot
וְשָׁמַע קוֹל סוֹפְרִים מַקְרִין\hdot
אִישׁ פְּלוֹנִי מְגָרֵשׁ אֶת פְּלוֹנִית\hdot
מִמָּקוֹם פְּלוֹנִי\hdot
וְאָמַר זֶה שְׁמִי וְזֶה שֵׁם אִשְׁתִּי\hdot
פָּסוּל לְגָרֵשׁ בּוֹ\hdot
יָתֵר מִכֵּן כָּתַב לְגָרֵשׁ (בּוֹ) אֶת אִשְׁתּוֹ וְנִמְלַךְ\hdot
מְצָאוֹ בֶן עִירוֹ וְאָמַר לוֹ שְׁמִי כִּשְׁמָךְ וְשֵׁם אִשְׁתִּי כְּשֵׁם אִשְׁתֶּךָ\hdot
פָּסוּל לְגָרֵשׁ בּוֹ\hdot
יָתֵר מִכֵּן\hdot
הָיוּ לוֹ שְׁתֵּי נָשִׁים וּשְׁמוֹתֵיהֶן שָׁווֹת\hdot
כָּתַב לְגָרֵשׁ בּוֹ אֶת הַגְּדוֹלָה\hdot
לֹא יְגָרֵשׁ בּוֹ אֶת הַקְּטַנָּה\hdot
יָתֵר מִכֵּן אָמַר לַלַּבְלָר כְּתוֹב\hdot
