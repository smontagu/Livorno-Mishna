\perek[381]{הַנּוֹטֵעַ}{לַסְּיָג\hdot}{וְלַקּוֹרוֹת\hdot
פָּטוּר מֵהָעָרְלָה\hdot
רִבִּי יוֹסֵי אוֹמֵר אֲפִילוּ אָמַר הַפְּנִימִי לְמַאֲכָל\hdot
וְהַחִיצוֹן לַסְּיָג\hdot

וְהַחִיצוֹן פָּטוּר׃}

\halacha{עֵת שֶׁבָּאוּ אֲבוֹתֵינוּ לָאָרֶץ\hdot
מְצָאוֹ נָטוּעַ פָּטוּר\hdot
נְטָעוֹ אַף עַל פִּי שֶׁלֹּא כִבְּשׁוּ חַיָּיב\hdot
הַנּוֹטֵעַ לָרַבִּים חַיָּיב\hdot
רִבִּי יְהוּדָה פּוֹטֵר\hdot
הַנּוֹטֵעַ בִּרְשׁוּת הָרַבִּים\hdot
וְהַנָּכְרִי שֶׁנָּטַע\hdot
וְהַגַּזְלָן שֶׁנָּטַע\hdot
וְהַנּוֹטֵעַ בַּסְּפִינָה\hdot
וְהָעוֹלֶה מֵאֵלָיו\hdot
חַיָּיב בָּעָרְלָה׃}

\halacha{אִילָן שֶׁנֶּעֱקַר וְהַסֶּלַע עִמּוֹ\hdot
שְׁטָפוֹ נָהָר וְהַסֶּלַע עִמּוֹ\hdot
אִם יָכוֹל לִחְיוֹת פָּטוּר\hdot
וְאִם לָאוּ חַיָּיב\hdot
נֶעֱקַר הַסֶּלַע מִצִּדּוֹ\hdot
אוֹ שֶׁזִּעְזְעַתּוּ הַמַּחֲרִישָׁה\hdot
אוֹ שֶׁזִּעְזְעוֹ וְעֲשָׂאוֹ בֶּעָפָר אִם יָכוֹל לִחְיוֹת פָּטוּר\hdot
וְאִם לָאוּ חַיָּיב׃}

\halacha[129]{אִילָן שֶׁנֶּעֱקַר\hdot
וְנִשְׁתַּיֵּיר בּוֹ שֹׁרֶשׁ פָּטוּר\hdot
וְכַמָּה יְהֵא הַשֹּׁרֶשׁ\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר מִשּׁוּם רִבִּי אֶלְעָזָר בֶּן יְהוּדָה אִישׁ בַּרְתּוּתָא כְּמַחַט שֶׁל מִתּוּחַ׃}

\halacha[2573]{אִילָן שֶׁנֶּעֱקַר וּבוֹ בְרֵיכָה\hdot
וְהוּא חַיָּה מִמֶּנָּהּ\hdot
חָֽזְרָה הַזְּקֵנָה לִהְיוֹת כַּבְּרֵיכָה\hdot
הִבְרִיכָהּ שָׁנָה אַחַר שָׁנָה וְנִפְסָקָה\hdot
מוֹנֶה מִשָּׁעָה שֶׁנִּפְסָקָה\hdot
סִפּוּק הַגְּפָנִים\hdot
וְסִפּוּק עַל גַּבֵּי סִפּוּק\hdot
אַף עַל פִּי שֶׁהִבְרִיכָן בָּאָרֶץ מוּתר\hdot
ּרִבִּי מֵאִיר אוֹמֵר מָקוֹם שֶׁכֹּחָהּ יָפֶה מוּתָר וּמָקוֹם שֶׁכֹּחָהּ רָע אָסוּר\hdot
(וְכֵן) בְּרֵיכָה שֶׁנִּפְסְקָה וְהִיא מְלֵאָה פֵירוֹת\hdot
אִם הוֹסִיף בְּמָאתַיִם אָסוּר׃}

\halacha[2875]{נְטִיעָה שֶׁל עָרְלָה\hdot
וְשֶׁל כִּלְאֵי הַכֶּרֶם שֶׁנִּתְעָֽרְבוּ בַנְּטִיעוֹת\hdot
הֲרֵי זֶה לֹא יִלְקוֹט\hdot
וְאִם לָקַט יַעֲלֶה בְּאֶחָד וּמָאתַיִם\hdot
וּבִלְבַד שֶׁלֹּא יִתְכַּוֵּין לִלְקוֹט\hdot
רִבִּי יוֹסֵי אוֹמֵר אַף יִתְכַּוֵּין לִלְקוֹט וְיַעֲלוּ בְּאֶחָד וּמָאתַיִם\hdot}

וּבָֽרְבָעִי\hdot
וּבַנָּזִיר\hdot

רִבִּי אֱלִיעֶזֶר אוֹמֵר הַמַּעֲמִיד בִּשְׂרַף הָעָרְלָה אָסוּר\hdot
אָמַר רִבִּי יְהוֹשֻׁעַ שָׁמַעְתִּי בְּפֵירוּשׁ\hdot
שֶׁהַמַּעֲמִיד בִּשְׂרַף הֶעָלִים\hdot
בִּשְׂרַף הָעִקָּרִים מוּתָר\hdot
בִּשְׂרַף הַפָּגִים אָסוּר\hdot
מִפְּנֵי שֶׁהֵם פֶּרִי׃}

\halacha[1000]{עַַנְקוּקָלוֹת וְהַחַרְצַנִּים וְהַזַגִּים\hdot
