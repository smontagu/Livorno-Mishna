\tractatehead{מסכת סוכה}{sukka}

\perek{סֻכָּה}{שֶׁהִיא גְבוֹהָה לְמַעְלָה מֵעֶשְׂרִים אַמָּה פְּסוּלָה\hdot
רִבִּי יְהוּדָה מַכְשִׁיר\hdot
וְשֶׁאֵינָהּ גְּבוֹהָה עֲשָׂרָה טְפָחִים\hdot
וְשֶׁאֵין לָהָּ שָׁלֹשׁ דְּפַנּוֹת וְשֶׁחַמָּתָהּ מְרֻבָּה מִצִּלָּתָהּ\hdot
פְּסוּלָה\hdot
סֻכָּה יְשָׁנָה\hdot
בֵּית שַׁמָּאי פּוֹסְלִין\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot
וְאֵיזוֹ הִיא סֻכָּה יְשָׁנָה\hdot
כָּל שֶׁעֲשָׂאָהּ קֹדֶם לַחַג שְׁלֹשִׁים יוֹם\hdot
אֲבָל אִם עֲשָׂאָהּ לְשֵׁם חַג\hdot
אֲפִילוּ מִתְּחִלַּת הַשָּׁנָה כְּשֵׁרָה׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ תַּחַת הָאִילָן כְּאִלּוּ עֲשָׂאָהּ בְּתוֹךְ הַבַּיִת\hdot
סֻכָּה עַל גַּבֵּי סֻכָּה\hdot
הָעֶלְיוֹנָה כְּשֵׁרָה\hdot
וְהַתַּחְתּוֹנָה פְּסוּלָה\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
אִם אֵין דְּיוּרִין בָּעֶלְיוֹנָה\hdot
הַתַּחְתּוֹנָה כְּשֵׁרָה׃}

\halacha{פֵּרַס עָלֶיהָ סָדִין מִפְּנֵי הַחַמָּה\hdot
אוֹ תַחְתֶּיהָ מִפְּנֵי הַנֶּשֶׁר\hdot
אוֹ שֶׁפֵּרַס עַל גַּבֵּי הַקִּנּוֹף פְּסוּלָה\hdot
אֲבָל פּוֹרֵס הוּא\hdot
עַל גַּבֵּי נַקְלִיטֵי הַמִּטָּה׃}

\halacha{הִדְלָה עָלֶיהָ אֶת הַגֶּפֶן\hdot
וְאֶת הַדָּלַעַת\hdot
וְאֶת הַקִּיסוֹם\hdot
וְסִכֵּךְ עַל גַּבָּהּ פְּסוּלָה\hdot
וְאִם הָיָה סִכּוּךְ הַרְבֶּה מֵהֶן\hdot
אוֹ שֶׁקִצְּצָן\hdot
כְּשֵׁרָה\hdot
זֶה הַכְּלָל\hdot
כָּל דָּבָר שֶׁהוּא מְקַבֵּל טֻמְאָה\hdot
וְאֵין גִּדּוּלוֹ מִן הָאָרֶץ\hdot
אֵין מְסַכְּכִין בּוֹ\hdot
וְכָל דָּבָר שֶׁאֵינוֹ מְקַבֵּל טֻמְאָה\hdot
וְגִדּוּלוֹ מִן הָאָרֶץ\hdot
מְסַכְּכִין בּוֹ׃}

\halacha{חֲבִילֵי קַשׁ\hdot
וַחֲבִילֵי עֵצִים\hdot
וַחֲבִילֵי זְרָדִין אֵין מְסַכְּכִין בָּהֶן\hdot
וְכֻלָּן שֶׁהִתִּירָן\hdot
כְּשֵׁרוֹת\hdot
וְכֻלָּן כְּשֵׁרוֹת לַדְּפָנוֹת׃}

\halacha{מְסַכְּכִין בִּנְסָרִים\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
וְרִבִּי מֵאִיר אוֹסֵר\hdot
נָתַן עָלֶיהָ נֶסֶר שֶׁהוּא רַחַב אַרְבָּעָה טְפָחִים\hdot
כְּשֵׁרָה\hdot
וּבִלְבַד שֶׁלֹּא יִישָׁן תַּחְתָּיו׃}

\halacha{תְּקָרָה שֶׁאֵין עָלֶיהָ מַעֲזִיבָה\hdot
רִבִּי יְהוּדָה אוֹמֵר בֵּית שַׁמָּאי אוֹמְרִים מְפַקְפֵּק וְנוֹטֵל אַחַת מִבִּנְתַּיִם\hdot
וּבֵית הִלֵּל אוֹמְרִים מְפַקְפֵּק אוֹ נוֹטֵל אַחַת מִבִּנְתַּיִם\hdot
רִבִּי מֵאִיר אוֹמֵר נוֹטֵל אַחַת מִבִּנְתַּיִם וְאֵינוֹ מְפַקְפֵּק׃}

\halacha{הַמְּקָרֶה סֻכָּתוֹ בִּשְׁפוֹדִין\hdot
אוֹ בָאֲרוּכוֹת הַמִּטָּה\hdot
אִם יֵשׁ רֵיוַח בֵּינֵיהֶן כְּמוֹתָן\hdot
כְּשֵׁרָה\hdot
הַחוֹטֵט בַּגָּדִישׁ לַעֲשׂוֹת בּוֹ סֻכָּה אֵינָהּ סֻכָּה׃}
