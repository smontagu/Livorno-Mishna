\tractatehead{מסכת תמיד}{tamid}

\perek{בִּשְׁלֹשָׁה}{מְקוֹמוֹת הַכֹּהֲנִים שׁוֹמְרִים בְּבֵית הַמִּקְדָּשׁ\hdot
בְּבֵית אַבְטִינָס\hdot
בְּבֵית הַנִּיצוֹץ\hdot
וּבְבֵית הַמּוֹקֵד\hdot
בֵּית אַבְטִינָס וּבֵית הַנִּיצוֹץ הָיוּ עֲלִיּוֹת\hdot
וְהָרוֹבִים שׁוֹמְרִים שָׁם\hdot
בֵּית הַמּוֹקֵד כִּיפָה\hdot
וּבַיִת גָּדוֹל הָיָה מֻקַּף רוֹבְדִים שֶׁל אֶבֶן\hdot
וְזִקְנֵי בֵית אָב יְשֵׁנִין שָׁם\hdot
וּמַפְתֵּחוֹת הָעֲזָרָה בְּיָדָם\hdot
וּפִרְחֵי כְהוּנָה אִישׁ כְּסָתוֹ בָּאָרֶץ\hdot
לֹא הָיוּ יְשֵׁנִין בְּבִגְדֵי קֹדֶשׁ\hdot
אֶלָּא פוֹשְׁטִין (אוֹתָן) וּמְקַפְּלִין אוֹתָן\hdot
וּמַנִּיחִין אוֹתָן תַּחַת רָאשֵׁיהֶן\hdot
וּמִתְכַּסִּין בִּכְסוּת עַצְמָן\hdot
אִירַע קֶרִי לְאַחַד מֵהֶן\hdot
יוֹצֵא וְהוֹלֵךְ לוֹ בִּמְסִיבָה הַהוֹלֶכֶת תַּחַת הַבִּירָה\hdot
וְנֵרוֹת דּוֹלְקִין מִכָּאן וּמִכָּאן\hdot
עַד שֶׁהוּא מַגִּיעַ לְבֵית הַטְּבִילָה\hdot
וּמְדוּרָה הָֽיְתָה שָׁם\hdot
וּבֵית כִּסֵּא שֶׁל כָּבוֹד\hdot
וְזֶה הָיָה כְבוֹדוֹ\hdot
מְצָאוֹ נָעוּל יוֹדֵע שֶׁיֵּשׁ שָׁם אָדָם\hdot
פָּתוּחַ יוֹדֵעַ שֶׁאֵין שָׁם אָדָם\hdot
יָרַד וְטָבַל\hdot
עָלָה וְנִסְתַּפַּג\hdot
וְנִתְחַמַּם כְּנֶגֶד הַמְּדוּרָה\hdot
בָּא וְיָשַׁב לוֹ אֵצֶל אֶחָיו הַכֹּהֲנִים\hdot
עַד שֶׁהַשְּׁעָרִים נִפְתָּחִים וְיוֹצֵא וְהוֹלֵך לוֹ׃}

\halacha{מִי שֶׁהוּא רוֹצֶה לִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
מַשְׁכִּים וְטוֹבֵל עַד שֶׁלֹּא יָבֹא הַמְּמֻנֶּה\hdot
וְכִי בְּאֵיזוֹ שָׁעָה הַמְּמֻנֶּה בָא\hdot
לֹא כָל הָעִתִּים שָׁווֹת\hdot
פְּעָמִים שֶׁהוּא בָא מִקְּרִיאַת הַגֶּבֶר\hdot
אוֹ סָמוּךְ לוֹ מִלְּפָנָיו\hdot
אוֹ מִלְּאַחֲרָיו\hdot
הַמְּמֻנֶּה בָא וְדוֹפֵק עֲלֵיהֶם\hdot
וְהֵם פּוֹתְחִים לוֹ\hdot
אָמַר לָהֶן מִי שֶׁטָּבַל יָבֹא וְיָפִיס\hdot
הֵפִיסוּ זָכָה מִי שֶׁזָּכָה׃}

\halacha{נָטַל אֶת הַמַּפְתֵּחַ וּפָתַח אֶת הָפִּשְׁפֵּשׁ\hdot
וְנִכְנַס מִבֵּית הַמּוֹקֵד לָעֲזָרָה\hdot
וְנִכְנְסוּ אַחֲרָיו\hdot
וּשְׁתֵּי אֲבוּקוֹת שֶׁל אוֹר בְּיָדָם\hdot
וְנֶחְלְקוּ לִשְׁתֵּי כִתּוֹת\hdot
אֵלּוּ הוֹלְכִין בָּאַכְסַדְרָה דֶּרֶךְ הַמִּזְרָח\hdot
וְאֵלּוּ הוֹלְכִין בָּאַכְסַדְרָה דֶּרֶךְ הַמַּעֲרָב\hdot
הָיוּ בוֹדְקִין וְהוֹלְכִין עַד שֶׁמַּגִּיעִין לִמְקוֹם (בֵּית) עוֹשֵׂי חֲבִיתִים\hdot
הִגִּיעוּ אֵלּוּ וָאֵלּוּ אָֽמְרוּ שָׁלוֹם\hdot
הַכֹּל שָׁלוֹם\hdot
הֶעֱמִידוּ עוֹשֵׂי חֲבִיתִים לַעֲשׂוֹת חֲבִיתִים׃}

\halacha{מִי שֶׁזָּכָה לִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
הוּא יִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
וְהֵם אוֹמְרִים לוֹ הִזָּהֵר שֶׁמֶא תִגַּע בַּכֶּלִי\hdot
עַד שֶׁתְּקַדֵּשׁ יָדֶיךָ וְרַגְלֶיךָ מִן הִכִּיּוֹר\hdot
וַהֲרֵי הַמַּחְתָּה נְתוּנָה בַמִּקְצוֹעַ\hdot
בֵּין הַכֶּבֶשׁ לַמִּזְבֵּחַ\hdot
בְּמַעֲרָבוֹ שֶׁל כֶּבֶשׁ\hdot
אֵין אָדָם נִכְנָס עִמּוֹ וְלֹא נֵר בְּיָדוֹ\hdot
אֶלָּא מְהַלֵּךְ לְאוֹר הַמַּעֲרָכָה\hdot
לֹא הָיוּ רוֹאִין אוֹתוֹ\hdot
וְלֹא שׁוֹמְעִין אֶת קוֹלוֹ\hdot
עַד שֶׁשּׁוֹמְעִין קוֹל הָעֵץ שֶׁעָשָׂה בֶן קָטִין מוּכָנִי לַכִּיּוֹר\hdot
וְהֵם אוֹמְרִים הִגִּיעַ עֵת\hdot
קִדֵּשׁ יָדָיו וְרַגְלָיו מִן הַכִּיּוֹר\hdot
נָטַל מַחְתַּת הַכֶּסֶף וְעָלָה לְרֹאשׁ הַמִּזְבֵּחַ\hdot
וּפִנָּה אֶת הַגֶּחָלִים הֵילָךְ (וְהֵילָךְ)\hdot
וְחִתָּה מִן הַמַּאֻכָּלוֹת הַפְּנִימִיּוֹת וְיָרַד\hdot
הִגִּיעַ לָרִצְפָה הָפַךְ פָּנָיו לַצָּפוֹן\hdot
וְהָלַךְ לְמִזְרָחוֹ שֶׁל כֶּבֶשׁ כְּעֶשֶׂר אַמּוֹת\hdot
צָבַר אֶת הַגֶּחָלִים עַל גַּבֵּי הָרִצְפָה\hdot
רָחוֹק מִן הַכֶּבֶשׁ שְׁלֹשָׁה טְפָחִים\hdot
מָקוֹם שֶׁנּוֹתְנִין שָׁם מֻרְאַת הָעוֹף\hdot
וְדִשּׁוּן מִזְבֵּחַ הַפְּנִימִי\hdot
וְהַמְּנוֹרָה׃}

\perek{רָאוּהוּ}{אֶחָיו שֶׁיָּרַד\hdot
וְהֵם רָצוּ וּבָאוּ\hdot
מִהֲרוּ וְקִדְּשׁוּ יְדֵיהֶן וְרַגְלֵיהֶן מִן הַכִּיּוֹר\hdot
נָֽטְלוּ אֶת הַמַּגְרֵפוֹת וְאֶת הַצִּנּוֹרוֹת\hdot
וְעָלוּ לְרֹאשׁ הַמְּזִבֵּחַ\hdot
הָאֵבָרִין וְהַפְּדָרִין שֶׁלֹּא נִתְאַכְּלוּ מִבָּעֶרֶב\hdot
סוֹנְקִין אוֹתָן לִצְדָדֵי הַמִּזְבֵּחַ\hdot
אִם אֵין הַצְּדָדִין מַחֲזִיקִין\hdot
סוֹדְרִין אוֹתָם בַּסּוֹבֵב עַל הָכֶּבֶשׁ׃}

\halacha{הֵחֵלוּ מַעֲלִין בָּאֵפֶר עַל גַּבֵּי הַתַּפּוּחַ\hdot
וְתַפּוּחַ הָיָה בְאֶמְצַע הַמִּזְבֵּחַ\hdot
פְּעָמִים עָלָיו כִּשְׁלֹשׁ מֵאוֹת כּוֹר\hdot
וּבָֽרְגָלִים לֹא הָיוּ מְדַשְּׁנִין אוֹתוֹ מִפְּנֵי שֶׁהוּא נוֹי לַמִּזְבֵּחַ\hdot
מִיָּמָיו לֹא נִתְעַצַּל הַכֹּהֵן מִלְּהוֹצִיא אֶת הַדָּשֶׁן׃}

\halacha{הֵחֵלוּ מַעֲלִין בַּגִּזְרִין לְסַדֵּר אֶת הַמַּעֲרָכָה\hdot
וְכִי כָּל הָעֵצִים כְּשֵׁרִים לַמַּעֲרָכָה\hdot
הֵן\hdot
כַּל הָעֵצִים כְּשֵׁרִים לַמַּעֲרָכָה\hdot
חוּץ מִשֶּׁל זַיִת וְשֶׁל גֶּפֶן\hdot
אֲבָל בְּאֵלּוּ רְגִילִין בְּמֻרְבִּיּוֹת שֶׁל תְּאֵנָה\hdot
וְשֶׁל אֱגוֹז וְשֶׁל עֵץ שָׁמֶן׃}

\halacha{סִדֵּר הַמַּעֲרָכָה גְדוֹלָה מִזְרָחָה\hdot
וַחֲזִיתָהּ מִזְרָחָה\hdot
וְרָאשֵׁי הַגִּזְרִין הַפְּנִימִים\hdot
הָיוּ נוֹגְעִים בַּתַּפּוּחַ\hdot
וְרֵיוַח הָיָה בֵּין הַגִּזְרִין\hdot
שֶׁהָיוּ מַצִּיתִין אֶת הָאֲלִיתָה מִשָּׁם׃}

\halacha{בֵּרְרוּ מִשָּׁם עֲצֵי תְאֵנָה יָפִין\hdot
לְסַדֵּר הַמַּעֲרָכָה שְׁנִיָּה לִקְטֹרֶת\hdot
מִכְּנֶגֶד קֶרֶן מַעֲרַבִית דְּרוֹמִית\hdot
מָשׁוּךְ מִן הַקֶּרֶן כְּלַפֵּי צָפוֹן\hdot
אַרְבַּע אַמּוֹת בְּעֹמֶד חָמֵשׁ סְאִין גֶּחָלִין\hdot
וּבְשַׁבָּת בְּעֹמֶד שְׁמֹנֶה סְאִין גֶּחָלִין\hdot
שֶׁשָּׁם הָיוּ נוֹתְנִין שְׁנֵי בָזִיכֵי לְבוֹנָה שֶׁל לֶחֶם הַפָּנִים\hdot
הָאֵבָרִים וְהַפְּדָרִים שֶׁלֹּא נִתְאַכְּלוּ מִבָּעֶרֶב\hdot
מַחֲזִירִין אוֹתָן לַמַּעֲרָכָה\hdot
הִצִּיתוּ שְׁתֵּי הַמַּעֲרָכוֹת בָּאֵשׁ\hdot
וְיָֽרְדוּ וּבָאוּ לָהֶם לְלִשְׁכַּת הַגָּזִית׃}

\perek{אָמַר}{לָהֶם הַמְּמֻנֶּה בּוֹאוּ וְהָפִיסוּ\hdot
מִי שׁוֹחֵט\hdot
מִי זוֹרֵק\hdot
מִי מְדַשֵּׁן מִזְבֵּחַ הַפְּנִימִי\hdot
מִי מִדַשֵּׁן אֶת הַמְּנוֹרָה\hdot
מְי מַעֲלֶה אֵבָרִים לַכֶּבֶשׁ\hdot
הָרֹאשׁ וְהָרֶגֶל\hdot
וּשְׁתֵּי הַיָּדַיִם\hdot
הָעֹקֶץ וְהָרֶגֶל\hdot
הֶחָזֶה וְהַגֵּרָה\hdot
וּשְׁתֵּי הַדְּפָנוֹת\hdot
הַקְּרָבַיִם וְהַסֹּלֶת\hdot
וְהַחֲבִיתִים\hdot
וְהַיַּיִן\hdot
הֵפִיסוּ זָכָה מִי שֶׁזָּכָה׃}

\halacha{אָמַר לָהֶם הַמְּמוּנֶה צְאוּ וּרְאוּ אִם הִגִּיעַ זְמַן הַשְּׁחִיטָה\hdot
אִם הִגִּיעַ\hdot
הָרוֹאֶה אוֹמֵר בַּרְקָאי\hdot
מַתְיָא בֶן שְׁמוּאֵל אוֹמֵר הֵאִיר פְּנֵי כָל הַמִּזְרָח עַד שֶׁהוּא בְחֶבְרוֹן\hdot
וְהוּא אוֹמֵר הֵן׃}

\halacha{אָמַר לָהֶם צְאוּ וְהָבִיאוּ טָלֶה\hdot
מִלִּשְׁכַּת הַטְּלָאִים\hdot
וַהֲרֵי לִשְׁכַּת הַטְּלָאִים\hdot
הָֽיְתָה בְמִקְצוֹעַ\hdot
צְפוֹנִית מַעֲרָבִית\hdot
וְאַרְבַּע לְשָׁכוֹת הָיוּ שָׁם\hdot
אַחַת לִשְׁכַּת הַטְּלָאִים\hdot
וְאַחַת לִשְׁכַּת הַחוֹתָמוֹת\hdot
וְאַחַת לִשְׁכַּת בֵּית הַמּוֹקֵד\hdot
וְאַחַת לִשְׁכָּה שֶׁהָיוּ עוֹשִׂין בָּהּ לֶחֶם הַפָּנִים׃}

\halacha{נִכְנְסוּ לְלִשְׁכַּת הַכֵּלִים\hdot
וְהוֹצִיאוּ מִשָּׁם\hdot
תִּשְׁעִים וּשְׁלֹשָׁה כְּלֵי כֶסֶף וּכְלֵי זָהָב\hdot
הִשְׁקוּ אֶת הַתָּמִיד בְּכוֹס שֶׁל זָהָב\hdot
אַף עַל פִּי שֶׁהוּא מְבֻקָּר מִבָּעֶרֶב\hdot
מְבַקְּרִין אוֹתוֹ לְאוֹר הָאֲבוּקוֹת׃}

\halacha{מִי שֶׁזָּכָה בַתָּמִיד\hdot
מוֹשְׁכוֹ וְהוֹלֵךְ לְבֵית הַמִּטְבְּחַיִם\hdot
וּמִי שֶׁזָּכוּ בָאֵבָרִים הוֹלְכִין אַחֲרָיו\hdot
בֵּית הַמִּטְבְּחַיִם הָיָה לִצְפוֹנוֹ שֶׁל מִזְבֵּחַ\hdot
וְעָלָיו שְׁמוֹנָה עֲמוּדִים נַנָּסִין\hdot
וּרְבִיעִין שֶׁל אֶרֶז עַל גַּבֵּיהֶן\hdot
וְאָנְקְלִיּוֹת שֶׁל בַּרְזֶל הָיוּ קְבוּעִין בָּהֶם\hdot
וּשְׁלֹשָׁה סְדָרִים הָיָה לְכָל אֶחָד וְאֶחָד\hdot
שֶׁבָּהֶן תּוֹלִין וּמַפְשִׁיטִין עַל שֻׁלְחָנוֹת שֶׁל שַׁיִשׁ שֶׁבֵּין הָעַמּוּדִים׃}

\halacha{מִי שֶׁזָּכוּ בַדִּשּׁוּן מִזְבֵּחַ הַפְּנִימִי וְהַמְּנוֹרָה הָיוּ מַקְדִּימִין\hdot
וְאַרְבָּעָה כֶלִים בְּיָדָם\hdot
הַטֶּנִי וְהַכּוֹז וּשְׁתֵּי מַפְתֵּחוֹת\hdot
הַטֶּנִי דוֹמֶה לְתַרְקָב (גָּדוֹל) שֶׁל זָהָב\hdot
מַחֲזִיק קַבַּיִּין וָחֵצִי\hdot
וְהַכּוֹז דּוֹמֶה לְקִתּוֹן גָּדוֹל שֶׁל זָהָב\hdot
וּשְׁתֵּי מַפְתֵּחוֹת\hdot
אַחַת יוֹרֵד לְאַמַּת הַשֶּׁחִי\hdot
וְאַחַת פּוֹתֵחַ כִּוּוּן׃}
