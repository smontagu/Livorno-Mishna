\tractatehead{מסכת תענית}{taanit}

\perek{מֵאֵימָתַי}{מַזְכִּירִין גְּבוּרוֹת גְּשָׁמִים\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר מִיּוֹם טוֹב הָרִאשׁוֹן שֶׁל חָג\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר מִיּוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג\hdot
אָמַר לוֹ רִבִּי יְהוֹשֻׁעַ הוֹאִיל וְאֵין הַגְּשָׁמִים סִימַן בְּרָכָה בֶּחָג לָמָּה הוּא מַזְכִּיר\hdot
אָמַר לוֹ רִבִּי אֱלִיעֶזֶר אַף אֲנִי לֹא אָמַרְתִּי לִשְׁאוֹל\hdot
אֶלָּא לְהַזְכִּיר מַשִּׁיב הָרוּחַ וּמוֹרִיד הַגֶּשֶׁם בְּעוֹנָתוֹ\hdot
אָמַר לוֹ אִם כֵּן לְעוֹלָם יְהֵא מַזְכִּיר׃}

\halacha{אֵין שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
אֶלָּא סָמוּךְ לַגְּשָׁמִים\hdot
רִבִּי יְהוּדָה אוֹמֵר הָעוֹבֵר לִפְנֵי הַתֵּיבָה בְּיוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג\hdot
הָאַחֲרוֹן מַזְכִּיר הָרִאשׁוֹן אֵינוֹ מַזְכִּיר\hdot
בְּיוֹם טוֹב הָרִאשׁוֹן שֶׁל פֶּסַח\hdot
הָרִאשׁוֹן מַזְכִּיר הָאַחֲרוֹן אֵינוֹ מַזְכִּיר\hdot
עַד אֵימָתַי שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
רִבִּי יְהוּדָה אוֹמֵר עַד שֶׁיַּעֲבוֹר הַפֶּסַח\hdot
רִבִּי מֵאִיר אוֹמֵר עַד שֶיֵּצֵא נִיסָן\hdot
שֶנֶּאֱמַר וַיּוֹרֶד לָכֶם גֶּשֶׁם מוֹרֶה וּמַלְקוֹשׁ בָּרִאשׁוֹן׃}

\halacha{בִּשְׁלֹשָׁה בְּמַרְחֶשְׁוָן שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר בְּשִׁבְעָה בוֹ\hdot
חֲמִשָּׁה עָשָׂר יוֹם אַחַר הֶחָג\hdot
כְּדֵי שֶׁיַּגִּיעַ אַחֲרוֹן שֶׁבְּיִשְׂרָאֵל לִנְהַר פְּרָת׃}

\halacha{הִגִּיעַ שִׁבְעָה עָשָׂר בְּמַרְחֶשְׁוָן וְלֹא יָֽרְדוּ גְשָׁמִים\hdot
הִתְחִילוּ הַיְּחִידִים מִתְעַנִּים שָׁלֹשׁ תַּעֲנִיּוֹת\hdot
אוֹכְלִים וְשׁוֹתִים מִשֶּׁחֲשֵׁיכָה\hdot
וּמוּתָרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה׃}

\halacha{הִגִּיעַ רֹאשׁ חוֹדֶשׁ כִּסְלֵו\hdot
וְלֹא יָֽרְדוּ גְשָׁמִים\hdot
בֵּית דִּין גּוֹזְרִין שָׁלֹשׁ תַּעֲנִיּוֹת עַל הַצִּבּוּר\hdot
אוֹכְלִים וְשׁוֹתִים מִשֶּׁחֲשֵׁיכָה\hdot
וּמוּתָרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה׃}

\halacha{עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
בֵּית דִּין גּוֹזְרִין שָׁלֹשׁ תַּעֲנִיּוֹת אֲחֵרוֹת עַל הַצִּבּוּר\hdot
אוֹכְלִים וְשׁוֹתִים מִבְּעוֹד יוֹם\hdot
וַאֲסוּרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה\hdot
וְנוֹעֲלִין אֶת הַמֶּרְחֲצָאוֹת\hdot
עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
בֵּית דִּין גּוֹזְרִין עֲֲלֵיהֶם עוֹד שֶׁבַע\hdot
שֶׁהֵן שְׁלֹשׁ עֶשְׂרֵה תַּעֲנִיּוֹת עַל הַצִּבּוּר\hdot
הֲרֵי אֵלּוּ יְתֵרוֹת עַל הָרִאשׁוֹנוֹת\hdot
שֶׁבְּאֵלּוּ מַתְרִיעִין וְנוֹעֲלִין אֶת הַחֲנִיּוֹת\hdot
בַּשֵּׁנִי מַטִּין עִם חֲשֵׁיכָה\hdot
וּבַחֲמִישִׁי מֻתָּרִין מִפְּנֵי כְבוֹד הַשַּׁבָּת׃}

\halacha{עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
מְמַעֲטִין בְּמַשָּׂא וּמַתָּן\hdot
בְּבִנְיַן וּבִנְטִיעָה בְּאֵירוּסִין וּבְנִשּׂוּאִין וּבִשְׁאֵלַת שָׁלוֹם בֵּין אָדָם לַחֲבֵירוֹ כִּבְנֵי אָדָם הַנְּזוּפִין לַמָּקוֹם\hdot
הַיְּחִידִים חוֹזְרִים וּמִתְעַנִּים עַד שֶׁיֵּצֵא נִיסָן\hdot
יָצָא נִיסָן וְיָֽרְדוּ גְשָׁמִים\hdot
סִימַן קְלָלָה\hdot
שֶׁנֶּאֱמַר הֲלֹא קְצִיר חִטִּים הַיּוֹם וְגוֹמֵר׃}

\perek{סֵדֶר}{תַּעֲנִיּוֹת כֵּיצַד\hdot
מוֹצִיאִין אֶת הַתֵּיבָה לִרְחוֹבָהּ שֶׁל עִיר\hdot
וְנוֹתְנִין אֵפֶר מַקְלֶה עַל גַּבֵּי הַתֵּיבָה\hdot
וּבְרֹאשׁ הַנָּשִׂיא\hdot
וּבְרֹאשׁ אָב בֵּית דִּין\hdot
וְכָל אֶחָד וְאֶחָד נוֹטֵל וְנוֹתֵן בְּרֹאשׁוֹ\hdot
הַזָּקֵן שֶׁבָּהֶן אוֹמֵר לִפְנֵיהֶן דִּבְרֵי כִבּוּשִׁין\hdot
אַחֵינוּ לֹא נֶאֱמַר בְּאַנְשֵׁי נִינְוֵה\hdot
וַיַּרְא אֱלֹהִים אֶת שַׂקָּם וְאֶת תַּעֲנִיתָם\hdot
אֶלָּא וַיַּרְא הָאֱלֹהִים אֶת מַעֲשֵׂיהֶם\hdot
כִּי שָׁבוּ מִדַּרְכָּם הָרָעָה\hdot
וּבְקַבָּלָה הוּא אוֹמֵר\hdot
וְקִרְעוּ לְבַבְכֶם וְאַל בִּגְדֵיכֶם׃}

\halacha{עָֽמְדוּ בַתְּפִלָּה\hdot
מוֹרִידִין לִפְנֵי הַתֵּיבָה\hdot
זָקֵן וְרָגִיל וְיֵשׁ לוֹ בָנִים\hdot
וּבֵיתוֹ רֵיקָם\hdot
כְּדֵי שֶׁיְּהֵא לִבּוֹ שָׁלֵם בַּתְּפִלָּה\hdot
וְאוֹמֵר לִפְנֵיהֶם אַרְבַּע וְעֶשְׂרִים בְּרָכוֹת\hdot
שְׁמֹנֶה עֶשְׂרֶה שֶׁבְּכָל יוֹם וּמוֹסִיף עֲלֵיהֶם עוֹד שֵׁשׁ׃}

\halacha{וְאֵלּוּ הֵן זִכְרוֹנוֹת וְשׁוֹפָרוֹת\hdot
אֶל יְיָ בַּצָּרָתָה לִּי קָרָאתִי וַיַּעֲַנֵנִי\hdot
אֶשָּׂא עֵינַי אֶל הֶהָרִים וְגוֹמֵר\hdot
מִמַּעֲמַקִּים קְרָאתִיךָ יְיָ\hdot
תְּפִלָּה לְעָנִי כִי יַעֲטוֹף\hdot
רִבִּי יְהוּדָה אוֹמֵר לֹא הָיָה צָרִיךְ לוֹמַר זִכְרוֹנוֹת וְשׁוֹפָרוֹת\hdot
אֶלָּא אוֹמֵר תַּחְתֵּיהֶן\hdot
רָעָב כִּי יִהְיֶה בָאָרֶץ\hdot
דֶּבֶר כִּי יִהְיֶה וְגוֹמֵר\hdot
אֲשֶׁר הָיָה דְּבַר יְיָ אֶל יִרְמְיָהוּ עַל דִּבְרֵי הַבַּצָּרוֹת\hdot
וְאוֹמֵר חוֹתָמֵיהֶן׃}
