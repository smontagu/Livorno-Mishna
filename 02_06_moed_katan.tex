\tractatehead{מסכת מועד קטן}{moedKatan}

\chead{יום ח}

\perek{מַשְׁקִין}{בֵּית הַשְׁלָחִין בְּמוֹעֵד וּבַשְּׁבִיעִית בֵּין מִמַּעְיָן שֶׁיָּצָא בַּתְּחִלָּה\hdot
בֵּין מִמַּעְיָן שֶׁלֹּא יָצָא בַּתְּחִלָּה\hdot
אֲבָל אֵין מַשְׁקִין לֹא מִמֵּי הַגְּשָׁמִים\hdot
וְלֹא מִמֵּי הַקִּילוֹן\hdot
וְאֵין עוֹשִׂין עֻגְּיוֹת לַגְּפָנִים׃}

\halacha{רִבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר אֵין עוֹשִׂין אֶת הָאַמָּה בַּתְּחִלָּה בְּמוֹעֵד וּבַשְּׁבִיעִית\hdot
וַחֲכָמִים אוֹמְרִים עוֹשִׂין אֶת הָאַמָּה בַּתְּחִלָּה בַּשְּׁבִיעִית\hdot
וּמְתַקְּנִין אֶת הַמְּקֻלְקָלוֹת בְּמוֹעֵד\hdot
וּמְתַקְּנִין אֶת קִלְקוּלֵי הַמַּיִם שֶׁבִּרְשׁוּת הָרַבִּים\hdot
וְחוֹטְטִין אוֹתָן\hdot
וּמְתַקְּנִין אֶת הַדְּרָכִים\hdot
וְאֶת הָֽרְחוֹבוֹת\hdot
וְאֶת מִקְווֹת הַמַּיִם\hdot
וְעוֹשִׂין כָּל צָרְכֵי הָרַבִּים\hdot
וּמְצַיְּנִין אֶת הַקְּבָרוֹת\hdot
וְיוֹצְאִין עַל הַכִּלְאָיִם׃}

\halacha{רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר מוֹשְׁכִין אֶת הַמַּיִם מֵאִילָן לְאִילָן\hdot
וּבִלְבַד שֶׁלֹּא יַשְׁקֶה אֶת כָּל הַשָּׂדֶה\hdot
זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד\hdot
לֹא יַשְׁקֵם בְּמוֹעֵד\hdot
וַחֲכָמִים מַתִּירִין בַּזֶּה וּבַזֶּה׃}

\halacha{צָדִין אֶת הָאִישׁוֹת וְאֶת הָעַכְבָּרִים מִשְּׂדֵה הָאִילָן וּמִשְּׂדֵה הַלָּבָן שֶׁלֹּא כְדַרְכּוֹ בְּמוֹעֵד וּבַשְּׁבִיעִית\hdot
וַחֲכָמִים אוֹמְרִים מִשְּׂדֵה הָאִילָן כְּדַרְכּוֹ\hdot
וּמִשְּׂדֵה הַלָּבָן שֶׁלֹּא כְדַרְכּוֹ\hdot
וּמְקָרִין אֶת הַפִּרְצָה בְּמועֵד\hdot
וּבַשְּׁבִיעִית בּוֹנֶה כְּדַרְכּוֹ׃}

\halacha{רִבִּי מֵאִיר אוֹמֵר רוֹאִין אֶת הַנְּגָעִים (בַּתְּחִלָּה) לְהָקֵל\hdot
אֲבָל לֹא לְהַחְמִיר\hdot
וַחֲכָמִים אוֹמְרִים לֹא לְהָקֵל וְלֹא לְהַחְמִיר\hdot
וְעוֹד אָמַר רִבִּי מֵאִיר מְלַקֵּט אָדָם עַצְמוֹת אָבִיו וְאִמּוֹ\hdot
מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ\hdot
רִבִּי יוֹסֵי אוֹמֵר אֵבֶל הוּא לוֹ\hdot
לֹא יְעוֹרֵר אָדָם עַל מֵתוֹ\hdot
וְלֹא יַסְפִּידֶנּוּ קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם׃}

\halacha{אֵין חוֹפְרִין כּוּכִין וּקְבָרוֹת בְּמוֹעֵד\hdot
אֲבָל מְחַנְּכִים אֶת הַכּוֹכִין בְּמוֹעֵד\hdot
וְעוֹשִׂין נִבְרֶכֶת בְּמוֹעֵד\hdot
וְאָרוֹן עִם הַמֵּת בֶּחָצֵר\hdot
רִבִּי יְהוּדָה אוֹסֵר אֶלָּא אִם כֵּן יֵשׁ עִמּוֹ נְסָרִים׃}

\halacha{אֵין נוֹשְׂאִין נָשִׁים בְּמוֵֹעד\hdot
לֹא בְתוּלוֹת\hdot
וְלֹא אַלְמָנוֹת\hdot
וְלֹא מְיַבְּמִין מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ\hdot
אֲבָל מַחֲזִיר הוּא אֶת גְּרוּשָׁתוֹ\hdot
וְעוֹשָׂה אִשָּׁה תַּכְשִׁיטֶיהָ בְּמוֹעֵד\hdot
רִבִּי יְהוּדָה אוֹמֵר לֹא תָסוּד מִפְּנֵי שֶׁנִּוּוּל הוּא לָהּ׃}

\halacha{הַהֶדְיוֹט תּוֹפֵר כְּדַרְכּוֹ וְהָאוֹמָן מַכְלִיב\hdot
וּמְסָֽרְגִין אֶת הַמִּטּוֹת\hdot
רִבִּי יוֹסֵי אוֹמֵר (אַף) מְמַתְּחִין׃}

\halacha{מַעֲמִידִין תַּנּוּר וְכִירַיִן וְרֵיחַיִם בְּמוֹעֵד\hdot
רִבִּי יְהוּדָה אוֹמֵר אֵין מַכְבִּישִׁין אֶת הָרֵיחַיִם בַּתְּחִלָּה׃}

\halacha{עוֹשִׂין מַעֲקֶה לַגַּג וְלַמַּרְפֶּסֶת מַעֲשֵׂה הֶדְיוֹט\hdot
אֲבָל לֹא מַעֲשֵׂה אוֹמָן\hdot
שָׁפִין אֶת הַסְּדָקִין וּמַעֲגִילִין אוֹתָן בְּמַעֲגִילָה בַּיָּד וּבָרֶגֶל\hdot
אֲבָל לֹא בַמַּחְלְצַיִם\hdot
הַצִּיר וְהַצִּינּוֹר וְהַקּוֹרָה וְהַמַּנְעוּל\hdot
וְהַמַּפְתֵּחַ שֶׁנִּשְׁבָּרוּ\hdot
מְתַקְּנָן בְּמוֹעֵד\hdot
וּבִלְבַד שֶׁלֹּא יְכַוֵּין מְלַאכְתּוֹ בְמוֹעֵד\hdot
וְכָל כַּבָּשִׁין שֶׁהוּא יָכוֹל לֶאֱכוֹל מֵהֶן בְּמוֹעֵד כּוֹבְשָׁן׃}

\perek{מִי}{שֶׁהָפַךְ אֶת זֵיתָיו וְאֵירְעוֹ אֵבֶל אוֹ אֹנֶס אוֹ שֶׁהִטְעוּהוּ פוֹעֲלִים\hdot
טוֹעֵן קוֹרָה רִאשׁוֹנָה וּמַנִּיחָה לְאַחַר הַמּוֹעֵד\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
רִבִּי יוֹסֵי אוֹמֵר זוֹלֵף וְגוֹמֵר וְגַף כְּדַרְכּוֹ׃}

\halacha{וְכֵן מִמִּי שֶׁהָיָה יֵינוֹ בְּתוֹךְ הַבּוֹר וְאֵירְעוֹ אֵבֶל אוֹ אֹנֶס אוֹ שֶׁהִטְעוּהוּ (פוֹעֲלִים)\hdot
זוֹלֵף וְגוֹמֵר וְגַף כְּדַרְכּוֹ\hdot
דִּבְרֵי רִבִּי יוֹסֵי\hdot
רִבִּי יְהוּדָה אוֹמֵר עוֹשֶׂה לוֹ לִמּוּדִים בִּשְׁבִיל שֶׁלֹּא יַחְמִיץ׃}

\halacha{מַכְנִיס אָדָם פֵּירוֹתָיו מִפְּנֵי הַגַּנָּבִים\hdot
וְשׁוֹלֶה פִשְׁתָּנוֹ מִן הַמִּשְׁרָה בִּשְׁבִיל שֶׁלֹּא תֵאָבֵד\hdot
וּבִלְבַד שֶׁלֹּא יְכַוֵּין אֶת מְלַאכְתּוֹ בְמוֹעֵד\hdot
וְכֻלָּן אִם כִּוְּנוּ מְלַאכְתָּן בְּמוֹעֵד יֵאָבֵדוּ׃}

\halacha{אֵין לוֹקְחִין בָּתִּים (אֲבָנִים) עֲבָדִים וּבְהֵמָה אֶלָּא לְצוֹרֶךְ הַמּוֹעֵד\hdot
אוֹ לְצוֹרֶךְ הַמּוֹכֵר שֶׁאֵין לוֹ מַה יֹּאכַל\hdot
אֵין מְפַנִּין מִבַּיִת לְבַיִת\hdot
אֲבָל מְפַנֶּה הוּא לַחֲצֵירוֹ\hdot
אֵין מְבִיאִין כֵּלִים מִבֵּית הָאוֹמָן\hdot
אִם חוֹשֵׁשׁ לָהֶם מְפַנָּן לְחָצֵר אַחֶרֶת׃}

\halacha{מְחַפִּין אֶת הַקְּצִיעוֹת בַּקַּשׁ\hdot
רִבִּי יְהוּדָה אוֹמֵר אַף מְעַבִּין\hdot
מוֹכְרֵי פֵירוֹת כְּסוּת וְכֵלִים\hdot
מוֹכְרִים בְּצִנְעָה לְצוֹרֶךְ הַמּוֹעֵד\hdot
הַצַּיָּדִין וְהַדְּשׁוּשׁוֹת וְהַגְּרוּסוֹת עוֹשִׂין בְּצִנְעָה לְצוֹרֶךְ הַמּוֹעֵד\hdot
רִבִּי יוֹסֵי אוֹמֵר הֵם הֶחְמִירוּ עַל עַצְמָן׃}

\perek{וְאֵלּוּ}{מְגַלְּחִין בְּמוֹעֵד\hdot
הַבָּא מִמְּדִינַת הַיָּם\hdot
וּמִבֵּית הַשִּׁבְיָה\hdot
וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִים\hdot
וְהַמְּנֻדֶּה שֶׁהִתִּירוּ לוֹ חֲכָמִים\hdot
וְכֵן מִי שֶׁנִּשְׁאַל לֶחָכָם וְהֻתָּר\hdot
וְהַנָּזִיר\hdot
וְהַמְּצוֹרָע\hdot
וְהָעוֹלֶה מִטֻּמְאָתוֹ לְטַהֳרָתוֹ׃}

\halacha{וְאֵלּוּ מְכַבְּסִין בְּמוֹעֵד\hdot
הַבָּא מִמְּדִינַת הַיָּם\hdot
וּמִבֵּית הַשִּׁבְיָה\hdot
וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִים\hdot
וְהַמְּנֻדֶּה שֶׁהִתִּירוּ לוֹ חֲכָמִים\hdot
וְכֵן מִי שֶׁנִּשְׁאַל לֶחָכָם וְהֻתָּר\hdot
מִטְפּחוֹת הַיָּדַיִם\hdot
וּמִטְפְּחוֹת הַסַּפָּרִים\hdot
וּמִטְפְּחוֹת הַסַּפָּג\hdot
הַזָּבִין וְהַזָּבוֹת וְהַנִּדּוֹת וְהַיּוֹלְדוֹת\hdot
וְכָל הָעוֹלִין מִטֻּמְאָה לְטַהֳרָה הֲרֵי אֵלּוּ מוּתָרִין\hdot
וּשְׁאַר כָּל אָדָם אֲסוּרִין׃}

\halacha{וְאֵלּוּ כוֹתְבִין בְּמוֹעֵד\hdot
קִדּוּשֵׁי נָשִׁים\hdot
וְגִיטִין\hdot
וְשׁוֹבָרִין\hdot
דַּיְתִיקִי\hdot
מַתָּנָה\hdot
וּפְרוֹזְבּוֹלִין\hdot
אִגְּרוֹת שׁוֹם\hdot
וְאִגְּרוֹת מָזוֹן\hdot
שִׁטְרֵי חֲלִיצָה וּמִיאוּנִין\hdot
וְשִׁטְרֵי בֵֵֵֵֵרוּרִין\hdot
וּגְזֵרוֹת בֵּית דִּין\hdot
וְאִגָּרוֹת שֶׁל רְשׁוּת׃}

\halacha{אֵין כּוֹתְבִין שִׁטְרֵי חוֹב בְּמוֹעֵד\hdot
וְאִם אֵינוֹ מַאֲמִינוֹ אוֹ שֶׁאֵין לוֹ מַה יֹּאכַל הֲרֵי זֶה יִכְתּוֹב\hdot
אֵין כּוֹתְבִין סְפָרִים תְּפִלִּין וּמְזוּזוֹת בְּמוֹעֵד\hdot
וְאֵין מַגִּיהִין אוֹת אַחַת אֲפִילוֹ בְּסֵפֶר עֶזְרָא\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
כּוֹתֵב אָדָם תְּפִלִּין וּמְזוּזוֹת לְעַצְמוֹ\hdot
וְטוֹוֶה עַל יְרֵיכוֹ תְּכֵלֶת לְצִיצִיתוֹ׃}

\halacha{הַקּוֹבֵר אֶת מֵתוֹ שְׁלֹשָׁה יָמִים קֹדֶם הָרֶגֶל\hdot
בָּֽטְלָה הֵימֶנּוּ גְּזֵרַת שִׁבְעָה\hdot
שְׁמֹנָה\hdot
בָּֽטְלָה הֵימֶנּוּ גְּזֵרַת שְׁלֹשִׁים\hdot
מִפְּנֵי שֶׁאָֽמְרוּ שַׁבָּת עוֹלָה וְאֵינָהּ מַפְסֶקֶת\hdot
רְגָלִים מַפְסִיקִין וְאֵינָן עוֹלִין׃}

\halacha{רִבִּי אֱלִיעֶזֶר אוֹמֵר מִשֶּׁחָרֵב בֵּית הַמִּקְדָּשׁ עֲצֶרֶת כְּשַׁבָּת\hdot
רַבָּן גָּמְלִיאֵל אוֹמֵר\hdot
