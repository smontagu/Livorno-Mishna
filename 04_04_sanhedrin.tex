\tractatehead{סנהדרין}{sanhedrin}

\perek{דִּינֵי}{מָמוֹנוֹת בִּשְׁלֹשָׁה\hdot
גְּזֵילוֹת וַחֲבָלוֹת בִּשְׁלֹשָׁה\hdot
נֶזֶק וַחֲצִי נֶזֶק\hdot
תַּשְׁלוּמֵי כֶפֶל וְתַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה בִּשְׁלֹשָׁה\hdot
הָאוֹנֵס וְהַמְּפַתֶּה וְהַמּוֹצִיא שֵׁם רָע בִּשְׁלֹשָׁה דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים מוֹצִיא שֵׁם רַע בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
מִשּׁוּם שֶׁיֵּשׁ בּוֹ דִּינֵי נְפָשׁוֹת׃}

\halacha{מַכּוֹת בִּשְׁלֹשָׁה\hdot
מִשּׁוּם רִבִּי יִשְׁמָעֵאל אָֽמְרוּ בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
עִבּוּר הַחֹדֶשׁ בִּשְׁלֹשָׁה\hdot
עִבּוּר הַשָּׁנָה בִּשְׁלֹשָׁה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר בִּשְׁלֹשָׁה מַתְחִילִין\hdot
וּבַחֲמִשָּׁה נוֹשְׂאִין וְנוֹתְנִין\hdot
וְגוֹמְרִין בְּשִׁבְעָה\hdot
וְאִם גָּֽמְרוּ בִשְׁלֹשָׁה מְעֻבֶּרֶת׃}

\halacha{סְמִיכַת זְקֵנִים וַעֲרִיפַת עֶגְלָה בִּשְׁלֹשָׁה\hdot
דִּבְרֵי רִבִּי שִׁמְעוֹן\hdot
וְרִבִּי יְהוּדָה אוֹמֵר בַּחֲמִישָּׁה\hdot
הַחֲלִיצָה וְהַמִּאוּנִין בִּשְׁלֹשָׁה\hdot
נֶטַע רְבָעִי וּמַעֲשֵׂר שֵׁנִי שֶׁאֵין דָּמָיו יְדוּעִין בִּשְׁלֹשָׁה\hdot
הַהֶקְדֵּשׁוֹת בִּשְׁלֹשָׁה\hdot
הָעֲרָכִין הַמְּטַלְטְלִין בִּשְׁלֹשָׁה\hdot
רִבִּי יְהוּדָה אוֹמֵר אֶחָד מֵהֶן כֹּהֵן\hdot
וְהַקַּרְקָעוֹת תִּשְׁעָה וְכֹהֵן\hdot
וְאָדָם כַּיּוֹצֵא בָהֶן׃}

\halacha{דִּינֵי נְפָשׁוֹת בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
הָרוֹבֵעַ וְהַנִּרְבָּע בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
שֶׁנֶּאֱמַר וְהָרַגְתָּ אֶת הָאִשָּׁה וְאֶת הַבְּהֵמָה\hdot
וְאוֹמֵר וְאֶת הַבְּהֵמָה תַּהֲרוֹגוּ\hdot
שׁוֹר הַנִּסְקָל בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
שֶׁנֶּאֱמַר הַשּׁוֹר יִסָּקֵל וְגַם בְּעָלָיו יוּמָת\hdot
כְּמִיתַת בְּעָלִים כַּךְ מִיתַת הַשּׁוֹר\hdot
הַזְּאֵב וְהָאֲרִי וְהַדּוֹב וְהַנָּמֵר וְהַבַּרְדְּלָס וְהַנָּחָשׁ מִיתָתָן בְּעֶשְׂרִים וּשְׁלֹשָׁה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר כָּל הַקּוֹדֵם לְהָרְגָן זָכָה\hdot
רִבִּי עֲקִיבָא אוֹמֵר מִיתָתָן בְּעֶשְׂרִים וּשְׁלֹשָׁה׃}

\halacha{אֵין דָּנִין לֹא אֶת הַשֵּׁבֶט וְלֹא אֶת נְבִיא הַשֶּׁקֶר\hdot
וְלֹא אֶת כֹּהֵן גָּדוֹל\hdot
אֶלָּא בְּבֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
וְאֵין מוֹצִיאִין לְמִלְחֶמֶת הָֽרְשׁוּת\hdot
אֶלָּא עַל פִּי בֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
אֵין מוֹסִיפִין עַל הָעִיר וְעַל הָעֲזָרוֹת\hdot
אֶלָּא עַל פִּי בֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
אֵין עוֹשִׂין סַנְהֶדְרָיוֹת לַשְּׁבָטִים\hdot
אֶלָּא עַל פִּי בֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
אֵין עוֹשִׂין עִיר הַנִּדַּחַת\hdot
אֶלָּא בְּבֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
וְאֵין עוֹשִׂין עִיר הַנִּדַּחַת בַּסְּפָר\hdot
וְלֹא שָׁלשׁ עִיר הַנִּדַּחַת\hdot
אֲבָל עוֹשִׂין אַחַת אוּ שְׁתַּיִם׃}

\halacha{סַנְהֶדְרֵי גְדוֹלָה הָֽיְתָה שֶׁל שִׁבְעִים וְאֶחָד\hdot
וּקְטַנָּה שֶׁל עֶשְׂרִים וּשְׁלֹשָׁה\hdot
וּמְנַיִן לַגְּדוֹלָה שֶׁהִיא שֶׁל שִׁבְעִים וְאֶחָד\hdot
שֶׁנֶּאֱמַר אֶסְפָה לִי שִׁבְעִים אִישׁ מִזִּקְנֵי יִשְׂרָאֵל\hdot
וּמֹשֶׁה עַל גַּבֵּיהֶן\hdot
הֲרֵי שִׁבְעִים וְאֶחָד\hdot
רִבִּי יְהוּדָה אוֹמֵר שִׁבְעִים\hdot
וּמְנַיִן לַקְּטַנָּה שֶׁהִיא שֶׁל עֶשְׂרִים וּשְׁלֹשָׁה\hdot
שֶׁנֶּאֱמַר וְשָֽׁפְטוּ הָעֵדָה\hdot
וְהִצִּילוּ הָעֵדָה\hdot
עֵדָה שׁוֹפֶטֶת\hdot
וְעֵדָה מַצֶּלֶת\hdot
הֲרֵי כַּאן עֶשְׂרִים\hdot
וּמְנַיִן לָעֵדָה שֶׁהִיא עֲשָׂרָה\hdot
שֶׁנֶּאֱמַר עַד מָתַי לָעֵדָה הָרָעָה הַזֹּאת\hdot
יָֽצְאוּ יְהוֹשֻׁעַ וְכָלֵב\hdot
וּמְנַיִן לְהָבִיא עוֹד שְׁלֹשָׁה\hdot
מִמַּשְׁמַע שֶׁנֶּאֱמַר לֹא תִהְיֶה אַחֲרֵי רַבִּים לְרָעוֹת\hdot
שׁוֹמֵעַ אֲנִי שֶׁאֶהְיֶה עִמָּהֶם לְטוֹבָה\hdot
אִם כֵּן לָמָּה נֶאֱמַר אַחֲרֵי רַבִּים לְהַטּוֹת\hdot
לֹא כַּהַטַּיָּֽתְךָ לְטוֹבָה\hdot
הַטַּיָּֽתְךָ לְרָעָה\hdot
הַטַּיָּֽתְךָ לְטוֹבָה עַל פִּי אֶחָד\hdot
הַטַּיָּֽתְךָ לְרָעָה עַל פִּי שְׁנַיִם\hdot
וְאֵין בֵּית דִּין שָׁקוּל\hdot
מוֹסִיפִין עֲלֵיהֶן עוֹד אֶחָד\hdot
הֲרֵי כָּאן עֶשְׂרִים ושְׁלֹשָׁה\hdot
וְכַמָּה יְהֵא בָעִיר וּתְהֵא רְאוּיָה לְסַנְהֶדְרִין מֵאָה וְעֶשְׂרִים\hdot
רִבִּי נְחֶמְיָה אוֹמֵר מָאתַיִם וּשְׁלֹשִׁים\hdot
כְּנֶגֶד שָׂרֵי עֲשָׂרוֹת׃}

\perek{כֹּהֵן}{גָּדוֹל דָּן וְדָנִין אוֹתוֹ\hdot
מֵעִיד וּמְעִידִין אוֹתוֹ\hdot
חוֹלֵץ וְחוֹלְצִין לְאִשְׁתּוֹ\hdot
וּמְיַבְּמִין אֶת אִשְׁתּוֹ\hdot
אֲבָל הוּא אֵינוֹ מְיַבֵּם\hdot
מִפְּנֵי שֶׁהוּא אָסוּר בְּאַלְמָנָה\hdot
מֵת לוֹ מֵת\hdot
אֵינוֹ יוֹצֵא אַחַר הַמִּטָּה\hdot
אֶלָּא הֵן נִכְסִין וְהוּא נִגְלֶה\hdot
הֵן נִגְלִין וְהוּא נִכְסֶה\hdot
וְיוֹצֵא עִמָּהֶן עַד פֶּתַח הָעִיר\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר אֵינוֹ יוֹצֵא מִן הַמִּקְדָּשׁ שֶׁנֶּאֱמַר וּמִן הַמִּקְדָּשׁ לֹא יֵצֵא\hdot
וּכְשֶׁהוּא מְנַחֵם אַחֵרִים\hdot
דֶּרֶךְ כָּל הָעָם עוֹבְרִין זֶה אַחַר זֶה\hdot
וְהַמְּמוּנֶה מְמַצְּעוֹ בֵּינוֹ לְבֵין הָעָם\hdot
וּכְשֶׁהוּא מִתְנַחֵם מֵאֲחֵרִים\hdot
כָּל הָעָם אוֹמְרִים לוֹ אָנוֹ כַפָּרָתָךְ\hdot
וְהוּא אוֹמֵר לָהֶן תִּתְבָּֽרְכוּ מִן הַשָּׁמַיִם\hdot
וּכְשֶׁמַּבְרִין אוֹתוֹ\hdot
כָּל הָעָם מְסֻבִּין עַל הָאָרֶץ\hdot
וְהוּא מֵסֵב עַל הַסַּפְסָל׃}

\halacha{הַמֶּלֶךְ לֹא דָן וְלֹא דָנִין אוֹתוֹ\hdot
לֹא מֵעִיד וְלֹא מְעִידִין אוֹתוֹ\hdot
לֹא חוֹלֵץ וְלֹא חוֹלְצִין לְאִשְׁתּוֹ\hdot
לֹא מְיַבֵּם וְלֹא מְיַבְּמִין לְאִשְׁתּוֹ\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם רָצָה לַחֲלוֹץ אוֹ לְיַבֵּם\hdot
זָכוּר לְטוֹב\hdot
אָֽמְרוּ לוֹ אֵין שׁוֹמְעִין לוֹ\hdot
וְאֵין נוֹשְׂאִין אַלְמָנָתוֹ\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
נוֹשֵׂא הַמֶּלֶךְ אַלְמָנָתוֹ שֶׁל מֶלֶךְ\hdot
שֶׁכֵּן מָצִינוּ בְּדָוִד שֶׁנָּשָׂא אַלְמָנָתוֹ שֶׁל שָׁאוּל\hdot
שֶׂנֶּאֱמַר וָאֶתְּנָה לָךְ אֶת בֵּית אֲדוֹנֶיךָ וְאֶת נְשֵׁי אֲדוֹנֶיךָ בְּחֵיקֶךָ׃}

\halacha{מֵת לוֹ מֵת\hdot
אֵינוֹ יוֹצֵא מִפֶּתַח פַּלְטְרִין שֶׁלּוֹ\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם רוֹצֶה לָצֵאת אַחַר הַמִּטָּה יוֹצֵא שֶׁכֵּן מָצִינוּ בְדָוִד שֶׁיָּצָא אַחַר מִטָּתוֹ שֶׁל אַבְנֵר\hdot
שֶׁנֶּאֱמַר וְהַמֶּלֶךְ דָּוִד הוֹלֵךְ אַחֲרֵי הַמִּטָּה\hdot
אָֽמְרוּ לוֹ לֹא הָיָה הַדָּבָר אֶלָּא לְפַיֵּס אֶת הָעָם\hdot
וּכְשֶׁמַּבְרִין אוֹתוֹ כָּל הָעָם מְסֻבִּין עַל הָאָרֶץ\hdot
וְהוּא מֵסֵב עַל הַדַּרְגָּשׁ׃}

\halacha{וּמוֹצִיא לְמִּלְחֶמֶת הָֽרְשׁוּת\hdot
עַל פִּי בֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד\hdot
וּפוֹרֵץ לַעֲשׂוֹת לוֹ דֶרֶךְ\hdot
וְאֵין מַמְחִין בְּיָדוֹ\hdot
דֶּרֶךְ הַמֶּלֶךְ אֵין לָהּ שִׁיעוּר\hdot
וְכָל הָעָם בּוֹזְזִין וְנוֹתְנִין לְפָנָיו\hdot
וְהוּא נוֹטֵל חֵלֶק בָּרֹאשׁ\hdot
לֹא יַרְבֶּה לוֹ נָשִׁים אֶלָּא שְׁמוֹנֶה עֶשְׂרֵה\hdot
רִבִּי יְהוּדָה אוֹמֵר מַרְבֶּה הוּא לוֹ\hdot
וּבִלְבַד שֶׁלֹּא יְהוּ מְסִירוֹת אֶת לִבּוֹ\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אֲפִילוּ אַחַת וּמְסִירָה אֶת לִבּוֹ\hdot
הֲרֵי זֶה לֹא יִשָּׂאֶנָּה\hdot

\halacha{מֵאֶימָתַי חַיָּב מִשֶּׁיֹּאכַל טַרְטִימַר בָּשָׂר\hdot
וְיִשְׁתֶּה חֲצִי לוֹג יַיִן הָאִיטַלְקִי\hdot
רִבִּי יוֹסֵי אוֹמֵר מָנֶה בָשָׂר וְלוֹג יַיִן\hdot
אָכַל בַּחֲבוּרַת מִצְוָה\hdot
אָכַל בְּעִבּוּר הַחֹדֶשׁ\hdot
אָכַל מַעֲשֵׂר שֵׂנִי בִּירוּשָׁלַיִם\hdot
אָכַל נְבֵלוֹת וּטְרֵפוֹת שְׁקָצִים וּרְמָשִׁים\hdot
(אָכַל טֶבֶל וּמַעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ\hdot
וּמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁלֹּא נִפְדּוּ\hdot)
אָכַל דָּבָר שֶׁהוּא מִצְוָה\hdot
וְדָבָר שֶׁהוּא עֲבֵירָה\hdot
אָכַל כָּל מַאֲכָל\hdot
וְלֹא אָכַל בָּשָׂר\hdot
שָׁתָה כָל מַשְׁקֶה וְלֹא שָׁתָה יַיִן\hdot
אֵינוֹ נֶעֱשָׂה בֵּן סוֹרֵר וּמוֹרֶה\hdot
עַד שֶׁיֹּאכַל בָּשָׂר וְיִשְׁתֶּה יַיִן\hdot
שֶׁנֶּאֱמַר זוֹלֵל וְסוֹבֵא\hdot
וְאַף עַל פִּי שֶׁאֵין רְאָיָה לַדָּבָר\hdot
זֵכֶר לַדָּבָר\hdot
שֶׁנֶּאֱמַר אַל תְּהִי בְּסוֹבְאֵי יַיִן בְּזוֹלְלֵי בָשָׂר לָמוֹ׃}
