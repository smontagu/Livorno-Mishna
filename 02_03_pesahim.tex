\tractatehead{מסכת פסחים}{pesahim}

\sechead{יום ז}

\perek[832]{אוֹר}{לְאַרְבָּעָה עָשָׂר בּוֹדְקִין אֶת הֶחָמֵץ לְאוֹר הַנֵּר\hdot
כָּל מָקוֹם שֶׁאֵין מַכְנִיסִין בּוֹ חָמֵץ\hdot
אֵין צָרִיךְ בְּדִיקָה\hdot
וְלָמָּה אָֽמְרוּ שְׁתֵּי שׁוּרוֹת בַּמַּרְתֵּף\hdot
מָקוֹם שֶׁמַּכְנִיסִין בּוֹ חָמֵץ\hdot
בֵּית שַׁמָּאי אוֹמְרִים שְׁתֵּי שׁוּרוֹת עַל פְּנֵי כָל הַמַּרְתֵּף\hdot
וּבֵית הִלֵּל אוֹמְרִים שְׁתֵּי שׁוּרוֹת הַחִיצוֹנוֹת שֶׁהֵן הָעֶלְיוֹנוֹת׃}

\halacha{אֵין חוֹשְׁשִין שֶׁמֶּא גָֽרְרָה חֻלְדָה מִבַּיִת לְבַיִת וּמִמָּקוֹם לְמָקוֹם\hdot
דְּאִם כֵּן מֵחָצֵר לֶחָצֵר וּמֵעִיר לָעִיר\hdot
אֵין לַדָּבָר סוֹף׃}

\halacha{רִבִּי יְהוּדָה אוֹמֵר בּוֹדְקִין אוֹר אַרְבָּעָה עָשָׂר וּבְאַרְבָּעָה עָשָׂר שַׁחְרֵית\hdot
וּבְשַׁעַת הַבִּיעוּר\hdot
וַחֲכָמִים אוֹמְרִים לֹא בָדַק אוֹר אַרְבַָּעָה עָשָׂר\hdot
יִבְדּוֹק בְּאַרְבָּעָה עָשָׂר\hdot
לֹא בָדַק בְּאַרְבָּעָה עָשָׂר\hdot
יִבְדּוֹק בְּתוֹךְ הַמּוֹעֵד\hdot
לֹא בָדַק בְּתוֹךְ הַמּוֹעֵד יִבְדּוֹק לְאַחַר הַמּוֹעֵד\hdot
וּמַה שֶּׁמְּשַׁיֵּר יַנִּיחֶנּוּ בְצִנְעָא כְּדֵי שֶׁלֹּא יְהֵא צָרִיךְ בְּדִיקָה אַחֲרָיו׃}

\halacha{רִבִּי מֵאִיר אוֹמֵר אוֹכְלִין כָּל חָמֵשׁ וְשׂוֹרְפִין בִּתְחִלַּת שֵׁשׁ
רִבִּי יְהוּדָה אוֹמֵר אוֹכְלִין כָּל אַרְבַּע\hdot
וְתוֹלִין כָּל חָמֵשׁ\hdot
וְשׂוֹרְפִין בִּתְחִלַּת שֵׁשׁ׃}
