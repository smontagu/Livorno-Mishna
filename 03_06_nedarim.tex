\perek{כָּל}{כִּנּוּיֵי נְדָרִים כִּנְדָרִים\hdot
וַחֲרָמִים כַּחֲרָמִים\hdot
וּשְׁבוּעוֹת כִּשְׁבוּעוֹת\hdot
וּנְזִירוֹת כִּנְזִירוֹת\hdot
הָאוֹמֵר לַחֲבֵירוֹ מוּדַרְנִי מִמָּךְ\hdot
מֻפְרַשְׁנִי מִמָּךְ\hdot
מְרוּחַקְנִי מִמָּךְ\hdot
שֶׁאֵינִי אוֹכֵל לָךְ\hdot
שֶׁאֵינִי טוֹעֵם לָךְ אָסוּר\hdot
מְנוּדֶה אֲנִי לָךְ\hdot
רִבִּי עֲקִיבָא הָיָה חוֹכֵךְ בַּזֶּה לְהַחְמִיר\hdot
כְּנִדְרֵי רְשָׁעִים נָדַר בַּנָּזִיר וּבְקָרְבָּן וּבִשְׁבוּעָה\hdot
כְּנִדְרֵי כְשֵׁרִים\hdot
לֹא אָמַר כְּלוּם\hdot
כְּנִדְבוֹתָם נָדַר בַּנָּזִיר וּבַקָּרְבָּן׃}

נָזִיק\hdot
נָזִיחַ\hdot
פָּזִיחַ\hdot

שְׁקוּקָה\hdot

הֲרֵי אֵלּוּ כִּנּוּיִן לַשְּׁבוּעָה׃}

\halacha{הָאוֹמֵר לֹא חוֹלִין לֹא אוֹכַל לָךְ\hdot
לֹא כָשֵׁר וְלֹא דָכֵי\hdot
טָהוֹר\hdot
וְטָמֵא\hdot
נוֹתָר\hdot
וּפִגּוּל\hdot
אָסוּר\hdot
כְּאִימְרָא\hdot
כְּדִירִים\hdot
כְּעֵצִים\hdot
כְּאִשִּׁים\hdot
כְּמִזְבֵּחַ\hdot
כְּהֵיכָל\hdot
כִּירוּשָׁלַיִם\hdot
נָדַר בְּאַחַד מִכָּל מְשַׁמְּשֵׁי הַמִּזְבֵּחַ\hdot
אַף עַל פִּי שֶׁלֹּא הִזְכִּיר קָרְבָּן\hdot
הֲרֵי זֶה נָדַר קָרְבָּן\hdot
רִבִּי יְהוּדָה אוֹמֵר הָאוֹמֵר יְרוּשָׁלַיִם לֹא אָמַר כְּלוּם׃}

\halacha{הָאוֹמֵר קָרְבָּן\hdot

תּוֹדָה\hdot

הַקָּרְבָּן\hdot

קָרְבָּן שֶאוֹכַל לָךְ\hdot

לְקָרְבָּן לֹא אוֹכַל לָךְ\hdot

הָאוֹמֵר לַחֲבֵירוֹ קוּנָם פִּי מְדַבֵּר עִמָּךְ\hdot

\perek{וְאֵלּוּ}{מֻתָּרִין חוֹלִין שֶׁאוֹכַל לָךְ\hdot
כִּבְשַׂר חֲזִיר\hdot
כַּעֲבוֹדַת כּוֹכָבִים\hdot
כְּעוֹרוֹת לְבוּבִין\hdot
כִּנְבֵלוֹת\hdot
(כִּטְרֵפוֹת)\hdot
כִּשְׁקָצִים\hdot
(כִּרְמָשִׁים)\hdot
כְּחַלַּת אַהֲרֹן וְכִתְרוּמָתוֹ מֻתָּר\hdot
הָאוֹמֵר לְאִשְׁתּוּ הֲרֵי אַתְּ עָלַי כְּאִמָּא\hdot
פּוֹתְחִין לוֹ פֶּתַח מִמָּקוֹם אַחֵר\hdot
שֶׁלֹּא יָקֵל רֹאשׁוֹ לְכַךְ\hdot
קוּנָם שֶׁאֵינִי יָשֵׁן\hdot
שֶׁאֵינִי מְדַבֵּר\hdot
שֶׁאֵינִי מְהַלֵּךְ\hdot
הָאוֹמֵר לְאִשָּׁה קוּנָם שֶׁאֵינִי מְשַׁמְּשָׁךְ\hdot
הֲרֵי זֶה בְּלֹא יָחֵל דְּבָרוֹ\hdot
שְׁבוּעָה שֶׁאֵינִי יָשֵׁן\hdot
שֶׁאֵינִי מְדַבֵּר\hdot
שֶׁאֵינִי מְהַלֵּךְ אָסוּר׃}

\halacha{קָרְבָּן לֹא אוֹכַל לָךְ\hdot
הַקָּרְבָּן שֶׁאוֹכַל לָךְ\hdot
לֹא קָרְבָּן לֹא אוֹכַל לָךְ\hdot
מֻתָּר\hdot
שְׁבוּעָה לֹא אוֹכַל לָךְ\hdot
שְׁבוּעָה שֶׁאוֹכַל לָךְ\hdot
לֹא שְׁבוּעָה לֹא אוֹכַל לָךְ\hdot
אָסוּר\hdot
זֶה חֹמֶר בַּשְּׁבוּעוֹת מִבַּנְּדָרִים\hdot
וְחֹמֶר בַּנְּדָרִים מִבַּשְּׁבוּעוֹת\hdot
כֵּיצַד\hdot
אָמַר קוּנָם סֻכָּה שֶׁאֵינִי עוֹשֶׂה\hdot
לוּלָב שֶׁאֵינִי נוֹטֵל\hdot
תְּפִילִין שֶׁאֵינִי מַנִּיחַ\hdot
בַּנְּדָרִים אָסוּר\hdot
בַּשְּׁבוּעוֹת מֻתָּר\hdot
שֶׁאֵין נִשְׁבָּעִין לַעֲבוֹר עַל הַמִּצְוֺת׃}

\halacha{יֵשׁ נֶדֶר בְּתוֹךְ נֶדֶר\hdot
וְאֵין שְׁבוּעָה בְּתוֹךְ שְׁבוּעָה\hdot

אָמַר הֲרֵינִי נָזִיר אִם אוֹכַל\hdot
הֲרֵינִי נָזִיר אִם אוֹכַל וְאָכַל\hdot
חַיָּיב עַל כָּל אַחַת וְאֶחָת\hdot
שְבוּעָה שֶׁלֹּא אוֹכַל\hdot

\halacha{סְתָם נְדָרִים לְהַחְמִיר וּפֵירוּשָׁם לְהָקֵל\hdot
כֵּיצַד\hdot
אָמַר הֲרֵי עָלַי כִּבְשַׂר מָלִיחַ\hdot
כְּיַיִן נֶסֶךְ\hdot
אִם שֶׁל שָׁמַיִם נָדַר\hdot
אָסוּר\hdot
וְאִם שֶׁל עֲבוֹדַת אֱלִילִים מֻתָּר\hdot
וְאִם סְתָם אָסוּר\hdot
הֲרֵי עָלַי כְּחֵרֶם אִם כְּחֵרֶם שֶׁל שָׁמַיִם אָסוּר\hdot
וְאִם כְּחֵרֶם שֶׁל כֹּהֲנִים מֻתָּר\hdot
וְאִם סְתָם אָסוּר\hdot
הֲרֵי עָלַי כְּמַעֲשֵׂר אִם כְּמַעְשַׂר בְּהֵמָה נָדַר אָסוּר\hdot
וְאִם שֶׁל דָּגָן מֻתָּר\hdot
וְאִם סְתָם אָסוּר\hdot
הֲרֵי עָלַי כִּתְרוּמָה\hdot
אְם כִּתְרוּמַת הַלִּשְׁכָּה נָדַר אָסוּר\hdot
וְאִם שֶׁל גּוֹרֶן מֻתָּר\hdot
וְאִם סְתָם אָסוּר\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר סְתַָם תְּרוּמָה בִּיהוּדָה אֲסוּרָה\hdot
וּבַגָּלִיל מוּתֶרֶת\hdot
שֶׁאֵין אַנְשֵׁי גָלִיל מַכִּירִין אֶת תְּרוּמַת הַלִּשְׁכָּה\hdot
סְתָם חֲרָמִים בִּיהוּדָה מֻתָּרִין וּבַגָּלִיל אֲסוּרִין\hdot
שֶׁאֵין אַנְשֵׁי גָלִיל מַכִּירִין אֶת חֶרְמֵי הַכֹּהֲנִים׃}

\halacha{נָדַר בְּחֵרֶם\hdot
וְאָמַר לֹא נָדַרְתִּי אֶלָּא בְּחֶרְמוֹ שֶׁל יָם\hdot
בְּקָרְבָּן וְאָמַר לֹא נָדַרְתִּי אֶלָּא בְּקָרְבְּנוֹת שֶׁל מְלָכִים\hdot
הֲרֵי עַצְמִי קָרְבָּן\hdot
וְאָמַר לֹא נָדַרְתִּי אֶלָּא בְּעֶצֶם שֶׁהִנַּחְתִּי לִי לִהְיוֹת נוֹדֵר בּוֹ\hdot
קוּנָם אִשְׁתִּי נֶהֱנֵית לִי\hdot
וְאָמַר לֹא נָדַרְתִּי אֶלָּא מֵאִשְׁתִּי הָרִאשׁוֹנָה שֶׁגֵּרַשְׁתִּי\hdot
עַל כֻּלָּן אֵין נִשְׁאָלִים לָהֶם\hdot
וְאִם נִשְׁאֲלוּ עוֹנְשִׁין אוֹתָן\hdot
וּמַחְמִירִין עֲלֵיהֶם\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים פּוֹתְחִין לָהֶן פֶּתַח מִמָּקוֹם אַחֵר וּמְלַמְּדִין אוֹתָם כְּדֵי שֶׁלֹּא יְנַהֲגוּ קַלּוּת רֹאשׁ בַּנְּדָרִים׃}
