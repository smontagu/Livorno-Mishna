\tractatehead{מסכת עדויות}{eduyot}

\perek{שַׁמָּאי}{אוֹמֵר\hdot
כָּל הַנָּשִׁים דַּיָּן שַׁעְתָּן\hdot
וְהִלֵּל אוֹמֵר מִפְּקִידָה לִפְקִידָה\hdot
אֲפִילוּ לְיָמִים הַרְבֶּה\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא מֵעֵת לְעֵת מְמָעֶטֶת עַל יָד מִפְּקִידָה לִפְקִידָה\hdot
וּמִפְּקִידָה לִפְקִידָה מְמָעֶטֶת עַל יָד מֵעֵת לְעֵת\hdot
כָּל אִשָּׁה שֶׁיֵּשׁ לָהּ וֶסֶת דַּיָּהּ שַׁעְתָּהּ\hdot
הַמְּשַׁמֶּשֶׁת בְּעֵדִים הֲרֵי זוֹ כִפְקִידָה\hdot
וּמְמָעֶטֶת עַל יָד מֵעֵת לְעֵת\hdot
וְעַל יָד מִפְּקִידָה לִפְקִידָה׃}

\halacha{שַׁמָּאי אוֹמֵר מִקַּב לַחַלָּה\hdot
וְהִלֵּל אוֹמֵר מִקַּבַּיִם\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא קַב וּמֶחְצָה חַיָּבִים בַּחַלָּה\hdot
וּמִשֶּׁהִגְדִּילוּ הַמִּדּוֹת אָֽמְרוּ\hdot
חֲמֵשֶׁת רְבִיעִים חַיָּבִין\hdot
רִבִּי יוֹסֵי אוֹמֵר חֲמִשָּׁה פְּטוּרִין\hdot
חֲמִשָּׁה וְעוֹד חַיָּבִין׃}

\halacha{הִלֵּל אוֹמֵר מְלֹא הִין מַיִם שְׁאוּבִין פּוֹסְלִין אֶת הַמִּקְוֶה\hdot
אֶלָּא שֶׁאָדָם חַיָּב לוֹמַר בִּלְשׁוֹן רַבּוֹ\hdot
וְשַׁמָּאי אוֹמֵר תִּשְׁעָה קַבִּין\hdot
וַחֲכָמִים אוֹמְרִים לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה\hdot
אֶלָּא עַד שֶׁבָּאוּ שְׁנֵי גַרְדִּיִּים מִשַּׁעַר הָאַשְׁפּוֹת שֶׁבִּירוּשָׁלַיִם\hdot
וְהֵעִידוּ מִשּׁוּם שְׁמַעְיָה וְאַבְטַלְיוֹן שֶׁשְּׁלֹשָׁה לוֹגִים מַיִם שְׁאוּבִין\hdot
פּוֹסְלִין אֶת הַמִּקְוֶה\hdot
וְקִיְּמוּ חֲכָמִים אֶת דִּבְרֵיהֶם׃}

\halacha{וְלָמָּה מַזְכִּירִין אֶת דִּבְרֵי שַׁמָּאי וְהִלֵּל לְבַטָּלָה\hdot
לְלַמֵּד לְדוֹרוֹת הַבָּאִים\hdot
שֶׁלֹּא יְהֵא אָדָם עוֹמֵד עַל דְּבָרָיו\hdot
שֶׁהֲרֵי אָבוֹת הָעוֹלָם\hdot
לֹא עָֽמְדוּ עַל דִּבְרֵיהֶם׃}

\halacha{וְלָמָּה מַזְכִּירִין דִּבְרֵי הַיָּחִיד בֵּין הַמְּרֻבִּין הוֹאִיל וְאֵין הֲלָכָה אֶלָּא כְדִבְרֵי הַמְּרֻבִּין\hdot
שֶׁאִם יִרְאֶה בֵית דִּין אֶת דִּבְרֵי הַיָּחִיד וְיִסְמוֹךְ עָלָיו\hdot
שֶׁאֵין בֵּית דִּין יָכוֹל לְבַטֵּל דִּבְרֵי בֵית דִּין חֲבֵירוֹ\hdot
עַד שֶׁיִּהְיֶה גָּדוֹל מִמֶּנּוּ בְּחָכְמָה וּבְמִנְיָן\hdot
הָיָה גָדוֹל מִמֶּנּוּ בְּחָכְמָה אֲבָל לֹא בְמִנְיָן\hdot
בְּמִנְיָן אֲבָל לֹא בְחָכְמָה\hdot
אֵינוֹ יָכוֹל לְבַטֵּל דְּבָרָיו\hdot
עַד שֶׁיִּהְיֶה גָדוֹל מִמֶּנּוּ בְּחָכְמָה וּבְמִנְיָן׃}

\halacha{אָמַר רִבִּי יְהוּדָה אִם כֵּן לָמָּה מַזְכִּירִין דִּבְרֵי הַיָּחִיד\hdot
בֵּין הַמְּרֻבִּין לְבַטָּלָה\hdot
שֶׁאִם יֹאמַר הָאָדָם כַּך אֲנִי מְקֻבַּל\hdot
יֹאמַר לוֹ כְּדִבְרֵי אִישׁ פְּלוֹנִי שָׁמַעְתָּ׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים רֹבַע עֲצָמוֹת מִן הָעֲצָמִים\hdot
בֵּין מִשְּׁנַיִם בֵּין מִשְּׁלֹשָׁה\hdot
וּבֵית הִלֵּל אוֹמְרִים רֹבַע עֲצָמוֹת מִן הַגְּוִיָּה\hdot
מֵרוֹב הַבִּנְיָן אוֹ מֵרוֹב הַמִּנְיָן\hdot
שַׁמָּאי אוֹמֵר אֲפִילוּ מֵעֶצֶם אֶחָד׃}

\halacha{כַּרְשִׁינֵי תְרוּמָה\hdot
בֵּית שַׁמָּאי אוֹמְרִים שׁוֹרִין וְשָׁפִין בְּטַהֳרָה\hdot
וּמַאֲכִילִין בְּטֻמְאָה\hdot
וּבֵית הִלֵּל אוֹמְרִים שׁוֹרִין בְּטָהֳרָה\hdot
וְשָׁפִין וּמַאֲכִילִין בְּטֻמְאָה\hdot
(שַׁמָּאי אוֹמֵר יֵאָֽכְלוּ צָרִיד\hdot)
רִבִּי עֲקִיבָא אוֹמֵר כָּל מַעֲשֵׂיהֶם בְּטֻמְאָה׃}

\halacha{הַפּוֹרֵט סֶלַע מִמָּעוֹת מַעֲשֵׂר שֵׁנִי בֵּית שַׁמָּאי אוֹמְרִים בְּכָל הַסֶּלַע מָעוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים בְּשֶׁקֶל כֶּסֶף וּבְשֶׁקֶל מָעוֹת\hdot
רִבִּי מֵאִיר אוֹמֵר אֵין מְחַלְּלִין כֶּסֶף וּפֵירוֹת עַל הַכֶּסֶף\hdot
וַחֲכָמִים מַתִּירִין׃}

\halacha{הַפּוֹרֵט סֶלַע שֶׁל מַעֲשֵׂר שֵׁנִי בִּירוּשָׁלַיִם\hdot
בֵּית שַׁמָּאי אוֹמְרִים בְּכָל הַסֶּלַע מָעוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים בְּשֶׁקֶל כֶּסֶף\hdot
וּבְשֶׁקֶל מָעוֹת\hdot
הַדָּנִים לִפְנֵי חֲכָמִים אוֹמְרִים בִּשְׁלֹשָׁה דִינָרִים כֶּסֶף\hdot
וּבְדִינַר מָעוֹת\hdot
רִבִּי עֲקִיבָא אוֹמֵר בִּשְׁלֹשָׁה דִינָרִים כֶּסֶף\hdot
וּבִרְבִיעִית (כֶּסֶף\hdot
בִּרְבִיעִת)מָעוֹת\hdot
רִבִּי טַרְפוֹן אוֹמֵר אַרְבַּע אַסְפְּרֵי כֶסֶף\hdot
שַׁמָּאי אוֹמֵר יַנִּיחֶנָּה בֶּחָנוּת וְיֹאכַל כַּנֶגְדָּהּ׃}

\halacha{כִּסֵּא שֶׁל כַּלָּה שֶׁנִּטְּלוּ חִפּוּיָו\hdot
בֵּית שַׁמָּאי מְטַּמְּאִין\hdot
וּבֵית הִלֵּל מְטַהֲרִין\hdot
שַׁמָּאי אוֹמֵר אַף מַלְבֵּן שֶׁל כִּסֵא טָמֵא\hdot
כִּסֵא שֶׁקְּבָעוֹ בַעֲרֵיבָה\hdot
בֵּית שַׁמָּאי מְטַמְּאִין וּבֵית הִלֵּל מְטַהֲרִין\hdot
שַׁמָּאי אוֹמֵר אַף הֶעָשׂוּי בָּהּ׃}

\halacha{אֵלּוּ דְבָרִים שֶׁחָֽזְרוּ בֵית הִלֵּל לְהוֹרוֹת כְּדִבְרֵי בֵית שַׁמָּאי\hdot
הָאִשָּׁה שֶׁבָּאָה מִמְּדִינַת הַיָּם\hdot
וְאָֽמְרָה מֵת בַּעְלִי תִּנָּשֵׂא\hdot
מֵת בַּעְלִי תִּתְיַבֵּם\hdot
דִּבְרֵי בֵית שַׁמָּאי\hdot
וּבֵית הִלֵּל אוֹמְרִים לֹא שָׁמַעְנוּ אֶלָּא בְּבֹאָהּ מִן הַקָּצִיר בִּלְבָד\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי אַחַת הַבָּאָה מִן הַקָּצִיר\hdot
וְאַחַת הַבָּאָה מִן הַזֵּיתִים\hdot
וְאַחַת הַבָּאָה מִמְּדִינַת הַיָּם\hdot
לֹא דִבְּרוּ בַקָּצִיר אֶלָּא בַּהוֹוֶה\hdot
חָֽזְרוּ בֵית הִלֵּל לְהוֹרוֹת כְּדִבְרֵי בֵית שַׁמָּאי\hdot
בֵּית שַׁמָּאי אוֹמְרִים תִּנָּשֵׂא וְתִטּוֹל כְּתוּבָתָהּ\hdot
וּבֵית הִלֵּל אוֹמְרִים תִּנָּשֵׂא וְלֹא תִטּוֹל כְּתוּבָתָהּ\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי הִתַּרְתֶּם אֶת הָעֶרְוָה הַחֲמוּרָה\hdot
לֹא תַתִּירוּ (אֶת) הַמָּמוֹן הַקַּל\hdot
אָֽמְרוּ לָהֶם בֵּית הִלֵּל מָצִינוּ שֶׁאֵין הָאַחִים נִכְנְסִין לְנַחֲלָה (עַל פִּיהָ\hdot)
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי\hdot
וַהֲלֹא מִסֵּפֶר כְּתוּבָתָהּ נִלְמוֹד\hdot
שֶׁהוּא כוֹתֶב לָהּ\hdot
שֶׁאִם תִּנָּֽשְׂאִי לְאַחֵר תִּטְּלִי מַה שֶּׁכָּתוּב לִיךְ\hdot
חָֽזְרוּ בֵית הִלֵּל (לְהוֹרוֹת) כְּדִבְרֵי בֵית שַׁמָּאי׃}

\halacha{מִי שֶׁחֶצְיוֹ עֶבֶד וְחֶצְיוֹ בֶן חוֹרִין עוֹבֵד אֶת רַבּוֹ יוֹם אֶחָד\hdot
וְאֶת עַצְמוֹ יוֹם אֶחָד\hdot
דִּבְרֵי בֵית הִלֵּל\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי\hdot
תְּקַנְתֶּם אֶת רַבּוֹ\hdot
וְאֶת עַצְמוֹ לֹא תְקַנְתֶּם\hdot
לִשָּׂא שִׁפְחָה אֵינוֹ יָכוֹל\hdot
בַּת חוֹרִין אֵינוֹ יָכוֹל\hdot
לִבָּטֵל\hdot
וַהֲלֹא לֹא נִבְרָא הָעוֹלָם אֶלָּא לְפִרְיָה וְרִבְיָה\hdot
שֶׁנֶּאֱמַר לֹא תֹהוּ בְרָאָהּ לָשֶׁבֶת יְצָרָהּ\hdot
אֶלָּא מִפְּנֵי תִקּוּן הָעוֹלָם\hdot
כּוֹפִין אֶת רַבּוֹ\hdot
וְעוֹשֶׂה אוֹתוֹ בֶּן חוֹרִין\hdot
וְכוֹתֵב שְׁטָר עַל חֲצִי דָמָיו\hdot
וְחָֽזְרוּ בֵית הִלֵּל (לְהוֹרוֹת) כְּדִבְרֵי בֵית שַׁמָּאי׃}

\halacha{כְּלִי חֶרֶס מַצִּיל עַל הַכֹּל כְּדִבְרֵי בֵית הִלֵּל\hdot
וּבֵית שַׁמָּאי אוֹמְרִים אֵינוֹ מַצִּיל אֶלָּא עַל הָאוֹכָלִין וְעַל הַמַּשְׁקִין\hdot
וְעַל כְּלֵי חֶרֶס\hdot
אָֽמְרוּ לָהֶם בֵּית הִלֵּל מִפְּנֵי מָה\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי מִפְּנֵי שֶׁהוּא טָמֵא עַל גַּב עַם הָאָרֶץ\hdot
וְאֵין כֶּלִי טָמֵא חוֹצֵץ\hdot
אָֽמְרוּ לָהֶם בֵּית הִלֵּל וַהֲלֹא טִהַרְתֶּם אוֹכָלִים וּמַשְׁקִין שֶׁבְּתוֹכוֹ\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי כְּשֶׁטִהַרְנוּ אוֹכָלִים וּמַשְׁקִין שֶׁבְּתוֹכוֹ\hdot
לְעַצְמוֹ טִהַרְנוּ\hdot
אֲבָל כְּשֶׁטִּהַרְתָּ אֶת הַכֶּלִי\hdot
טִהַרְתָּ לָךְ וְלוֹ\hdot
חָֽזְרוּ בֵית הִלֵּל לְהוֹרוֹת כְּדִבְרֵי בֵית שַׁמָּאי׃}
