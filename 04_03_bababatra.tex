\tractatehead{מסכת בבא בתרא}{baba batra}

\perek{הַשּׁוּתָפִין}{שֶׁרָצוּ לַעֲשׂוֹת מְחִיצָה בֶּחָצֵר בּוֹנִין אֶת הַכֹּתֶל בָּאֶמְצַע\hdot
מָקוֹם שֶׁנָּהֲגוּ לִבְנוֹת גָּוִיל\hdot
גָּזִית\hdot
כְּפִיסִין\hdot
לְבֵנִים בּוֹנִים\hdot
הַכֹּל כְּמִנְהַג הַמּדִינָה\hdot
בַּגָּוִיל\hdot
זֶה נוֹתֵן שְׁלֹשָׁה טְפָחִים וְזֶה נוֹתֵן שְׁלֹשָׁה טְפָחִים\hdot
בַּגָּזִית זֶה נוֹתֵן טְפָחַיִם וּמֶחְצָה\hdot
וְזֶה נוֹתֵן טְפָחַיִם וּמֶחְצָה\hdot
בַּכְּפִיסִין\hdot
זֶה נוֹתֵן טְפָחַים\hdot
וְזֶה נוֹתֵן טְפָחַים\hdot
בַּלְּבֵנִים\hdot
זֶה נוֹתֵן טֶפַח וּמֶחְצָה\hdot
וְזֶה נוֹתֵן טֶפַח וּמֶחְצָה\hdot
לְפִיכַךְ אִם נָפַל הַכֹּתֶל\hdot
הַמָּקוֹם וְהָאֲבָנִים שֶׁל שְׁנֵיהֶם׃}

\halacha{וְכֵן בַּגִּנָּה\hdot
מָקוֹם שֶׁנָּהֲגוּ לִגְדּוֹר\hdot
(מְחַיְּבִין אוֹתוֹ) אֲבָל בַּבִּקְעָה\hdot
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לִגְדּוֹר\hdot
אֵין מְחַיְּבִין אוֹתוֹ\hdot
אֶלָּא אִם רָצָה כּוֹנֵס לְתוֹךְ שֶׁלּוֹ וּבוֹנֶה\hdot
וְעוֹשֶׂה חָזִית מִבַּחוּץ\hdot
לְפִיכַךְ אִם נָפַל הַכֹּתֶל הַמָּקוֹם וְהָאֲבָנִים שֶׁלּוֹ\hdot
אִם עָשׂוּ מִדַּעַת שְׁנֵיהֶם\hdot
בּוֹנִין אֶת הַכֹּתֶל  בָּאֶמְצַע\hdot
וְעוֹשִׂין חָזִית מִכָּאן וּמִכָּאן\hdot
לְפִיכַךְ אִם נָפַל הַכֹּתֶל\hdot
הַמָּקוֹם וְהָאֲבָנִים שֶׁל שְׁנֵיהֶם׃}

\halacha{הַמַּקִּיף אֶת חֲבֵירוֹ מִשָּׁלֹשׁ רוּחוֹתָיו\hdot
וְגָדַר אֶת הָרִאשׁוֹנָה\hdot
וְאֶת הַשְּׁנִיָּה\hdot
וְאֶת הַשְּׁלִישִׁית\hdot
אֵין מְחַיְּבִין אוֹתוֹ\hdot
רִבִּי יוֹסֵי אוֹמֵר אִם עָמַד וְגָדַר אֶת הָֽרְבִיעִית\hdot
מְגַלְגְּלִים עָלָיו אֶת הַכֹּל׃}

\halacha{כֹּתֶל חָצֵר שֶׁנָּפַל\hdot
מְחַיְּבִין אוֹתוֹ לִבְנוֹתוֹ\hdot
עַד אַרְבַּע אַמּוֹת\hdot
בְּחֶזְקַת שֶׁנָּתַן עַד שֶׁיָּבִיא רְאָיָה שֶׁלֹּא נָתַן\hdot
מֵאַרְבַּע אַמּוֹת וּלמַעְלָה\hdot
אֵין מְחַיְּבִין אוֹתוֹ\hdot
סָמַךְ לוֹ כֹּתֶל אַחֵר\hdot
אַף עַל פִּי שֶׁלֹּא נָתַן עָלָיו אֶת הַתְּקָרָה מְגַלְגְּלִין עָלָיו אֶת הַכֹּל\hdot
בְּחֶזְקַת שֶׁלֹּא נָתַן\hdot
עַד שֶׁיָּבִיא רְאָיָה שֶׁנָּתַן׃}

\halacha{כּוֹפִין אוֹתוֹ לִבְנוֹת בֵּית שַׁעַר\hdot
וְדֶלֶת לֶחָצֵר\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר לֹא כָל הַחֲצֵירוֹת רְאוּיוֹת לְבֵית שַׁעַר\hdot
כּוֹפִין אוֹתוֹ לִבְנוֹת לָעִיר\hdot
חוֹמָה דְּלָתַיִם וּבְרִיחַ\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר לֹא כָל הָעֲיָרוֹת רְאוּיוֹת לַחוֹמָה\hdot
כַּמָּה יְהֵא בָעִיר\hdot
וְיהֵא כְאַנְשֵׁי הָעִיר\hdot
שְׁנֵים עָשָׂר חֹדֶשׁ\hdot
קָנָה בָהּ בֵּית דִּירָה\hdot
הֲרֵי הוּא כְּאַנְשֵׁי הָעִיר מִיָּד׃}

\halacha{אֵין חוֹלְקִין אֶת הֶחָצֵר\hdot
עַד שְׁיְּהֵא בָהּ אַרְבַּע אַמּוֹת לַזֶּה וְאַרְבַּע אַמּוֹת לַזֶּה\hdot
וְלֹא אֶת הַשָּׂדֶה עַד שֶׁיְּהֵא בָהּ תִּשְׁעָה קַבִּין לַזֶּה\hdot
וְתִשְׁעָה קַבִּין לַזֶּה\hdot
רִבִּי יְהוּדָה אוֹמֵר עַד שֶׁיְּהֵא בָהּ תִּשְׁעָה חֲצָאֵי קַבִּין לַזֶּה\hdot
וְתִשְׁעָה חֲצָאֵי קַבִּין לַזֶּה\hdot
וְלֹא אֶת הַגִּנָּה עַד שֶׁיְּהֵא בָהּ חֲצִי קַב לַזֶּה\hdot
וַחֲצִי קַב לַזֶּה\hdot
רִבִּי עֲקִיבָא אוֹמֵר בֶּית רוֹבַע\hdot
וְלֹא אֶת הַטְּרַקְלִין\hdot
וְלֹא אֶת הַמֹּרֶן\hdot
וְלֹא אֶת הַשּׁוֹבָךְ\hdot
וְלֹא אֶת הַטָּלֵית\hdot
וְלֹא אֶת הַמֶּרְחָץ\hdot
וְלֹא אֶת בֵּית הַבָּד\hdot
(וְלֹא אֶת בֵּית הַשְּׁלָחִין) עַד שֶׁיְּהֵא בָהֶן כְּדַי לַזֶּה וּכְדַי לַזֶּה\hdot
זֶה הַכְּלָל כָּל שֶׁיֵּחָלֵק וּשְׁמוֹ עָלָיו\hdot
חוֹלְקִין\hdot
וְאֵם לָאוּ אֵין חוֹלְקִין\hdot
אֵימָתַי בִּזְמַן שֶׁאֵין שְׁנֵיהֶם רוֹצִים\hdot
אֲבָל בִּזְמַן שֶׁשְׁנֵיהֶם רוֹצִים\hdot
אֲפִילוּ בְּפָחוּת מִכָּאן יַחֲלוֹקוּ\hdot
וְכִתְבֵי הַקֹּדֶשׁ אַף עַל פִּי שֶׁשְּׁנֵיהֶם רוֹצִים לֹא יַחֲלוֹקוּ׃}

\perek{לֹא}{יַחְפּוֹר אָדָם בּוֹר סָמוּךְ לְבוֹרוֹ שֶׁל חֲבֵירוֹ\hdot
וְלֹא שִׂיחַ\hdot
וְלֹא מְעָרָה\hdot
וְלֹא אַמַּת הַמַּיִם\hdot
וְלֹא נִבְרֶכֶת כוֹבְסִין\hdot
אֶלָּא אִם כֵּן הִרְחִיק מִכֹּתֶל חֲבֵירוֹ שְׁלֹשָׁה טְפָחִים וְסַד בַּסִּיד\hdot
מַרְחִיקִין אֶת הַגֶּפֶת
וְאֶת הַזֶּבֶל\hdot
וְאֶת הַמֶּלַח\hdot
וְאֶת הַסִּיד וְאֶת הַסְּלָעִים\hdot
מִכָּתְלוֹ שֶׁל חֲבֵירוֹ שְׁלֹשָׁה טְפָחִים\hdot
וְסַד בְּסִיד\hdot
מַרְחִיקִין אֶת הַזְּרָעִים\hdot
וְאֶת הַמַּחֲרִישָׁה\hdot
וְאֶת מֵי רַגְלַיִם\hdot
מִן הַכֹּתֶל שְׁלֹשָׁה טְפָחִים\hdot
וּמַרֲחִיקִין אֶת הָרֵיחַיִם שְׁלֹשָׁה מִן הַשֶׁכֶב\hdot
שֶׁהֵן אַרְבַּע מִן הָרֶכֶב\hdot
וְאֶת הַתַּנּוּר שְׁלֹשָׁה מִן הַכִּלְיָא שֶׁהֵן אַרְבַּע מִן הַשָּׂפָה׃}

\halacha{לֹא יַעֲמִיד אָדָם תַּנּוּר בְּתוֹךְ הַבַּיִת אֶלָּא אִם כֵּן יֵשׁ עַל גַּבָּיו גּוֹבָה אַרְבַּע אַמּוֹת\hdot
הָיָה מַעֲמִידוֹ בַּעֲלִיָּה\hdot
צָרִיךְ שֶׁיְּהֵא תַחְתָּיו מַעֲזִיבָה  שְׁלֹשָׁה טְפָחִים\hdot
וּבַכִּירָה טֶפַח\hdot
וְאִם הִזִּיק מְשַׁלֵּם מַה שֶּׁהִזִּיק\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר\hdot
לֹא אָֽמְרוּ כָּל הַשִּיעוּרִין הָאֵלּוּ\hdot
אֶלָּא שֶׁאִם הִזִּיק פָּטוּר מִלְּשַׁלֵּם׃}
