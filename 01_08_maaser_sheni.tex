\tractatehead{מסכת מעשר שני}{maaserSheni}

\perek{מַעֲשֵׂר}{שֵׁנִי אֵין מוֹכְרִין אוֹתוֹ\hdot
וְאֵין מְמַשְׁכְּנִין אוֹתוֹ\hdot
וְאֵין מַחֲלִיפִין אוֹתוֹ וְלֹא שׁוֹקְלִין כְּנֶגְדּוֹ וְלֹא יֹאמַר אָדָם לַחֲבֵירוֹ בִּירוּשָׁלַיִם הֵילָךְ יַיִן וְתֵן לִי שֶׁמֶן\hdot
וְכֵן שְׁאַר כָּל הַפֵּירוֹת\hdot
אֲבָל נוֹתְנִין זֶה לָזֶה מַתְּנַת חִנָּם׃}

\halacha{מַעַשַׂר בְּהֵמָה אֵין מוֹכְרִין אוֹתוֹ תָּמִים חַי\hdot
וְלֹא בַעַל מוּם חַי וְשָׁחוּט\hdot
וְאֵין מְקַדְּשִׁין בּוֹ אֶת הָאִשָּׁה\hdot
הַבְּכוֹר מוֹכְרִים אוֹתוֹ תָּמִים חַי\hdot
וּבַעַל מוּם חַי וְשָׁחוּט\hdot
וּמְקַדְּשִׁין בּוֹ אֶת הָאִשָּׁה\hdot
אֵין מְחַלְּלִין מַעַשֵׂר שֵׁנִי עַל אָסִימוֹן וְלֹא עַל הַמַּטְבֵּעַ שֶׁאֵינוֹ יוֹצֵא\hdot
וְלֹא עַל הַמָּעוֹת שֶׁאֵינָן בִּרְשׁוּתוֹ׃}

\halacha[257]{הַלּוֹקֵחַ בְּהֵמָה לְזִבְחֵי שְׁלָמִים אוֹ חַיָּה לִבְשַׂר תַּאֲוָה\hdot
יָצָא הָעוֹר לַחוֹלִין אַף עַל פִּי שֶׁהָעוֹר מְרֻבֶּה עַל הַבָּשָׂר\hdot
כַּדֵּי יַיִן סְתוּמוֹת מָקוֹם שֶׁדַּרְכָּן לִמָּכֵר סְתוּמוֹת יָצָא קַנְקָן לַחוֹלִין\hdot
הָאֱגוֹזִים וְהַשְּׁקֵדִים יָֽצְאוּ קְלִיפֵהֶם לַחוֹלִין\hdot
הַתֶּמֶד עַד שׁלֹּא הֶחֱמִיץ אֵינוֹ נִלְקַח בְּכֶסֶף מַעֲשֵׂר\hdot
וּמִשֶּׁהֶחֱמִיץ נִלְקַח בְּכֶסֶף מַעֲשֵׂר׃}

\halacha{הַלּוֹקֵחַ חַיָּה לְזִבְחֵי שְׁלָמִים\hdot
בְּהֵמָה לִבְשַׂר תַּאֲוָה\hdot
לֹא יָצָא הָעוֹר לַחוֹלִין\hdot
כַּדֵּי יַיִן פְּתוּחוֹת אוֹ סְתוּמוֹת\hdot
מָקוֹם שֶׁדַּרְכָּן לִמָּכֵר פְּתוּחוֹת\hdot
לֹא יָצָא קַנְקָן לַחוֹלִין\hdot
סַלֵּי זֵיתִים וְסַלֵּי עֲנָבִים עִם הַכֶּלִי\hdot
לֹא יָֽצְאוּ דְּמֵי הַכֶּלִי לַחוֹלִין׃}

\halacha[334]{הַלּוֹקֵחַ מַיִם וּמֶלַח וּפֵירוֹת הַמְּחֻבָּרִים לַקַּרְקַע אוֹ פֵירוֹת שֶׁאֵינָן יְכוֹלִין לְהַגִּיעַ לִירוּשָׁלַיִם\hdot
לֹא קָנָה מַעֲשֵׂר\hdot
הַלּוֹקֵחַ פֵּירוֹת שׁוֹגֵג יַחְזְרוּ דָמִים לִמְקוֹמָן\hdot
מֵזִיד יַעֲלוּ וְיֵאָֽכְלוּ בַמָּקוֹם\hdot
וְאִם אֵין מִקְדָּשׁ יִרְקָבוּ׃}

\halacha{הַלּוֹקֵחַ בְּהֵמָה שׁוֹגֵג יַחְזְרוּ דָמֶיהָ לִמְקוֹמָן\hdot
מֵזִיד תַּעֲלֶה וְתֵאָכֵל בַּמָּקוֹם\hdot
וְאִם אֵין מִקְדָּשׁ תִּקָּבֵר עַל יְדֵי עוֹרָהּ׃}

\halacha{אֵין לוֹקְחִין עֲבָדִים שְׁפָחוֹת וְקַרְקָעוֹת\hdot
וּבְהֵמָה טְמֵאָה מִדְּמֵי מַעֲשֵׂר שֵׁנִי\hdot
וְאִם לָקַח יֹאכַל כְּנֶגְדָּן\hdot
אֵין מְבִיאִין קִנֵּי זָבִים\hdot
וְקִנֵּי זָבוֹת\hdot
וְקִנֵּי יוֹלְדוֹת\hdot
חַטָּאוֹת וַאֲשָׂמוֹת מִדְּמֵי מַעֲשֵׂר שֵׁנִי\hdot
וְאִם הֵבִיא יֹאכַל כְּנֶגְדָּם\hdot
זֶה הַכְּלָל כָּל שֶׁהוּא חוּץ לַאֲכִילָה וְלִשְׁתִּיָּה וּלְסִיכָה מִדְּמֵי מַעֲשֵׂר שֵׁנִי\hdot
יֹאכַל כְּנֶגְדּוֹ׃}

\perek[562]{מַעֲשֵׂר}{שֵׁנִי נִתָּן לַאֲכִילָה וְלִשְׁתִּיָּה וּלְסִיכָה\hdot
לֶאֱכוֹל דָּבָר שֶׁדַּרְכּוֹ לֶאֱכוֹל\hdot
(וְלִשְׁתּוֹת דָּבָר שֶׁדַּרְכּוֹ לִשְׁתּוֹת\hdot)
וְלָסוּךְ דָּבָר שֶׁדַּרְכּוֹ לָסוּךְ\hdot
לֹא יָסוּךְ יַיִן וְחֹמֶץ\hdot
אֲבָל סַךְ הוּא אֶת הַשֶּׁמֶן\hdot
אֵין מְפַטְּמִין שֶׁמֶן שֶׁל מַעֲשֵׂר שֵׁנִי\hdot
וְאֵין לוֹקְחִין בִּדְמֵי מַעֲשֵׂר שֵׁנִי שֶׁמֶן מְפֻטָּם\hdot
אֲבָל מְפַטֵּם הוּא אֶת הַיַּיִן\hdot
נָפַל לְתוֹכוֹ דְּבַשׁ וּתְבָלִין וְהִשְׁבִּיחוּ\hdot
הַשֶּׁבַח לְפִי חֶשְׁבּוֹן\hdot
דָּגִים שֶׁנִּתְבַּשְּׁלוּ עִם הַקַּפְלוֹטוֹת שֶׁל מַעֲשֵׂר שֵׁנִי וְהִשְׁבִּיחוּ\hdot
הַשֶּׁבַח לְפִי חֶשְׁבּוֹן\hdot
עִסָּה שֶׁל מַעֲשֵׁר שֵׁנִי שֶׁאֲפָאָהּ וְהִשְׁבִּיחָה הַשֶּׁבַח לַשֵּׁנִי\hdot
זֶה הַכְּלָל כָּל שֶׁשִּׁבְחוֹ נִכָּר\hdot
הַשֶּׁבַח לְפִי חֶשְׁבּוֹן וְכָל שֶׁאֵין שִׁבְחוֹ נִכָּר הַשֶּׁבַח לַשֵּׁנִי׃}

\halacha[812]{רִבִּי שִׁמְעוֹן אוֹמֵר אֵין סָכִין שֶׁמֶן שֶׁל מַעֲשֵׂר שֵׁנִי בִּירוּשָׁלַיִם וַחֲכָמִים מַתִּירִין\hdot
אָֽמְרוּ לוֹ לְרִבִּי שִׁמְעוֹן אִם הֵקֵל בַּתְּרוּמָה חֲמוּרָה\hdot
לֹא נָקֵל בְּמַעֲשֵׂר שֵׁנִי הַקַּל\hdot
אָמַר לָהֵם מַה לֹּא אִם הֵקֵל בַּתְּרוּמָה הַחֲמוּרָה מָקוֹם שֶׁהֵקֵל בַּכַּרְשִׁינִים וּבַתִּלְתָּן\hdot
נֵקֵל בְּמַעֲשֵׂר שֵׁנִי הַקַּל מָקוֹם שֶׁלֹּא הֵקֵל בַּכַּרְשִׁינִים וּבַתִּלְתָּן׃}

\halacha{תִּלְתָּן שֶׁל מַעֲשֵׂר שֵׁנִי תֵּאָכֵל צִמְחוֹנִים\hdot
וְשֶׁל תְּרוּמָה\hdot
בֵּית שַׁמָּאי אוֹמְרִים כָּל מַעֲשֶׂיהָ בְּטָהֳרָה חוּץ מֵחֲפִיפָתָהּ\hdot
וּבֵית הִלֵּל אוֹמְרִים כָּל מַעֲשֶׁיהָ בְּטוּמְאָה חוּץ מִשְּׁרִיָּתָהּ׃}

\halacha[1279]{כַּרְשִׁינֵי מַעֲשֵׂר שֵׁנִי יֵאָֽכְלוּ צִמְחוֹנִים\hdot
וְנִכְנָסִין לִירוּשָׁלַיִם וְיוֹצְאִין\hdot
נִטְמָאוּ\hdot
רִבִּי טַרְפוֹן אוֹמֵר יִתְחַלְּקוּ לְעִסּוֹת וַחֲכָמִים אוֹמִרִים יִפָּדוּ\hdot
וְשֶׁל תְּרוּמָה בֵּית שַׁמָּאי אוֹמְרִים שׁוֹרִין וְשָׁפִין בְּטָהֳרָה וּמַאֲכִילִין בְּטוּמְאָה\hdot
וּבֵית הִלֵּל אוֹמְרִים שׁוֹרִין בְּטָהֳרָה וְשָׁפִין וּמַאֲכִילִין בְּטוּמְאָה\hdot
שַׁמָּאי אוֹמֵר יֵאָֽכְלוּ צָרִיד\hdot
רִבִּי עֲקִיבָא אוֹמֵר כָּל מַעֲשֵׂיהֶן בְּטוּמְאָה׃}

\halacha{מָעוֹת חוֹלִין\hdot
וּמָעוֹת מַעֲשֵׂר שֵׁנִי שֶׁנִּתְפַּזְּרוּ\hdot
מַה שֶּׁלָּקַט לָקַט לְמַעֲשֵׂר שֵׂנִי עַד שֶׁיַּשְׁלִים\hdot
וְהַשְּׁאָר חוֹלִין\hdot
אִם בָּלַל וְחָפַן לְפִי חֶשְׁבּוֹן\hdot
זֶה הַכְּלָל הַמִּתְלַקְּטִים לְמַעֲשֵׂר שֵׁנִי\hdot
וְהַנִּבְלָלִין לְפִי חֶשְׁבּוֹן׃}

\halacha[506]{סֶלַע שֶׁל מַעֲשֵׂר שֵׁנִי וְשֶׁל חוֹלִין שֶׁנִּתְעָֽרְבוּ\hdot
מֵבִיא בְּסֶלַע מָעוֹת וְאוֹמֵר סֶלַע שֶׁל מַעֲשֵׂר שֵׁנִי בְּכָל מָקוֹם שֶׁהִיא\hdot
מְחֻלֶּלֶת עַל הַמָּעוֹת הָאֵלּוּ\hdot
וּבוֹרֵר אֶת הַיָּפָה שֶׁבָּהֶן וּמְחַלְּלָן עָלֶיהָ\hdot
מִפְּנֵי שֶׁאָֽמְרוּ מְחַלְּלִין כֶּסֶף עַל נְחֹשֶׁת מִדְּוֹחַק\hdot
וְלֹא שֶׁיִּתְקַיֵּים כֵּן אֶלָּא חוֹזֵר וּמְחַלְּלָם עַל הַכֶּסֶף׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים לֹא יַעֲשֶׂה אָדָם אֶת סְלָעָיו דִּינָרֵי זָהָב\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
אָמַר רִבִּי עֲקִיבָא אֲנִי עָשִׂיתִי לְרַבָּן גַּמְלִיאֵל וּלְרִבִּי יְהוֹשֻׁעַ אֶת כַּסְפָּן דִּינָרֵי זָהָב׃}

\halacha[149]{הַפּוֹרֵט סֶלַע מִמָּעוֹת מַעֲשֵׂר שֵׁנִי\hdot
בֵּית שַׁמָּאי אוֹמְרִים בְּכָל הַסֶּלַע מָעוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים בְּשֶׁקֶל כֶּסֶף וּבְשֶׁקֶל מָעוֹת\hdot
רִבִּי מֵאִיר אוֹמֵר אֵין מְחַלְּלִין כֶּסֶף וּפֵירוֹת עַל הַכֶּסֶף וַחֲכָמִים מַתִּירִין׃}

\halacha[1484]{הַפּוֹרֵט סֶלַע שֶׁל מַעֲשֵׂר שֵׁנִי בִירוּשָׁלַיִם\hdot
בֵּית שַׁמָּאי אוֹמְרִים בְּכָל הַסֶּלַע מָעוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים בְּשֶׁקֶל כֶּסֶף וּבְשֶׁקֶל מָעוֹת\hdot
הַדָּנִים לִפְנֵי חֲכָמִים אוֹמְרִים\hdot
בִּשְׁלֹשָׁה דִינָרִים כֶּסֶף וּבְדִינָר מָעוֹת\hdot
רִבִּי עֲקִיבָא אוֹמֵר בִּשְׁלֹשָׁה דִינָרִים כֶּסֶף (וּבִרְבִיעִית כֶּסֶף\hdot)
וּבִרְבִיעִית מָעוֹת\hdot
רִבִּי טַרְפוֹן אוֹמֵר אַרְבָּעָה אַסְפְּרֵי כֶסֶף\hdot
שַׁמָּאי אוֹמֵר יַנִּיחֶנָּה בֶּחָנוּת וְיֹאכַל כְּנֶגְדָּהּ׃}

\halacha{מִי שֶׁהָיוּ מִקְצַת בָּנָיו טְמֵאִין\hdot
וּמִקְצָתָן טְהוֹרִין\hdot
מַנִּיחַ אֶת הַסֶּלַע וְאוֹמֵר\hdot
מַה שֶּׁהַטְּהוֹרִים שׁוֹתִים\hdot
סֶלַע זוֹ מְחֻלֶּלֶת עָלָיו\hdot
נִמְצְאוּ טְהוֹרִים וּטְמֵאִין שׁוֹתִין מִכַּד אֶחָד׃}
