\tractatehead{מסכת תענית}{taanit}

\perek{מֵאֵימָתַי}{מַזְכִּירִין גְּבוּרוֹת גְּשָׁמִים\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר מִיּוֹם טוֹב הָרִאשׁוֹן שֶׁל חָג\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר מִיּוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג\hdot
אָמַר לוֹ רִבִּי יְהוֹשֻׁעַ הוֹאִיל וְאֵין הַגְּשָׁמִים סִימַן בְּרָכָה בֶּחָג לָמָּה הוּא מַזְכִּיר\hdot
אָמַר לוֹ רִבִּי אֱלִיעֶזֶר אַף אֲנִי לֹא אָמַרְתִּי לִשְׁאוֹל\hdot
אֶלָּא לְהַזְכִּיר מַשִּׁיב הָרוּחַ וּמוֹרִיד הַגֶּשֶׁם בְּעוֹנָתוֹ\hdot
אָמַר לוֹ אִם כֵּן לְעוֹלָם יְהֵא מַזְכִּיר׃}

\halacha{אֵין שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
אֶלָּא סָמוּךְ לַגְּשָׁמִים\hdot
רִבִּי יְהוּדָה אוֹמֵר הָעוֹבֵר לִפְנֵי הַתֵּיבָה בְּיוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג\hdot
הָאַחֲרוֹן מַזְכִּיר הָרִאשׁוֹן אֵינוֹ מַזְכִּיר\hdot
בְּיוֹם טוֹב הָרִאשׁוֹן שֶׁל פֶּסַח\hdot
הָרִאשׁוֹן מַזְכִּיר הָאַחֲרוֹן אֵינוֹ מַזְכִּיר\hdot
עַד אֵימָתַי שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
רִבִּי יְהוּדָה אוֹמֵר עַד שֶׁיַּעֲבוֹר הַפֶּסַח\hdot
רִבִּי מֵאִיר אוֹמֵר עַד שֶׁיֵּצֵא נִיסָן\hdot
שֶׁנֶּאֱמַר וַיּוֹרֶד לָכֶם גֶּשֶׁם מוֹרֶה וּמַלְקוֹשׁ בָּרִאשׁוֹן׃}

\halacha{בִּשְׁלֹשָׁה בְּמַרְחֶשְׁוָן שׁוֹאֲלִין אֶת הַגְּשָׁמִים\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר בְּשִׁבְעָה בוֹ\hdot
חֲמִשָּׁה עָשָׂר יוֹם אַחַר הֶחָג\hdot
כְּדֵי שֶׁיַּגִּיעַ אַחֲרוֹן שֶׁבְּיִשְׂרָאֵל לִנְהַר פְּרָת׃}

\halacha{הִגִּיעַ שִׁבְעָה עָשָׂר בְּמַרְחֶשְׁוָן וְלֹא יָֽרְדוּ גְשָׁמִים\hdot
הִתְחִילוּ הַיְּחִידִים מִתְעַנִּים שָׁלֹשׁ תַּעֲנִיּוֹת\hdot
אוֹכְלִים וְשׁוֹתִים מִשֶּׁחֲשֵׁיכָה\hdot
וּמוּתָרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה׃}

\halacha{הִגִּיעַ רֹאשׁ חוֹדֶשׁ כִּסְלֵו\hdot
וְלֹא יָֽרְדוּ גְשָׁמִים\hdot
בֵּית דִּין גּוֹזְרִין שָׁלֹשׁ תַּעֲנִיּוֹת עַל הַצִּבּוּר\hdot
אוֹכְלִים וְשׁוֹתִים מִשֶּׁחֲשֵׁיכָה\hdot
וּמוּתָרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה׃}

\halacha{עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
בֵּית דִּין גּוֹזְרִין שָׁלֹשׁ תַּעֲנִיּוֹת אֲחֵרוֹת עַל הַצִּבּוּר\hdot
אוֹכְלִים וְשׁוֹתִים מִבְּעוֹד יוֹם\hdot
וַאֲסוּרִין בִּמְלָאכָה\hdot
וּבִרְחִיצָה\hdot
וּבְסִיכָה\hdot
וּבִנְעִילַת הַסַּנְדָּל\hdot
וּבְתַשְׁמִישׁ הַמִּטָּה\hdot
וְנוֹעֲלִין אֶת הַמֶּרְחֲצָאוֹת\hdot
עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
בֵּית דִּין גּוֹזְרִין עֲלֵיהֶם עוֹד שֶׁבַע\hdot
שֶׁהֵן שְׁלֹשׁ עֶשְׂרֵה תַּעֲנִיּוֹת עַל הַצִּבּוּר\hdot
הֲרֵי אֵלּוּ יְתֵרוֹת עַל הָרִאשׁוֹנוֹת\hdot
שֶׁבְּאֵלּוּ מַתְרִיעִין וְנוֹעֲלִין אֶת הַחֲנִיּוֹת\hdot
בַּשֵּׁנִי מַטִּין עִם חֲשֵׁיכָה\hdot
וּבַחֲמִישִׁי מֻתָּרִין מִפְּנֵי כְבוֹד הַשַּׁבָּת׃}

\halacha{עָֽבְרוּ אֵלּוּ וְלֹא נֶעֱנוּ\hdot
מְמַעֲטִין בְּמַשָּׂא וּמַתָּן\hdot
בְּבִנְיַן וּבִנְטִיעָה בְּאֵירוּסִין וּבְנִשּׂוּאִין וּבִשְׁאֵלַת שָׁלוֹם בֵּין אָדָם לַחֲבֵירוֹ כִּבְנֵי אָדָם הַנְּזוּפִין לַמָּקוֹם\hdot
הַיְּחִידִים חוֹזְרִים וּמִתְעַנִּים עַד שֶׁיֵּצֵא נִיסָן\hdot
יָצָא נִיסָן וְיָֽרְדוּ גְשָׁמִים\hdot
סִימַן קְלָלָה\hdot
שֶׁנֶּאֱמַר הֲלֹא קְצִיר חִטִּים הַיּוֹם וְגוֹמֵר׃}

\perek{סֵדֶר}{תַּעֲנִיּוֹת כֵּיצַד\hdot
מוֹצִיאִין אֶת הַתֵּיבָה לִרְחוֹבָהּ שֶׁל עִיר\hdot
וְנוֹתְנִין אֵפֶר מַקְלֶה עַל גַּבֵּי הַתֵּיבָה\hdot
וּבְרֹאשׁ הַנָּשִׂיא\hdot
וּבְרֹאשׁ אָב בֵּית דִּין\hdot
וְכָל אֶחָד וְאֶחָד נוֹטֵל וְנוֹתֵן בְּרֹאשׁוֹ\hdot
הַזָּקֵן שֶׁבָּהֶן אוֹמֵר לִפְנֵיהֶן דִּבְרֵי כִבּוּשִׁין\hdot
אַחֵינוּ לֹא נֶאֱמַר בְּאַנְשֵׁי נִינְוֵה\hdot
וַיַּרְא אֱלֹהִים אֶת שַׂקָּם וְאֶת תַּעֲנִיתָם\hdot
אֶלָּא וַיַּרְא הָאֱלֹהִים אֶת מַעֲשֵׂיהֶם\hdot
כִּי שָׁבוּ מִדַּרְכָּם הָרָעָה\hdot
וּבְקַבָּלָה הוּא אוֹמֵר\hdot
וְקִרְעוּ לְבַבְכֶם וְאַל בִּגְדֵיכֶם׃}

\halacha{עָֽמְדוּ בַתְּפִלָּה\hdot
מוֹרִידִין לִפְנֵי הַתֵּיבָה\hdot
זָקֵן וְרָגִיל וְיֵשׁ לוֹ בָנִים\hdot
וּבֵיתוֹ רֵיקָם\hdot
כְּדֵי שֶׁיְּהֵא לִבּוֹ שָׁלֵם בַּתְּפִלָּה\hdot
וְאוֹמֵר לִפְנֵיהֶם אַרְבַּע וְעֶשְׂרִים בְּרָכוֹת\hdot
שְׁמֹנֶה עֶשְׂרֶה שֶׁבְּכָל יוֹם וּמוֹסִיף עֲלֵיהֶם עוֹד שֵׁשׁ׃}

\halacha{וְאֵלּוּ הֵן זִכְרוֹנוֹת וְשׁוֹפָרוֹת\hdot
אֶל יְיָ בַּצָּרָתָה לִּי קָרָאתִי וַיַּעֲנֵנִי\hdot
אֶשָּׂא עֵינַי אֶל הֶהָרִים וְגוֹמֵר\hdot
מִמַּעֲמַקִּים קְרָאתִיךָ יְיָ\hdot
תְּפִלָּה לְעָנִי כִי יַעֲטוֹף\hdot
רִבִּי יְהוּדָה אוֹמֵר לֹא הָיָה צָרִיךְ לוֹמַר זִכְרוֹנוֹת וְשׁוֹפָרוֹת\hdot
אֶלָּא אוֹמֵר תַּחְתֵּיהֶן\hdot
רָעָב כִּי יִהְיֶה בָאָרֶץ\hdot
דֶּבֶר כִּי יִהְיֶה וְגוֹמֵר\hdot
אֲשֶׁר הָיָה דְּבַר יְיָ אֶל יִרְמְיָהוּ עַל דִּבְרֵי הַבַּצָּרוֹת\hdot
וְאוֹמֵר חוֹתָמֵיהֶן׃}

\halacha{עַל הָרִאשׁוֹנָה הוּא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת אַבְרָהָם בְּהַר הַמּוֹרִיָּה\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ גּוֹאֵל יִשְׂרָאֵל\hdot
עַל הַשְּׁנִיָּה הוּא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת אֲבוֹתֵינוּ עַל יַם סוּף\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ זוֹכֵר הַנִּשְׁכָּחוֹת\hdot
עַל הַשְּׁלִישִׁית הוּא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת יְהוֹשֻׁעַ בַּגִּלְגָּל\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ שׁוֹמֵעַ תְּרוּעָה\hdot
עַל הָרְבִיעִית הוּא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת שְׁמוּאֵל בַּמִּצְפָּה\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ שׁוֹמֵעַ צְעָקָה\hdot
עַל הַחֲמִישִׁית הוֹא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת אֵלִיָּהוּ בְּהַר הַכַּרְמֶל\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ שׁוֹמֵעַ תְּפִלָּה\hdot
עַל הַשִּׁשִּׁית הוּא אוֹמֵר\hdot
מִי שֶׁעָנָה אֶת יוֹנָה מִמְּעֵי הַדָּגָה\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ הָעוֹנֶה בְּעֵת צָרָה\hdot
עַל הַשְּׁבִיעִית הוּא אוֹמֵר מִי שֶׁעָנָה אֶת דָּוִד וְאֶת שְׁלֹמֹה בְנוֹ בִּירוּשָׁלַיִם\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
בָּרוּךְ אַתָּה יְיָ הַמְּרַחֵם עַל הָאָרֶץ׃}

\halacha{מַעֲשֶׂה בִּימֵי רִבִּי חֲלַפְתָּא וְרִבִּי חֲנַנְיָה בֶּן תְּרַדְיוֹן שֶׁעָבַר אֶחָד לִפְנֵי הַתֵּיבָה\hdot
וְגָמַר אֶת הַבְּרָכָה כֻּלָּהּ\hdot
וְלֹא עָנוּ אַחֲרָיו אָמֵן\hdot
תִּקְעוּ הַכֹּהֲנִים תִּקְעוּ\hdot
מִי שֶׁעָנָה אֶת אַבְרָהָם בְּהַר הַמּוֹרִיָּה\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
הָרִיעוּ בְּנֵי אַהֲרֹן הָרִיעוּ\hdot
מִי שֶׁעָנָה אֶת אֲבוֹתֵינוּ עַל יַם סוּף\hdot
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה\hdot
וּכְשֶׁבָּא הַדָּבָר אֵצֶל חֲכָמִים אָֽמְרוּ\hdot
לֹא הָיוּ נוֹהֲגִין כֵּן אֶלָּא בְשַׁעַר מִזְרָח (וּבְהַר הַבַּיִת)׃}

\halacha{שָׁלֹשׁ תַּעֲנִיּות הָרִאשׁוֹנוֹת\hdot
אַנְשֵׁי מִשְׁמָר מִתְעַנִּין וְלֹא מַשְׁלִימִין\hdot

רִבִּי יוֹסֵי אוֹמֵר לְעֶזְרָה\hdot
וְלֹא לִצְעָקָה\hdot
שִׁמְעוֹן הַתֵּימָנִו אוֹמֵר אַף עַל הַדֶּבֶר\hdot

\halacha{עַל כָּל צָרָה שֶׁלֹּא תָבֹא עַל הַצִּבּוּר מַתְרִיעִין עֲלֵיהֶן\hdot
חוּץ מֵרוֹב גְּשָׁמִים\hdot

בִּשְׁבִיל שֶׁלֹּא יִמּוֹקוּ\hdot

הִתְחִילוּ לִירֵד בְּזַעַף\hdot

וּבְיוֹם הַכִּפּוּרִים׃}

\halacha{אֵלּוּ הֵן מַעֲמָדוֹת\hdot
לְפִי שֶׁנֶּאֱמַר צַו אֶת בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵיהֶם אֶת קָרְבָּנְי לַחְמִי\hdot
וְכִי הֵיאַךְ קָרְבָּנוֹ שֶׁל אָדָם קָרֵב\hdot
וְהוּא אֵינוֹ עוֹמֵד עַל גַּבָּיו\hdot
הִתְקִינוּ נְבִיאִים הָרִאשׁוֹנִים אַרְבַּע וְעֶשְׂרִים מִשְׁמָרוֹת\hdot
עַל כָּל מִשְׁמָר וּמִשְׁמָר הָיָה מַעֲמָד בִּירוּשָׁלַיִם\hdot
שֶׁל כֹּהֲנִים שֶׁל לְוִיִּם וְשֶׁל יִשְׂרְאֵלִים\hdot
הִגִּיעַ זְמַן הַמִּשְׁמָר לַעֲלוֹת\hdot
כֹּהֲנִים וּלְוִיִּם עוֹלִים לִירוּשָׁלַיִם\hdot
וְיִשְׂרָאֵל שֶׁבְּאוֹתוֹ מִשְׁמָר מִתְכַּנְּסִין לְעָרֵיהֶן וְקוֹרְאִין בְּמַעֲשֵׁה בְרֵאשִׁית׃}

\halacha{(וְאַנְשֵׁי הַמַּעֲמָד הָיוּ מִתְעַנִּין אַרְבָּעָה יָמִים בַּשָּׁבוּעַ\hdot
מִיּוֹם שֵׁנִי וְעַד יוֹם חֲמִישִׁי\hdot
וְלֹא הָיוּ מִתְעַנִּין עֶרֶב שַׁבָּת מִפְּנֵי כְבוֹד הַשַּׁבָּת\hdot
וְלֹא בְאֶחָד בְּשַׁבָּת כְּדֵי שֶׁלֹּא יֵצְאוּ מִמְּנוּחָה וְעֹנֶג\hdot
לִיגִיעָה וְתַעֲנִית וְיָמוּתוּ)\hdot
בַּיּוֹם הָרִאשׁוֹן בְּרֵאשִׁית\hdot
וִיהִי רְקִיעַ\hdot
בַּשֵּׁנִי יְהִי רְקִיעַ\hdot
וְיִקָּווּ הַמַּיִם\hdot
בַּשְּׁלִישִׁי יְקָּווּ הַמַּיִם\hdot
וִיהִי מְאוֹרוֹת\hdot
בָּֽרְבִיעִי יְהִי מְאֹרֹת\hdot
וְיִשְׁרְצוּ הַמַּיִם\hdot
בַּחֲמִישִׁי יִשְׁרְצוּ הַמַּיִם\hdot
וְתוֹצֵא הָאָרֶץ\hdot
בַּשִּׁשִּׁי תּוֹצֵא הָאָרֶץ\hdot
וַיְכֻלּוּ הַשָּׁמַיִם\hdot
פָּרָשָׁה גְדוֹלָה קוֹרִין אוֹתָהּ בִּשְׁנַיִם\hdot
וְהַקְּטַנָּה בַּיָּחִיד\hdot
בַּשַּׁחְרִית\hdot
בַּמּוּסָף\hdot
וּבַמִּנְחָה\hdot
נִכְנָסִין וְקוֹרִין עַל פִּיהֶן כְּקוֹרִין אֶת שְׁמַע\hdot
עֶרֶב שַׁבָּת בַּמִּנְחָה לֹא הָיוּ נִכְנָסִין מִפְּנֵי כְבוֹד הַשַּׁבָּת׃}

\halacha{כָּל יוֹם שֶׁיֵּשׁ בּוֹ הַלֵּל אֵין בּוֹ מַעֲמָד בַּשַּׁחְרִית\hdot
קָרְבַּן מוּסָף אֵין בּוֹ בַנְּעִילָה\hdot
קָרְבַּן עֵצִים אֵין בּוֹ בַמִּנְחָה\hdot
דִּבְרֵי רִבִּי עֲקִיבָא\hdot
אָמַר לוֹ בֶן עֲזָאי\hdot
כַּךְ הָיָה רִבִּי יְהוֹשֻׁעַ שׁוֹנֶה\hdot
קָרְבַּן מוּסָף אֵין בּוֹ בַמִּנְחָה\hdot
קָרְבַּן עֵצִים אֵין בּוֹ בַנְּעִילָה\hdot
חָזַר רִבִּי עֲקִיבָא לִהְיוֹת שׁוֹנֶה כְבֶן עֲזָאי׃}

\halacha{זְמַן עֲצֵי כֹהֲנִים וְהָעָם תִּשְׁעָה\hdot
בְּאֶחָד בְּנִיסָן בְּנֵי אָרַח בֶּן יְהוּדָה\hdot
בְּעֶשְׂרִים בְּתַמּוּז בְּנֵי דָוִד בֶּן יְהוּדָה\hdot
בַּחֲמִשָּׁה בְאָב בְּנֵי פַרְעוֹשׁ בֶּן יְהוּדָה\hdot
בְּשִׁבְעָה בוֹ בְּנֵי יוֹנָדָב בֶּן רֵכָב\hdot
בַּעֲשָׂרָה בוֹ בְּנֵי סַנְאָה בֶּן בִּנְיָמִין\hdot
בַּחֲמִשָּׁה עָשָׂר בּוֹ בְּנֵי זַתּוּא בֶּן יְהוּדָה\hdot
וְעִמָּהֶן כֹּהֲנִים וּלְוִיִּם\hdot
וְכָל מִי שֶׁטָּעָה בְשִׁבְטוֹ\hdot
וּבְנֵי גוֹנְבֵי עֵלִי\hdot
בְּנֵי קוֹצְעֵי קְצִיעוֹת\hdot
בְּעֶשְׁרִים בּוֹ בְּנֵי פַחַת מוֹאָב בֶּן יְהוּדָה\hdot
בְּעֶשְׁרִים בְּאֶלוּל בְּנֵי עֲדִין בֶּן יְהוּדָה\hdot
בְּאֶחָד בְּטֵבֵת שָׁבוּ בְנֵי פַרְעוֹשׁ שְׁנִיָּה\hdot
בְּאֶחָד בְּטֵבֵת לֹא הָיָה בוֹ מַעֲמָד שֶׁהָיָה בוֹ הַלֵּל וְקָרְבַּן מוּסָף וְקָרְבַּן עֵצִים׃}

\halacha{חֲמִשָּׁה דְבָרִים אֵרְעוּ אֶת אֲבוֹתֵינוּ בְּשִׁבְעָה עָשָׂר בְּתַמּוּז\hdot
וַחֲמִשָּׁה בְּתִשְׁעָה בְּאָב\hdot
בְּשִׁבְעָה עָשָׂר בְּתַמּוּז נִשְׁתַּבְּרוּ הַלֻּחוֹת\hdot
וּבֻטַּל הַתָּמִיד\hdot
וְהֻבְקְעָה הָעִיר\hdot
וְשָׂרַף אַפּוֹסְטְמוֹס אֶת הַתּוֹרָה וְהֶעֱמִיד צֶלֶם בַּהֵיכָל\hdot
בְּתִשְׁעָה בְאָב נִגְזַר עַל אֲבוֹתֵינוּ שֶׁלֹּא יִכָּֽנְסוּ לָאָרֶץ\hdot
וְחָרֵב הַבַּיִת בָּרִאשׁוֹנָה וּבַשְּׁנִיָּה\hdot
וְנִלְכְּדָה בִתֵּר\hdot
וְנֶחֱרְשָׁה הָעִיר\hdot
מִשֶּׁנִּכְנַס אָב מְמַעֲטִין בְּשִׂמְחָה׃}

\halacha{שַׁבָּת שֶׁחַל תִּשְׁעָה בְאָב לִהְיוֹת בְּתוֹכָהּ\hdot
אֲסוּרִין מִלְּסַפֵּר וּמִלְּכַבֵּס\hdot
וּבַחֲמִישִׁי מֻתָּרִין\hdot
מִפְּנֵי כְבוֹד הַשַּׁבָּת\hdot
עֶרֶב תִּשְׁעָה בְאָב לֹא יֹאכַל אָדָם שְׁנֵי תַבְשִׁילִין\hdot
לֹא יֹאכַל בָּשָׂר וְלֹא יִשְׁתֶּה יַיִן\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר יְשַׁנֶּה\hdot
רִבִּי יְהוּדָה מְחַיֵּיב בִּכְפִיַּת הַמִּטָּה\hdot
וְלֹא הוֹדוּ לוֹ חֲכָמִים׃}

\halacha{אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל\hdot
לֹא הָיוּ יָמִים טוֹבִים לְיִשְׂרָאֵל כַּחֲמִשָּׁה עָשָׂר בְּאָב\hdot
וּכְיוֹם הַכִּפּוּרִים\hdot
שֶׁבָּהֶן בְּנוֹת יְרוּשָׁלַיִם יוֹצְאוֹת בִּכְלֵי לָבָן שְׁאוּלִין\hdot
שֶׁלֹּא לְבַיֵּשׁ אֶת מִי שֶׁאֵין לוֹ\hdot
כָּל הַכֵּלִים טְעוּנִין טְבִילָה\hdot
וּבְנוֹת יְרוּשָׁלַיִם יוֹצְאוֹת וְחוֹלוֹת בַּכְּרָמִים\hdot
וּמַה הָיוּ אוֹמְרוֹת\hdot
בָּחוּר שָׁא נָא עֵינֶיךָ וּרְאֵה\hdot
מָה אַתָּה בּוֹרֵר לָךְ\hdot
אַל תִּתֵּן עֵינֶיךָ בַּנּוֹי תֵּן עֵינֶיךָ בַּמִּשְׁפָּחָה\hdot
שֶׁקֶר הַחֵן וְהֶבֶל הַיּוֹפִי אִשָּׁה יִרְאַת יְיָ הִיא תִּתְהַלָּל\hdot
וְאוֹמֵר תְּנוּ לָהּ מִפְּרִי יָדֶיהָ וִיהַלְלוּהָ בַּשְּׁעָרִים מַעֲשֶׂיהָ\hdot
וְכֵן הוּא אוֹמֵר צְאֶינָה וּרְאֶינָה בְּנוֹת צִיּוֹן בַּמֶּלֶךְ שְׁלֹמֹה\hdot
בַּעֲטָרָה שֶׁעִטְּרָה לּוֹ אִמּוֹ בְּיוֹם חֲתֻנָּתוֹ וּבְיוֹם שִׂמְחַת לִבּוֹ\hdot
בְּיוֹם חֲתֻנָּתוֹ זֶה מַתָּן תּוֹרָה\hdot
וּבְיוֹם שִׂמְחַת לִבּוֹ\hdot
זֶה בִנְיַן בֵּית הַמִּקְדָּשׁ\hdot
שֶׁיִּבָּנֶה בִּמְהֵרָה בְיָמֵינוּ אָמֵן׃\selik}
