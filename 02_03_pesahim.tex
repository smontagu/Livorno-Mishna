\sechead{יום ז}

\perek[832]{אוֹר}{לְאַרְבָּעָה עָשָׂר בּוֹדְקִין אֶת הֶחָמֵץ לְאוֹר הַנֵּר\hdot
כָּל מָקוֹם שֶׁאֵין מַכְנִיסִין בּוֹ חָמֵץ\hdot
אֵין צָרִיךְ בְּדִיקָה\hdot
וְלָמָּה אָֽמְרוּ שְׁתֵּי שׁוּרוֹת בַּמַּרְתֵּף\hdot
מָקוֹם שֶׁמַּכְנִיסִין בּוֹ חָמֵץ\hdot
בֵּית שַׁמָּאי אוֹמְרִים שְׁתֵּי שׁוּרוֹת עַל פְּנֵי כָל הַמַּרְתֵּף\hdot
וּבֵית הִלֵּל אוֹמְרִים שְׁתֵּי שׁוּרוֹת הַחִיצוֹנוֹת שֶׁהֵן הָעֶלְיוֹנוֹת׃}

\halacha{אֵין חוֹשְׁשִין שֶׁמֶּא גָֽרְרָה חֻלְדָה מִבַּיִת לְבַיִת וּמִמָּקוֹם לְמָקוֹם\hdot
דְּאִם כֵּן מֵחָצֵר לֶחָצֵר וּמֵעִיר לָעִיר\hdot
אֵין לַדָּבָר סוֹף׃}

\halacha{רִבִּי יְהוּדָה אוֹמֵר בּוֹדְקִין אוֹר אַרְבָּעָה עָשָׂר וּבְאַרְבָּעָה עָשָׂר שַׁחְרֵית\hdot
וּבְשַׁעַת הַבִּיעוּר\hdot
וַחֲכָמִים אוֹמְרִים לֹא בָדַק אוֹר אַרְבַָּעָה עָשָׂר\hdot
יִבְדּוֹק בְּאַרְבָּעָה עָשָׂר\hdot
לֹא בָדַק בְּאַרְבָּעָה עָשָׂר\hdot
יִבְדּוֹק בְּתוֹךְ הַמּוֹעֵד\hdot
לֹא בָדַק בְּתוֹךְ הַמּוֹעֵד יִבְדּוֹק לְאַחַר הַמּוֹעֵד\hdot
וּמַה שֶּׁמְּשַׁיֵּר יַנִּיחֶנּוּ בְצִנְעָא כְּדֵי שֶׁלֹּא יְהֵא צָרִיךְ בְּדִיקָה אַחֲרָיו׃}

\halacha{רִבִּי מֵאִיר אוֹמֵר אוֹכְלִין כָּל חָמֵשׁ וְשׂוֹרְפִין בִּתְחִלַּת שֵׁשׁ
רִבִּי יְהוּדָה אוֹמֵר אוֹכְלִין כָּל אַרְבַּע\hdot
וְתוֹלִין כָּל חָמֵשׁ\hdot
וְשׂוֹרְפִין בִּתְחִלַּת שֵׁשׁ׃}

\halacha[136]{וְעוֹד אָמַר רִבִּי יְהוּדָה שְׁתֵּי חַלּוֹת שֶׁל תּוֹדָה פְּסוּלוֹת וּמוּנָחוֹת עַל גַּג הָאִצְטְבָא\hdot
כָּל זְמַן שֶׁמּוּנָחוֹת כָּל הָעָם אוֹכְלִין\hdot
נִטְּלָה אֶחָת תּוֹלִין\hdot
לֹא אוֹכְלִין וְלֹא שׂוֹרְפִין\hdot
נִטְּלוּ שְׁתֵּיהֶן הִתְחִילוּ כָל הָעָם שׂוֹרְפִים\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר חוֹלִין נֶאֱכָלִין כָּל אַרְבַּע\hdot
וּתְרוּמָה כָּל חָמֵשׁ\hdot
וְשׁוֹרְפִין בִּתְחִלַּת שֵׁשׁ׃}

\halacha[334]{רִבִּי חֲנִינָא סְגַן הַכֹּהֲנִים אוֹמֵר מִימֵיהֶן שֶׁל כֹּהֲנִים לֹא נִמְנְעוּ מִלִּשְׂרוֹף אֶת הַבָּשָׂר שֶׁנִּטְמָא בְּוָלֶד הַטֻּמְאָה עִם הַבָּשָׂר שֶׁנִּטְמָא בְּאָב הַטֻּמְאָה\hdot
אַף עַל פִּי שֶׁמּוֹסִיפִין טֻמְאָה עַל טֻמְאָתוֹ\hdot
הוֹסִיף רִבִּי עֲקִיבָא וְאָמַר מִימֵיהֶם שֶׁל כֹּהֲנִים לֹא נִמְנְעוּ מִלְּהַדְלִיק אֶת הַשֶּׁמֶן שֶׁנִּפְסַל בִּטְבוּל יוֹם\hdot
בְּנֵר שֶׁנִּטְמָא בִּטְמֵא מֵת\hdot
אַף עַל פִּי שֶׁמּוֹסִיפִין טֻמְאָה עַל טֻמְאָתוֹ׃}

\halacha[417]{אָמַר רִבִּי מֵאִיר מִדִּבְרֵיהֶם לָמַדְנוּ שֶׁשּׂוֹרְפִין תְּרוּמָה טְהוֹרָה עִם הַטְּמֵאָה בַּפֶּסַח\hdot
אָמַר לוֹ רִבִּי יוֹסֵי אֵינָהּ הִיא הַמִּדָּה\hdot
וּמוֹדִים רִבִּי אֱלִיעֶזֶר וְרִבִּי יְהוֹשֻׁעַ שֶׁשּׂוֹרְפִין זוֹ לְעַצְמָהּ וְזוֹ לְעַצְמָהּ\hdot
עַל מָה נֶחֱלָקוּ עַל הַתְּלוּיָה וְעַל הַטְּמֵאָה\hdot
שֶׁרִבִּי אֱלִיעֶזֶר אוֹמֵר תִּשָּׂרֵף זוֹ לְעַצְמָהּ וְזוֹ לְעַצְמָהּ\hdot
וְרִבִּי יְהוֹשֻׁעַ אוֹמֵר שְׁתֵּיהֶן כְּאֶחָת׃}

\perek[2856]{כָּל}{שָׁעָה שֶׁמּוּתָר לֶאֱכוֹל\hdot
מַאֲכִיל לַבְּהֵמָה לַחַיָּה וְלָעוֹפוֹת\hdot
וּמוֹכְרוֹ לַנָּכְרִי\hdot

מוּתָר בַּהֲנָאָה\hdot

שֶׁנֶּאֱמַר לֹא יֵרָאֶה לְךָ שְׂאוֹר׃}

\halacha{נָכְרִי שֶׁהִלְוָה אֶת יִשְׂרָאֵל עַל חֶמְצוֹ\hdot
אַחַר הַפֶּסַח מוּתָר בַּהֲנָאָה\hdot
וְיִשְׂרָאֵל שֶׁהִלְוָה אֶת הַנָּכְרִי עַל חֶמְצוֹ\hdot
אַחַר הַפֶּסַח אָסוּר בַּהֲנָאָה\hdot
חָמֵץ שֶׁנָּֽפְלָה עָלָיו מַפּוֹלֶת הֲרֵי הוּא כִמְבוֹעָר\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר כָּל שֶׁאֵין הַכֶּלֶב יָכוֹל לְחַפֵּשׂ אַחֲרָיו׃}

\halacha[787]{הָאוֹכֵל תְּרוּמַת חָמֵץ בַּפֶּסַח\hdot
