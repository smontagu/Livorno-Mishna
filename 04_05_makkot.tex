\tractatehead{מסכת מכות}{makkot}

\perek{כֵּיצַד}{הָעֵדִים נַעֲשִׂים זוֹמְמִין\hdot
מְעִידִין אָנוּ בְּאִישׁ פְּלוֹנִי\hdot
שֶׁהוּא בֶן גְּרוּשָׁה אוֹ בֶן חֲלוּצָה\hdot
אֵין אוֹמְרִין יֵעָשֶׂה זֶה בֶּן גְּרוּשָׁה אוֹ בֶן חֲלוּצָה תַּחְתָּיו\hdot
אֶלָּא לוֹקֶה אַרְבָּעִים\hdot
מְעִידִין אָנוּ בְּאִישׁ פְּלוֹנִי\hdot
שֶׁהוּא חַיָּב לִגְלוֹת\hdot
אֵין אוֹמְרִים יִגְלֶה זֶה תַחְתָּיו\hdot
אֶלָּא לוֹקֶה אַרְבָּעִים\hdot
מְעִידִין אָנוּ בְּאִישׁ פְּלוֹנִי שֶׁגֵּרַשׂ אֶת אִשְׁתּוֹ וְלֹא נָתַן לָהּ כְּתוּבָתָהּ\hdot
וַהֲלֹא בֵּין הַיּוֹם וּבֵין מָחָר סוֹפוֹ לִתֵּן לָהּ כְּתוּבָתָהּ\hdot
אוֹמְדִים כַּמָּה רוֹצֶה לִתֵּן בִּכְתוּבָתָהּ שֶׁל זוֹ שֶׁאֵם נִתְאַלְמְנָה אוֹ נִתְגָּֽרְשָׁה\hdot
וְאִם מֵתָה יִירָשֶׁנָּה בַעְלָהּ\hdot
מְעִידִין אָנוּ בְּאִישׁ פְּלוֹנִי שֶׁהוּא חַיָּב לַחֲבֵירוֹ אֶלֶף זוּז\hdot
עַל מְנַת לִתְּנָן לוֹ מִכָּאן וְעַד שְׁלֹשִׁים יוֹם\hdot
וְהוּא אוֹמֵר מִכָּאן וְעַד עֶשֶׂר שָׁנִים\hdot
אוֹמְדִין כַּמָּה אָדָם רוֹצֶה לִתֵּן וְיִהְיוּ בְיָדוֹ אֶלֶף זוּז\hdot
בֵּין נוֹתְנָן מִכָּאן וְעַד שְׁלֹשִׁים יוֹם\hdot
בֵּין נוֹתְנָן מִכָּאן וְעַד עֶשֶׁר שָׁנִים׃}

\halacha{מְעִידִין אָנוּ בְּאִישׁ פְּלוֹנִי\hdot
שֶׁחַיָּב לַחֲבֵירוֹ מָאתַיִם זוּז\hdot
וְנִמְצְאוּ זוֹמְמִין\hdot
לוֹקִין וּמְשַׁלְּמִין\hdot
שֶׁלֹּא הַשֵּׁם הַמְּבִיאָן לִידֵי מַכּוֹת\hdot
מְבִיאָן לִידֵי תַשְׁלוּמִין\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים כָּל הַמְּשַׁלֵּם אֵינוֹ לוֹקֶה׃}

\halacha{מְעִידִין אֲנִי בְּאִישׁ פְּלוֹנִי שֶׁהוּא חַיָּב מַלְקוּת אַרְבָּעִים\hdot
וְנִמְצְאוּ זוֹמְמִין\hdot
לוֹקִין שְׁמוֹנִים\hdot
מִשּׁוּם לֹא תַעֲנֶה בְרֵעֲךָ עֵד שָׁקֶר\hdot
וּמִשּׁוּם וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים אֵין לוֹקִין אֶלָּא אַרְבָּעִים\hdot
מְשַׁלְּשִׁין בַּמָּמוֹן\hdot
וְאֵין מְשַׁלְּשִׁין בַּמַּכּוֹת\hdot
כֵּיצַד\hdot
הֵעִידוּהוּ שֶׁהוּא חַיָּב לַחֲבֵירוֹ מָאתַיִם זוּז\hdot
וְנִמְצְאוּ זוֹמְמִין\hdot
מְשַׁלְּשִׁין בֵּינֵיהֶם\hdot
אֲבָל אִם הֵעִידוּהוּ שֶׁהוּא חַיָּב מַלְקוּת אַרְבָּעִים\hdot
וְנִמְצְאוּ זוֹמְמִין\hdot
כָּל אֶחָד וְאֶחָד לוֹקֶה אַרְבָּעִים׃}

\halacha{אֵין הָעֵדִים נַעֲשִׂים זוֹמְמִין\hdot
עַד שֶׁיָּזֹמּוּ אֶת עַצְמָן\hdot
כֵּיצַד\hdot
אָֽמְרוּ מְעִידיִן אָנוּ בְּאִישׁ פְּלוֹנִי\hdot
שֶׁהָרַג אֶת הַנֶּפֶשׁ\hdot
אָֽמְרוּ לָהֶן הֵיאַךְ אַתֶּם מְעִידִין שֶׁהֲרֵי נֶהֱרַג זֶה\hdot
אוֹ הַהוֹרֵג הָיָה עִמָּנוּ אוֹתוֹ הַיּוֹם בְּמָקוֹם פְּלוֹנִי\hdot
אֵין אֵלּוּ זוֹמְמִין\hdot
אֲבָל אָֽמְרוּ לָהֶן הֵיאַךְ אַתֶּם מְעִידִין\hdot
שֶׁהֲרֵי הֱיִיתֶם עִמָּנוּ אוֹתוֹ הַיּוֹם בְּמָקוֹם פְּלוֹנִי\hdot
הֲרֵי אֵלּוּ זוֹמְמִין\hdot
וְנֶהֱרָגִין עַל פִּיהֶם׃}

\halacha{בָּאוּ אַחֵרִים וְהֵזִימוּם\hdot
בָּאוּ אַחֵרִים וְהֵזִימוּם\hdot
אֲפִילוּ מֵאָה\hdot
כֻּלָּם יֵהָרֵגוּ\hdot
רִבִּי יְהוּדָה אוֹמֵר אִיסְטָסִית הִיא זוֹ וְאֵינָהּ נֶהֱרֶגֶת אֶלָּא כַּת הָרִאשׁוֹנָה בִּלְבָד׃}

\halacha{אֵין הָעֵדִים זוֹמְמִין נֶהֱרָגִין עַד שֶׁיִּגָּמֵר הַדִּין\hdot
שֶׁהֲרֵי הַצָּדוֹקין אוֹמְרִים\hdot
עַד שֶׁיֵּהָרֵג שֶׁנֶּאֱמַר נֶפֶשׁ תַּחַת נָפֶשׁ\hdot
אָֽמְרוּ לָהֶם חֲכָמִים וֲהַלֹא כְּבָר נֶאֱמַר\hdot
וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לְאָחִיו\hdot
וַהֲרֵי אָחִיו קַיָּם\hdot
אִם כֵּן לָמָּה נֶאֱמַר נֶפֶשׁ תַּחַת נָפֶשׁ\hdot
יָכוֹל מִשָּׁעָה שֶׁקִּבְּלוּ עֵדוּתָן יֵהָרֵגוּ\hdot
תַּלְמוּד לוֹמַר נֶפֶשׁ תַּחַת נָפֶשׁ\hdot
הָא אֵינָן נֶהֱרָגִין עַד שֶׁיִּגָּמֵר הַדִּין׃}

\halacha{עַל פִּי שְׁנַיִם עֵדִים\hdot
אוֹ שְׁלֹשָׁה עֵדִים יוּמָת הַמֵּת\hdot
אִם מִתְקַיֶּמֶת הָעֵדוּת בִּשְׁנַיִם לָמָּה פָּרַט הַכָּתוּב בִּשְׁלֹשָׁה\hdot
אֶלָּא לְהַקִּישׁ שְׁלֹשָׁה לִשְׁנַיִם\hdot
מַה שְּׁלֹשָׁה מְזַמִּים אֶת הַשְּׁנַיִם\hdot
אַף הַשְּׁנַיִם יָזּמּוּ אֶת הַשְּׁלֹשָׁה\hdot
וּמְנַיִן אֲפִילוּ מֵאָה\hdot
תָּלְמוּד לוֹמַר עֵדִים\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר מַה שְּׁנַיִם אֵינָן נֶהֱרָגִין\hdot
עַד שֶׁיִּהְיוּ שְׁנֵיהֶם זוֹמְמִין\hdot
אַף שְׁלֹשָׁה אֵינָן נֶהֱרָגִין\hdot
עַד שֶׁיִּהְיוּ שְׁלֹשְׁתָּן זוֹמְמִין\hdot
וּמְנַיִן אֲפִילוּ מֵאָה\hdot
תַּלְמוּד לוֹמַר עֵדִים\hdot
רִבִּי עֲקִיבָא אוֹמֵר לֹא בָא הַשְּׁלִישִׁי אֶלָּא לְהַחְמִיר עָלָיו\hdot
וְלַעֲשׂוֹת דִּינוֹ כַּיּוֹצֵא בָאֵלּוּ\hdot
וְאִם כֵּן עָנַשׁ הַכָּתוּב לַנִּטְפָּל לְעוֹבְרֵי עֲבֵרָה כְּעוֹבְרֵי עֲבֵרָה\hdot
עַל אַחַת כַּמָּה וְכַמָּה שֶׁיְּשַׁלֵּם שָׂכָר\hdot
לַנִּטְפָּל לְעוֹשֵׂי מִצְוָה\hdot
כְּעוֹשֵׂי מִצְוָה׃}

\halacha{מַה שְּׁנַיִם נִמְצָא אֶחָד מֵהֶן קָרוֹב אוֹ פָסוּל עֵדוּתָן בְּטֵלָה\hdot
אַף שְׁלֹשָׁה נִמְצָא אֶחָד מֵהֶן קָרוֹב אוֹ פָסוּל\hdot
עֵדוּתָן בְּטֵלָה\hdot
מְנַיִן אֲפִילוּ מֵאָה\hdot
תַּלְמוּד לוֹמַר עֵדִים\hdot
אָמַר רִבִּי יוֹסֵי בַּמֶּה דְבָרִים אֲמוּרִים בְּדִינֵי נְפָשׁוֹת\hdot
אֲבָל בְּדִינֵי מָמוֹנוֹת\hdot
תִּתְקַיֵּם הָעֵדוּת בַּשְּׁאָר\hdot
רִבִּי אוֹמֵר אֶחָד דִּינֵי מָמוֹנוֹת וְאֶחָד דִּינֵי נְפָשׁוֹת\hdot
בִּזְמָן שֶׁהִתְרוּ בָהֶן\hdot
אֲבָל בִּזְמָן שֶׁלֹּא הִתְרוּ בָהֶן\hdot
מַה יַּעֲשׂוּ שְׁנֵי אָחִין\hdot
שֶׁרָאוּ בְאֶחָד שֶׁהָרַג אֶת הַנֶּפֶשׁ׃}

\halacha{הָיוּ שְׁנַיִם רוֹאִין אוֹתוֹ מֵחַלּוֹן זֶה\hdot
וּשְׁנַיִם רוֹאִין אוֹתוֹ מֵחַלּוֹן זֶה\hdot
וְאֶחָד מַתְרֶה בוֹ בָאֶמְצַע\hdot
בִּזְמָן שֶׁמִּקְצָתָן רוֹאִין אֵלּוּ אֶת אֵלּוּ\hdot
הֲרֵי אֵלּוּ עֵדוּת אֶחָת\hdot
וְאִם לָאוּ הֲרֵי אֵלּוּ שְׁתֵּי עֵדוּיוֹת\hdot
לְפִיכַךְ אִם נִמְצֵאת אַחַת מֵהֶן זוֹמֶמֶת\hdot
הוּא וְהֵן נֶהֱרָגִין\hdot
וְהַשְּׁנִיָּה פְּטוּרָה\hdot
רִבִּי יוֹסֵי אוֹמֵר לְעוֹלָם אֵינָן נֶהֱרָגִין עַד שֶׁיִּהְיוּ פִּי שְׁנֵי עֵדָיו מַתְרִין בּוֹ\hdot
שֶׁנֶּאֱמַר עַל פִּי שְׁנַיִם עֵדִים\hdot
דָּבָר אַחֵר עַל פִּי שְׁנַיִם עֵדִים\hdot
שֶׁלֹּא תְהֵא סַנְהֶדְרִין שׁוֹמַעַת\hdot
מִפִּי הַתֻּרְגְּמָן׃}
