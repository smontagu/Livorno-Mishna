\tractatehead{מסכת כריתות}{keritot}

\perek{שְׁלֹשִׁים}{וָשֵׁשׁ כְּרִיתוֹת בַּתּוֹרָה\hdot
הַבָּא עַל הָאֵם\hdot
וְעַל אֵשֶׁת הָאָב\hdot
וְעַל הַכַּלָּה\hdot
הַבָּא עַל הַזָּכוּר\hdot
וְעַל הַבְּהֵמָה\hdot
וְהָאִשָּׁה הַמְּבִיאָה אֶת הַבְּהֵמָה עָלֶיהָ\hdot
הַבָּא עַל אִשָּׁה וּבִתָּהּ\hdot
וְעַל אֵשֶׁת אִישׁ\hdot
הַבָּא עַל אֲחוֹתוֹ\hdot
וְעַל אֲחוֹת אָבִיו\hdot
וְעַל אֲחוֹת אִמּוֹ\hdot
וְעַל אֲחוֹת אִשְׁתּוֹ\hdot
וְעַל אֵשֶׁת אָחִיו\hdot
וְעַל אֵשֶׁת אֲחִי אָבִיו\hdot
(וְעַל) הַנִּדָּה\hdot
הַמְּגַדֵּף\hdot
וְהָעוֹבֵד עֲבוֹדַת אֱלִילִים\hdot
וְהַנּוֹתֵן מִזַּרְעוֹ לַמֹּלֶךְ\hdot
וּבַעַל אוֹב\hdot
הַמְּחַלֵּל אֶת הַשַּׁבָּת\hdot
וְטָמֵא שֶׁאָכַל אֶת הַקֹּדֶשׁ\hdot
וְהַבָּא לַמִּקְדָּשׁ טָמֵא\hdot
הָאוֹכֵל חֵלֶב\hdot
וָדָם\hdot
נוֹתָר\hdot
וּפִגּוּל\hdot
הַשּׁוֹחֵט וְהַמַּעֲלֶה בַחוּץ\hdot
הָאוֹכֵל חָמֵץ בַּפֶּסַח\hdot
הָאוֹכֵל וְהָעוֹשֶׁה מְלָאכָה בְּיוֹם הַכִּפּוּרִים\hdot
הַמְּפַטֵּם אֶת הַשֶּׁמֶן\hdot
וְהַמְּפַטֵּם אֶת הַקְּטֹרֶת\hdot
וְהַסַּךְ בְּשֶׁמֶן הַמִּשְׁחָה\hdot
הַפֶּסַח וְהַמִּילָה בְּמִצְוַת עֲשֵׂה׃}

\halacha{עַל אֵלּוּ חַיָּבִים עַל זְדוֹנָם כָּרֵת\hdot
וְעַל שִׁגְגָתָן חַטָּאת\hdot
וְעַל לֹא הוֹדַע שֶׁלָּהֶן אָשָׁם תָּלוּי\hdot
חוּץ מִן הַמְּטַמֵּא מִקְדָּשׁ וְקָדָשָׁיו\hdot
מִפְּנֵי שֶׁהוּא בְּעוֹלֶה וְיוֹרֵד דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים אַף הַמְּגַדֵּף\hdot
שֶׁנֶּאֱמַר תּוֹרָה אַחַת יִהְיֶה לָכֶם לָעוֹשֶׂה בִשְׁגָגָה\hdot
יָצָא מְגַדֵּף שֶׁאֵינוֹ עוֹשֶׂה מַעֲשֶׂה׃}

\halacha{יֵשׁ מְבִיאוֹת קָרְבָּן וְנֶאֱכָל\hdot
(וְיֵשׁ) מְבִיאוֹת וְאֵינוֹ נֶאֱכָל\hdot
וְיֵשׁ שֶׁאֵינָן מְבִיאוֹת\hdot
אֵלּוּ מְבִיאוֹת קָרְבָּן וְנֶאֱכָל\hdot
הַמַפֶּלֶת כְּמִין בְּהֵמָה חַיָּה וְעוֹף דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים עַד שֶׁיְּהֵא בוֹ מִצּוּרַת הָאָדָם\hdot
הַמַּפֶּלֶת סַנְדָּל אוֹ שִׁלְיָא וְשָׁפִיר מְרֻקָּם\hdot
וְהַיּוֹצֵא מְחֻתָּךְ\hdot
וְכֵן שִׁפְחָה שֶׁהִפִּילָה\hdot
מְבִיאָה קָרְבָּן וְנֶאֱכָל׃}

\halacha{אֵלּוּ מְבִיאוֹת וְאֵינוֹ נֶאֱכָל\hdot
הַמַּפֶּלֶת וְאֵין יָדוּעַ מָה הִפִּילָה\hdot
וְכֵן שְׁתֵּי נָשִׁים שֶׁהִפִּילוּ\hdot
אַחַת מִמִּין פָּטוּר וְאַחַת מִמִּין חוֹבָה\hdot
אָמַר רִבִּי יוֹסֵי אֵימָתַי בִּזְמָן שֶׁהָֽלְכוּ זוֹ לְמִזְרַח וְזוֹ לְמַעֲרָב\hdot
אֲבָל אִם הָיוּ שְׁתֵּיהֶן עוֹמְדוֹת כְּאֶחָת\hdot
מְבִיאוֹת קָרְבָּן וְנֶאֱכָל׃}

\halacha{אֵלּוּ שֶׁאֵינָן מְבִיאוֹת\hdot
הַמַּפֶּלֶת שָׁפִיר מָלֵא מַיִם\hdot
מָלֵא דָם\hdot
מָלֵא גְנִינִים\hdot
הַמַּפֶּלֶת כְּמִין דָּגִים וַחֲגָבִים\hdot
שְׁקָצִים וּרְמָשׁים\hdot
הַמַפֶּלֶת יוֹם אַרְבָּעִים\hdot
וְיוֹצֵא דֹפֶן\hdot
רִבִּי שִׁמְעוֹן מְחַיֵּב בְּיוֹצֵא דֹפֶן׃}

\halacha{הַמַּפֶּלֶת אוֹר לִשְׁמוֹנִים וְאֶחָד\hdot
בֵּית שַׁמָּאי פּוֹטְרִין מִן הַקָּרְבָּן\hdot
וּבֵית הִלֵּל מְחַיְּבִין\hdot
אָֽמְרוּ בֵית הִלֵּל לְבֵית שַׁמָּאי מַאי שְׁנָא אוֹר לִשְׁמוֹנִים וְאֶחָד\hdot
מִיּוֹם שְׁמוֹנִים וְאֶחָד\hdot
אִם שָׁוֶה לוֹ לַטֻּמְאָה\hdot
לֹא יְשַׁוֶּה לוֹ לַקָּרְבָּן\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי לֹא\hdot
אִם אֲמַרְתֶּם בַּמַּפֶּלֶת יוֹם שְׁמוֹנִים וְאֶחָד\hdot
שֶׁכֵּן יָצָא בְּשָׁעָה שֶׁהִיא רְאוּיָה לְהָבִיא בָהּ קָרְבָּן\hdot
תֹּאמְרוּ בַּמַּפֶּלֶת אוֹר לִשְׁמוֹנִים וְאֶחָד\hdot
שֶׁלֹּא יָצָא בְּשָׁעָה שֶׁהִיא רְאוּיָה לְהָבִיא בָהּ קָרְבָּן\hdot
אָֽמְרוּ לָהֶם בֵּית הִלֵּל\hdot
וַהֲרֵי הַמַּפֶּלֶת יוֹם שְׁמוֹנִים וְאֶחָד\hdot
שֶׁחַל לִהְיוֹת בְּשַׁבָּת תּוֹכִיחַ\hdot
שֶׁלֹּא יָצָא בְּשָׁעָה שֶׁהִיא רְאוּיָה לְהָבִיא בָהּ קָרְבָּן\hdot
וְחַיֶּבֶת בַּקָּרְבָּן\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי לֹא\hdot
אִם אֲמַרְתֶּם בַּמַּפֶּלֶת יוֹם שְׁמוֹנִים וְאֶחָד\hdot
שֶׁחַל לִהְיוֹת בְּשַׁבָּת
שֶׁאַף עַל פִּי שֶׁאֵינוֹ רָאוּי לְקָרְבַּן יָחִיד רָאוּי לְקָרְבַּן צִבּוּר\hdot
תֹּאמְרוּ בַּמַּפֶּלֶת אוֹר לִשְׁמוֹנִים וְאֶחָד\hdot
שֶׁאֵין הַלַּיְלָה רָאוּי לֹא לְקָרְבַּן יָחִיד וְלֹא לְקָרְבַּן צִבּוּר\hdot
הַדָּמִים אֵינָן מוֹכִיחִין שֶׁהַמַּפֶּלֶת בְּתוֹךְ מְלוֹאוֹת\hdot
דָּמֶיהָ טְמֵאִין\hdot
וּפְטוּרָה מִן הַקָּרְבָּן׃}

\halacha{הָאִשָּׁה שֶׁיֵּשׁ עָלֶיהָ סָפֵק חָמֵשׁ לֵידוֹת\hdot
וְסָפֵק חָמֵשׁ זִיבוֹת\hdot
מְבִיאָה  קָרְבָּן אֶחָד וְאוֹכֶלֶת בַּזְּבָחִים\hdot
וְאֵין הַשְּׁאָר עָלֶיהָ חוֹבָה\hdot
חָמֵשׁ לֵידוֹת וַדָּאוֹת\hdot
חָמֵשׁ זִיבוֹת וַדָּאוֹת מְבִיאָה קָרְבָּן אֶחָד וְאוֹכֶלֶת בַּזְּבָחִים\hdot
וְהַשְּׁאָר עָלֶיהָ חוֹבָה\hdot
מַעֲשֶׂה שֶׁעָֽמְדוּ קִנִּים בִּירוּשָׁלַים בְּדִינָרֵי זָהָב\hdot
אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל הַמָּעוֹן הַזָּה לֹא אָלִין הַלַּיְלָה עַד שֶׁיְּהוּ בְדִינָרִין\hdot
נִכְנַס לְבֵית דִּין וְלִמֵּד\hdot
הָאִשָּׁה שֶׁיֵּשׁ עָלֶיהָ חָמֵשׁ לֵידוֹת וַדָּאוֹת\hdot
חָמֵשׁ זִיבוֹת וַדָּאוֹת\hdot
מְבִיאָה קָרְבָּן אֶחָד\hdot
וְאוֹכֶלֶת בַּזְּבָחִים\hdot
וְאֵין הַשְּׁאָר עָלֶיהָ חוֹבָה\hdot
וְעָֽמְדוּ קִנִּים בּוֹ בַיּוֹם בִּרְבַעְתָּיִם׃}
