\tractatehead{מסכת חגיגה}{hagiga}

\perek{הַכֹּל}{חַיָּיבִים בָֽרְאִיָּה\hdot
חוּץ מֵחֵרֵשׁ שׁוֹטֶה וְקָטָן\hdot
וְטוֹמְטוֹם וְאַנְדְרוֹגִינוֹס\hdot
וְנָשִׁים וַעֲבָדִים שֶׁאֵינָם מְשׁוּחְרָרִים\hdot
הַחִגֵּר\hdot
וְהַסּוֹמֵא\hdot
וְהַחוֹלֶה וְהַזָּקֵן\hdot
וּמִי שֶׁאֵינוֹ יָכוֹל לַעֲלוֹת בְּרַגְלָיו\hdot
אֵיזֶהוּ קָטָן\hdot
כָּל שֶׁאֵינוֹ יָכוֹל לִרְכּוֹב עַל כְּתֵפָיו שֶׁל אָבִיו\hdot
וְלַעֲלוֹת מִירוּשָׁלַיִם לְהַר הַבַּיִת\hdot
דִּבְרֵי בֵית שַׁמָּאי\hdot
וּבֵית הִלֵּל אוֹמְרִים כָּל שֶׁאֵינוֹ יָכוֹל לֶאֱחוֹז בְּיָדוֹ שֶׁל אָבִיו\hdot
וְלַעֲלוֹת מִירוּשָׁלַיִם לְהַר הַבַּית\hdot
שֶׁנֶּאֱמַר שָׁלֹשׁ רְגָלִים׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים הָֽרְאִיָּה שְׁתֵּי כֶסֶף\hdot
וְהַחֲגִיגִה מְעָה כֶסֶף\hdot
וּבֵית הִלֵּל אוֹמְרִים הָֽרְאִיָּה מְעָה כֶסֶף וְהַחֲגִיגָה שְׁתֵּי כֶסֶף׃}

\halacha{עוֹלוֹת בְּמוֹעֵד בָּאוֹת מִן הַחוֹלִין\hdot
וְהַשְּׁלָמִים מִן הַמַּעֲשֵׂר\hdot
יוֹם טוֹב הָרִאשׁוֹן שֶׁל פֶּסַח\hdot
בֵּית שַׁמָּאי אוֹמְרִים מִן הַחוֹלִין\hdot
וּבֵית הִלֵּל אוֹמְרִים מִן הַמַּעֲשֵׂר׃}

\halacha{יִשְׂרָאֵל יוֹצְאִין יְדֵי חוֹבָתָן בִּנְדָרִים\hdot
וּבִנְדָבוֹת\hdot
וּבְמַעְשַׂר בְּהֵמָה\hdot
וְהַכֹּהֲנִים בַּחַטָּאוֹת\hdot
וּבַאֲשָׁמוֹת\hdot
וּבַבְּכוֹר וּבַחָזֶה וָשׁוֹק\hdot
אֲבָל לֹא בָעוֹפוֹת\hdot
וְלֹא בַמְּנָחוֹת׃}

\halacha{מִי שֶׁיֵּשׁ לוֹ אוֹכְלִים מְרֻבִּים וּנְכָסִים מוּעֲטִים\hdot
מֵבִיא שְׁלָמִים מְרֻבִּים וְעוֹלוֹת מוּעֲטוֹת\hdot
נְכָסִים מְרֻבִּים וְאוֹכְלִין מוּעֲטִין\hdot
מֵבִיא עוֹלוֹת מְרֻבּוֹת וּשְׁלָמִים מוּעֲטִין\hdot
זֶה וְזֶה מוּעָט\hdot
עַל זֶה אָֽמְרוּ מְעָה כֶסֶף וּשְׁתֵּי כֶסֶף\hdot
זֶה וְזֶה מְרֻבֶּה\hdot
עַל זֶה נֶאֱמַר אִישׁ כְּמַתְּנַת יָדוֹ כְּבִרְכַּת יְיָ אֱלֹהֶיךָ אֱשֶׁר נָתַן לָךְ׃}

\halacha{מִי שֶׁלֹּא חַג בְּיוֹם טוֹב הָרִאשׁוֹן שֶׁל חַג\hdot
חוֹגֵג אֶת כָּל הָרֶגֶל\hdot
וְיוֹם טוֹב הָאֲחָרוֹן שֶׁל חַג\hdot
עָבַר הָרֶגֶל וְלֹא חַג\hdot
אֵינוֹ חַיָּיב בְּאַחֲרָיוּתוֹ\hdot
עַל זֶה נֶאֱמַר מְעֻוָּת לֹא יוּכַל לִתְקוֹן\hdot
וְחֶסְרוֹן לֹא יוּכַל לְהִמָּנוֹת׃}

\halacha{רִבִּי שִׁמְעוֹן בֶּן מְנַסְיָא אוֹמֵר אֵיזֶה מְעֻוָּת שֶׁאֵינוֹ יָכוֹל לִתְקוֹן\hdot
זֶה הַבָּא עַל הָעֶרְוָה וְהוֹלִיד מִמֶּנָּה מַמְזֵר\hdot
אִם תֹּאמַר בְּגוֹנֵב וְגוֹזֵל\hdot
יָכוֹל הוּא לְהַחְזִיר וְיִתָּקֵן\hdot
רִבִּי שִׁמְעוֹן בֶּן יוֹחָאי אוֹמֵר אֵין קוֹרִין מְעֻוָּת\hdot
אֶלָּא לְמִי שֶׁהָיָה מְתֻקָּן מִתְּחִלָּה וְנִתְעַוָּת\hdot
וְאֵיזֶה\hdot
זֶה תַּלְמִיד חָכָם הַפּוֹרֵשׂ מִן הַתּוֹרָה׃}

\halacha{הֵתֵר נְדָרִים פּוֹרְחִין בַּאֲוֵיר\hdot
וְאֵין לָהֶם עַל מַה שֶּׁיִּסְמוֹכוּ\hdot
הִלְכוֹת שַׁבָּת\hdot
חֲגִיגוֹת\hdot
וְהַמְּעִילוֹת\hdot
הֲרֵי הֵם כַּהֲרָרִים הַתְּלוּיִן בַּשַּׂעְרָה\hdot
שֶׁהֵן מִקְרָא מוּעַט וְהִלְכוֹת מְרֻבּוֹת\hdot
הַדִּינִין וְהָעֲבוֹדוֹת\hdot
הַטַּהֲרוֹת\hdot
וְהַטֻּמְאוֹת\hdot
וַעֲרָיוֹת\hdot
יֵשׁ לָהֶן עַל מַה שֶּׁיִּסְמוֹכוּ\hdot
הֵן הֵן גּוּפֵי תוֹרָה׃}

\perek{אֵין}{דּוֹרְשִׁין בָּעֲרָיוֹת בִּשְׁלֹשָׁה\hdot
וְלֹא בְמַעֲשֵׂה בְרֵאשִׁית בִּשְׁנַיִם\hdot
וְלֹא בַמֶּרְכָּבָה בְּיָחִיד\hdot
אֶלָּא אִם כֵּן הָיָה חָכָם וּמֵבִין מִדַּעְתּוֹ\hdot
כָּל הַמִּסְתַּכֵּל בְּאַרְבָּעָה דְבָרִים רָאוּי לוֹ כְּאִלּוּ לֹא בָא לָעוֹלָם\hdot
מַה לְּמַעְלָה מַה לְּמַטָּה\hdot
מַה לְּפָנִים וּמַה לְאָחוֹר\hdot
וְכָל שֶׁלֹּא חַס עַל כְּבוֹד קוֹנוֹ רָאוּי לוֹ שָלֹּא בָא לָעוֹלָם׃}

\halacha{יוֹסֵי בֶן יוֹעֶזֶר אוֹמֵר שֶׁלֹּא לִסְמוֹךְ\hdot
יוֹסֵי בֶן יוֹחָנָן אוֹמֵר לִסְמוֹךְ\hdot
יְהוֹשֻׁעַ בֶּן פְּרַחְיָה אוֹמֵר שֶׁלֹּא לִסְמוֹךְ\hdot
נִתָּאי הָאַרְבְּלִי אוֹמֵר לִסְמוֹךְ\hdot
יְהוּדָה בֶן טַבָּאי אוֹמֵר שֶׁלֹּא לִסְמוֹךְ\hdot
שִׁמְעוֹן בֶּן שָׁטָח אוֹמֵר לִסְמוֹךְ\hdot
שְׁמַעְיָה אוֹמֵר לִסְמוֹךְ\hdot
אַבְטַלְיוֹן אוֹמֵר שֶׁלֹּא לִסְמוֹךְ\hdot
הִלֵּל וּמְנַחֵם לֹא נֶחֱלָקוּ\hdot
יָצָא מְנַחֵם נִכְנַס שַׁמָּאי\hdot
שַׁמָּאי אוֹמֵר שֶׁלֹּא לִסְמוֹךְ\hdot
הִלֵּל אוֹמֵר לִסְמוֹךְ\hdot
הָרִאשׁוֹנִים הָיוּ נִשִׂיאִים וּשְׁנִיִּם לָהֶם אָבוֹת בֵּית דִּין׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים מְבִיאִין שְׁלָמִים וְאֵין סוֹמְכִין עֲלֵיהֶן\hdot
אֲבָל לֹא עוֹלוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים מְבִיאִין שְׁלָמִים וְעוֹלוֹת\hdot
וְסוֹמְכִין עֲלֵיהֶן׃}

\halacha{עֲצֶרֶת שֶׁחַל לִהְיוֹת בְּעֶרֶב שַׁבָּת\hdot
בֵּית שַׁמָּאי אוֹמְרִים יוֹם טְבוֹחַ אַחַר הַשַּׁבָּת\hdot
וּבֵית הִלֵּל אוֹמְרִים אֵין לָהּ יוֹם טְבוֹחַ\hdot
וּמוֹדִים שֶׁאִם חַל לִהְיוֹת בְּשַׁבָּת שֶׁיּוֹם טְבוֹחַ אַחַר שַׁבָּת\hdot
וְאֵין כֹּהֵן גָּדוֹל מִתְלַבֵּש בְּכֵלָיו\hdot
וּמוּתָרִין בְּהֵסְפֵּד וּבְתַעֲנִית\hdot
שֶׁלֹּא לְקַיֵּם דִּבְרֵי הָאוֹמְרִים עֲצֶרֶת אַחַר הַשַּׁבָּת׃}

\halacha{נוֹטְלִין לַיָּדַיִם לְחוֹלִין וְלַמַּעֲשֵׂר וְלַתְּרוּמָה\hdot
וְלַקֹּדֶשׁ מַטְבִּילִין\hdot
וְלַחַטָּאת אִם נִטְמְאוּ יָדָיו\hdot
נִטְמָא גוּפוֹ׃}

\halacha{הַטּוֹבֵל לְחוֹלִין וְהֻחְזַק לְחוֹלִין\hdot
אָסוּר לַמַּעֲשֵׂר\hdot
טָבַל לַמַּעֲשֵׂר וְהֻחְזַק לַמַּעֲשֵׂר\hdot
אָסוּר לַתְּרוּמָה\hdot
טָבַל לַתְּרוּמָה וְחֻחְזַק לַתְּרוּמָה\hdot
אָסוּר לַקֹּדֶשׁ\hdot
טָבַל לַקֹּדֶשׁ וְהֻחְזַק לַקֹּדֶשׁ\hdot
אָסוּר לַחַטָּאת\hdot
טָבַל לַחָמוּר מֻתָּר לַקָּל\hdot
טָבַל וְלֹא הֻחְזַק כְּאִלּוּ לֹא טָבַל׃}

\halacha{בִּגְדֵי עַם הָאָרֶץ מִדְרָס לַפְּרוּשִׁין\hdot
בִּגְדֵי פְרוּשִׁין מִדְרָס לְאוֹכְלֵי תְרוּמָה\hdot
בִּגְדֵי אוֹכְלֵי תְרוּמָה מִדְרָס לַקֹּדֶשׁ\hdot
בִּגְדֵי אוֹכְלֵי קֹדֶשׁ מִדְרָס לַחַטָּאת\hdot
יוֹסֵף בֶּן יוֹעֶזֶר הָיָה חָסִיד שֶׁבַּכְּהוּנָה\hdot
וְהָֽיְתָה מִטְפַחְתּוֹ מִדְרָס לַקֹּדֶשׁ\hdot
יוֹחָנָן בֶּן גֻדְגָדָא הָיָה אוֹכֵל עַל טַהֲרַת הַקֹּדֶשׁ כָּל יָמָיו\hdot
וְהָֽיְתָה מִטְפַחְתּוֹ מִדְרָס לַחַטָּאת׃}

\perek{חֹמֶר}{בַּקֹּדֶשׁ מִבַּתְּרוּמָה\hdot
שֶׁמַּטְבִּילִין כֵּלִים בְּתוֹךְ כֵּלִים לַתְּרוּמָה\hdot
אֲבָל לֹא לַקֹּדֶשׁ\hdot
אֲחוֹרַיִם וְתוֹךְ וּבֵית הַצְּבִיטָה בַּתְּרוּמָה אֲבָל לֹא בַקֹּדֶשׁ\hdot
הַנּוֹשֵׂא אֶת הַמִּדְרָס נוֹשֵׂא אֶת הַתְּרוּמָה\hdot
אֲבָל לֹא אֶת הַקֹּדֶשׁ\hdot
בִּגְדֵי אוֹכְלֵי תְרוּמָה מִדְרָס לַקֹּדֶשׁ\hdot
לֹא כְמִדַּת הַקֹּדֶשׁ מִדַּת הַתְּרוּמָה\hdot
שֶׁבַּקֹּדֶשׁ מַתִּיר וּמְנַגֵּב וּמַטְבִּיל וְאַחַר כַּךְ קוֹשֵׁר\hdot
וּבַתְּרוּמָה קוֹשֵׁר וְאַחַר כַּךְ מַטְבִּיל׃}

\halacha{כֵּלִים הַנִּגְמָרִים בְּטַהֲרָה צְרִיכִין טְבִילָה לַקֹּדֶשׁ אֲבָל לֹא לַתְּרוּמָה\hdot
הַכֶּלִי מְצָרֵף מַה שֶּׁבְּתוֹכוֹ לַקֹּדֶשׁ\hdot
אֲבָל לֹא לַתְּרוּמָה\hdot
הָֽרְבִיעִי בַּקֹּדֶשׁ פָּסוּל\hdot
וְהַשְּׁלִישִׁי בַּתְּרוּמָה\hdot
וּבַתְּרוּמָה אִם נִטְמֵאת אַחַת מִיָּדָיו\hdot
חֲבֶרְתָּהּ טְהוֹרָה\hdot
וּבַקֹּדֶשׁ מַטְבִּיל שְׁתֵּיהֶן\hdot
שֶׁהַיָּד מְטַמְּאָה אֶת חֲבֶרְתָּהּ בַּקֹּדֶשׁ\hdot
אֲבָל לֹא בַתְּרוּמָה׃}

\halacha{אוֹכְלִין אוֹכָלִים נְגוּבִים בְּיָדַיִם מְסוֹאָבוֹת בַּתְּרוּמָה\hdot
אֲבָל לֹא בַקֹּדֶשׁ\hdot
הָאוֹנֵן וּמְחֻסַּר כִּפּוּרִים צְרִיכִין טְבִילָה לַקֹּדֶשׁ אֲבָל לֹא לַתְּרוּמָה׃}

\halacha{חֹמֶר בַּתְּרוּמָה\hdot
שֶׁבִּיהוּדָה נֶאֱמָנִין עַל טַהֲרַת יַיִן וְשֶׁמֶן כָּל יְמוֹת הַשָּׁנָה\hdot
וּבְשַׁעַת הַגִּתּוֹת וְהַבַּדִּים אַף עַל הַתְּרוּמָה\hdot
עָֽבְרוּ הַגִּתּוֹת וְהַבַּדִּים וְהֵבִיא לוֹ חָבִית שֶׁל יַיִן שֶׁל תְּרוּמָה לֹא יְקַבְּלֶנָּה מִמֶּנּוּ\hdot
אֲבָל מַנִּיחָהּ לְגַת הַבָּאָה\hdot
וְאִם אָמַר לוֹ הִפְרַשְׁתִּי לְתוֹכָהּ רְבִיעִית קֹדֶשׁ נֶאֱמָן\hdot
כַּדֵּי יַיִן וְכַדֵּי שֶׁמֶן הַמְּדֻמָּעוֹת נֶאֱמָנִין עֲלֵיהֶם בְּשַׁעַת הַגִּתּוֹת וְהַבַּדִּים\hdot
וְקוֹדֶם לַגִּתּוֹת שִׁבְעִים יוֹם׃}
