\tractatehead{מסכת כלים}{kelim}

\perek{אֲבוֹת}{הַטֻּמְאוֹת\hdot
הַשֶּׁרֶץ\hdot
וְשִׁכְבַת זֶרַע\hdot
וּטְמֵא מֵת\hdot
וְהַמְּצוֹרָע בִּימֵי סָפְרוֹ\hdot
וּמֵי חַטָּאת שֶׁאֵין בָּהֶם כְּדֵי הַזָּיָה\hdot
הֲרֵי אֵלּוּ מְטַמְּאִין\hdot
אָדָם וְכֵלִים בְּמַגַּע\hdot
וּכְלֵי חֶרֶשׁ בָּאַוֵּיר\hdot
וְאֵינָם מְטַמְּאִין בְּמַשָּׂא׃}

\halacha{לְמַעְלָה מֵהֶם נְבֵלָה\hdot
וּמֵי חַטָּאת שֶׁיֵּשׁ בָּהֶם כְּדֵי הַזָּיָה\hdot
שֶׁהֵם מְטַמְּאִין אֶת הָאָדָם בְּמַשָּׂא\hdot
לְטַמֵּא בְגָדִים (בְּמַגַּע)\hdot
וַחֲשׂוּכֵי בְגָדִים בְּמַגַּע׃}

\halacha{לְמַעְלָה מֵהֶם בּוֹעֵל נִדָּה\hdot
שֶׁהוּא מְטַמֵּא מִשְׁכָּב תַּחְתּוֹן כָּעֶלְיוֹן\hdot
לְמַעְלָה מֵהֶן זוֹבוֹ שֶׁל זָב\hdot
וְרוֹקוֹ\hdot
וְשִׁכְבַת זַרְעוֹ\hdot
וּמֵימֵי רַגְלָיו\hdot
וְדַם הַנִּדָּה\hdot
שֶׁהֵן מְטַמְּאִין בְּמַגַּע וּבְמַשָּׂא\hdot
לְמַעְלָה מֵהֶן מֶרְכָּב\hdot
שֶׁהוּא מְטַמֵּא תַּחַת אֶבֶן מַסְמָא\hdot
לְמַעְלָה מִן הַמֶּרְכָּב מִשְׁכָּב\hdot
שֶׁשָּׁוֶה מַגָּעוֹ לְמַשָּׂאוֹ\hdot
לְמַעְלָה מִן הַמִּשְׁכָּב הַזָּב\hdot
שֶׁהַזָּב עוֹשֶׂה מִשְׁכָּב\hdot
וְאֵין מִשְׁכָּב עוֹשֶׂה מִשְׁכָּב׃}

\halacha{לְמַעְלָה מִן הַזָּב זָבָה\hdot
שֶׁהִיא מְטַמְּאָה אֶת בּוֹעֲלָהּ\hdot
לְמַעְלָה מִן הַזָּבָה מְצוֹרָע\hdot
שֶׁהוּא מְטַמֵּא בְּבִיאָה\hdot
לְמַעְלָה מִן הַמְּצוֹרָע עֶצֶם כַּשְּׂעוֹרָה\hdot
שֶׁהוּא מְטַמֵּא טֻמְאַת שִׁבְעָה\hdot
חָמוּר מִכֻּלָּם הַמֵּת\hdot
שֶׁהוּא מְטַמֵּא בָאֹהֶל\hdot
מַה שֶּׁאֵין כֻּלָּם מְטַמְּאִין׃}

\halacha{עֶשֶׂר טֻמְאוֹת פּוֹרְשׁוֹת מִן הָאָדָם\hdot
מְחֻסַּר כִּפּוּרִים אָסוּר בַּקֹּדֶשׁ\hdot
וּמוּתָר בַּתְּרוּמָה וּבַמַּעֲשֵׂר\hdot
חָזַר לִהְיוֹת טְבוּל יוֹם אָסוּר בַּקֹּדֶשׁ וּבַתְּרוּמָה\hdot
וּמוּתָר בַּמַּעֲשֵׂר\hdot
חָזַר לִהְיוֹת בַּעַל קֶרִי\hdot
אָסוּר בִּשְׁלֹשְׁתָּן\hdot
חָזַר לִהְיוֹת בּוֹעֵל נִדָּה\hdot
מְטַמֵּא מִשְׁכָּב תַּחְתּוֹן כָּעֶלְיוֹן\hdot
חָזַר לִהְיוֹת זָב\hdot
שֶׁרָאָה שְׁתֵּי רְאִיּוֹת\hdot
מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב\hdot
וְצָרִיךְ בִּיאַת מַיִם חַיִּים\hdot
וּפָטוּר מִן הַקָּרְבָּן\hdot
רָאָה שָׁלֹשׁ\hdot
חַיָּב בַּקָּרְבָּן\hdot
חָזַר לִהְיוֹת מְצוֹרָע מֻסְגָּר\hdot
מְטַמֵּא בְּבִיאָה\hdot
וּפָטוּר מִן הַפְּרִיעָה וּמִן הַפְּרִימָה\hdot
וּמִן הַתִּגְלַחַת\hdot
וּמִן הַצִּפֳּרִים\hdot
אִם הָיָה מֻחְלָט חַיָּב בְּכֻלָּן\hdot
פֵּירַשׁ מִמֶּנּוּ אֵבֶר שֶׁאֵין עָלָיו בָּשָׂר כָּרָאוּי\hdot
מְטַמֵּא בְּמַגַּע וּבְמַשָּׂא\hdot
וְאֵינוֹ מְטַמֵּא בָאֹהֶל\hdot
וְאִם יֵשׁ עָלָיו בָּשָׂר כָּרָאוּי\hdot
מְטַמֵּא בְמַגַּע וּבְמַשָּׂא וּבָאֹהֶל\hdot
שִׁיעוּר בָּשָׂר כָּרָאוּי\hdot
כְּדֵי לְהַעֲלוֹת אֲרוּכָה\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם יֵשׁ בְּמָקוֹם אֶחָד\hdot
כְּדֵי לְהַקִּיפוֹ בְּחוּט עֶרֶב\hdot
יֵשׁ בּוֹ לְהַעֲלוֹת אֲרוּכָה׃}

\halacha{עֶשֶׂר קְדוּשׁוֹת הֵן\hdot
אֶרֶץ יִשְׂרָאֵל מְקֻדֶּשֶׁת מִכָּל הָאֲרָצוֹת\hdot
וּמָה הִיא קְדוּשָׁתָהּ\hdot
שֶׁמְּבִיאִים מִמֶּנּוּ הָעֹמֶר וְהַבִּכּוּרִים וּשְׁתֵּי הַלֶּחֶם\hdot
מַה שֶּׁאֵין מְבִיאִים כֵּן מִכָּל הָאֲרָצוֹת׃}

\halacha{עֲיָרוֹת הַמֻּקָּפוֹת חוֹמָה מְקֻדָּשׁוֹת מִמֶּנָּה\hdot
שֶׁמְּשַׁלְּחִין מִתּוֹכָן הַמְּצוֹרָעִים\hdot
וּמְסַבְּכִין לְתוֹכָן מֵת\hdot
עַד שֶׁיִּרְצוּ\hdot
יָצָא אֵין מַחֲזִירִין אוֹתוֹ׃}

\halacha{לִפְנִים מִן הַחוֹמָה מְקֻדָּשׁ מֵהֶם\hdot
שֶׁאוֹכְלִים שָׁם קָדָשִׁים קַלִּים וּמַעֲשֵׂר שֵׁנִי\hdot
הַר הַבַּיִת מְקֻדָּשׁ מִמֶּנּוּ\hdot
שֶׁאֵן זָבִים וְזָבוֹת נִדּוֹת וְיוֹלְדוֹת נִכְנָסִים לְשָׁם\hdot
הַחֵיל מְקֻדָּשׁ מִמֶּנּוּ\hdot
שֶׁאֵין הַנָּכְרִי וּטְמֵא מֵת נִכְנָסִים לְשָׁם\hdot
עֲזָרַת נָשִׁים מְקֻדֶּשֶׁת מִמֶּנּוּ\hdot
שֶׁאֵין טְבוּל יוֹם נִכְנָס לְשָׁם\hdot
וְאֵין חַיָּבִים עָלֶיהָ חַטָּאת\hdot
עֲזָרַת יִשְׁרָאֵל מְקֻדֶּשֶׁת מִמֶּנָּה\hdot
שֶׁאֵין מְחֻסַּר כִּפּוּרִים נִכְנָס לְשָׁם\hdot
וְחַיָּבִין עָלֶיהָ חַטָּאת\hdot
עֲזָרַת כֹּהֲנִים מְקֻדֶּשֶׁת מִמֶּנָּה\hdot
שֶׁאֵין יִשְׂרָאֵל נִכְנָסִים לְשָׁם\hdot
אֶלָּא בְּשַׁעַת צָרְכֵיהֶם\hdot
לִסְמִיכָה וְלִשְׁחִיטָה וְלִתְנוּפָה׃}

\halacha{בֵּין הָאוּלָם וְלַמִּזְבֵּחַ מְקֻדָּשׁ מִמֶּנָּה\hdot
שֶׁאֵין בַּעֲלֵי מוּמִים וּפְרוּעֵי רֹאשׁ נִכְנָסִים לְשָׁם\hdot
הַהֵיכָל מְקֻדָּשׁ מִמֶּנּוּ\hdot
שֶׁאֵין נִכְנָס לְשָׁם\hdot
אֶלָּא רְחוּץ יָדַיִם וְרַגְלַיִם\hdot
קֹדֶשׁ הַקֳדָשִׁים מְקֻדָּשׁ מֵהֶם\hdot
שֶׁאֵין נִכְנָס לְשָׁם אֶלָּא כֹּהֵן גָּדוֹל\hdot
בְּיוֹם הַכִּפּוּרִים בְּשַׁעַת הָעֲבוֹדָה\hdot
אָמַר רִבִּי יוֹסֵי בַּחֲמִשָּׁה דְבָרִים בֵּין הָאוּלָם וְלַמְּזִבֵּחַ\hdot
שָׁוֶה לַהֵיכָל\hdot
שֶׁאֵין בַּעֲלֵי מוּמִים\hdot
וּפְרוּעֵי רֹאשׁ\hdot
וּשְׁתוּיֵי יַיִן\hdot
וְשֶׁלֹּא רְחוּץ יָדַיִם וְרַגְלַיִם\hdot
נִכְנָסִים לְשָׁם\hdot
וּפוֹרְשִׁין מִבֵּין הָאוּלָם וְלַמִּזְבֵּחַ\hdot
בְּשַׁעַת הַקְטָרָה׃}

\perek{כְּלֵי}{עֵץ וּכְלֵי עוֹר\hdot
וּכְלֵי עֶצֶם\hdot
וּכְלֵי זְכוּכִית\hdot
פְּשׁוּטֵיהֶם טְהוֹרִים\hdot
וּמְקַבְּלֵיהֶן טְמֵאִים\hdot
נִשְׁבְּרוּ טָהָרוּ\hdot
חָזַר וְעָשָׁה מֵהֶם כֵּלִים\hdot
מְקַבְּלִים טֻמְאָה מִכַּאן וּלְהַבָּא\hdot
כְּלֵי חֶרֶס וּכְלֵי נֶתֶר טֻמְאָתָן שָׁוָה\hdot
מִטַּמְּאִין\hdot
וּמְטַמְּאִין בַּאֲוֵיר וּמִטְּמְּאִין מֵאֲחוֹרֵיהֶן\hdot
וְאֵינָן מִטַּמְּאִין מִגַּבֵּיהֶן\hdot
וּשְׁבִירָתָן הִיא טַהֲרָתָן׃}

\halacha{הַדַּקִּין שֶׁבִּכְלֵי חֶרֶס\hdot
וְקַרְקְרוֹתֵיהֶן וְדַפְנוֹתֵיהֶן\hdot
יוֹשְׁבִין שֶׁלֹּא מֻסְמָכִין\hdot
שִׁיעוּרָן מִכְּדֵי סִיכַת קָטָן וְעַד לוֹג\hdot
מִלּוֹג וְעַד סֵאָה בָּֽרְבִיעִית\hdot
מִסֵּאָה וְעַד סָאתַיִם בַּחֲצִי לוֹג\hdot
מְסָּאתַיִם וְעַד שָׁלֹשׁ וְעַד חָמֵשׁ סְאִין בַּלּוֹג\hdot
דִּבְרֵי רִבִּי יִשְׁמָעֵאל\hdot
רִבִּי עֲקִיבָא אוֹמֵר אֲנִי אֵינִי נוֹתֵן בָּהֶן מִדָּה\hdot
אֶלָּא הַדַּקִּין שֶׁבִּכְלֵי חֶרֶס\hdot
וְקַרְקְרוֹתֵיהֶן וְדַפְנוֹתֵיהֶן\hdot
יוֹשְׁבִין שֶׁלֹּא מֻסְמָכִין\hdot
שִׁיעוּרָן מִכְּדֵי סִיכַת קָטָן וְעַד קְדֵרוֹת הַדַּקּוֹת\hdot
מֵהַקְּדֵירוֹת הַדַּקּוֹת\hdot
וְעַד חֲבִיּוֹת לוּדִיּוֹת בָּֽרְבִיעִית\hdot
מִלּוּדִיּוֹת וְעַד לַחֲמִיּוֹת\hdot
בַּחֲצִי לוֹג\hdot
מִלַּחֲמִיּוֹת וְעַד חֲצָבִים גּדֹולִים בַּלּוֹג\hdot
רַבָּן יוֹחָנָן בֶּן זַכָּאי אוֹמֵר חֲצָבִים גְּדוֹלִים\hdot
שִׁיעוּרָן בִּשְׁנֵי לוֹגִין\hdot
הַפָּכִים הַגְּלִילִים וְהַחֲבִיּוֹנוֹת\hdot
שִׁעוּר קַרְקְרוֹתֵיהֶן כָּל שֶׁהֵן\hdot
וְאֵין לָהֶם דְּפָנוֹת׃}

\halacha{הַטְּהוֹרִין שֶׁבִּכְלֵי חֶרֶס\hdot
טַבְלָא שֶׁאֵין לָהּ לַזְבֵּז\hdot
מְחִתָּה פְרוּצָה\hdot
וְאָבוּב שֶׁל קַלָּאִין\hdot
וְסִלּוֹנוֹת אַף עַל פִּי כְפוּפִין\hdot
וְאַף עַל פִּי מְקַבְּלִין\hdot
וְכָבְכָּב שֶׁעֲשָׂאוֹ לְסַל הַפַּת\hdot
וְטָפִי שֶׁהִתְקִינוּ לַעֲנָבִים\hdot
וְחָבִית שֶׁל שְׁיָטִין\hdot
וְחָבִית דְּפוּנָה בְשׁוּלֵי הַמַּחַץ\hdot
וְהַמִּטָּה וְהַכִּסֵּא וְהַסַּפְסָל\hdot
וְהַשֻּׁלְחָן\hdot
וְהַסְּפִינָה\hdot
וְהַמְּנוֹרָה שֶׁל חֶרֶס\hdot
הֲרֵי אֵלּוּ טְהוֹרִים\hdot
זֶה הַכְּלָל\hdot
כָּל שֶׁאֵין לוֹ תוֹךְ בִּכְלֵי חֶרֶס\hdot
אֵין לוֹ אֲחוֹרַיִם׃}

אֵין מַקִּיפִין לֹא בְּבַעַץ\hdot
וְלֹא בְּעוֹפֶרֶת\hdot
מִפְּנֵי שֶׁהוּא פָּתִיל וְאֵינוֹ צָמִיד\hdot
אֵין מַקִּיפִין לֹא בִדְבֵילָה שְׁמֵנָה\hdot
וְלֹא בְבָצֵק שֶׁנִּלּוֹשׁ בְּמֵי פֵירוֹת\hdot
