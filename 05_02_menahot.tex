\tractatehead{מסכת מנחות}{menahot}

\perek{כָּל}{הַמְּנָחוֹת שֶׁנִּקְמְצוּ שֶׁלֹּא לִשְׁמָן\hdot
כְּשֵׁרוֹת\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִמִּנְחַת חוֹטֵא וּמִנְחַת קְנָאוֹת\hdot
מִנְחַת חוֹטֵא וּמִנְחַת קְנָאוֹת שֶׁקְּמָצָן שֶׁלֹּא לִשְׁמָן\hdot
נָתַן בַּכֶּלִי וְהָלַךְ וְהִקְטִיר שֶׁלֹּא לִשְׁמָן\hdot
אוֹ לִשְׁמָן וְשֶׁלֹּא לִשְׁמָן\hdot
אוֹ שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן פְּסוּלוֹת\hdot
כֵּיצַד לִשְׁמָן\hdot
וְשֶׁלֹּא לִשְׁמָן\hdot
לְשֵׁם מִנְחַת חוֹטֵא\hdot
וּלְשֵׁם מִנְחַת נְדָבָה\hdot
אוֹ שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן\hdot
לְשֵׁם מִנְחַת נְדָבָה\hdot
וּלְשֵׁם מִנְחַת חוֹטֵא׃}

\halacha{אֶחָד מִנְחַת חוֹטֵא\hdot
וְאֶחָד כָּל הַמְּנָחוֹת שֶׁקְּמָצָן זָר\hdot
אוֹנֵן\hdot
טְבוּל יוֹם\hdot
מְחֻסַּר בְּגָדִים\hdot
מְחֻסַּר כִּפּוּרִים\hdot
שֶׁלֹּא רְחוּץ יָדַיִם וְרַגְלַיִם\hdot
עָרֵל טָמֵא יוֹשֵׁב\hdot
עוֹמֵד עַל גַּבֵּי כֵלִים\hdot
עַל גַּבֵּי בְהֵמָה עַל גַּבֵּי רַגְלֵי חֲבֵרוֹ פָּסַל\hdot
קָמַץ בַּשְּׂמֹאל פָּסַל\hdot
בֵּן בְּתֵירָא אוֹמֵר יַחֲזִיר וְיַחֲזוֹר וְיִקְמוֹץ בְּיָמִין\hdot
קָמַץ וְעָלָה בְיָדוֹ צְרוֹר\hdot
אוֹ גַרְגִּיר מֶלַח\hdot
אוֹ קֹרֶט שֶׁל לְבוֹנָה פָּסַל\hdot
מִפְּנֵי שְׁאָֽמְרוּ הַקֹּמֶץ הַיָּתֵר וְהֶחָסֵר פָּסוּל\hdot
אֵיזֶהוּ הַיָּתֵר שֶׁקְּמָצוֹ מְבוֹרָץ\hdot
וְחָסֵר שֶׁקְּמָצוֹ בְּרָאשֵׁי אֶצְבְּעוֹתָיו\hdot
כֵּיצַד הוּא עוֹשֶׂה\hdot
פּוֹשֵׁט אֶת אֶצְבְּעוֹתָיו עַל פַּס יָדוֹ׃}

\halacha{רִבָּה שַׁמְנָהּ\hdot
וְחִסֵּר שַׁמְנָהּ\hdot
חִסֵּר לְבוֹנָתָהּ פְּסוּלָה\hdot
הַקּוֹמֵץ אֶת הַמִּנְחָה לֶאֱכוֹל שִׁירֶיהָ בַּחוּץ\hdot
אוֹ כַּזַּיִת מִשִּׁירֶיהָ בַּחוּץ\hdot
לְהַקְטִיר קֻמְצָהּ בַּחוּץ\hdot
אוֹ כַּזַּיִת מִקֻּמְצָהּ בַּחוּץ\hdot
אוֹ לְהַקְטִיר לְבוֹנָתָהּ בַּחוּץ\hdot
פָּסוּל וְאֵין בּוֹ כָּרֵת\hdot
לֶאֱכוֹל שִׁירֶיהָ לְמָחָר אוֹ כַּזַּיִת מִשִּׁירֶיהָ לְמָחָר\hdot
לְהַקְטִיר קֻמְצָהּ לְמָחָר אוֹ כַּזַּיִת מִקֻּמְצָהּ לְמָחָר\hdot
אוֹ לְהַקְטִיר לְבוֹנָתָהּ לְמָחָר\hdot
פּגּוּל וְחַיָּבִין עָלָיו כָּרֵת\hdot
זֶה הַכְּלָל\hdot
כָּל הַקּוֹמֵץ וְהַנּוֹתֵן בֶּכֶּלִי\hdot
וְהַמְּהַלֵּךְ וְהַמַּקְטִיר\hdot
לֶאֱכוֹל דָּבָר שֶׁדַּרְכּוֹ לֶאֱכוֹל\hdot
וּלְהַקְטִיר דָּבָר שֶׁדַּרְכּוֹ לְהַקְטִיר\hdot
חוּץ לִמְקוֹמוֹ פָּסוּל וְאֵין בּוֹ כָּרֵת\hdot
חוּץ לִזְמַנּוֹ פִּגּוּל וְחַיָּבִין עָלָיו כָּרֵת\hdot
וּבִלְבַד שֶׁיִּקָּרֵב הַמַּתִּיר כְּמִצְוָתוֹ\hdot
כֵּיצַד קָרַב הַמַּתִּיר כְּמִצְוָתוֹ\hdot
קָמַץ בִּשְׁתִּיקָה וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ לְהַקְטִיר חוּץ לִזְמַנּוֹ\hdot
אוֹ שֶׁקָּמַץ חוּץ לִזְמַנּוֹ וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר בִּשְׁתִּיקָה\hdot
אוֹ שֶׁקָּמַץ וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר חוּץ לִזְמַנּוֹ\hdot
זֶה הוּא שֶׁקָרַב הַמַּתִּיר כְּמִצְוָתוֹ׃}

\halacha{כֵּיצַד לֹא קָרַב הַמַּתִּיר כְּמִצְוָתוֹ\hdot
קָמַץ חוּץ לִמְקוֹמוֹ\hdot
וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר חוּץ לִזְמַנּוֹ\hdot
אוֹ שֶׁקָּמַץ חוּץ לִזְמַנּוֹ\hdot
וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר חוּץ לִמְקוֹמוֹ\hdot
אוֹ שֶׁקָּמַץ וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר חוּץ לִמְקוֹמוֹ\hdot
מִנְחַת חוֹטֵא וּמִנְחַת קְנָאוֹת שֶׁקְּמָצָן שֶׁלֹּא לִשְׁמָן\hdot
וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר חוּץ לִזְמַנָּן\hdot
אוֹ שֶׁקָּמַץ חוּץ לִזְמַנָּן\hdot
וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר שֶׁלֹּא לִשְׁמָן\hdot
אוֹ שֶׁקָּמַץ וְנָתַן בַּכֶּלִי\hdot
וְהָלַךְ וְהִקְטִיר שֶׁלֹּא לִשְׁמָן\hdot
זֶה הוּא שֶׁלֹּא קָרַב הַמַּתִּיר כְּמִצְוָתוֹ\hdot
לֶאֱכוֹל כַּזַּיִת בַּחוּץ וְכַזַּיִת לְמָחָר\hdot
כַּזַּיִת לְמָחָר וְכַזַּיִת בַּחוּץ\hdot
כַּחֲצִי זַיִת בַּחוּץ וְכַחֲצִי זַיִת לְמָחָר\hdot
כַּחֲצִי זַיִת לְמָחָר וְכַחֲצִי זַיִת בַּחוּץ\hdot
 פָּסוּל וְאֵין בּוֹ כָרֵת\hdot
אָמַר רִבִּי יְהוּדָה זֶה הַכְּלָל אִם מַחֲשֶׁבֶת הַזְּמָן קָֽדְמָה לְמַחֲשֶׁבֶת הַמָּקוֹם\hdot
פִּגּוּל וְחַיָּבִין עָלָיו כָּרֵת\hdot
וְאִם מַחֲשֶׁבֶת הַמָּקוֹם\hdot
קָֽדְמָה לְמַחֲשֶׁבֶת הַזְּמָן\hdot
פָּסוּל וְאֵין בּוֹ כָרֵת\hdot
וַחֲכָמִים אוֹמְרִים זֶה וְזֶה פָּסוּל וְאֵין בּוֹ כָרֵת\hdot
לֶאֱכוֹל כַּחֲצִי זַיִת\hdot
וּלְהַקְטִיר כַּחֲצִי זַיִת כָּשֵׁר\hdot
שֶׁאֵין אֲכִילָה וְהַקְטָרָה מִצְטָֽרְפִין׃}
