\tractatehead{מסכת נזיר}{nazir}

\perek{כָּל}{כִּנּוּיֵי נְזִירוּת\hdot
כִּנְזִירוּת\hdot
הָאוֹמֵר אֲהֵא הֲרֵי זֶה נָזִיר\hdot
אוֹ אֲהֵא נָוֶה\hdot
נָזִיר\hdot
נָזִיק\hdot
נָזִיחַ\hdot
פָּזִיחַ\hdot
הֲרֵי זֶה נָזִיר\hdot
הֲרֵינִי כַזֶּה\hdot
הֲרֵינִי מְסַלְסֵל\hdot
הֲרֵינִי מְכַלְכֵּל\hdot
הֲרֵי עָלַי לִשְׁלוֹחַ פֶּרַע\hdot
הֲרֵי זֶה נָזִיר\hdot
הֲרֵי עָלַי צִפּוֹרִים\hdot
רִבִּי מֵאִיר אוֹמֵר נָזִיר\hdot
וַחֲכָמִים אוֹמְרִים אֵינוֹ נָזִיר׃}

\halacha{הֲרֵינִי נָזִיר מִן הַחַרְצַנִּים וּמִן הַזַּגִּים וּמִן הַתִּגְלַחַת\hdot
וּמִן הַטֻּמְאָה\hdot
הֲרֵי זֶה נָזִיר\hdot
וְכָל דִּקְדוּקֵי נְזִירוּת עָלָיו\hdot
הֲרֵינִי כְשִׁמְשׁוֹן\hdot
כְּבֶן מָנוֹחַ\hdot
כְּבַעַל דְּלִילָה\hdot
כְּמִי שֶׁעָקַר דַּלְתּוֹת עַזָּה\hdot
כְּמִי שֶׁנָּֽקְרוּ פְלִשְׁתִּים אֶת עֵינָיו\hdot
הֲרֵי זֶה נְזִיר שְׁמְשׁוֹן\hdot
מַה בֵּין נְזִיר עוֹלָם לִנְזִיר שִׁמְשׁוֹן\hdot
נְזִיר עוֹלָם הִכְבִּיד שְׂעָרוֹ מֵיקֵל בְּתַעַר\hdot
וּמֵבִיא שָׁלֹשׁ בְּהֵמוֹת\hdot
וְאִם נִטְמָא מֵבִיא קָרְבַּן טֻמְאָה\hdot
נְזִיר שִׁמְשׁוֹן הִכְבִּיד שְׂעָרוֹ אֵינוֹ מֵיקֵל\hdot
וְאִם נִטְמָא אֵינוֹ מֵבִיא קָרְבָּן טֻמְאָה׃}

\halacha{סְתָם נְזִירוּת שְׁלֹשִׁים יוֹם\hdot
אָמַר הֲרֵינִי נָזִיר אַחַת גְּדוֹלָה\hdot
הֲרֵינִי נָזִיר אַחַת קְטַנָּה\hdot
אֲפִילוּ מִכָּאן וְעַד סוֹף הָעוֹלָם\hdot
נָזִיר שְׁלֹשִׁים יוֹם\hdot
הֲרֵינִי נָזִיר וְיוֹם אֶחָד\hdot
הֲרֵינִי נָזִיר וְשָׁעָה אַחַת\hdot
הֲרֵינִי נָזִיר אַחַת וּמֶחְצָה\hdot
הֲרֵי זֶה נָזִיר שְׁתַּיִם\hdot
הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם וְשָׁעָה אַחַת\hdot
נָזִיר שְׁלֹשִׁים וְאֶחָד יוֹם\hdot
שֶׁאֵין נוֹזְרִים לְשָׁעוֹת׃}

\halacha{הֲרֵינִי נָזִיר כִּשְׂעַר רֹאשִׁי\hdot
וְכַעֲפַר הָאָרֶץ\hdot
וּכְחוֹל הַיָּם\hdot
הֲרֵי זֶה נְזִיר עוֹלָם\hdot
וּמְגַלֵּחַ אַחַת לִשְׁלֹשִׁים יוֹם\hdot
רִבִּי אוֹמֵר אֵין זֶה מְגַלֵּחַ אַחַת לִשְׁלֹשִׁים יוֹם\hdot
וְאֵיזֶה שֶׁמְּגַלֵּחַ אַחַת לִשְׁלֹשִׁים יוֹם\hdot
הָאוֹמֵר הֲרֵי עָלַי נְזִירוּת כִּשְׂעַר רֹאשִׁי\hdot
וְכַעֲפַר הָאָרֶץ\hdot
וּכְחוֹל  הַיָּם׃}

\halacha{הֲרֵינִי נָזִיר מְלֹא הַבַּיִת\hdot
אוֹ מְלֹא הַקֻּפָּה\hdot
בּוֹדְקִין אוֹתוֹ\hdot
אִם אָמַר אַחַת גְּדוֹלַה נָזַרְתִּי\hdot
נָזִיר שְׁלֹשִׁים יוֹם\hdot
וְאִם אָמַר סְתָם נָזַרְתִּי\hdot
רוֹאִין אֶת הַקֻּפָּה כְּאִלּוּ הִיא מְלֵאָה חַרְדָּל\hdot
וְנָזִיר כָּל יָמָיו׃}

\halacha{הֲרֵינִי נָזִיר מִכָּאן עַד מָקוֹם פְּלוֹנִי\hdot
אוֹמְדִים כַּמָּה יָמִים מִכָּאן עַד מָקוֹם פְּלוֹנִי\hdot
אִם פָּחוּת מִשְּׁלֹשִׁים יוֹם נָזִיר שְׁלֹשִׁים יוֹם\hdot
וְאִם לָאוּ נָזִיר כְּמִנְיַן הַיָּמִים׃}

\halacha{הֲרֵינִי נָזִיר כְּמִנְיַן יְמוֹת הַחַמָּה\hdot
מוֹנֶה נְזִירוּת כְּמִנְיַן יְמוֹת הַחַמָּה\hdot
אָמַר רִבִּי יְהוּדָה מַעֲשֶׂה הָיָה\hdot
וְכֵיוַן שֶׁהִשְׁלִים מֵת׃}

\perek{הֲרֵינִי}{נָזִיר מִן הַגְּרוֹגְרוֹת\hdot
וּמִן הַדְּבֵילָה\hdot
בֵּית שַׁמָּאי אוֹמְרִים נָזִיר\hdot
וּבֵית הִלֵּל אוֹמְרִים אֵינוֹ נָזִיר\hdot
אָמַר רִבִּי יְהוּדָה אַף כְּשֶׁאָֽמְרוּ בֵית שַׁמָּאי לֹא אָֽמְרוּ\hdot
אֶלָּא בְּאוֹמֵר הֲרֵי הֵן עָלַי כְּקָרְבָּן׃}

\halacha{אָמַר\hdot
אָֽמְרָה פָרָה זוֹ הֲרֵינִי נְזִירָה אִם עוֹמֶדֶת (אָנִי)\hdot
אָמַר הַדֶּלֶת הַזֶּה\hdot
הֲרֵינִי נָזִיר אִם נִפְתַּח אָנִי\hdot
בֵּית שַׁמָּאי אוֹמְרִים נָזִיר\hdot
וּבֵית הִלֵּל אוֹמְרִים אֵינוֹ נָזִיר\hdot
אָמַר רִבִּי יְהוּדָה\hdot
אַף כְּשֶׁאָֽמְרוּ בֵית  שַׁמָּאי לֹא אָֽמְרוּ\hdot
אֶלָּא בְּאוֹמֵר הֲרֵי פָרָה זוֹ קָרְבָּן עָלַי\hdot
אִם עוֹמֶדֶת הִיא׃}

\halacha{מָֽזְגוּ לוֹ אֶת הַכּוֹס\hdot
וְאָמַר הֲרֵינִי נָזִיר מִמֶּנּוּ\hdot
הֲרֵי זֶה נָזִיר\hdot
מַעֲשֶׂה בְּאִשָּׁה אַחַת שֶׁהָֽיְתָה שִׁכּוֹרָה\hdot
וּמָֽזְגוּ לָהּ אֶת הַכּוֹס\hdot
וְאָֽמְרָה הֲרֵינִי נְזִירָה מִמֶּנּוּ\hdot
אָֽמְרוּ חֲכָמִים לֹא נִתְכַּוְּנָה זוֹ\hdot
אֶלָּא לוֹמַר הֲרֵי הוּא עָלַי קָרְבָּן׃}

\halacha{הֲרֵינִי נָזִיר\hdot
עַל מְנַת  שֶׁאֶהֵא שׁוֹתֶה יַיִן וּמִטַּמֵּא לַמֵּתִים\hdot
הֲרֵי זֶה נָזִיר\hdot
וְאָסוּר בְּכֻלָּן\hdot
יוֹדֵעַ אֲנִי שֶׁיֵּשׁ נְזִירוּת\hdot
אֲבָל אֵינִי יוֹדֵעַ שֶׁהַנָּזִיר אָסוּר בַּיַּיִן\hdot
הֲרֵי זֶה אָסוּר\hdot
וְרִבִּי שִׁמְעוֹן מַתִּיר\hdot
יוֹדֵעַ אֲנִי שֶׁהַנָּזִיר אָסוּר בַּיַּיִן\hdot
אֲבָל סָבוּר הָיִיתִי שֶׁחֲכָמִים מַתִּירִין לִי\hdot
מִפְּנֵי שֶׁאֵין אֲנִי יָכוֹל לִחְיוֹת אֶלָּא בַיַּיִן\hdot
אוֹ מִפְּנֵי שֶׁאֲנִי קוֹבֵר אֶת הַמֵּתִים\hdot
הֲרֵי זֶה מוּתָר\hdot
וְרִבִּי שִׁמְעוֹן אוֹסֵר׃}

\halacha{הֲרֵינִי נָזִיר וְעָלַי לְגַלֵּחַ נָזִיר\hdot
וְשָׁמַע חֲבֵירוֹ וְאָמַר וַאֲנִי וְעָלַי לְגַלֵּחַ נָזִיר\hdot
אִם הָיוּ פִקְחִים מְגַלְּחִים זֶה אֶת זֶה\hdot
וְאִם לָאוּ מְגַלְּחִים נְזִירִים אֲחֵרִים׃}

\halacha{הֲרֵי עָלַי לְגַלֵּחַ חֲצִי נָזִיר\hdot
וְשָׁמַע חֲבֵירוֹ וְאָמַר\hdot
וַאֲנִי וְעָלַי לְגַלֵּחַ חֲצִי נָזִיר\hdot
זֶה מְגַלֵּחַ נָזִיר שָׁלֵם\hdot
וְזֶה מְגַלֵּחַ נָזִיר שָׁלֵם\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים זֶה מְגַלֵּחַ חֲצִי נָזִיר\hdot
וְזֶה מְגַלֵּחַ חֲצִי נָזִיר׃}

\halacha{הֲרֵינִי נָזִיר לִכְשֶׁיִּהְיֶה לִי בֵן\hdot
וְנוֹלַד לוֹ בֵן\hdot
הֲרֵי זֶה נָזִיר\hdot
נוֹלַד לוֹ בַת טוֹמְטוֹם וְאַנְדְרוֹגִינוֹס\hdot
אֵינוֹ נָזִיר\hdot
אִם אָמַר כְּשֶׁאֶרְאֶה (כְּשֶׁיִּהְיֶה) לִי וָלֶד\hdot
אֲפִילוּ נוֹלַד לוֹ בַת טוֹמְטוֹם וְאַנְדְרוֹגִינוֹס\hdot
הֲרֵי זֶה נָזִיר׃}

\halacha{הִפִּילָה אִשְׁתּוֹ אֵינוֹ נָזִיר\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר יֹאמַר\hdot
אִם הָיָה בֶן קַיָּמָא\hdot
הֲרֵי אֲנִי נְזִיר חוֹבָה\hdot
וְאִם לָאוּ\hdot
הֲרֵי אֲנִי נְזִיר נְדָבָה\hdot
חָֽזְרָה וְיָֽלְדָה\hdot
הֲרֵי זֶה נָזִיר\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר יֹאמַר\hdot
אִם הָרִאשׁוֹן בֶּן קַיָּמָא הָרִאשׁוֹן חוֹבָה\hdot
וְזוֹ נְדָבָה\hdot
וְאִם לָאוּ הָרִאשׁוֹן נְדָבָה\hdot
וְזוֹ חוֹבָה׃}
