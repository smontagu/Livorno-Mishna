\tractatehead{מסכת בכורות}{bechorot}

\perek{הַלּוֹקֵחַ}{עוֹבַר חֲמוֹרוֹ שֶׁל נָכְרִי\hdot
וְהַמּוֹכֵר לוֹ אַף עַל פִּי שֶׁאֵינוֹ רַשָּׁאי\hdot
וְהַמִּשְׁתַּתֵּף לוֹ וְהַמְּקַבֵּל מִמֶּנּוּ\hdot
וְהַנּוֹתֵן לוֹ בְקַבָּלָה\hdot
פָּטוּר מִן הַבְּכוֹרָה\hdot
שֶׁנֶּאֱמַר בְּיִשְׂרָאֵל\hdot
אֲבָל לֹא בָאֲחֵרִים\hdot
כֹּהֲנִים וּלְוִיִּם פְּטוּרִין מִקַּל וָחֹמֶר\hdot
אִם פָּֽטְרוּ שֶׁל יִשְׂרָאֵל בַּמִּדְבָּר\hdot
דִּין הוּא שֶׁיִּפְטְרוּ אֶת עַצְמָן׃}

\halacha{פָּרָה שֶׁיָּֽלְדָה כְּמִין חֲמוֹר\hdot
וַחֲמוֹר שֶׁיָּֽלְדָה כְּמִין סוּס\hdot
פָּטוּר מִן הַבְּכוֹרָה\hdot
שֶׁנֶּאֱמַר פֶּטֶר חֲמוֹר פֶּטֶר חֲמוֹר שְׁנֵי פְעָמִים\hdot
עַד שֶׁיְּהֵא הַיּוֹלֵד חֲמוֹר\hdot
וְהַנּוֹלָד חֲמוֹר\hdot
וּמַה הֵם בַּאֲכִילָה\hdot
בְּהֵמָה טְהוֹרָה שֶׁיָּֽלְדָה כְּמִין בְּהֵמָה טְמֵאָה\hdot
מֻתָּר בַּאֲכִילָה\hdot
וּטְמֵאָה שֶׁיָּֽלְדָה כְּמִין בְּהֵמָה טְהוֹרָה\hdot
אָסוּר בַּאֲכִילָה\hdot
שֶׁהַיּוֹצֵא מִן הַטָּמֵא טָמֵא\hdot
וְהַיּוֹצֵא מִן הַטָּהוֹר טָהוֹר\hdot
דָּג טָמֵא שֶׁבָּלַע דָּג טָהוֹר\hdot
מֻתָּר בַּאֲכִילָה\hdot
וְטָהוֹר שֶׁבָּלַע דָּג טָמֵא אָסוּר בַּאֲכִילָה לְפִי שֶׁאֵינוֹ גִּדּוּלָיו׃}
