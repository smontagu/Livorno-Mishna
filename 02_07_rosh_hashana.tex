\tractatehead{מסכת ראש השנה}{roshHashana}

\perek{אַרְבָּעָה}{רָאשֵׁי שָׁנִים הֵם\hdot
בְּאֶחָד בְּנִיסָן רֹאשׁ הַשָּׁנָה לַמְּלָכִים וְלָֽרְגָלִים\hdot
בְּאֶחָד בֶּאֱלוּל רֹאשׁ הַשָּׁנָה לְמַעֲשֵׂר בְּהֵמָה\hdot
רִבִּי אֶלְעָזָר וְרִבִּי שִׁמְעוֹן אוֹמְרִים בְּאֶחָד בְּתִשְׁרִי\hdot
בְּאֶחָד בְּתִשְׁרִי רֹאשׁ הַשָּׁנָה לַשָּׁנִים וְלַשְּׁמִטִּין וְלַיּוֹבְלוֹת לַנְּטִיעָה וְלַיְּרָקוֹת\hdot
בְּאֶחָד בִּשְׁבָט רֹאשׁ הַשָּׁנָה לָאִילָן כְּדִבְרֵי בֵית שַׁמָּאי\hdot
בֵּית הִלֵּל אוֹמְרִים בַּחֲמִשָּׁה עָשָׂר בּוֹ׃}

\halacha{בְּאַרְבָּעָה פְרָקִים הָעוֹלָם נִדּוֹן\hdot
בַּפֶּסַח עַל הַתְּבוּאָה\hdot
בַּעֲצֶרֶת עַל פֵּירוֹת הָאִילָן\hdot
בְּרֹאשׁ הַשָּׁנָה כָּל בָּאֵי הָעוֹלָם עוֹבְרִין לְפָנָיו כִּבְנֵי מְרוֹן\hdot
שֶׁנֶּאֱמַר הַיּוֹצֵר יַחַד לִבָּם הַמֵּבִין אֶל כָּל מַעֲשֵׂיהֶם\hdot
וּבַחַג נִדּוֹנִים עַל הַמָּיִם׃}

\halacha{עַל שִׁשָּׁה חֳדָשִׁים הַשְּׁלוּחִין יוֹצְאִין\hdot
עַל נִיסָן מִפְּנֵי הַפֶּסַח\hdot
עַל אָב מִפְּנֵי הַתַּעֲנִית\hdot
עַל אֱלוּל מִפְּנֵי רֹאשׁ הַשָּׁנָה\hdot
עַל תִּשְׁרִי מִפְּנֵי תְּקָנַת הַמּוֹעֲדוֹת\hdot
עַל כִּסְלֵו מִפְּנֵי חֲנֻכָּה\hdot
וְעַל אֲדָר מִפְּנֵי הַפּוּרִים\hdot
וּכְשֶׁהָיָה בֵית הַמִּקְדָּשׁ קַיָּם יוֹצְאִין אַף עַל אִיָּיר מִפְּנֵי פֶסַח קָטָן׃}

\halacha{עַל שְׁנֵי חֳדָשִׁים מְחַלְּלִין אֶת הַשַּׁבָּת\hdot
עַל נִיסָן וְעַל תִּשְׁרִי\hdot
שֶׁבָּהֵן הַשְּׁלוּחִין יוֹצְאִין לְסוּרִיָּא\hdot
וּבָהֵן מְתַקְּנִין אֶת הַמּוֹעֲדוֹת\hdot
וּכְשֶׁהָיָה בֵית הַמִּקְדָּשׁ קַיָּם\hdot
מְחַלְּלִין אַף עַל כֻּלָּן\hdot
מִפְּנֵי תְקָנַת הַקָּרְבָּן׃}

\halacha{בֵּין שֶׁנִּרְאֶה בֶעָלִיל\hdot
בֵּין שֶׁלֹּא נִרְאֶה בֶעָלִיל\hdot
מְחַלְּלִין עָלָיו אֶת הַשַּׁבָּת\hdot
רִבִּי יוֹסֵי אוֹמֵר אִם נִרְאֶה בֶעָלִיל\hdot
אֵין מְחַלְּלִין עָלָיו אֶת הַשַּׁבָּת׃}

\halacha{מַעֲשֶׂה שֶׁעָֽבְרוּ יוֹתֵר מֵאַרְבָּעִים זוֹג וְעִכְּבָן רִבִּי עֲקִיבָא בְּלוּד\hdot
שָׁלַח לוֹ רַבָּן גַּמְלִיאֵל\hdot
אִם מְעַכֵּב אַתָּה אֶת הָרַבִּים\hdot
נִמְצֵאתָ מַכְשִׁילָן לֶעָתִיד לָבוֹא׃}

\halacha{אָב וּבְנוֹ שֶׁרָאוּ אֶת הַחֹדֶשׁ יֵלֵכוּ\hdot
לֹא שֶמִּצְטָֽרְפִין זֶה עִם זֶה\hdot
אֶלָּא שֶׁאִם יִפָּסֵל אֶחָד מֵהֶן\hdot
יִצְטָרֵף הַשֵּׁנִי עִם אַחֵר\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אָב וּבְנוֹ וְכָל הַקְּרוֹבִין\hdot
כְּשֵׁרִין לְעֵדוּת הַחֹדֶשׁ\hdot
אָמַר רִבִּי יוֹסֵי מַעֲשֶׂה בְּטוֹבִיָּה הָרוֹפֵא\hdot
שֶׁרָאָה אֶת הַחֹדֶשׁ בִּירוּשָׁלַיִם\hdot
הוּא וּבְנוֹ וְעַבְדּוֹ מְשׁוּחֲרָר\hdot
וְקִבְּלוּ הַכֹּהֲנִים אוֹתוֹ וְאֶת בְּנוֹ\hdot
וּפָֽסְלוּ אֶת עַבְדּוֹ\hdot
וּכְשֶׁבָּאוּ לִפְנֵי בֵית דִּין\hdot
קִבְּלוּ אוֹתוֹ וְאֶת עַבְדּוֹ\hdot
וּפָֽסְלוּ אֶת בְּנוֹ׃}

\halacha{אֵלּוּ הֵן הַפְּסוּלִין\hdot
הַמְּשַׂחֵק בְּקֻבְיָא\hdot
וּמַלְוֶה בְרִבִּית\hdot
וּמַפְרִיחֵי יוֹנִים\hdot
וְסוֹחֲרֵי שְׁבִיעִית\hdot
וַעֲבָדִים\hdot
זֶה הַכְּלָל\hdot
כָּל עֵדוּת שֶׁאֵין הָאִשָּׁה כְּשֵׁרָה לָהּ\hdot
אַף הֵן אֵינָן כְּשֵׁרִים לָהּ׃}

\halacha{מִי שֶׁרָאָה אֶת הַחֹדֶשׁ\hdot
וְאֵינוֹ יָכוֹל לְהַלֵּךְ\hdot
מוֹלִיכִין אוֹתוֹ עַל הַחֲמוֹר\hdot
אֲפִילוּ בַמִּטָּה\hdot
וְאִם צוֹדֶה לָהֶם\hdot
לוֹקְחִין בְּיָדָם מַקְלוֹת\hdot
וְאִם הָֽיְתָה דֶרֶךְ רְחוֹקָה\hdot
לוֹקְחִין בְּיָדָם מְזוֹנוֹת\hdot
שֶׁעַל מַהֲלַךְ לַיְלָה וָיוֹם\hdot
מְחַלְּלִין אֶת הַשַּׁבָּת\hdot
וְיוֹצְאִין לְעֵדוּת הַחֹדֶשׁ\hdot
שֶנֶּאֱמַר אֵלֶּה מוֹעֲדֵי יְיָ מִקְרָאֵי קֹדֶשׁ\hdot
אֲשֶׁר תִּקְרְאוּ אֹתָם בְּמוֹעֲדָם׃}

\perek{אִם}{אֵינָן מַכִּירִין אוֹתוֹ\hdot
מְשַׁלְּחִין אַחֵר עִמּוֹ לְהָעִידוֹ\hdot
בָּרִאשׁוֹנָה הָיוּ מְקַבְּלִין עֵדוּת הַחׁדֶשׁ מִכָּל אָדָם\hdot
מִשֶׁקִּלְקְלוּ הָאֲפִיקוֹרוֹסִים\hdot
הִתְקִינוּ שֶׁלֹּא יְהוּ מְקַבְּלִין אֶלָּא מִן הַמַכִּירִים׃}

\halacha{בָּרִאשׁוֹנָה הָיוּ מַשִּׂיאִין מַשּׂוּאוֹת\hdot
מִשֶׁקִּלְקְלוּ הַכּוּתִים\hdot
הִתְקִינוּ שֶׁיְּהוּ שְׁלוּחִין יוֹצְאִין׃}
