\tractatehead{מסכת בבא קמא}{baba kama}

\perek{אַרְבָּעָה}{אָבוֹת נְזִיקִין\hdot
הַשּׁוֹר\hdot
וְהַבּוֹר\hdot
וְהַמַּבְעֶה\hdot
וְהַהֶבְעֵר\hdot
לֹא הֲרֵי הַשּׁוֹר כַּהֲרֵי הַמַּבְעֶה\hdot
וְלֹא הֲרֵי הַמַּבְעֶה כַּהֲרֵי הַשּׁוֹר\hdot
וְלֹא זֶה וְזֶה שֶׁיֵּשׁ בָּהֶן רוּחַ חַיִּים\hdot
כַּהֲרֵי הָאֵשׁ שֶׁאֵין בּוֹ רוּחַ חַיִּים\hdot
וְלֹא זֶה וְזֶה שֶׁדַּרְכָּן לִילֵךְ וּלְהַזִּיק\hdot
כַּהֲרֵי הַבּוֹר שֶׁאֵין דַּרְכּוֹ לִילֵךְ וּלְהַזִּיק\hdot
הַצַּד הַשָּׁוֶה שֶׁבָּהֶן\hdot
שֶׁדַּרְכָּן לְהַזִּיק\hdot
וּשְׁמִירָתָן עָלֶיךָ\hdot
וּכְשֶׁהִזִּיק\hdot
חָב הַמַּזִּיק לְשַׁלֵּם תַּשְׁלוּמֵי נֶזֶק\hdot
בְּמֵיטַב הָאָרֶץ׃}

\halacha{כָּל שֶׁחַבְתִּי בִשְׁמִירָתוֹ\hdot
הִכְשַׁרְתִּי אֶת נִזְקוֹ\hdot
הִכְשַׁרְתִּי בְמִקְצַת נִזְקוֹ\hdot
חַבְתִּי בְתַשְׁלוּמֵי נִזְקוֹ\hdot
כְּהֶכְשֵׁר כָּל נִזְקוֹ\hdot
נְכָסִים שֶׁאֵין בָּהֶן מְעִילָה\hdot
נְכָסִים שֶׁל בְּנֵי בְרִית\hdot
נְכָסִים הַמְּיוּחָדִים (וּבְכָל מָקוֹם) חוּץ מֵרְשׁוּת הַמְּיוּחֶדֶת לַמַּזִּיק וּרְשׁוּת הַנִּזָּק וְהַמַּזִּיק\hdot
וּכְשֶׁהִזִּיק חָב הַמַּזִּיק לְשַׁלֵּם תַּשְׁלוּמֵי נֶזֶק\hdot
בְּמֵיטַב הָאָרֶץ׃}

\halacha{שׁוֹם כֶּסֶף\hdot
וְשָׁוֵה כֶסֶף\hdot
בִּפְנֵי בֵית דִּין וְעַל פִּי עֵדִים בְּנֵי חוֹרִין וּבְנֵי בְרִית\hdot
וְהַנָּשִׁים בִּכְלַל הַנֶּזֶק\hdot
וְהַנֶּזֶק וְהַמַּזִיק בַּתַּשְׁלוּמִין׃}

\halacha{חֲמִשָּׁה תַּמִּין\hdot
וַחֲמִשָּׁה מוּעָדִין\hdot
הַבְּהֵמָה אֵינָהּ מוּעֶדֶת לֹא לִגַּח וְלֹא לִגּוֹף\hdot
וְלֹא לָשׁוּךְ\hdot
וְלֹא לִרְבּוֹץ\hdot
וְלֹא לִבְעוֹט\hdot
הַשֵּׁן מוּעָד לֶאֱכוֹל אֶת הָרָאוּי לָהּ\hdot
הָרֶגֶל מוּעֶדֶת לִשְׁבּוֹר בְּדֶרֶךְ הִלּוּכָהּ\hdot
וְשׁוֹר הַמּוּעָד\hdot
וְשׁוֹר הַמַּזִיק בִּרְשׁוּת הַנִּזָּק\hdot
וְהָאָדָם\hdot
הַזְּאֵב וְהָאֲרִי וְהַדּוֹב וְהַנָּמֵר וְהַבַּרְדְּלָס וְהַנָּחָשׁ הֲרֵי אֵלּוּ מוּעָדִין\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר בִּזְמָן שֶׁהֵן בְּנֵי תַרְבּוּת אֵינָן מוּעָדִין\hdot
וְהַנָּחָשׁ מוּעָד לְעוֹלָם\hdot
מַה בֵּין תָּם לַמּוּעָד\hdot
(אֶלָּא) שֶׁהַתָּם מְשַׁלֵּם חֲצִי נֶזֶק מִגּוּפוֹ\hdot
וּמוּעָד מְשַׁלֵּם נֶזֶק שָׁלֵם מִן הָעֲלִיָּה׃}
