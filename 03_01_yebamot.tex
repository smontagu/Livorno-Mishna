\tractatehead{מסכת יבמות}{yebamot}

\chead{יום יוד}

\perek{חֲמֵשׁ}{עֶשְׂרֵה נָשִׁים פּוֹטְרוֹת צָרוֹתֵיהֶן\hdot
וְצָרוֹת צָרוֹתֵיהֶן\hdot
מִן הַחֲלִיצָה וּמִן הַיִּבּוּם\hdot
עַד סוֹף הָעוֹלָם\hdot
וְאֵלּוּ הֵן\hdot
בִּתּוֹ\hdot
וּבַת בִּתּוֹ\hdot
וּבַת בְּנוֹ\hdot
בַּת אִשְׁתּוֹ\hdot
וּבַת בְּנָהּ\hdot
וּבַת בִּתָּהּ\hdot
חֲמוֹתוֹ\hdot
וְאֵם חֲמוֹתוֹ\hdot
וְאֶם חָמִיו\hdot
אֲחוֹתוֹ מֵאִמּוֹ\hdot
וַאֲחוֹת אִמּוֹ\hdot
וַאֲחוֹת אִשְׁתּוֹ\hdot
וְאֵשֶׁת אָחִיו מֵאִמּוּ\hdot
וְאֵשֶׁת אָחִיו שֶׁלֹּא הָיָה בְעוֹלָמוֹ\hdot
וְכַלָּתוֹ\hdot
הֲרֵי אֵלּוּ פּוֹטְרוֹת צָרוֹתֵיהֶן\hdot
וְצָרוֹת צָרוֹתֵיהֶן מִן הַחֲלִיצָה וּמִן הַיִּבּוּם עַד סוֹף הָעוֹלָם\hdot
וְכֻלָּן אִם מֵתוּ\hdot
אוֹ מִאֲנוּ\hdot
אוֹ נִתְגָּֽרְשׁוּ\hdot
אוֹ שֶׁנִּמְצְאוּ אַיְלוֹנִיּוֹת\hdot
צָרוֹתֵיהֶן מֻתָּרוֹת\hdot
וְאֵי אַתָּה יָכוֹל לוֹמַר בַּחֲמוֹתוֹ (וּבְאֵם חֲמוֹתוֹ) וְבְאֵם חָמִיו\hdot
שֶׁנִּמְצְאוּ אַיּלוֹנִיּוֹת אוֹ שֶׁמִּאֲנוּ׃}

\halacha{כֵּיצִד פּוֹטְרוֹת צָרוֹתֵיהֶן\hdot
הָֽיְתָה בִתּוֹ\hdot
אוֹ אַחַת מִכָּל הָעֲרָיוֹת הָאֵלּוּ\hdot
נְשׂוּאָה לְאָחִיו\hdot
וְלוֹ אִשָּׁה אַחֶרֶת וּמֵת\hdot
כַּשֵּׁם שֶׁבִּתּוֹ פְטוּרָה\hdot
כַּךְ צָרָתָהּ פְּטוּרָה\hdot
הָֽלְכָה צָרַת בִּתּוֹ\hdot
וְנִשֵּׂאת לְאָחִיו הַשֵּׁנִי\hdot
וְלוֹ אִשָּׁה אַחֶרֶת וּמֵת\hdot
כַּשֵּׁם שֶׁצָּרַת בִּתּוֹ פְטוּרָה\hdot
כַּךְ צָרַת צָרָתָהּ פְּטוּרָה\hdot
אֲפִילוּ הֵן מֵאָה\hdot
כֵּיצַד אִם מֵתוּ צָרוֹתֵיהֶן מֻתָּרוֹת\hdot
הָֽיְתָה בִתּוֹ אוֹ אַחַת מִכָּל הָעֲרָיוֹת הָאֵלּוּ\hdot
נְשׂוּאָה לְאָחִיו\hdot
וְלוֹ אִשָּׁה אָחֶרֶת\hdot
מֵתָה בִתּוֹ אוֹ נִתְגָּֽרְשָׁה\hdot
וְאַחַר כַּךְ מֵת אָחִיו\hdot
צָרָתָהּ מֻתֶּרֶת\hdot
וְכָל הַיְּכוֹלָה לְמָאֵן וְלֹא מִיאֲנָה\hdot
צָרָתָהּ חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת׃}

\halacha{שֵׁשׁ עֲרָיוֹת חֲמוּרוֹת מֵאֵלּוּ\hdot
מִפְּנֵי שֶׁנְּשֹוּאוֹת לַאֲחֵרִים צָרוֹתֵיהֶן מֻתָּרוֹת\hdot
אִמּוֹ\hdot
וְאֵשֶׁת אָבִיו\hdot
וַאֲחוֹת אָבִיו\hdot
אֲחוֹתוֹ מֵאָבִיו\hdot
וְאֵשֶׁת אֲחִי אָבִיו\hdot
וְאֵשֶׁת אָחִיו מֵאָבִיו׃}

\halacha{בֶּית שַׁמָּאי מַתִּירִין הַצָּרוֹת לָאַחִים\hdot
וּבֵית הִלֵּל אוֹסְרִין\hdot
חָֽלְצוּ\hdot
בֵּית שַׁמָּאי פּוֹסְלִין מִן הַכְּהֻנָּה\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot
נִתְיַבְּמוּ בֵּית שַׁמָּאי מַכְשִׁירִין\hdot
וּבֵית הִלֵּל פּוֹסְלִין\hdot
אַף עַל פִּי שֶׁאֵלּוּ (אוֹסְרִין וְאֵלּוּ מַתִּירִין\hdot
אֵלּוּ) פּוֹסְלִין וְאֵלּוּ מַכְשִׁירִין\hdot
לֹא נִמְנְעוּ בֵּית שַׁמָּאי מִלִּשָּׂא נָשִׁים מִבֵּית הִלֵּל\hdot
וְלֹא בֵית הִלֵּל מִבֵּית שַׁמָּאי\hdot
כָּל הַטַּהֲרוֹת וְהַטֻּמְאוֹת שֶׁהָיוּ אֵלּוּ מְטַהֲרִין וְאֵלּוּ מְטַמְּאִין\hdot
לֹא נִמְנְעוּ עוֹשִׂין טַהֲרוֹת\hdot
אֵלּו עַל גַּבֵּי אֵלּוּ׃}

\perek{כֵּיצַד}{אֵשֶׁת אָחִיו שֶׁלֹּא הָיָה בְעוֹלָמוֹ\hdot
שְׁנֵי אַחִים וּמֵת אַחַד מֵהֶם\hdot
וְנוֹלָד לָהֶם אָח\hdot
וְאַחַר כַּךְ יִבֵּם הַשֵּׁנִי אֶת אֵשֶׁת אָחִיו וּמֵת\hdot
הָרִאשׁוֹנָה יוֹצֵאה מִשּׁוּם אֵשֶׁת אָחִיו שֶׁלֹּא הָיָה בְעוֹלָמוֹ\hdot
וְהַשְּׁנִיָּה מִשּׁוּם צָרָתָהּ\hdot
עָשָׂה בָהּ מַאֲמָר וּמֵת\hdot
שְׁנִיָּה חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת׃}

\halacha{שְֹנֵי אַחִים וּמֵת אַחַד מֵהֶן\hdot
וְיִבֵּם הַשֵּׁנִי אֶת אֵשֶׁת אָחִיו\hdot
וְאַחַר כַּךְ נוֹלָד לָהֶן אָח וּמֵת\hdot
הָרִאשׁוֹנָה יוֹצְאָה מִשּׁוּם אֵשֶׁת אָחִיו שֶׁלֹּא הָיָה בְעוֹלָמוֹ\hdot
וְהַשְּׁנִיָּה מִשּׁוּם צָרָתָהּ\hdot
עָשָׂה בָהּ מַאֲמָר וּמֵת\hdot
הַשְּׁנִיָּה חוֹלֶצֶת וְלֹא מִתְיַבֵֶּמֶת\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר מְיַבֵּם לְאֵיזוֹ מֵהֶן שֶׁיִּרְצֶה\hdot
אוֹ חוֹלֵץ לְאֵיזוֹ מֵהֶן שֶׁיִּרְצֶה׃}

\halacha{כְּלַל אָֽמְרוּ בַיְּבָמָה\hdot
כָּל שֶׁהִיא אֲסוּרָה אִסּוּר עֶרְוָה לֹא חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת\hdot
(אֲסוּרָה) אִסּוּר מִצְוָה וְאִסּוּר קְדוּשָׁה\hdot
חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת\hdot
אֲחוֹתָהּ שֶׁהִיא יְבִמְתָּהּ\hdot
חוֹלֶצֶת אוֹ מִתְיַבֶּמֶת׃}

\halacha{אִסּוּר מִצְוָה שְׁנִיּוֹת מִדִּבְרֵי סוֹפְרִים\hdot
אִסּוּר קְדוּשָׁה אַלְמָנָה לְכֹהֵן גָּדוֹל\hdot
גְּרוּשָׁה וַחֲלוּצָה לְכֹהֵן הֶדְיוֹט\hdot
מַמְזֶרֶת וּנְתִינָה לְיִשְׂרָאֵל\hdot
וּבַת יִשְׂרָאֵל לְנָתִין וּמַמְזֵר׃}

\halacha{מִי שֶׁיֵּשׁ לוֹ אָח מִכָּל מָקוֹם\hdot
זוֹקֵק אֶת אֵשֶׁת אָחִיו לְיִבּוּם\hdot
וְאָחִיו לְכָל דָּבָר\hdot
חוּץ מִמִּי שֵֶׁיֵּשׁ לוֹ מִן הַשִּׁפְחָה וּמִן הַנָּכְרִית\hdot
מִי שֶׁיֵּשׁ לוֹ בֵן מִכָּל מָקוֹם\hdot
פּוֹטֵר אֵשֶׁת אָבִיו מִן הַיִּבּוּם\hdot
וְחַיָּיב עַל מַכָּתוֹ וְעַל קִלְלָתוֹ\hdot
וּבְנוֹ לְכָל דָּבָר\hdot
חוּץ מִמִּי שֶׁיֵּשׁ לוֹ מִן הַשִּׁפְחָה וּמִן הַנָּכְרִית׃}

\halacha{מִי שֶׁקִּדֵּשׁ אַחַת מִשְּׁתֵּי אֲחִיּוֹת\hdot
וְאֵינוֹ יוֹדֵעַ אֵיזֶה מֵהֶן קִדֵּשׁ\hdot
נוֹתֵן גֵּט לְזוֹ וְגֵט לְזוֹ\hdot
מֵת וְלוֹ אָח אֶחָד\hdot
חוֹלֵץ לִשְׁתֵּיהֶן\hdot
הָיוּ לוֹ שְׁנַיִם\hdot
אֶחָד חוֹלֵץ וְאֶחָד מְיַבֵּם\hdot
קִדְּמוּ וְכִנְּסוּ\hdot
אֵין מוֹצִיאִין מִיָּדָם׃}

\halacha{שְׁנַיִם שֶׁקִּדְּשׁוּ שְׁתֵּי אֲחִיּוֹת\hdot
זֶה אֵינוֹ יוֹדֵעַ אֵיזוֹ קִדֵּשׁ\hdot
וְזֶה אֵינוֹ יוֹדֵעַ אֵיזוֹ קִדֵּשׁ\hdot
זֶה נוֹתֵן שְׁנֵי גִטִּין\hdot
וְזֶה נוֹתֵן שְׁנֵי גִטִּין\hdot
מֵתוּ\hdot
לַזֶּה אָח וְלַזֶּה אָח\hdot
זֶה חוֹלֵץ לִשְׁתֵּיהֶן\hdot
וְזֶה חוֹלֵץ לִשְׁתֵּיהֶן\hdot
לַזֶּה אֶחָד\hdot
וְלַזֶּה שְׁנַיִם\hdot
הַיָּחִיד חוֹלֵץ לִשְׁתֵּיהֶן\hdot
וְהַשְּׁנַיִם אֶחָד חוֹלֵץ\hdot
וְאֶחָד מְיַבֵּם\hdot
קִדְּמוּ וְכִנְּסוּ\hdot
אֵין מוֹצִיאִין מִיָּדָם\hdot
לַזֶּה שְׁנַיִם וְלַזֶּה שְׁנַיִם\hdot
אָחִיו שֶׁל זֶה חוֹלֵץ לְאֶחָת\hdot
וְאָחִיו שֶׁל זֶה חוֹלֵץ לְאֶחָת\hdot
אָחִיו שֶׁל זֶה מְיַבֵּם חֲלוּצָתוֹ שֶׁל זֶה\hdot
וְאָחִיו שֶׁל זֶה מְיַבֵּם חֲלוּצָתוֹ שֶׁל זֶה\hdot
קִדְּמוּ שְׁנַיִם וְחָֽלְצוּ\hdot
לֹא יְיַבְּמוּ הַשְּׁנַיִם\hdot
אֶלָּא אֶחָד חוֹלֵץ וְאֶחָד מְיַבֵּם\hdot
קִדְּמוּ וְכִנְּסוּ\hdot
אֵין מוֹצִיאִין מִיָּדָם׃}

\halacha{מִצְוָה בַגָּדוֹל לְיַבֵּם\hdot
וְאִם קָדַם הַקָּטָן זָכָה\hdot
הַנִּטְעָן עַל הַשִּׁפְחָה וְנִשְׁתַּחְרְרָה\hdot
אוֹ עַל הַנָּכְרִית וְנִתְגַּיְּרָה\hdot
הֲרֵי זֶה לֹא יִכְנוֹס\hdot
וְאִם כָּנַס אֵין מוֹצִיאִין מִיָּדוֹ\hdot
הַנִּטְעָן עַל אֵשֶׁת אִישׁ\hdot
וְהוֹצִיאוּהָ מִתַּחַת יָדוֹ\hdot
אַף עַל פִּי שֶׁכָּנַס יוֹצִיא׃}

\halacha{הַמֵּבִיא גֵט מִמְּדִינַת הַיָּם וְאָמַר בְּפָנַי נִכְתָּב וּבְפָנַי נֶחְתָּם\hdot
לֹא יִשָּׂא אֶת אִשְׁתּוֹ\hdot
מֵת\hdot
הֲרַגְתִּיו\hdot
הֲרַגְנֻהוּ\hdot
לֹא יִשָּׂא אֶת אִשְׁתּוֹ\hdot
רִבִּי יְהוּדָה אוֹמֵר הֲרַגְתִּיו\hdot
לֹא תִנָּשֵׂא אִשְׁתּוֹ\hdot
הֲרַגְנֻהוּ\hdot
תִּנָּשֵׂא אִשְׁתּוֹ׃}

\halacha{הֶחָכָם שֶׁאָסַר אֶת הָאִשָּׁה בְּנֶדֶר עַל בַּעְלָהּ\hdot
הֲרֵי זֶה לֹא יִשָּׂאֶנָּה\hdot
מִאֲנָה\hdot
אוֹ שֶׁחָֽלְצָה בְפָנָיו\hdot
יִשָּׂאֶנָּה\hdot
מִפְּנֵי שֶׁהוּא בֵית דִּין\hdot
וְכֻלָּם\hdot
שֶׁהָיוּ לָהֶם נָשִׁים וּמֵתוּ\hdot
מוּתָרוֹת לִנָּשֵׂא לָהֶם\hdot
וְכֻלָּן שֶׁנִּשְּׂאוּ לַאֲחֵרִים וְנִתְגָּֽרְשׁוּ\hdot
אוֹ שֶׁנִּתְאַלְמְנוּ\hdot
מוּתָרוֹת לִנָּשֵׂא לָהֶן\hdot
וְכֻלָּן מוּתָרוֹת לִבְנֵיהֶם\hdot
