\tractatehead{מסכת תמורה}{temura}

\perek{הַכֹּל}{מְמִירִין\hdot
אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים\hdot
לֹא שֶׁאָדָם רַשָּׁאי לְהָמִיר\hdot
אֶלָּא שֶׁאִם הֵמִיר מוּמָר\hdot
וְסוֹפֵג אֶת הָאַרְבָּעִים\hdot
הַכֹּהֲנִים מְמִירִים אֶת שֶׁלָּהֶם\hdot
וְיִשְׂרָאֵל מְמִירִים אֶת שֶׁלָּהֶן\hdot
אֵין הַכֹּהֲנִים מְמִירִים לֹא בַחַטָּאת וְלֹא בָאָשָׁם וְלֹא בַבְּכוֹר\hdot
אָמַר רִבִּי יוֹחָנָן בֶּן נוּרִי וְכִי מִפְּנֵי מָה אֵין מְמִירִים בַּבְּכוֹר\hdot
אָמַר לוֹ רִבִּי עֲקִיבָא חַטָּאת וְאָשָׁם מַתָּנָה לַכֹּהֵן\hdot
וְהַבְּכוֹר מַתָּנָה לַכֹּהֵן\hdot
מַה חַטָּׂאת וְאָשָׁם אֵין מְמִירִים בּוֹ\hdot
אַף הַבְּכוֹר לֹא יְמִירֶנּוּ בוֹ\hdot
אָמַר לוֹ רִבִּי יוֹחָנָן בֶּן נוּרִי מַה לִּי אֵינוֹ מֵמִיר בְּחַטָּאת וְאָשָׁם\hdot
שֶׁאֵין זָכִין בָּהֶן בְּחַיֵּיהֶן\hdot
תֹּאמַר בַּבְּכוֹר שֶׁזָּכִין בּוֹ בְּחַיָּיו\hdot
אָמַר לוֹ רִבִּי עֲקִיבָא וַהֲלֹא כְּבָר נֶאֱמַר\hdot
וְהָיָה הוּא וּתְמוּרָתוֹ יִהְיֶה קֹדֶשׁ\hdot
הֵיכָן קְדוּשָׁה חָלָה עָלָיו בְּבֵית הַבְּעָלִים\hdot
אַף תְּמוּרָה בְּבֵית הַבְּעָלִים׃}

\halacha{מְמִירִין מִן הַבָּקָר עַל הַצֹּאן\hdot
וּמִן הַצֹּאן עַל הַבָּקָר\hdot
מִן הַכְּבָשִׂים עַל הָעִזִּים\hdot
וּמִן הָעִזִּים עַל הַכְּבָשִׁים\hdot
מִן הַזְּכָרִים עַל הַנְּקֵבוֹת\hdot
וּמִן הַנְּקֵבוֹת עַל הַזְּכָרִים\hdot
מִן הַתְּמִימִים עַל בַּעֲלֵי מוּמִין\hdot
וּמִבַּעֲלֵי מוּמִין עַל הַתְּמִימִים\hdot
שֶׁנֶּאֱמַר לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אוֹתוֹ טוֹב בְּרָע\hdot
אוֹ רַע בְּטוֹב\hdot
אֵיזֶהוּ טוֹב בְּרָע\hdot
בַּעֲלֵי מוּמִין שֶׁקָּדַם הֶקְדֵּשָׁן אֶת מוּמָן\hdot
מְמִירִים אֶחָד בִּשְׁנַיִם\hdot
וּשְׁנַיִם בְּאֶחָד\hdot
אֶחָד בְּמֵאָה\hdot
וּמֵאָה בְאֶחָד\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אֵין מְמִירִין אֶלָּא אֶחָד בְּאֶחָד\hdot
שֶׁנֶּאֱמַר וְהָיָה הוּא וּתְמוּרָתוֹ\hdot
מַה הוּא מְיוּחָד\hdot
אַף תְּמוּרָתוֹ מְיוּחֶדֶת׃}
