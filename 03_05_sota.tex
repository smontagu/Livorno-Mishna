\tractatehead{מסכת סוטה}{sota}

\perek{הַמְּקַנֵּא}{לְאִשְׁתּוֹ\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר מְקַנֵּא לָהּ עַל פִּי שְׁנַיִם\hdot
וּמַשְׁקָהּ  עַל פִּי עֵד אֶחָד\hdot
אוֹ עַל פִּי עַצְמוֹ\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר מְקַנֵּא לָהּ עַל פִּי שְׁנַיִם וּמַשְׁקָהּ עַל פִּי שְׁנַיִם׃}

\halacha{כֵּיצַד מְקַנֵּא לָהּ\hdot
אָמַר לָהּ בִּפְנֵי שְׁנַיִם\hdot
אַל תְּדַבְּרִי עִם אִישׁ פְּלוֹנִי\hdot
וְדִבְּרָה עִמּוֹ\hdot
עֲדַיִן הִיא מֻתֶּרֶת לְבֵיתָהּ\hdot
וּמֻתֶּרֶת לֶאֱכוֹל בַּתְּרוּמָה\hdot
נִכְנְסָה עִמּוֹ לְבֵית הַסֵּתֶר\hdot
וְשָׁהֲתָה עִמּוֹ כְּדֵי טֻמְאָה\hdot
אֲסוּרָה לְבֵיתָהּ וַאֲסוּרָה לֶאֱכוֹל בַּתְּרומָה\hdot
וְאִם מֵת חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת׃}

\halacha{וְאֵלּוּ אֲסוּרוֹת מִלֶּאֱכוֹל בַּתְּרוּמָה\hdot
הָאוֹמֶרֶת טְמֵאָה אֲנִי לָךְ\hdot
וְשֶׁבָּאוּ עֵדִים שֶׁהִיא טְמֵאָה\hdot
וְהָאוֹמֶרֶת אֲנִי שׁוֹתָה\hdot
וְשֶׁבַּעְלָהּ אֵינוֹ רוֹצֶה לְהַשְׁקוֹתָהּ\hdot
וְשֶׁבַּעְלָהּ בָּא עֲלֶיהָ בַּדֶּרֶךְ\hdot
כֵּיצַד עוֹשֶׂה לָהּ\hdot
מוֹלִיכָהּ לְבֵית דִּין שֶׁבְּאוֹתוֹ מָקוֹם\hdot
וּמוֹסְרִין לוֹ שְׁנֵי תַלְמִידֵי חֲכָמִים שֶׁמֶא יָבֹא עָלֶיהָ בַּדֶּרֶךְ\hdot
רִבִּי יְהוּדָה אוֹמֵר בַּעְלָהּ נֶאֱמָן עָלֶיהָ׃}

\halacha{הָיוּ מַעֲלִין אוֹתָהּ לְבֵית דִּין הַגָּדוֹל שֶׁבִּירוּשָׁלַיִם\hdot
וּמְאַיְּמִין עָלֶיהָ כְּדֶרֶךְ שֶׁמְּאַיְּמִין עַל עֵדֵי נְפָשׁוֹת\hdot
וְאוֹמְרִים לָהּ בִּתִּי הַרְבֶּה יַיִן עוֹשֶׂה\hdot
הַרְבֶּה שְׂחוֹק עוֹשֶׂה\hdot
הַרְבֶּה יַלְדּוּת עוֹשָׂה\hdot
הַרְבֶּה שְׁכֵנִים הָרָעִים עוֹשִׂים\hdot
עֲשִׂי לִשְׁמוֹ הַגָּדוֹל שֶׁנִּכְתַּב בִּקְדֻשָּׁה שֶׁלֹּא יִמָּחֶה עַל הַמַּיִם\hdot
וְאוֹמְרִים לְפָנֶיהָ דְּבָרִים שֶׁאֵינָהּ כְּדַי לְשׁוֹמְעָן\hdot
הִיא וְכָל מִשְׁפַּחַת בֵּית אָבִיהָ׃}

\halacha{אִם אָֽמְרָה טְמֵאָה אָנִי\hdot
שׁוֹבֶרֶת כְּתוּבָתָהּ וְיוֹצֵאת\hdot
וְאִם אָֽמְרָה טְהוֹרָה אָנִי\hdot
מַעֲלִין אוֹתָהּ לְשַׁעַר הַמִּזְרַח\hdot
שֶׁעַל פֶּתַח שַׁעַר נִקָּנוֹר\hdot
שֶׁשָּׁם מַשְׁקִין אֶת הַסּוֹטוֹת\hdot
וּמְטַהֲרִין אֶת הַיּוֹלְדוֹת\hdot
וּמְטַהֲרִין אֶת הַמְּצוֹרָעִים\hdot
וְכֹהֵן אוֹחֵז  בִּבְגָדֶיהָ\hdot
אִם נִקְרָעוּ נִקְרָעוּ\hdot
וְאִם נִפְרָמוּ נִפְרָמו\hdot
עַד שֶׁהוּא מְגַלֶּה אֶת לִבָּהּ\hdot
וְסוֹתֵר אֶת שְׂעָרָהּ\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם הָיָה לִבָּהּ נָאֶה\hdot
לֹא הָיָה מְגַלֵּהוּ\hdot
ואִם הָיָה שְׁעָרָהּ נָאֶה\hdot
לֹא הָיָה סוֹתְרוֹ׃}

\halacha{הָֽיְתָה מְכֻּסָּה בִלְבָנִים\hdot
מְכַסָּהּ בִּשְׁחְוֹרִים\hdot
הָיָה עָלֶיהָ כְּלֵי זָהָב וְקַטְלִיאוֹת\hdot
נְזָמִים וְטַבָּעוֹת\hdot
מַעֲבִירָם מִמֶּנָּה כְּדֵי לְנַוְּלָהּ\hdot
וְאַחַר כַּךְ מֵבִיא חֶבֶל מִצְרִי וְקוֹשֵׁר לְמַעְלָה מִדַּדֶּיהָ\hdot
וְכָל הָרוֹצֶה לִרְאוֹת\hdot
בָּא לִרְאוֹת חוּץ מֵעֲבָדֶיהָ וְשִׁפְחוֹתֶיהָ\hdot
מִפְּנֵי שֶׁלִּבָּהּ גַּס בָּהֶן\hdot
וְכָל הַנָּשִׁים מֻתָּרוֹת לִרְאוֹתָהּ\hdot
שֶׁנֶּאֱמַר וְנִוָּֽסְרוּ כָל הַנָּשִׁים וְלֹא תַעֲשֶׂינָה כּזִמַתְכֶנָה׃}

\halacha{בַּמִּדָּה שֶׁאָדָם מוֹדֵד בָּהּ מוֹדְדִין לוֹ\hdot
הִיא קִשְּׁטָה אֶת עַצְמָהּ לַעֲבֵירָה\hdot
הַמָּקוֹם נִוְּלָהּ\hdot
הִיא גִלְּתָה אֶת עַצְמָהּ לַעֲבֵירָה\hdot
הַמָּקוֹם גִּלָּה עָלֶיהָ\hdot
הַיָּרֵךְ הִתְחִילָה בַעֲבֵירָה תְּחִלָּה\hdot
וְאַחַר כַּךְ הַבָּטֶן\hdot
וּשְׁאַר כָּל הַגּוּף לֹא פָלַט׃}

\halacha{שִׁמְשׁוֹן הָלַךְ אַחַר עֵינָיו\hdot
לְפִיכַךְ נָֽקְרוּ פְלִשְׁתִּים אֶת עֵינָיו\hdot
שֶׁנֶּאֱמַר וַיֹּאחֲזוּהוּ פְלִשְׁתִּים וַיְּנַקְּרוּ אֶת עֵינָיו\hdot
אַבְשָׁלוֹם נִתְגָּאָה בְשַׂעֲרוֹ\hdot
לְפִיכַךְ נִתְלָה בְשַׂעֲרוֹ\hdot
וּלְפִי שֶׁבָּא עַל עֶשֶׂר פִּלַגְשֵׁי אָבִיו\hdot
לְפִיכַךְ נָֽתְנוּ בוֹ עֶשֶׂר לֻנְבִיּוֹת\hdot
שֶׁנֶּאֱמַר וַיָּסֹבּוּ עֲשָׂרָה נְעָרִים נוֹשְׂאֵי כְלֵי יוֹאָב\hdot
וּלְפִי שֶׁגָּנַב  שְׁלֹשָׁה לְבָבוֹת\hdot
לֵב אָבִיו\hdot
וְלֵב בֵּית  דִּין\hdot
ולֵב יִשְׂרָאֵל\hdot
שֶׁנֶּאֱמַר וַיְּגַנֵּב אַבְשָׁלוֹם אֶת לֵב אַנְשֵׁי יִשְׂרָאֵל\hdot
לְפִיכַךְ נִתְקְעוּ בוֹ שְׁלֹשָׁה שְׁבָטִים\hdot
שֶׁנֶּאֱמַר וַיִּקַח שְׁלֹשָׁה שְׁבָטִים בְּכַפּוֹ וַיִּתְקָעֵם בְּלֵב אַבְשָׁלוֹם׃}

\halacha{וְכֵן לְעִנְיַן הַטּוֹבָה\hdot
מִרְיָם הִמְתִּינָה לְמֹשֶׁה שָׁעָה אַחַת\hdot
שֶׁנֶּאֱמַר וַתֵּתַצַּב אֲחוֹתוֹ מֵרָחוֹק\hdot
לְפִיכַךְ נִתְעַכְּבוּ לָהּ יִשְׁרָאֵל שִׁבְעָה יָמִים בַּמִּדְבָּר שֶׁנֶּאֱמַר וְהָעָם לֹא נָסַע עַד הֵאָסֵף מִרְיָם\hdot
יוֹסֵף זָכָה לִקְבּוֹר אֶת אָבִיו\hdot
וְאֵין בְּאֶחָיו גָּדוֹל מִמֶּנּוּ\hdot
שֶׁנֶּאֱמַר וַיַּעַל יוֹסֵף לִקְבּוֹר אֶת אָבִיו וְגוֹמֵר\hdot
וַיַּעַל עִמּוֹ גַּם רֶכֶב וְגַם פָּרָשִׁים\hdot
מִי לָנוּ גָּדוֹל מִיּוֹסֵף\hdot
שֶׁלֹּא נִתְעַסַּק בּוֹ אֶלָּא מֹשֶׁה\hdot
מֹשֶׁה זָכָה  בְּעַצְמוֹת יוֹסֵף\hdot
וְאֵין בְּיִשְׂרָאֵל  גָּדוֹל מִמֶּנּוּ\hdot
שֶׁנֶּאֱמַר וַיִּקַּח מֹשֶׁה אֶת עַצְמוֹת יוֹסֵף עִמּוֹ\hdot
מִי לָנוּ גָּדוֹל מִמֹּשֶׁה שֶׁלֹּא נִתְעָסַק בּוֹ אֶלָּא הַמָּקוֹם\hdot
שֶׁנֶּאֱמַר וַיִּקְבּוֹר אוֹתוֹ בַxגַּיְא\hdot
לֹא עַל מֹשֶׁה  בִלְבַד אָֽמְרוּ\hdot
אֶלָּא עַל כָּל הַצַּדִּיקִים\hdot
שֶׁנֶּאֱמַר וְהָלַךְ  לְפָנֶיךָ צִדְקֶךָ כְּבוֹד יְיָ יַאַסְפֶךָ׃}

\perek{הָיָה}{מֵבִיא אֶת מִנְחָתָהּ\hdot
בְּתוֹךְ כְּפִיפָה מִצְרִית וְנוֹתְנָהּ עַל יָדֶיהָ\hdot
כְּדֵי לְיַגְּעָהּ\hdot
כָּל הַמְּנָחוֹת תְּחִלָּתָן וְסוֹפָן בִּכְלֵי שָׁרֵת\hdot
וְזוֹ תְחִלָּתָהּ בִּכְפִיפָה מִצְרִית\hdot
וְסוֹפָהּ בִּכְלֵי שָׁרֵת\hdot
כָּל הַמְּנָחוֹת טְעוּנוֹת שֶׁמֶן וּלְבוֹנָה\hdot
וְזוֹ אֵינָהּ טְעוּנָה\hdot
לֹא שֶׁמֶן וְלֹא לְבוֹנָה\hdot
כָּל הַמְּנָחוֹת בָּאוֹת מִן הַחִטִּין\hdot
וְזוֹ בָאָה מִן הַשְּׂעוֹרִין\hdot
מִנְחַת הָעֹמֶר אַף עַל פִּי שֶׁבָּאָה מִן הַשְּׂעוֹרִין הִיא הָֽיְתָה בָאָה גֶרֶשׁ\hdot
וְזוֹ בָאָה קֶמַח\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר כְּשֵׁם שֶׁמַּעֲשֶׂיהָ מַעֲשֵׂה בְהֵמָה\hdot
כַּךְ קָרְבָּנָהּ מַאֲכַל בְּהֵמָה׃}

\halacha{הָיָה מֵבִיא פַּיְלִי שֶׁל חֶרֶס חֲדָשָׁה\hdot
וְנוֹתֵן לְתוֹכָהּ חֲצִי לוֹג מַיִם מִן הַכִּיּוֹר\hdot
רִבִּי יְהוּדָה אוֹמֵר רְבִיעִית\hdot
כְּשֵׁם שֶׁמְּמַעֵט בַּכְּתָב כַּךְ מְמַעֵט בַּמַּיִם\hdot
נִכְנָס לַהֵיכָל וּפָנָה לִימִינוֹ\hdot
וּמָקוֹם הָיָה שָׁם אַמָּה עַל אַמָּה\hdot
וְטַבְלָא שֶׁל שַׁיִשׁ\hdot
וְטַבַּעַת הָֽיְתָה קְבוּעָה בָהּ\hdot
וּכְשֶׁהוּא מַגְבִּיהָהּ נוֹטֵל עָפָר מִתַּחְתֶיהָ\hdot
וְנוֹתֵן כְּדֵי שֶׁיֵּרָאֶה עַל פְּנֵי הַמַּיִם\hdot
שֶׁנֶּאֱמַר וּמִן הֶעָפָר אֲשֶׁר יִהְיֶה בְּקַרְקַע הַמִּשְׁכָּן יִקַּח הַכֹּהֵן וְנָתַן אֶל הַמַּיִם׃}

\halacha{בָּא לוֹ לִכְתּוֹב אֶת הַמְּגִלָּה\hdot
מֵאֶיזֶה מָקוֹם הוּא כוֹתֵב\hdot
מֵאִם לֹא שָׁכַב אִישׁ וְגוֹמֵר\hdot
וְאַתּ כִּי שָׂטִית תַּחַת אִישֵׁךְ וְגוֹמֵר\hdot
וְאֵינוֹ כוֹתֵב וְהִשְׁבִּיעַ הַכֹּהֵן אֶת הָאִשָּׁה\hdot
וְכוֹתֵב יִתֵּן יְיָ אוֹתָךְ לְאָלָה וְלִשְׁבֻעָה\hdot
וּבָאוּ הַמַּיִם הַמְּאָֽרְרִים הָאֵלֶּה בְּמֵעַיִךְ לַצְבּוֹת בֶּטֶן וְלַנְפִּיל יָרֵךְ\hdot

וַחֲכָמִים אוֹמְרִים יָכוֹל הוּא לְהַפְרִישָׁהּ\hdot
וּלְהַחֲזִירָהּ לְאַחַר זְמָן\hdot
אַיְלוֹנִית וּזְקֵנָה וְשֶׁאֵינָהּ רְאוּיָה לִילֵד\hdot
לֹא שׁוֹתוֹת וְלֹא נוֹטְלוֹת כְּתוּבָה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר יָכוֹל הוּא לִישָׂא אִשָּׁה אַחֶרֶת\hdot
וְלִפְרוֹת וּלְרַבּוֹת הֵימֶנָּה\hdot
וּשְׁאַר כָּל הַנָּשִׁים אוֹ שׁוֹתוֹת אוֹ לֹא נוֹטְלוֹת כְּתוּבָה׃}

\halacha{אֵשֶׁת כֹּהֵן שׁוֹתָה\hdot
וּמֻתֶּרֶת לְבַעְלָהּ\hdot
אֵשֶׁת סָרִיס שׁוֹתָה\hdot
עַל יְדֵי כָל הָעֲרָיוֹת מְקַנִּין\hdot
חוּץ מִן הַקָּטָן\hdot
וּמִמִּי שֶׁאֵינוֹ אִישׁ׃}
