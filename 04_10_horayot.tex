\tractatehead{מסכת הוריות}{horayot}

\perek{הוֹרוּ}{בֵית דִּין לַעֲבוֹר עַל אַחַת מִכָּל מִצְוֺת הָאֲמוּרוֹת בַּתּוֹרָה\hdot
וְהָלַךְ הַיָּחִיד וְעָשָׂה שׁוֹגֵג עַל פִּיהֶם\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה עִמָּהֶן בֵּין שֶׁעָשׂוּ וְעָשָׂה אַחֲרֵיהֶן\hdot
בֵּין שֶׁלֹּא עָשׂוֹ וְעָשָׂה פָּטוּר\hdot
מִפְּנֵי שֶׁתָּלָה בְּבֵית דִּין\hdot
הוֹרוּ בֵית דִּין וְיָדַע אֶחָד מֵהֶן שֶׁטָּעוּ\hdot
אוֹ תַלְמִיד וְהוּא רָאוּי לְהוֹרָאָה\hdot
וְהָלַךְ וְעָשָׂה עַל פִּיהֶן\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה עִמָּהֶן\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה אַחֲרֵיהֶן\hdot
בֵּין שֶׁלֹּא עָשׂוֹ וְעָשָׂה\hdot
הֲרֵי זֶה חַיָּב\hdot
מִפְּנֵי שֶׁלֹּא תָלָה בְּבֵית דִּין\hdot
זֶה הַכְּלָל הַתּוֹלֶה בְעַצְמוֹ חַיָּיב\hdot
וְהַתּוֹלֶה בְּבֵית דִּין פָּטוּר׃}

\halacha{הוֹרוּ בֵית דִּין וְיָֽדְעוּ שֶׁטָּעוּ וְחָֽזְרוּ בָהֶן\hdot
בֵּין שֶׁהֵבִיאוּ כַפָּרָתָן וּבֵין שֶׁלֹּא הֵבִיאוּ כַפָּרָתָן\hdot
וְהָלַךְ וְעָשָׂה עַל פִּיהֶן\hdot
רִבִּי שִׁמְעוֹן פּוֹטֵר\hdot
וְרִבִּי אֱלִיעֶזֶר אוֹמֵר סָפֵק\hdot
אֵיזֶהוּ סָפֵק\hdot
יָשַׁב לוֹ בְּתוֹךְ בֵּיתוֹ חַיָּב\hdot
הָלַךְ לוֹ לִמְדִינַת הַיָּם פָּטוּר\hdot
אָמַר רִבִּי עֲקִיבָא מוֹדֶה אֲנִי בְזֶה שֶׁהוּא קָרוֹב לַפִּטּוּר מִן הַחוֹבָה\hdot
אָמַר לוֹ בֶן עַזָּאי מַה שְּׁנָא זֶה מִן הַיּוֹשֵׁב בְּבֵיתוֹ\hdot
שֶׁהַיּוֹשֵׁב בְּבֵיתוֹ אֶפְשַׁר הָיָה לוֹ שֶׁיִּשְׁמַע\hdot
וְזֶה לֹא הָיָה אֶפְשַׁר לוֹ שֶׁיִּשְׁמַע׃}

\halacha{הוֹרו בֵית דִּין לַעֲקוֹר אֶת כָּל הַגּוּף\hdot
אָֽמְרוּ אֵין נִדָּה בַתּוֹרָה\hdot
אֵין שַׁבָּת בַּתּוֹרָה\hdot
אֵין עֲבוֹדַת אֱלִילִים  בַּתּוֹרָה\hdot
הֲרֵי אֵלּוּ פְטוּרִין\hdot
הוֹרוּ לְבַטֵּל מִקְצַת וּלְקַיֵּם מִקְצַת\hdot
הֲרֵי אֵלּוּ חַיָּבִין\hdot
כֵּיצַד\hdot
אָֽמְרוּ יֵשׁ נִדָּה בַתּוֹרָה\hdot
אֲבָל הַבָּא עַל שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם פָּטוּר\hdot
יֵשׁ שַׁבָּת בַּתּוֹרָה\hdot
אֲבָל הַמּוֹצִיא מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הָרַבִּים פָּטוּר\hdot
יֵשׁ עֲבוֹדַת אֱלִילִים בַּתּוֹרָה\hdot
אֲבָל הַמִּשְׁתַּחֲוֶה פָּטוּר\hdot
הֲרֵי אֵלּו חַיָּבִין\hdot
שֶׁנֶּאֱמַר וְנֶעְלַם דָּבָר\hdot
דָּבָר וְלֹא כָל הַגּוּף׃}

\halacha{הוֹרוּ בֵית דִּין וְיָדַע אֶחָד מֵהֶן שֶׁטָּעוּ\hdot
וְאָמַר לָהֶן טוֹעִין אַתֶּם\hdot
אוֹ שֶׁלֹּא הָיָה מֻפְלֶא שֶׁל בֵּית דִּין שָׁם\hdot
אוֹ שֶׁהָיָה אֶחָד מֵהֶן גֵּר\hdot
אוֹ מַמְזֵר\hdot
אוֹ נְתִין\hdot
אוֹ זָקֵן שֶׁלֹּא רָאָה (לוֹ) בָנִים\hdot
הֲרֵי אֵלּוּ פְטוּרִין\hdot
שֶׁנֶּאֱמַר כָּאן עֵדָה\hdot
וְנֶאֱמַר לְהַלָּן עֵדָה\hdot
מַה עֵדָה הָאֲמוּרָה לְהַלָּן עַד שֶׁיִּהְיוּ כֻלָּם רְאוּיִם לְהוֹרָאָה\hdot
אַף עֵדָה הָאֲמוּרָה כָּאן עַד שֶׁיִּהְיוּ כֻלָּן רְאוּיִם לְהוֹרָאָה\hdot
הוֹרוּ בֵית דִּין שׁוֹגְגִין וְעָשׂוּ כָל הַקָּהָל שׁוֹגְגִין\hdot
מְבִיאִין פָּר\hdot
מְזִידִין וְעָשׂוּ שׁוֹגְגִין\hdot
מְבִיאִין כִּשְׂבָּה וּשְׂעִירָה\hdot
שׁוֹגְגִין וְעָשׂוּ מְזִידִין\hdot
הֲרֵי אֵלּוּ פְטוּרִין׃}

\halacha{הוֹרוּ בֵית דִּין וְעָשׂוּ כָל הַקָּהָל אוֹ רֻבָּן עַל פִּיהֶם\hdot
מְבִיאִין פַּר\hdot
וּבַעֲבוֹדַת אֱלִילִים מְבִיאִין פַר וְשָׂעִיר\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר שְׁנֵים עָשָׂר שְׁבָטִים\hdot
מְבִיאִים שְׁנֵים עָשָׂר פָּרִים\hdot
וּבַעֲבוֹדַת אֱלִילִים מְבִיאִין שְׁנֵים עָשָׂר פָּרִים וּשְׁנֵים עָשָׂר שְׂעִירִים\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר שְׁלֹשָׁה עָשָׂר פָּרִים\hdot
וּבַעֲבוֹדַת אֱלִילִים\hdot
שְׁלֹשָׁה עָשָׂר פָּרִים\hdot
וּשְׁלֹשָׁה עָשָׂר שְׂעִירִים\hdot
פַּר וְשָׂעִיר לְכָל שֵׁבֶט וְשֵׁבֶט\hdot
פַּר וְשָׂעִיר לְבֵית דִּין\hdot
הוֹרוּ בֵית דִּין וְעָשׂוּ שִׁבְעַת שׁבָטִים אוֹ רֻבָּן עַל פִּיהֶן\hdot
מְבִיאִין פַּר\hdot
וּבַעֲבוֹדַת אֱלִילִים מְבִיאִין פַּר וְשָׂעִיר\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר שִׁבְעָה שְׁבָטִים שֶׁחָֽטְאוּ\hdot
מְבִיאִין שִׁבְעָה פָרִים\hdot
וּשְׁאַר שְׁבָטִים שֶׁלֹּא חָֽטְאוּ\hdot
מְבִיאִין עַל יְדֵיהֶן פַּר\hdot
שֶׁאַף אֵלּוּ שֶׁלֹּא חָֽטְאוּ\hdot
מְבִיאִין עַל יְדֵי הַחוֹטְאִין\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר שְׁמֹנָה פָרִים\hdot
וּבַעֲבוֹדַת אֱלִילִים שְׁמֹנָה פָרִים\hdot
וּשְׁמוֹנָה שְׁעִירִים\hdot
פַּר וְשָׂעִיר לְכָל שֵׁבֶט וְשֵׁבֶט\hdot
פַּר וְשָׂעִיר לְבֵית דִּין\hdot
הוֹרוּ בֵית דִּין שֶׁל אַחַד מִן הַשְּׁבָטִים\hdot
וְעָשָׂה אוֹתוֹ הַשֵּׁבֶט עַל פִּיהֶם אוֹתוֹ הַשֵּׁבֶט הוּא חַיָּב\hdot
וּשְׁאַר הַשְּׁבָטִים פְּטוּרִין\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
וַחֲכָמִים אוֹמְרִים אֵין חַיָּבִים\hdot
אֶלָּא עַד הוֹרַיַּת בֵּית דִּין הַגָּדוֹל בִּלְבָד\hdot
שֶׁנֶּאֱמַר וְאִם כָּל עֲדַת יִשְׂרָאֵל יִשְׁגּוּ\hdot
וְלֹא עֲדַת אוֹתוֹ שֵׁבֶט׃}

\perek{הוֹרָה}{כֹהֵן מָשִׁיחַ לְעַצְמוֹ שׁוֹגֵג\hdot
וְעָשָׂה שׁוֹגֵג מֵבִיא פָר\hdot
שׁוֹגֵג וְעָשָׂה מֵזִיד\hdot
מֵזִיד וְעָשָׂה שׁוֹגֵג פָּטוּר\hdot
שֶׁהוֹרָאַת כֹּהֵן מָשִׁיחַ לְעַצְמוֹ\hdot
כּהוֹרָאַת בֵּית דִּין לַצִּיבּוּר׃}

\halacha{הוֹרָה בִּפְנֵי עַצְמוֹ\hdot
וְעָשָׂה בִּפְנֵי עַצְמוֹ\hdot
מִתְכַּפֵּר לוֹ בִּפְנֵי עַצְמוֹ\hdot
הוֹרָה עַל הַצִּבּוּר\hdot
וְעָשָׂה עִם הַצִּבּוּר\hdot
מִתְכַּפֵּר לוֹ עִם הַצִּבּוּר\hdot
שֶׁאֵין בֵּית דִּין חַיָּבִין\hdot
עַד שֶׁיּוֹרוּ לְבַטֵּל מִקְצַת וּלְקַיֵּם מִקְצַת\hdot
וְכֵן הַמָּשִׁיחַ\hdot
וְלֹא בַעֲבוֹדַת אֱלִילִים\hdot
עַד שֶׁיּוֹרוּ לְבַטֵּל מִקְצַת וּלְקַיֵּם מִקְצַת}

\halacha{אֵין חַיָּבִין אֶלָּא עַל הֶעְלֵם דָּבָר עִם שִׁגְגַת הַמַּעֲשֶׂה\hdot
וְכֵן הַמָּשִׁיחַ\hdot
וְלֹא בַעֲבוֹדַת אֱלִילִים\hdot
אֵין חַיָּבִין אֶלָּא עַל הֶעְלֵם דָּבָר עִם שִׁגְגַת הַמַּעֲשֶׂה\hdot
אֵין בֵּית דִּין חַיָּבִין\hdot
עַד שֵׁיּוֹרוּ בְּדָבָר שֶׁזְּדוֹנוֹ כָרֵת וְשִׁגְגָתוֹ חַטָּאת\hdot
וְכֵן הַמָּשִׁיחַ\hdot
וְלֹא בַעֲבוֹדַת אֱלִילִים\hdot
עַד שֶׁיּוֹרוּ עַל דָּבָר שֶׁזְּדוֹנוֹ כָרֵת וְשִׁגְגָתוֹ חַטָּאת׃}

\halacha{אֵין חַיָּבין עַל עֲשֵׂה\hdot
וְעַל לֹא תַעֲשֶׂה שֶׁבַּמִּקְדָּשׁ\hdot
וְאֵין מְבִיאִין אָשָׁם תָּלוּי\hdot
עַל עֲשֵׂה וְעַל לֹא תַעֲשֶׂה שֶׁבַּמִּקְדָּשׁ\hdot
אֲבָן חַיָּבִין עַל עֲשֵׂה\hdot
וְעַל לֹא תַעֲשֶׂה שֶׁבַּנִּדָּה\hdot
וּמְבִיאִין אָשָׁם תָּלוּי עַל עֲשֵׂה\hdot
וְעַל לֹא תַעֲשֶׂה שֶׁבַּנִּדָּה\hdot
אֵיזוֹ מִצְוַת עֲשֵׂה שֶׁבַּנִּדָּה\hdot
פְּרוֹשׁ מִן הַנִּדָּה\hdot
וּמִצְוַת לֹא תַעֲשֶׂה לֹא תָבֹא אֶל הַנִּדָּה׃}

\halacha{אֵין חַיָּבִין עַל שְׁמִיעַת הַקּוֹל וְעַל בִּטּוּי שְׂפָתַיִם\hdot
וְעַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו\hdot
וְהַנָּשִׂיא כַּיּוֹצֵא בָהֶם\hdot
דִּבְרֵי רִבִּי יוֹסֵי הַגָּלִילִי\hdot
רִבִּי עֲקִיבָא אוֹמֵר הַנָּשִׂיא חַיָּב בְּכֻלָּן\hdot
חוּץ מִשְּׁמִיעַת הַקּוֹל\hdot
שֶׁהַמֶּלֶךְ לֹא דָן וְלֹא דָנִין אוֹתוֹ\hdot
(לֹא מֵעִיד וְלֹא מְעִידִין אוֹתוֹ)׃}

\halacha{כָּל הַמִּצְוֺת שֶׁבַּתּוֹרָה שֶׁחַיָּבִין עַל זְדוֹנָן כָּרֵת וְעַל שִׁגְגָתָן חַטָּאת\hdot
  הַיָּחִיד מֵבִיא כִשְׂבָּה וּשְׂעִירָה\hdot
וְנָשִׂיא שָׂעִיר\hdot
וּמָשִׁיחַ וּבֵית  דִּין מְבִיאִין פָּר\hdot
וּבַעֲבוֹדַת אֱלִילִים\hdot
