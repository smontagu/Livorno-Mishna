\tractatehead{מסכת קידושין}{kidushin}

\perek{הָאִשָּׁה}{נִקְנֵית בִּשְׁלֹשָׁה דְרָכִים\hdot
וְקוֹנָה אֶת עַצְמָהּ בִּשְׁתֵּי דְרָכִים\hdot
נִקְנֵית בְּכֶסֶף בִּשְׁטָר וּבְבִיאָה\hdot
בְּכֶסֶף בֵּית שַׁמָּאי אוֹמְרִים בְּדִינָר וּבְשָׁוֶה דִינָר\hdot
וּבֵית הִלֵּל אוֹמְרִים בִּפְרוּטָה וּבְשָׁוֶה פְרוּטָה\hdot
וְכַמָּה הִיא פְרוּטָה\hdot
אֶחָד מִשְּׁמֹנָה בְּאִיסַר הָאִטַּלְקִי\hdot
וְקוֹנָה אֶת עַצְמָהּ\hdot
בְּגֵט\hdot
וּבְמִיתַת הַבַּעַל\hdot
הַיְּבָמָה נִקְנֵית בְּבִיאָה\hdot
וְקוֹנָה אֶת עַצְמָהּ בַּחֲלִיצָה\hdot
וּבְמִיתַת הַיָּבָם׃}

\halacha{עֶבֶד עִבְרִי נִקְנֶה בְּכֶסֶף וּבִשְׁטָר\hdot
וְקוֹנֶה אֶת עַצְמוֹ בְּשָׁנִים וּבְיוֹבֵל\hdot
וּבְגִרְעוֹן כֶּסֶף\hdot
יְתֵירָה עָלָיו הָאָמָה הָעִבְרִיָּה\hdot
שֶׁקּוֹנָה אֶת עַצְמָהּ בְּסִימָנִין\hdot
הַנִּרְצָע נִקְנֶה בִרְצִיעָה\hdot
וְקוֹנֶה אֶת עַצְמוֹ בַּיּוֹבל\hdot
וּבְמִיתַת הָאָדוֹן׃}

\halacha{עֶבֶד כְּנַעֲנִי נִקְנֶה בְּכֶסֶף וּבִשְׁטָר וּבַחֲזָקָה\hdot
וְקוֹנֶה אֶת עַצְמוֹ בְכֶסֶף\hdot
עַל יְדֵי אַחֵרִים\hdot
וּבִשְׁטָר עַל יְדֵי עַצְמוֹ\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים בְּכֶסֶף עַל יְדֵי עַצְמוֹ\hdot
וּבִשְׁטָר עַל יְדֵי אַחֵרִים\hdot
וּבִלְבַד שֶׁיְּהֵא הַכֶּסֶף מִשֶּׁל אַחֵרִים׃}

\halacha{בְּהֵמָה גַּסָּה נִקְנֵית בִּמְסִירָה\hdot
וְהַדַּקָּה בְּהַגְבָּהָה\hdot
דִּבְרֵי רִבִּי מֵאִיר וְרִבִּי אֱלִיעֶזֶר\hdot
וַחֲכָמִים אוֹמְרִים בְּהֵמָה דַּקָּה נִקְנֵית בִּמְשִׁיכָה׃}

\halacha{נְכָסִים שֶׁיֵּשׁ לָהֶם אַחֲרָיוּת\hdot
נִקְנִים בְּכֶסֶף וּבִשְׁטָר וּבַחֲזָקָה\hdot
וְשֶׁאֵין לָהֶם אַחֲרָיוּת\hdot
אֵין נִקְנִין אֶלָּא בִּמְשִׁיכָה\hdot
נִכָסִים שֶׁאֵין לָהֶם אַחֲרָיוּת\hdot
נִקְנִין עִם נְכָסִים שֶׁיֵּשׁ לָהֶם אַחֲרָיוּת\hdot
בְּכֶסֶף וּבִשְׁטָר וּבַחֲזָקָה\hdot
וְזוֹקְקִין נְכָסִים שֶׁאֵין לָהֶן אַחֲרָיוּת אֶת הַנְּכָסִים שֶׁיֵּשׁ לָהֶן אַחֲרָיוּת\hdot
לִשָּׁבַע עֲלֵיהֶן׃}

\halacha{כָּל הַנַּעֲשָׂה דָמִים בְּאַחֵר\hdot
כֵּיוַן שֶׁזָּכָה זֶה נִתְחַיָּב זֶה בַּחֲלִיפָיו\hdot
כֵּיצַד\hdot
הֶחֱלִיף שׁוֹר בְּפָרָה\hdot
אוֹ חֲמוֹר בְּשׁוֹר\hdot
כֵּיוַן שֶׁזָּכָה זֶה\hdot
נִתְחַיָּב זֶה בַּחֲלִיפָיו\hdot
רְשׁוּת הַגָּבוֹהַּ בְּכֶסֶף\hdot
וּרְשׁוּת הַהֶדְיוֹט בַּחֲזָקָה\hdot
אֲמִירָתוֹ לַגָּבוֹהַּ\hdot
כִּמְסִירָתוֹ לַהֶדְיוֹט׃}

\halacha{כָּל מִצְוֺת הַבֵּן עַל הָאָב\hdot
אֲנָשִׁים חַיָּבִין וְנָשִׁים פְּטוּרוֹת\hdot
וְכָל מִצְוֺת הָאָב עַל הַבֵּן\hdot
אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים חַיָּבִים\hdot
וְכָל מִצְוֺת עֲשֵׂה שֶׁהַזְּמָן גְּרָמָהּ\hdot
אֲנָשִׁים חַיָּבִים וְנָשִׁים פְּטוּרוֹת\hdot
וְכָל מִצְוֺת עֲשֵׂה שֶׁלֹּא הַזְּמָן גְּרָמָהּ\hdot
אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים חַיָּבִים\hdot
וְכָל מִצְוֺת לֹא תַעֲשֶׂה\hdot
בֵּין שֶׁהַזְּמָן גְּרָמָהּ בֵּין שֶׁלֹּא הַזְּמָן גְּרָמָהּ\hdot
אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים חַיָּבִין\hdot
חוּץ מִבַּל תַּשְׁחִית\hdot
וּבַל תַּקִּיף\hdot
וּבַל תִּטַּמֵּא לַמֵּתִים׃}

\halacha{הַסְּמִיכוֹת וְהַתְּנוּפוֹת\hdot
וְהַהַגָּשׁוֹת וְהַקְּמִיצוֹת\hdot
וְהַהַקְטָרוֹת וְהַמְּלִיקוֹת\hdot
וְהַהַזָּאוֹת וְהַקַּבָּלוֹת\hdot
נוֹהֲגִין בָּאֲנָשִׁים וְלֹא בַנָּשִׁים\hdot
חוּץ מִמִּנְחַת סוֹטָה וּנְזִירָה\hdot
שֶׁהֵן מְנִיפוֹת׃}

\halacha{כָּל מִצְוָה שֶׁאֵינָהּ תְּלוּיָה בָאָרץ\hdot
נוֹהֶגֶת בָּאָרץ וּבְחוּצָה לָאָרֶץ\hdot
וְכָל הַמִּצְוָה שֶׁהִיא תְלוּיָה בָאָרֶץ\hdot
אֵינָהּ נוֹהֶגֶת אֶלָּא בָאָרֶץ חוּץ מִן הָעָרְלָה וְכִלְאַיִם\hdot
רִבִּי אֶלִיעֶזֶר אוֹמֵר אַף מִן הֶחָדָשׁ׃}

\halacha{כָּל הָעוֹשֶׂה מִצְוָה אֶחָת\hdot
מְטִיבִין לוֹ\hdot
וּמַאֲרִיכִין לוֹ יָמָיו\hdot
וְנוֹחֵל אֶת הָאָרֶץ\hdot
וְכָל שֶׁאֵינוֹ עוֹשֶׂה מִצְוָה אֶחָת\hdot
אֵין מְטִיבִין לוֹ\hdot
וְאֵין מַאֲרִיכִין לוֹ יָמָיו\hdot
וְאֵינוֹ נוֹחֵל אֶת הָאָרֶץ\hdot
וְכָל שֶׁיֶּשְׁנוֹ בַּמִּקְרָא בַּמִּשְׁנָה וּבְדֶרֶךְ אֶרֶץ לֹא בִמְהֵרָה הוּא חוֹטֵא\hdot
שֶׁנֶּאֱמַר וְהַחוּט הַמְּשֻׁלָּשׁ לֹא בִמְהֵרָה יִנָּתֵק\hdot
וְכָל שֶׁאֵינוֹ לֹא בְמִּקְרָא וְלֹא בְמִשְׁנָה וְלֹא בְדֶרֶךְ אֶרֶץ אֵינוֹ מִן הַיִּשּׁוּב׃}

\perek{הָאִישׁ}{מְקַדֵּשׁ בּוֹ וּבִשְׁלוּחוֹ\hdot
הָאִשָּׁה מִתְקַדֶּשֶׁת בָּהּ וּבִשְׁלוּחָהּ\hdot
הָאִישׁ מְקַדֵּשׁ אֶת בִּתּוֹ כְּשֶׁהִיא נַעֲרָה בּוֹ וּבִשְׁלוּחוֹ\hdot
הָאוֹמֵר לָאִשָּׁה הִתְקַדְּשִׁי לִי בִּתְמָרָה זוֹ\hdot
הִתְקַדְּשִׁי לִי בְזוֹ\hdot
אִם יֵשׁ בְּאַחַת מֵהֶן שָׁוֶה פְרוּטָה מְקֻדֶּשֶׁת וְאִם לָאוּ אֵינָהּ מְקֻדֶּשֶׁת\hdot
בְּזוֹ וּבְזוֹ וּבְזוֹ אִם יֵשׁ שָׁוֶה פְרוּטָה בְּכֻלָּן מְקֻדֶּשֶׁת\hdot
וְאִם לָאוּ אֵינָהּ מְקֻדֶּשֶׁת\hdot
הָֽיְתָה אוֹכֶלֶת רִאשׁוֹנָה רִאשׁוֹנָה\hdot
אֵינָהּ מְקֻדֶּשֶׁת\hdot
עֶד שְׁיְּהֵא בְּאַחַת מֵהֶן שָׁוֶה פְרוּטָה׃}

\halacha{הִתְקַדְּשִׁי לִי בְּכוֹס זֶה שֶׁל יַיִן\hdot
וְנִמְצָא שֶׁל דְּבָשׁ\hdot
שֶׁל דְּבַשׁ וְנִמְצָא שֶׁל יַיִן\hdot
בְּדִינָר זֶה שֶׁל כֶּסֶף\hdot
וְנִמְצָא שֶׁל זָהָב\hdot
שֶׁל זָהָב וְנִמְצָא שֶׁל כֶּסֶף\hdot
עַל מְנַת שֶׁאֲנִי עָשִׁיר\hdot
וְנִמְצָא עָנִי\hdot
עָנִי וְנִמְצָא עָשִׁיר\hdot
אֵינָהּ מְקֻדֶשֶׁת\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אִם הִטְעָהּ לְשֶׁבַח מְקֻדֶּשֶׁת׃}

\halacha{עַל מְנַת שֶׁאֲנִי כֹהֵן וְנִמְצָא לֵוִי\hdot
לֵוִי וְנִמְצָא כֹהֵן\hdot
נָתִין וְנִמְצָא מַמְזֵר\hdot
מַמְזֵר וְנִמְצָא נָתִין\hdot
בֶּן עִיר וְנִמְצָא בֶן כְּרָךְ\hdot
בֶּן כְּרָךְ וְנִמְצָא בֶן עִיר\hdot
עַל מְנָת שֶׁבֵּיתִי קָרוֹב לַמֶּרְחָץ\hdot
וְנִמְצָא רָחוֹק\hdot
רָחוֹק וְנִמְצָא קָרוֹב\hdot
עַל מְנַת שֶׁיֵּשׁ לִי בַת אוֹ שִׁפְחָה גוֹדֶלֶת וְאֵין לוֹ\hdot
אוֹ עַל מְנַת שֶׁאֵין לִי וְיֵשׁ לוֹ\hdot
עַל מְנַת שֶׁאֵין לִי בָנִים וְיֵשׁ לוֹ\hdot
אוֹ עַל מְנַת שֶׁיֵּשׁ לִי וְאֵין לוֹ\hdot
עַל כֻּלָּם אַף עַל פִּי שֶׁאָֽמְרָה בְּלִבִּי הָיָה לְהִתְקַדֵּשׁ לוֹ\hdot
אַף עַל פִּי כֵן אֵינָהּ מְקֻדֶּשֶׁת\hdot
וְכֵן הִיא שֶׁהִטְעַתּוּ׃}

\halacha{הָאוֹמֵר לִשְׁלוּחוֹ\hdot
צֵא וְקַדֵּשׁ לִי אִשָּׁה פְלוֹנִית בְּמָקוֹם פְּלוֹנִי\hdot
וְהָלַךְ וְקִדְּשָׁהּ בְּמָקוֹם אַחֵר\hdot
אֵינָהּ מְקֻדֶּשֶׁת\hdot
הֲרֵי הִיא בְּמָקוֹם פְּלוֹנִי\hdot
וְקִדְּשָׁהּ בְּמָקוֹם אַחֵר\hdot
הֲרֵי זוֹ מְקֻדֶּשֶׁת׃}

\halacha{הַמְּקַדֵּשׁ אֶת הָאִשָּׁה עַל מְנַת שֶׁאֵין עָלֶיהָ נְדָרִים\hdot
וְנִמְצְאוּ עָלֶיהָ נְדָרִים\hdot
אֵינָהּ מְקֻדֶּשֶׁת\hdot
כְּנָסָהּ סְתָם וְנִמְצְאוּ עָלֶיהָ נְדָרִים\hdot
תֵּצֵא שֶׁלֹּא בִכְתוּבָה\hdot
עַל מְנַת שֶׁאֵין בָּהּ מוּמִין\hdot
וְנִמְצְאוּ בָהּ מוּמִין אֵינָהּ מְקֻדֶּשֶׁת\hdot
כְּנָסָהּ סְתָם\hdot
וְנִמְצְאוּ בָהּ מוּמִין\hdot
תֵּצֵא שֶׁלֹּא בִכְתוּבָה\hdot
כָּל הַמּוּמִין הַפּוֹסְלִים בַּכֹּהֲנִים\hdot
פּוֹסְלִים בַּנָּשִׁים׃}

\halacha{הַמְּקַדֵּשׁ שְׁתֵּי נָשִׁים בְּשָׁוֶה פְרוּטָה\hdot
אוֹ אִשָּׁה אַחַת בְּפָחוּת מִשָּׁוֶה פְרוּטָה\hdot
אַף עַל פִּי שֶׁשָּׁלַח  סִבְלוֹנוֹת לְאַחַר מִכָּאן\hdot
אֵינָהּ מְקֻדֶּשֶׁת\hdot
שֶׁמֵּחֲמַת קִדּוּשִׁין הָרִאשׁוֹנֵים שָׁלַח\hdot
וְכֵן קָטָן שֶׁקִּדֵּשׁ׃}

\halacha{הַמְּקַדֵּשׁ אִשָּׁה וּבִתָּהּ\hdot
אוֹ אִשָּׁה וַאֲחוֹתָהּ כְּאֶחָת\hdot
אֵינָן מְקֻדָּשׁוֹת\hdot

\halacha{הַמְּקַדֵּשׁ בָּעָרְלָה\hdot

\halacha{הַמְּקַדֵּשׁ בִּתְרוּמוֹת\hdot

\halacha{הָאוֹמֵר לָאִשָּׁה הֲרֵי אַתְּ מְקֻדֶּשֶׁת לִי\hdot

עַל מְנַת שֶׁאַרְאֵךְ מָאתַיִם זוּז\hdot
הֲרֵי זוֹ מְקֻדֶּשֶׁת וְיַרְאֶה לָהּ\hdot
וְאִם הֶרְאָהּ עַל הַשֻּׁלְחָן אֵינָהּ מְקֻדֶּשֶׁת׃}

\halacha{עַל מְנַת שֶׁיֵּשׁ לִי בֵּית כּוֹר עָפָר הֲרֵי זוֹ מְקֻדֶּשֶׁת וְיֵשׁ לוֹ\hdot
עַל מְנַת שֶׁיֵּשׁ לִי בְּמָקוֹם פְּלוֹנִי\hdot

עַל מְנַת שֶׁאַרְאֵךְ בֵּית כּוֹר עָפָר\hdot

\halacha{רִבִּי מֵאִיר אוֹמֵר כָּל תְּנָאי שֶׁאֵינוֹ כִתְנָאי בְּנֵי גָד וּבְנֵי רְאוּבֵן אֵינוֹ תְנָאי\hdot
שֶׁנֶּאֱמַר וַיֹּאמֶר מֹשֶׁה אֲלֵהֶם אִם יַעַבְרוּ בְנֵי גָד וּבְנֵי רְאוּבֵן\hdot

\halacha{הַמְּקַדֵּשׁ אֶת הָאִשָּׁה\hdot

וַהֲרֵי הִיא לְוִיָּה\hdot

עֲשִׁירָה וַהֲרֵי הִיא עֲנִיָּה הֲרֵי זוֹ מְקֻדֶּשֶׁת\hdot
