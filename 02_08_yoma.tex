\perek[611]{שִׁבְעַת}{יָמִים קוֹדֶם יוֹם הַכִּפּוּרִים\hdot
מַפְרִישִׁין כֹּהֵן גָּדוֹל מִבֵּיתוֹ לְלִשְׁכַּת פְּלַהְדְּרִין\hdot
וּמַתְקִינִין לוֹ כֹּהֵן אַחֵר תַּחְתָּיו\hdot
שֶׁמֶא יֶאֱרַע בּוֹ פִּסּוּל\hdot
רִבִּי יְהוּדָה אוֹמֵר אַף אִשָּׁה אַחֶרֶת מַתְקִינִין לוֹ\hdot
שֶׁמֶא תָמוּת אִשְׁתּוֹ\hdot
שֶׁנֶּאֱמַר וְכִפֵּר בַּעֲדוֹ וּבְעַד בֵּיתוֹ\hdot
בֵּיתוֹ זוֹ אִשְׁתּוֹ\hdot
אָֽמְרוּ לוֹ אִם כֵּן אֵין לַדָּבָר סוֹף׃}

\halacha{כָּל שִׁבְעַת הַיָּמִים הוּא זוֹרֵק אֶת הַדָּם\hdot
וּמַקְטִיר אֶת הַקְּטֹרֶת\hdot
וּמֵטִיב אֶת הַנֵּרוֹת\hdot
וּמַקְרִיב אֶת הָרֶגֶל\hdot
וּשְׁאַר כָּל הַיָּמִים אִם רָצָה לְהַקְרִיב\hdot
מַקְרִיב\hdot
שֶׁכֹּהֵן גָּדוֹל מַקְרִיב חֵלֶק בְּרֹאשׁ וְנוֹטֵל חֵלֶק בָּרֹאשׁ׃}

\halacha[168]{מָֽסְרוּ לוֹ זְקֵנִים מִזִּקְנֵי בֵית דִּין\hdot
וְקוֹרִין לְפָנָיו בְּסֵדֶר הַיּוֹם\hdot
וְאוֹמְרִים לוֹ אִישִׁי כֹהֵן גָּדוֹל\hdot
קְרָא אַתָּה בְּפִיךָ שֶׁמֶא שָׁכַחְתָּ\hdot
אוֹ שֶׁמֶא לֹא לָמַדְתָּ\hdot
עֶרֶב יוֹם הַכִּפּוּרִים שַׁחְרִית\hdot
מַעֲמִידִין אוֹתוֹ בְשַׁעַר הַמִּזְרָח\hdot
וּמַעֲבִירִין לְפָנָיו פָּרִים וְאֵילִים וּכְבָשִׂים\hdot
כְּדֵי שֶׁיְּהֵא מַכִּיר וְרָגִיל בַּעֲבוֹדָה׃}

\halacha{כָּל שִׁבְעַת הַיָּמִים לֹא הָיוּ מוֹנְעִין מִמֶּנּוּ מַאֲכָל וּמִשְׁתֶּה\hdot
עֶרֶב יוֹם הַכִּפּוּרִים עִם חֲשֵׁיכָה\hdot
לֹא הָיוּ מַנִּיחִים אוֹתוֹ לֶאֱכוֹל הַרְבֶּה\hdot
מִפְּנֵי שֶׁהַמַּאֲכָל מֵבִיא אֶת הַשֵּׁינָה׃}

\halacha{מְסָרוּהוּ זִקְנֵי בֵית דִּין לְזִקְנֵי כְהוּנָה\hdot
וְהֶעֱלוּהוּ לַעֲלִיַּת בֵּית אַבְטִינָס\hdot
וְהִשְׁבִּיעוּהוּ\hdot
וְנִפְטָרוּ וְהָלְכוּ לָהֶם וְאוֹמְרִים לוֹ\hdot
אִישִׁי כֹהֵן גָּדוֹל אָנוּ שְׁלוּחֵי בֵית דִּין\hdot
וְאַתָּה שְׁלוּחֵינוּ וְשָׁלִיחַ בֵּית דִּין\hdot
מַשְׁבִּיעִין אָנוּ עָלֶיךָ בְּמִי שֶׁשִּׁכֵּן אֶת שְׁמוֹ בַּבַּיִת הַזֶּה\hdot
שֶׁלֹּא תְשַׁנֶּה דָבָר מִכּל מַה שֶּׁאָמַרְנוּ לָךְ\hdot
הוּא פּוֹרֵשׁ וּבוֹכֶה\hdot
וְהֵן פּוֹרְשִׁים וּבוֹכִים׃}

\halacha[909]{אִם הָיָה חָכָם דּוֹרֵשׁ\hdot
וְאִם לָאוּ\hdot
תַּלְמִידֵי חֲכָמִים דּוֹרְשִׁין לְפָנָיו\hdot
אִם רָגִיל לִקְרוֹת קוֹרֵא\hdot
וְאִם לָאוּ\hdot
קוֹרִין לְפָנָיו\hdot
וּבַמֶּה קוֹרִין לְפָנָיו\hdot
בְּאִיּוֹב וּבְעֶזְרָא\hdot
וּבְדִבְרֵי הַיָּמִים\hdot
זְכַרְיָה בֶּן קְבוּטָל אוֹמֵר פְּעָמִים הַרְבֶּה קָרִיתִי לְפָנָיו בְּדָנִיֵּאל׃}

\halacha[1122]{בִּקֵּש לְהִתְנַמְנֵם פִּרְחֵי כְהוּנָה מַכִּין לְפָנָיו בְּאֶצְבַע צְרָדָה\hdot
וְאוֹמְרִים לוֹ\hdot
אִישִׁי כֹהֵן גָּדוֹל\hdot
עֲמוֹד וְהָפֵג אַחַת עַל הָרִצְפָּה\hdot
וּמַעֲסִיקִין אוֹתוֹ\hdot
עַד שֶׁיַּגִּיעַ זְמַן הַשְּׁחִיטָה׃}

\halacha[208]{בְּכָל יוֹם תּוֹרְמִין אֶת הַמִּזְבֵּחַ\hdot
בִּקְרִיאַת הַגֶּבֶר\hdot
אוֹ סָמוּךְ לוֹ\hdot
בֵּין לְפָנָיו בֵּין לְאַחֲרָיו\hdot
וּבְיוֹם הַכִּפּוּרִים מֵחֲצוֹת\hdot
וּבָֽרְגָלִים מֵאַשְׁמוֹרֶת הָרִאשוֹנָה\hdot
וְלֹא הָֽיְתָה קְרִיאַת הַגֶּבֶר מַגַּעַת\hdot
עַד שֶׁהָֽיְתָה עֲזָרָה מְלֵאָה מִיִּשְׂרָאֵל׃}

\perek{בָּרִאשׁוֹנָה}{כָּל מִי שֶׁרוֹצֶה לִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
תּוֹרֵם\hdot
וּבִזְמָן שֶׁהֵן מְרֻבִּין\hdot
רָצִין וְעוֹלִין בַּכֶּבֶשׁ\hdot
וְכָל הַקּוֹדֵם אֶת חֲבֵירוֹ לְתוֹךְ אַרְבַּע אַמּוֹת זָכָה\hdot
אִם הָיוּ שְׁנַיִם שָׁוִין\hdot
הַמְּמוּנֶה אוֹמֵר לָהֶן הַצְבִּיעוּ\hdot
וּמָה הֵן מוֹצִיאִין אַחַת אוֹ שְׁתַּיִם\hdot
וְאֵין מוֹצִיאִין הַגּוּדָל בַּמִּקְדָּשׁ׃}

\halacha{מַעֲשֶׂה שֶׁהָיוּ שְׁנַיִם שָׁוִים\hdot
וְרָצִין וְעוֹלִין בַּכֶּבֶשׁ\hdot
וְדָחַף אֶחָד מֵהֶן אֶת חֲבֵרוֹ\hdot
וְנָפַל וְנִשְׁבְּרָה רַגְלוֹ\hdot
וְכֵיוַן שֶׁרָאוּ בֵית דִּין שְׁבָאִין לִידֵי סַכָּנָה\hdot
הִתְקִינוּ שֶׁלֹּא יְהוּ תוֹרְמִין אֶת הַמִּזְבֵּחַ אֶלָּא בַּפַּיָּס\hdot
אַרְבָּעָה פַּיָּסוֹת הָיוּ שָׁם\hdot
וְזֶה פַּיָּס הָרִאשׁוֹן׃}

\halacha[1917]{הַפַּיָּס הַשֵּׁנִי\hdot
מִי שׁוֹחֵט\hdot
מִי זוֹרֵק\hdot

וּמִי מְדַשֵּׁן אֶת הַמְּנוֹרָה\hdot
וּמִי מַעֲלֶה אֵבָרִים לַכֶּבֶשׁ\hdot
הָרֹאשׁ וְהָרֶגֶל\hdot

וְהַקְּרָבַיִם\hdot
וְהַסֹּלֶת\hdot
וְהַחֲבִתִּין\hdot
וְהַיַּיִן\hdot
שְׁלֹשָׁה עָשָׂר כֹּהֲנִים זוֹכִים בּוֹ\hdot

\halacha{הַפַּיָּס הַשְּׁלִישִׁי חֲדָשִׁים לִקְטֹרֶת בּוֹאוּ וְהָפִיסוּ\hdot
וְהָֽרְבִיעִי חֲדָשִׁים עִם יְשָׁנִים\hdot
מִי מַעֲלֶה אֵיבָרִים מִן הַכֶּבֶשׁ לַמִּזְבֵּחַ׃}

\halacha[401]{תָּמִיד קָרֵב בְּתִשְׁעָה\hdot

\perek[471]{אָמַר}{לָהֶם הַמְּמוּנֶה\hdot

\halacha[192]{וְלָמָּה הֻצְרְכוּ לְכַךְ\hdot

אֲפִילוּ טָהוֹר עַד שֶׁיִּטְבּוֹל\hdot

\halacha{פֵּרְסוּ סָדִין שֶׁל בּוּץ בֵּינוֹ לְבֵין הָעָם\hdot
פָּשַׁט\hdot
יָרַד וְטָבַל\hdot
עָלָה וְנִסְתַּפָּג\hdot
הֵבִיאוּ לוֹ בִגְדֵי זָהָב וְלָבַשׁ\hdot
וְקִדֵּש יָדָיו וְרַגְלָיו\hdot
הֵבִיאוּ לוֹ אֶת הַתָּמִיד קְרָצוֹ\hdot
וּמִרֵק אַחֵר שְׁחִיטָה עַל יָדוֹ\hdot
קִבֵּל אֶת הַדָּם וּזְרָקוֹ\hdot
נִכְנָס לְהַקְטִיר קְטֹרֶת שֶׁל שַׁחַר\hdot
וּלְהֵטִיב אֶת הַנֵּרוֹת\hdot
וּלְהַקְרִיב אֶת הָרֹאשׁ\hdot
וְאֶת הָאֵיבָרִים\hdot
וְאֶת הַחֲבִיתִּין\hdot
וְאֶת הַיַּיִן׃}
