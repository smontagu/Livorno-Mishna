\tractatehead{מסכת כלים}{kelim}

\perek{אֲבוֹת}{הַטֻּמְאוֹת\hdot
הַשֶּׁרֶץ\hdot
וְשִׁכְבַת זֶרַע\hdot
וּטְמֵא מֵת\hdot
וְהַמְּצוֹרָע בִּימֵי סָפְרוֹ\hdot
וּמֵי חַטָּאת שֶׁאֵין בָּהֶם כְּדֵי הַזָּיָה\hdot
הֲרֵי אֵלּוּ מְטַמְּאִין\hdot
אָדָם וְכֵלִים בְּמַגַּע\hdot
וּכְלֵי חֶרֶשׁ בָּאַוֵּיר\hdot
וְאֵינָם מְטַמְּאִין בְּמַשָּׂא׃}

\halacha{לְמַעְלָה מֵהֶם נְבֵלָה\hdot
וּמֵי חַטָּאת שֶׁיֵּשׁ בָּהֶם כְּדֵי הַזָּיָה\hdot
שֶׁהֵם מְטַמְּאִין אֶת הָאָדָם בְּמַשָּׂא\hdot
לְטַמֵּא בְגָדִים (בְּמַגַּע)\hdot
וַחֲשׂוּכֵי בְגָדִים בְּמַגַּע׃}

\halacha{לְמַעְלָה מֵהֶם בּוֹעֵל נִדָּה\hdot
שֶׁהוּא מְטַמֵּא מִשְׁכָּב תַּחְתּוֹן כָּעֶלְיוֹן\hdot
לְמַעְלָה מֵהֶן זוֹבוֹ שֶׁל זָב\hdot
וְרוֹקוֹ\hdot
וְשִׁכְבַת זַרְעוֹ\hdot
וּמֵימֵי רַגְלָיו\hdot
וְדַם הַנִּדָּה\hdot
שֶׁהֵן מְטַמְּאִין בְּמַגַּע וּבְמַשָּׂא\hdot
לְמַעְלָה מֵהֶן מֶרְכָּב\hdot
שֶׁהוּא מְטַמֵּא תַּחַת אֶבֶן מַסְמָא\hdot
לְמַעְלָה מִן הַמֶּרְכָּב מִשְׁכָּב\hdot
שֶׁשָּׁוֶה מַגָּעוֹ לְמַשָּׂאוֹ\hdot
לְמַעְלָה מִן הַמִּשְׁכָּב הַזָּב\hdot
שֶׁהַזָּב עוֹשֶׂה מִשְׁכָּב\hdot
וְאֵין מִשְׁכָּב עוֹשֶׂה מִשְׁכָּב׃}
