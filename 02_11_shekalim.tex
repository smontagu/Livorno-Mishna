\tractatehead{מסכת שקלים}{shekalim}

\perek[157]{בְּאֶחָד}{בַּאֲדָר מַשְׁמִיעִין עַל הַשְּׁקָלִים\hdot
וְעַל הַכִּלְאַיִם\hdot
בַּחֲמִשָּׁה עָשָׂר בּוֹ\hdot
קוֹרִין אֶת הַמְּגִילָה בַּכְרָכִים\hdot
וּמְתַקְּנִין אֶת הַדְּרָכִים\hdot
וְאֶת הָרְחוֹבוֹת\hdot
וְאֶת מִקְוְאוֹת הַמַּיִם\hdot
וְעוֹשִׂין כָּל צָרְכֵי הָרַבִּים\hdot
וּמְצַיְּנִין אֶת הַקְּבָרוֹת\hdot
וְיוֹצְאִין אַף עַל הַכִּלְאַיִם׃}

\halacha[1152]{אָמַר רִבִּי יְהוּדָה בָּרִאשׁוֹנָה הָיוּ עוֹקְרִין וּמַשְׁלִיכִין לִפְנֵיהֶם\hdot
מִשֶּׁרַבּוּ עוֹבְרֵי עֲבֵירָה\hdot
הָיוּ עוֹקְרִין וּמַשְׁלִיכִין עַל הַדְּרָכִים\hdot
הִתְקִינוּ שֶׁיְּהוּ מַפְקִירִין כָּל הַשָּׂדֶה כֻּלָּהּ׃}

\halacha[744]{בַּחֲמִשָּׁה עָשָׂר בּוֹ\hdot
שֻׁלְחָנוֹת הָיוּ יוֹשְׁבִין בַּמְּדִינָה\hdot
בְּעֶשְׂרִים וַחֲמִשָּׁה יָֽשְׁבוּ בַמִּקְדָּשׁ\hdot
מִשֶּׁיָּֽשְׁבוּ בַמִּקְדָּשׁ\hdot
הִתְחִילוּ לְמַשְׁכֵּן\hdot
אֶת מִי מְמַשְׁכְּנִין\hdot
לְוִיִּם וְיִשְׂרָאֵלִים\hdot
גֵּרִים וַעֲבָדִים מְשוּחְרָרִים\hdot
אֲבָל לֹא נָשִׁים\hdot
וַעֲבָדִים\hdot
וּקְטַנִּים\hdot
כָּל קָטָן שֶׁהִתְחִיל אָבִיו לִשְׁקוֹל עַל יָדוֹ\hdot
שׁוּב אֵינוֹ פוֹסֵק\hdot
וְאֵין מְמַשְׁכְּנִין אֶת הַכֹּהֲנִים\hdot
מִפְּנֵי דַרְכֵי שָׁלוֹם׃}
