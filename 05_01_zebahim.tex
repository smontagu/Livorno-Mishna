\tractatehead{מסכת זבחים}{zebahim}

\chead{יום יח}

\perek{כָּל}{הַזְּבָחִים שֶׁנִּזְבְּחוּ שֶׁלֹּא לִשְׁמָן כְּשֵׁרִים\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִן הַפֶּסַח וּמִן הַחַטָּאת\hdot
הַפֶּסַח בִּזְמַנּוֹ\hdot
וְהַחַטָּאת בְּכָל זְמָן\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אַף הָאָשָׁם\hdot
הַפֶּסַח בִּזְמַנּוֹ\hdot
וְהַחַטָּאת וְהָאָשָׁם בְּכָל זְמָן\hdot
אָמַר רִבִּי אֱלִיעֶזֶר הַחַטָּאת בָּאָה עַל חֵטְא\hdot
וְהָאָשָׁם בָּא עַל חֵטְא\hdot
מַה חַטָּאת פְּסוּלָה שֶׁלֹּא לִשְׁמָהּ\hdot
אַף הָאָשָׁם פָּסוּל שֶׁלֹּא לִשְׁמוֹ׃}

\halacha{יוֹסֵי בֶּן חוֹנִי אוֹמֵר הַנִּשְׁחָטִים לְשֵׁם פֶּסַח וּלְשֵׁם חַטָּאת פְּסוּלִין\hdot
שִׁמְעוֹן אֲחִי עֲזַרְיָה אוֹמֵר שְׁחָטָן לְשֵׁם גָּבוֹהַּ מֵהֶם כְּשֵׁרִין\hdot
לְשֵׁם נָמוּךְ מֵהֶם פְּסוּלִין\hdot
כֵּיצַד קָדְשֵׁי קֳדָשִׁים שֶׁשְּׁחָטָן לְשֵׁם קֳדָשִׁים קַלִּים\hdot
פּסוּלִין\hdot
קֳדָשִׁים קַלִּים שֶׁשְּׁחָטָן לְשֵׁם קָדְשֵׁי קֳדָשִׁים\hdot
כְּשֵׁרִין\hdot
הַבְּכוֹר וְהַמַּעֲשֵׂר שֶׁשְּׁחָטָן לְשֵׁם שְׁלָמִים כְּשֵׁרִין\hdot
וּשְׁלָמִים שֶׁשְּׁחָטָן לְשֵׁם בְּכוֹר וּלְשֵׁם מַעֲשֵׂר פְּסוּלִין׃}

\halacha{הַפֶּסַח שֶׁשְּׁחָטוֹ בְּשַׁחְרִית בְּאַרְבָּעָה עָשָׂר שֶׁלֹּא לִשְׁמוֹ\hdot
רִבִּי יְהוֹשֻׁעַ מַכְשִׁיר כְּאִלּוּ נִשְׁחַט בִּשְׁלֹשָׁה עָשָׂר\hdot
בֶּן בְּתֵירָא פּוֹסֵל כְּאִלּוּ נִשְׁחַט בֵּין הָאַרְבָּעִים\hdot
אָמַר שִׁמְעוֹן בֶּן עֲזָאי מְקֻבָּל אֲנִי מִפִּי שִׁבְעִים וּשְׁנַיִם זְקֵנִים\hdot
בְּיוֹם שֶׁהוֹשִׁיבוּ רִבִּי אֶלְעָזָר בֶּן עֲזַרְיָה בִּישִׁיבָה שֶׁכָּל הַזְּבָחִים הַנֶּאֱכָלִים שֶׁנִּזְבְּחוּ שֶׁלֹּא לִשְׁמָן כְּשֵׁרִין\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִן הַפֶּסַח וּמִן הַחַטָּאת\hdot
וְלֹא הוֹסִיף בֶּן עֲזָאי אֶלָּא הָעוֹלָה\hdot
וְלֹא הוֹדוּ לוֹ חֲכָמִים׃}

\halacha{הַפֶּסַח וְהַחַטָּאת שֶׁשְּׁחָטָן שֶׁלֹּא לִשְׁמָן\hdot
קִבֵּל וְהָלַךְ וְזָרַק שֶׁלֹּא לִשְׁמָן\hdot
אוֹ לִשְׁמָן וְשֶׁלֹּא לִשְׁמָן\hdot
אוֹ שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן\hdot
פְּסוּלִין\hdot
כֵּיצַד לִשְׁמָן וְשֶׁלֹּא לִשְׁמָן\hdot
לְשֵׁם פֶּסַח וּלְשֵׁם שְׁלָמִים\hdot
שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן\hdot
לְשֵׁם שְׁלָמִים וּלְשֵׁם פֶּסַח\hdot
שֶׁהַזֶּבַח נִפְסָל בְּאַרְבָּעָה דְבָרִים\hdot
בִּשְׁחִיטָה\hdot
וּבְקִבּוּל\hdot
וּבְהִלּוּךְ\hdot
וּבִזְרִיקָה\hdot
רִבִּי שִׁמְעוֹן מַכְשִׁיר בְּהִלּוּךְ\hdot
שֶׁהָיָה רִבִּי שִׁמְעוֹן אוֹמֵר אֵי אֶפְשָׁר שֶׁלֹּא בִשְׁחִיטָה\hdot
וְשֶׁלֹּא בְקַבָּלָה\hdot
וְשֶׁלֹּא בִזְרִיקָה\hdot
אֲבָל אֶפְשָׁר שֶׁלֹּא בְהִלּוּךְ\hdot
שֶׁשּׁוֹחֵט בְּצַד הַמִּזְבֵּחַ וְזוֹרֵק\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר הַמְּהַלֵּךְ בְּמָקוֹם שֶׁהוּא צָרִיךְ לְהַלֵּךְ הַמַּחֲשָׁבָה פוֹסֶלֶת\hdot
וּבִמְקוֹם שֶׁאֵינוֹ צָרְיךְ לְהַלֵּךְ\hdot
אֵין הַמַּחֲשָׁבָה פוֹסֶלֶת׃}
