\tractatehead{מסכת בבא מציעא}{baba mesia}

\perek{שְׁנַיִם}{אוֹחֲזִין בְּטָלֵית\hdot
זֶה אוֹמֵר אֲנִי מְצָאתִיהָ\hdot
וְזֶה אוֹמֵר אֲנִי מְצָאתִיהָ\hdot
זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
וְזֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
זֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מֵחֶצְיָהּ\hdot
וְזֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מֵחֶצְיָהּ וְיַחֲלוֹקוּ\hdot
זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
וְזֶה אוֹמֵר חֶצְיָהּ שֶׁלִּי\hdot
הָאוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מִשְּׁלֹשָׁה חֲלָקִים\hdot
וְהָאוֹמֵר חֶצְיָהּ שֶׁלִּי יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מֵרָבִיעַ\hdot
זֶה נוֹטֵל שְׁלֹשָׁה חֲלָקִים\hdot
וְזֶה נוֹטֵל רָבִיעַ׃}

\halacha{הָיוּ שְׁנַיִם רוֹכְבִין עַל גַּבֵּי בְהֵמָה\hdot
אוֹ שֶׁהָיָה אֶחָד רוֹכֵב וְאֶחָד מַנְהִיג\hdot
זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
וְזֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי\hdot
זֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מֵחֶצְיָהּ\hdot
וְזֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוּת מֵחֶצְיָהּ וְיַחֲלוֹקוּ\hdot
בִּזְמָן שֶׁהֵן מוֹדִים\hdot
אוֹ שֶׁיֵּשׁ לָהֶן עֵדִים\hdot
חוֹלְקִים בְּלֹא שְׁבוּעָה׃}

\halacha{הָיָה רוֹכֵב עַל גַּבֵּי בְהֵמָה\hdot
וְרָאָה אֶת הַמְּצִיאָה וְאָמַר לַחֲבֵירוֹ תְּנָהּ לִי\hdot
נְטָלָהּ וְאָמַר אֲנִי זָכִיתִי בָהּ\hdot
זָכָה בָהּ\hdot
אִם מִשֶּׁנְּתָנָהּ לוֹ אָמַר אֲנִי זָכִיתִי בָהּ תְּחִלָּה\hdot
לֹא אָמַר כְּלוּם׃}

\halacha{רָאָה אֶת הַמְּצִיאָה\hdot
וְנָפַל לוֹ עָלֶיהָ\hdot
וּבָא אַחֵר וְהֶחֱזִיק בָּהּ\hdot
זֶה שֶׁהֶחֱזִיק בָּהּ זָכָה בָהּ\hdot
רָאָה אוֹתָן רָצִין אַחַר מְצִיאָה\hdot
אַחַר צְבִי שָׁבוּר\hdot
אַחַר גּוֹזלָוֹת שֶׁלֹּא פָֽרְחוּ\hdot
וְאָמַר זָֽכְתָה לִי שָׂדִי\hdot
זָֽכְתָה לוֹ\hdot
הָיָה צְבִי רָץ כְּדַרְכּוֹ\hdot
אוֹ שֶׁהָיוּ גוֹזָלוֹת מַפְרִיחִין\hdot
וְאָמַר זָֽכְתָה לִי שָׂדִי\hdot
לֹא אָמַר כְּלוּם׃}

\halacha{מְצִיאַת בְּנוֹ וּבִתּוֹ הַקְּטַנִּים\hdot
מְצִיאַת עַבְדּוֹ וְשִׁפְחָתוֹ הַכְּנַעֲנִים\hdot
מְצִיאַת אִשְׁתּוֹ\hdot
הֲרֵי אֵלּוּ שֶׁלּוֹ\hdot
מְצִיאַת בְּנוֹ וּבִתּוֹ הַגְּדוֹלִים\hdot
מְצִיאַת עַבְדּוֹ וְשִׁפְחָתוֹ הָעִבְרִים\hdot
מְצִיאַת אִשְׁתּוֹ שֶׁגֵּרְשָׁהּ\hdot
אַף עַל פִּי שֶׁלֹּא נָתַן לָהּ כְּתוּבָתָהּ\hdot
הֲרֵי אֵלּוּ שֶׁלָּהֶן׃}

\halacha{מָצָא שִׁטְרֵי חוֹב אִם יֵשׁ בָּהֶן אַחֲרָיוּת נִכָסִים לֹא יַחֲזִיר\hdot
שֶׁבֵּית דִּין נִפְרָעִין מֵהֶן\hdot
אֵין בָּהֶן אַחֲרָיוּת נִכָסִים יַחֲזִיר\hdot
שֶׁאֵין בֵּית דִּין נִפְרָעִין מֵהֶן\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים בֵּין כַּךְ וּבֵין כַּךְ לֹא יַחֲזִיר\hdot
מִפְּנֵי שֶׁבֵּית דִּין נִפְרָעִין מֵהֶן׃}

\halacha{מָצָא גִטֵּי נָשִׁים\hdot
וְשִׁחְרוּרֵי עֲבָדִים\hdot
דַּיְתִּיקִי\hdot
מַתָּנָה וְשׁוֹבָרִים\hdot
הֲרֵי זֶה לֹא יַחֲזִיר\hdot
שֶׁאֲנִי אוֹמֵר כְּתוּבִים הָיוּ\hdot
וְנִמְלַךְ עֲלֵיהֶם שֶׁלֹּא לִתְּנָם׃}

\halacha{מָצָא אִגְּרוֹת שׁוֹם\hdot
וְאִגְּרוֹת מָזוֹן\hdot
שִׁטְרֵי חֲלִיצָה וּמִיאוּנִין\hdot
וְשִׁטְרֵי בֵירוּרִין\hdot
וְכָל מַעֲשֵׁה בֵית דִּין\hdot
הֲרֵי זֶה יַחֲזִיר\hdot
מָצָא בַחֲפִיסָה\hdot
אוֹ בִדְלוֹסְקָמָא\hdot
תַּכְרִיךְ שֶׁל שְׁטָרוֹת\hdot
אוֹ אֲגֻדָּה שֶׁל שְׁטָרוֹת\hdot
הֲרֵי זֶה יַחֲזִיר\hdot
וְכַמָּה הִיא אֲגֻדָּה שֶׁל שְׁטָרוֹת\hdot
שְׁלֹשָׁה קְשׁוּרִין זֶה בַזֶּה\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר אֶחָד הַלֹּוֶה מִשְּׁלֹשָׁה\hdot
יַחֲזִיר לַלֹּוֶה\hdot
שְׁלֹשָׁה הַלּוֹוִין מֵאֶחָד\hdot
יַחֲזִיר לַמַּלְוֶה\hdot
מָצָא שְׁטָר בֵּין שְׁטָרוֹתָיו\hdot
וְאֵינוֹ יוֹדֵעַ מַה טִּיבוֹ\hdot
יְהֵא מֻנָּח עַד שֶׁיָּבֹא אֵלִיָּהוּ\hdot
אִם יֵשׁ עִמָּהֶן סִמְפוֹנוֹת\hdot
יַעֲשֶׂה מַה שֶּׁבַּסִּמְפוֹנוֹת׃}

\perek{אֵלּוּ}{מְצִיאוֹת שֶׁלּוֹ\hdot
וְאֵלּוּ חַיָּב לְהַכְרִיז\hdot
אֵלּוּ מְצִיאוֹת שֶׁלּוֹ\hdot
מָצָא פֵירוֹת מְפֻזָּרִין\hdot
מָעוֹת מְפֻזָּרוֹת\hdot
כְּרֵיכוֹת בִּרְשׁוּת הָרַבִּים\hdot
וְעִגּוּלֵי דְבֵילָה\hdot
כִּכָּרוֹת שֶׁל נַחְתּוֹם\hdot
מַחְרוּזוֹת שֶׁל דָּגִים\hdot
וַחֲתִיכוֹת שֶׁל בָּשָׂר\hdot
וְגִזֵּי צֶמֶר הַבָּאוֹת מִמְּדִינָתָן\hdot
וַאֲנִיצֵי פִשְׁתָּן\hdot
וּלְשׁוֹנוֹת שֶׁל אַרְגָּמָן\hdot
הֲרֵי אֵלּוּ שֶׁלּוֹ\hdot
(דִּבְרֵי רִבִּי מֵאִיר) רִבִּי יְהוּדָה אוֹמֵר כָּל שֶׁיֵּשׁ בּוֹ שִׁנּוּי חַיָּב לְהַכְרִיז\hdot
כֵּיצַד\hdot
מָצָא עִגּוּל וּבְתוֹכוֹ חֶרֶס\hdot
כִּכָּר וּבְתוֹכוֹ מָעוֹת\hdot
רִבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר כָּל כְּלִי אַנְפֻּרְיָא אֵינוֹ חַיָּב לְהַכְרִיז׃}
