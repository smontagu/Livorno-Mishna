\tractatehead{מסכת תמיד}{tamid}

\perek{בִּשְׁלֹשָׁה}{מְקוֹמוֹת הַכֹּהֲנִים שׁוֹמְרִים בְּבֵית הַמִּקְדָּשׁ\hdot
בְּבֵית אַבְטִינָס\hdot
בְּבֵית הַנִּיצוֹץ\hdot
וּבְבֵית הַמּוֹקֵד\hdot
בֵּית אַבְטִינָס וּבֵית הַנִּיצוֹץ הָיוּ עֲלִיּוֹת\hdot
וְהָרוֹבִים שׁוֹמְרִים שָׁם\hdot
בֵּית הַמּוֹקֵד כִּיפָה\hdot
וּבַיִת גָּדוֹל הָיָה מֻקַּף רוֹבְדִים שֶׁל אֶבֶן\hdot
וְזִקְנֵי בֵית אָב יְשֵׁנִין שָׁם\hdot
וּמַפְתֵּחוֹת הָעֲזָרָה בְּיָדָם\hdot
וּפִרְחֵי כְהוּנָה אִישׁ כְּסָתוֹ בָּאָרֶץ\hdot
לֹא הָיוּ יְשֵׁנִין בְּבִגְדֵי קֹדֶשׁ\hdot
אֶלָּא פוֹשְׁטִין (אוֹתָן) וּמְקַפְּלִין אוֹתָן\hdot
וּמַנִּיחִין אוֹתָן תַּחַת רָאשֵׁיהֶן\hdot
וּמִתְכַּסִּין בִּכְסוּת עַצְמָן\hdot
אִירַע קֶרִי לְאַחַד מֵהֶן\hdot
יוֹצֵא וְהוֹלֵךְ לוֹ בִּמְסִיבָה הַהוֹלֶכֶת תַּחַת הַבִּירָה\hdot
וְנֵרוֹת דּוֹלְקִין מִכָּאן וּמִכָּאן\hdot
עַד שֶׁהוּא מַגִּיעַ לְבֵית הַטְּבִילָה\hdot
וּמְדוּרָה הָֽיְתָה שָׁם\hdot
וּבֵית כִּסֵּא שֶׁל כָּבוֹד\hdot
וְזֶה הָיָה כְבוֹדוֹ\hdot
מְצָאוֹ נָעוּל יוֹדֵע שֶׁיֵּשׁ שָׁם אָדָם\hdot
פָּתוּחַ יוֹדֵעַ שֶׁאֵין שָׁם אָדָם\hdot
יָרַד וְטָבַל\hdot
עָלָה וְנִסְתַּפַּג\hdot
וְנִתְחַמַּם כְּנֶגֶד הַמְּדוּרָה\hdot
בָּא וְיָשַׁב לוֹ אֵצֶל אֶחָיו הַכֹּהֲנִים\hdot
עַד שֶׁהַשְּׁעָרִים נִפְתָּחִים וְיוֹצֵא וְהוֹלֵך לוֹ׃}

\halacha{מִי שֶׁהוּא רוֹצֶה לִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
מַשְׁכִּים וְטוֹבֵל עַד שֶׁלֹּא יָבֹא הַמְּמֻנֶּה\hdot
וְכִי בְּאֵיזוֹ שָׁעָה הַמְּמֻנֶּה בָא\hdot
לֹא כָל הָעִתִּים שָׁווֹת\hdot
פְּעָמִים שֶׁהוּא בָא מִקְּרִיאַת הַגֶּבֶר\hdot
אוֹ סָמוּךְ לוֹ מִלְּפָנָיו\hdot
אוֹ מִלְּאַחֲרָיו\hdot
הַמְּמֻנֶּה בָא וְדוֹפֵק עֲלֵיהֶם\hdot
וְהֵם פּוֹתְחִים לוֹ\hdot
אָמַר לָהֶן מִי שֶׁטָּבַל יָבֹא וְיָפִיס\hdot
הֵפִיסוּ זָכָה מִי שֶׁזָּכָה׃}

\halacha{נָטַל אֶת הַמַּפְתֵּחַ וּפָתַח אֶת הָפִּשְׁפֵּשׁ\hdot
וְנִכְנַס מִבֵּית הַמּוֹקֵד לָעֲזָרָה\hdot
וְנִכְנְסוּ אַחֲרָיו\hdot
וּשְׁתֵּי אֲבוּקוֹת שֶׁל אוֹר בְּיָדָם\hdot
וְנֶחְלְקוּ לִשְׁתֵּי כִתּוֹת\hdot
אֵלּוּ הוֹלְכִין בָּאַכְסַדְרָה דֶּרֶךְ הַמִּזְרָח\hdot
וְאֵלּוּ הוֹלְכִין בָּאַכְסַדְרָה דֶּרֶךְ הַמַּעֲרָב\hdot
הָיוּ בוֹדְקִין וְהוֹלְכִין עַד שֶׁמַּגִּיעִין לִמְקוֹם (בֵּית) עוֹשֵׂי חֲבִיתִים\hdot
הִגִּיעוּ אֵלּוּ וָאֵלּוּ אָֽמְרוּ שָׁלוֹם\hdot
הַכֹּל שָׁלוֹם\hdot
הֶעֱמִידוּ עוֹשֵׂי חֲבִיתִים לַעֲשׂוֹת חֲבִיתִים׃}

\halacha{מִי שֶׁזָּכָה לִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
הוּא יִתְרוֹם אֶת הַמִּזְבֵּחַ\hdot
וְהֵם אוֹמְרִים לוֹ הִזָּהֵר שֶׁמֶא תִגַּע בַּכֶּלִי\hdot
עַד שֶׁתְּקַדֵּשׁ יָדֶיךָ וְרַגְלֶיךָ מִן הִכִּיּוֹר\hdot
וַהֲרֵי הַמַּחְתָּה נְתוּנָה בַמִּקְצוֹעַ\hdot
בֵּין הַכֶּבֶשׁ לַמִּזְבֵּחַ\hdot
בְּמַעֲרָבוֹ שֶׁל כֶּבֶשׁ\hdot
אֵין אָדָם נִכְנָס עִמּוֹ וְלֹא נֵר בְּיָדוֹ\hdot
אֶלָּא מְהַלֵּךְ לְאוֹר הַמַּעֲרָכָה\hdot
לֹא הָיוּ רוֹאִין אוֹתוֹ\hdot
וְלֹא שׁוֹמְעִין אֶת קוֹלוֹ\hdot
עַד שֶׁשּׁוֹמְעִין קוֹל הָעֵץ שֶׁעָשָׂה בֶן קָטִין מוּכָנִי לַכִּיּוֹר\hdot
וְהֵם אוֹמְרִים הִגִּיעַ עֵת\hdot
קִדֵּשׁ יָדָיו וְרַגְלָיו מִן הַכִּיּוֹר\hdot
נָטַל מַחְתַּת הַכֶּסֶף וְעָלָה לְרֹאשׁ הַמִּזְבֵּחַ\hdot
וּפִנָּה אֶת הַגֶּחָלִים הֵילָךְ (וְהֵילָךְ)\hdot
וְחִתָּה מִן הַמַּאֻכָּלוֹת הַפְּנִימִיּוֹת וְיָרַד\hdot
הִגִּיעַ לָרִצְפָה הָפַךְ פָּנָיו לַצָּפוֹן\hdot
וְהָלַךְ לְמִזְרָחוֹ שֶׁל כֶּבֶשׁ כְּעֶשֶׂר אַמּוֹת\hdot
צָבַר אֶת הַגֶּחָלִים עַל גַּבֵּי הָרִצְפָה\hdot
רָחוֹק מִן הַכֶּבֶשׁ שְׁלֹשָׁה טְפָחִים\hdot
מָקוֹם שֶׁנּוֹתְנִין שָׁם מֻרְאַת הָעוֹף\hdot
וְדִשּׁוּן מִזְבֵּחַ הַפְּנִימִי\hdot
וְהַמְּנוֹרָה׃}
