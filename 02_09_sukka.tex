\tractatehead{מסכת סוכה}{sukka}

\perek{סֻכָּה}{שֶׁהִיא גְבוֹהָה לְמַעְלָה מֵעֶשְׂרִים אַמָּה פְּסוּלָה\hdot
רִבִּי יְהוּדָה מַכְשִׁיר\hdot
וְשֶׁאֵינָהּ גְּבוֹהָה עֲשָׂרָה טְפָחִים\hdot
וְשֶׁאֵין לָהָּ שָׁלֹשׁ דְּפַנּוֹת וְשֶׁחַמָּתָהּ מְרֻבָּה מִצִּלָּתָהּ\hdot
פְּסוּלָה\hdot
סֻכָּה יְשָׁנָה\hdot
בֵּית שַׁמָּאי פּוֹסְלִין\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot
וְאֵיזוֹ הִיא סֻכָּה יְשָׁנָה\hdot
כָּל שֶׁעֲשָׂאָהּ קֹדֶם לַחַג שְׁלֹשִׁים יוֹם\hdot
אֲבָל אִם עֲשָׂאָהּ לְשֵׁם חַג\hdot
אֲפִילוּ מִתְּחִלַּת הַשָּׁנָה כְּשֵׁרָה׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ תַּחַת הָאִילָן כְּאִלּוּ עֲשָׂאָהּ בְּתוֹךְ הַבַּיִת\hdot
סֻכָּה עַל גַּבֵּי סֻכָּה\hdot
הָעֶלְיוֹנָה כְּשֵׁרָה\hdot
וְהַתַּחְתּוֹנָה פְּסוּלָה\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
אִם אֵין דְּיוּרִין בָּעֶלְיוֹנָה\hdot
הַתַּחְתּוֹנָה כְּשֵׁרָה׃}
