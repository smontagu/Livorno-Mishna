\perek[597]{חֲמִשָּׁה}{לֹא}{יִתְרוֹמוּ\hdot
וְאִם תָֽרְמוּ אֵין תְּרוּמָתָן תְּרוּמָה\hdot
הַחֵרֵשׁ וְהַשּׁוֹטֶה וְהַקָּטָן וְהַתּוֹרֵם אֶת שֶׁאֵינוֹ שֶׁלּוֹ\hdot
נָכְרִי שֶׁתָּרַם אֶת שֶׁל יִשְׂרָאֵל אֲפִילוּ בִרְשׁוּת אֵין תְּרוּמָתוֹ תְרוּמָה׃}

\halacha{חֵרֵשׁ הַמְּדַבֵּר וְאֵינוֹ שׁוֹמֵעַ לֹא יִתְרוֹם וְאִם תָּרַם תְּרוּמָתוֹ תְּרוּמָה\hdot
חֵרֵשׁ שֶׁדִּבְּרוּ בוֹ חֲכָמִים בְּכָל מָקוֹם\hdot
שֶׁאֵינוֹ לֹא שׁוֹמֵעַ וְלֹא מְדַבֵּר׃}

\halacha[1038]{קָטָן שֶׁלֹּא הֵבִיא שְׁתֵּי שְׂעָרוֹת רִבִּי יְהוּדָה אוֹמֵר תְּרוּמָתוֹ תְרוּמָה\hdot
רִבִּי יוֹסֵי אוֹמֵר אִם עַד שֶׁלֹּא בָא לְעוֹנַת נְדָרִים אֵין תְּרוּמָתוֹ תְרוּמָה
וּמִשֶּׁבָּא לְעוֹנַת נְדָרִים תְּרוּמָתוֹ תְרוּמָה׃}

\halacha{אֵין תּוֹרְמִין זֵיתִים עַל הַשֶּׁמֶן וְלֹא עֲנָבִים עַל הַיַּיִן וְאִם תָּֽרְמוּ בֵּית שַׁמָּאי אוֹמְרִים תְּרוּמַת עַצְמָן בָּהֶם\hdot
וּבֵית הִלֵּל אוֹמְרִים אֵין תְּרוּמָתָן תְּרוּמָה׃}

\halacha[336]{אֵין תּוֹרְמִין מִן הַלֶּקֶט וּמִן הַשִּׁכְחָה וּמִן הַפֵּאָה וּמִן הַהֶפְקֶר\hdot
וְלֹא מִמַּעֲשֵׂר רִאשׁוֹן (שֶׁנִּטְּלָה תְּרוּמָתוֹ) וְלֹא מִמַּעֲשֵׂר שֵׁנִי וְהֶקְדֶּשׁ (שֶׁנִּפְדּוּ) וְלֹא מִן הַחַיָּיב עַל הַפָּטוּר וְלֹא מִן הַפָּטוּר עַל הַחַיָּיב\hdot
וְלֹא מִן הַתָּלוּשׁ עַל הַמְּחֻבָּר וְלֹא מִן הַמְּחֻבָּר עַל הַתָּלוּשׁ וְלֹא מִן הֶחָדָשׁ עַל הַיָּשָׁן וְלֹא מִן הַיָּשָׁן עַל הֶחָדָשׁ\hdot
וְלֹא מִפֵּירוֹת הָאָרֶץ עַל פֵּירוֹת חוּצָה לָאָרֶץ וְלֹא מִפֵּירוֹת חוּצָה לָאָרֶץ עַל פֵּירוֹת הָאָרֶץ\hdot
וְאִם תָּֽרְמוּ אֵין תְּרוּמָתָן תְּרוּמָה׃}

\halacha[291]{חֲמִשָּׁה לֹא יִתְרוֹמוּ וְאִם תָּֽרְמוּ תְּרוּמָתָן תְּרוּמָה\hdot
הָאִלֵּם וְהַשִּׁכּוֹר וְהֶעָרוֹם וְהַסּוֹמֵא וּבַעַל קֶרִי לֹא יִתְרוֹמוּ וִאִם תָּֽרְמוּ תְּרוּמָתָן תְּרוּמָה׃}

\halacha[182]{אֵין תּוֹרְמִין לֹא בְמִדָּה וְלֹא בְמִשְׁקָל וְלֹא בְמִנְיָן\hdot
אֲבָל תּוֹרֵם הוּא אֶת הַמּדָוּד וְאֶת הַשָּׁקוּל וְאֶת הַמָּנוּי\hdot
אֵין תּוֹרְמִין בְּסַל וּבְקֻפָּה שֶׁהֵם שֶׁל מִדָּה\hdot
אֲבָל תּוֹרֵם הוּא בָהֶן חֶצְיָין וְשָׁלִישָן\hdot
לֹא יִתְרוֹם בַּסֵּאָה חֶצְיָהּ\hdot
שֶׁחֶצְיָהּ מִדָּה׃}

\halacha{אֵין תּוֹרְמִין שֶׁמֶן עַל זֵיתִים הַנִּכְתָּשִׁין וְלֹא יַיִן עַל עֲנָבִים הַנִּדְרָכוֹת\hdot
וְאִם תָּרַם תְּרוּמָתוֹ תְרוּמָה וְיַחֲזוֹר וְיִתְרוֹם\hdot
הָרִאשׁוֹנָה מְדַמַּעַת בִּפְנֵי עַצְמָהּ וְחַיָּיבִין עָלֶיהָ חוֹמֶשׁ אֲבָל לֹא שְׁנִיָּה׃}

\halacha[729]{וְתוֹרְמִין שֶׁמֶן עַל זֵיתִים הַנִּכְבָּשִׁין\hdot
וְיַיִן עַל עֲנָבִים לַעֲשׂוֹתָן צִמּוּקִין\hdot
הֲרֵי שֶׁתָּרַם שֶׁמֶן עַל זֵיתִים לַאֲכִילָה וְזֵיתִים עַל זֵיתִים לַאֲכִילָה וְיַיִן עַל עֲנָבִים לַאֲכִילָה וַעֲנָבִים עַל עֲנָבִים לַאֲכִילָה\hdot
וְנִמְלַךְ לְדוֹרְכָן\hdot
אֵינוֹ צָרִיךְ לִתְרוֹם׃}

\halacha[185]{אֵין תּוֹרְמִין מִדָּבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ\hdot
עַל דָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ\hdot
וְלֹא מִדָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ\hdot
עַל דָּבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ\hdot
וְלֹא מִדָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ עַל דָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ\hdot
וְאִם תָּֽרְמוּ תְּרוּמָתָן תְּרוּמָה׃}

\perek[406]{אֵין}{תּוֹרְמִין}{מִטָּהוֹר עַל הַטָּמֵא\hdot
וְאִם תָּֽרְמוּ תְּרוּמָתָן תְּרוּמָה\hdot
בֶּאֱמֶת אָֽמְרוּ הָעִגּוּל שֶׁל דְּבֵילָה שֶׁנִּטְמָא מִקְצָתוֹ\hdot
תּוֹרֵם מִן הַטָּהוֹר שֶׁיֵּשׁ בּוֹ עַל הַטָּמֵא שֶׁיֵּשׁ בּוֹ\hdot
וְכֵן אֲגֻדָּה שֶׁל יָרָק וְכֵן עֲרֵמָה\hdot
הָיוּ שְׁנֵי עֲגוּלִין שְׁתֵּי אֲגֻּדּוֹת\hdot
שְׁתֵּי עֲרֵמוֹת\hdot
אַחַת טְמֵאָה וְאַחַת טְהוֹרָה\hdot
לֹא יִתְרוֹם מִזֶּה עַל זֶה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר תּוֹרְמִין מִן הַטָּהוֹר עַל הַטָּמֵא׃}

\halacha[146]{אֵין תּוֹרְמִין מִן הַטָּמֵא עַל הַטָּהוֹר\hdot
 וְאִם תָּרַם שׁוֹגֵג תְּרוּמָתוֹ תְרוּמָה\hdot
וּמֵזִיד לֹא עָשָׂה כְלוּם\hdot
וְכֵן בֶּן לֵוִי שֶׁהָיָה לוֹ מַעֲשֵׂר טֶבֶל הָיָה מַפְרִישׁ עָלָיו וְהוֹלֵךְ\hdot
שׁוֹגֵג מַה שֶּׁעָשָׂה עָשׂוּי\hdot
מֵזִיד לֹא עָשָׁה כְלוּם\hdot
רִבִּי יְהוּדָה אוֹמֵר עִם הָיָה יוֹדֵעַ בּוֹ בַּתְּחִלָּה\hdot
אַף עַל פִּי שֶׁהוּא שׁוֹגֵג לֹא עָשָׂה כְלוּם׃}

\halacha{הַמַּטְבִּיל כֵּלִים בְּשַׁבָּת שׁוֹגֵג יִשְׁתַּמֵּשׁ בָּהֶם\hdot
מֵזִיד לֹא יִשְׁתַּמֵּשׁ בָּהֶם\hdot
הַמְּעַשֵּׂר וְהַמְּבַשֵּׁל בְּשַׁבָּת\hdot
שׁוֹגֵג יֹאכַל\hdot
מֵזִיד לֹא יֹאכַל\hdot
הַנּוֹטֵעַ בְּשַׁבָּת\hdot
שׁוֹגֵג יְקַיֵּים\hdot
מֵזִיד יַעֲקוֹר וּבַשְּׁבִיעִית בֵּין שׁוֹגֵג בֵּין מֵזִיד יַעֲקוֹר׃}

\halacha[760]{אין תּוֹרְמִין מִמִּין עַל שֶׁאֵינו מִינוֹ\hdot
וְאִם תָּרַם אֵין תְּרוּמָתוֹ תְרוּמָה\hdot
כָּל מִין חִטִּים אֶחָד\hdot
כָּל מִין תְּאֵנִים וּגְרוֹגְרוֹת וּדְבֵילָה אֶחָד\hdot
וְתוֹרֵם מִזֶּה עַל זֶה\hdot
כָּל מָקוֹם שֶׁיֵּשׁ כֹּהֵן תּוֹרֵם מִן הַיָּפֶה\hdot
וְכָל מָקוֹם שֶׁאֵין כֹּהֵן תּוֹרֵם מִן הַמִּתְקַיֵּים\hdot
רִבִּי יְהוּדָה אוֹמֵר לְעוֹלָם הוּא תוֹרֵם מִן הַיָּפֶה׃}

\halacha{תּוֹרְמִין בָּצֵל קָטָן שָׁלֵם וְלֹא חֲצִי בָצֵל גָּדוֹל\hdot
רִבִּי יְהוּדָה אוֹמֵר לֹא כִי אֶלָּא חֲצִי בָצֵל גָּדוֹל\hdot
וְכֵן הָיָה רִבִּי יְהוּדָה אוֹמֵר תּוֹרְמִין בְּצָלִים מִבְּנֵי הַמְּדִינָה עַל הַכּוֹפְרִים\hdot
אֲבָל לֹא מִן הַכּוֹפְרִים עַל בְּנֵי הַמְּדִינָה\hdot
מִפְּנֵי שֶׁהוּא מַאֲכַל פּוֹלִיטִיקִין׃}

\halacha[165]{וְתוֹרְמִין זֵיתֵי שֶׁמֶן עַל זֵיתֵי כֶבֶשׁ\hdot
וְלֹא זֵיתֵי כֶבֶשׁ עַל זֵיתֵי שֶׁמֶן\hdot
וְיַיִן שֶׁאֵינוֹ מְבֻשָּׁל עַל הַמְּבֻשָּׁל\hdot
וְלֹא מִן הַמְּבֻשָּׁל עַל שֶׁאֵינוֹ מְבֻשָּׁל\hdot
זֶה הַכְּלָל כָּל שֶׁהוּא כִּלְאַיִם בַּחֲבֵירוֹ לֹא יִתְרוֹם מִזֶּה עַל זֶה\hdot
אֲפִילוּ מִן הַיָּפֶה עַל הָרָע\hdot
וְכָל שֶׁאֵינוּ כִּלְאַיִם בַּחֲבֵירוֹ תּוֹרֵם מִן הַיָּפֶה עַל הָרָע\hdot
אֲבָל לֹא מִן הָרָע עַל הַיָּפֶה\hdot
וְאִם תָּרַם מִן הָרָע עַל הַיָּפֶה תְּרוּמָתוֹ תְרוּמָה\hdot
חוּץ מִן הַזּוֹנִין עַל הַחִטִּים שֶׁאֵינָן אֹכֶל\hdot
וְהַקִּישׁוּת\hdot
וְהַמְּלַפָפוֹן\hdot
מִין אֶחָד\hdot
רִבִּי יְהוּדָה אוֹמֵר שְׁנֵי מִינִין׃}

\perek[273]{הַתּוֹרֵם}{קִישׁוּת\hdot}{וְנִמְצֵאת מָרָה\hdot
אֲבַטִּיחַ וְנִמְצֵאת סְרוּחָה\hdot
תְּרוּמָה וְיַחֲזוֹר וְיִתְרוֹם\hdot
הַתּוֹרֵם חָבִית שֶׁל יַיִן וְנִמְצֵאת שֶׁל חוֹמֶץ\hdot
אִם יָדוּעַ שֶׁהָֽיְתָה שֶׁל חֹמֶץ עַד שֶׁלֹּא תְּרָמָהּ אֵינָהּ תְּרוּמָה\hdot
אִם מִשֶּׁתְּרָמָהּ הֶחֱמִיצָה הֲרֵי זוֹ תְרוּמָה\hdot

הָרִאשׁוֹנָה אֵינָהּ מְדַמַּעַת בִּפְנֵי עַצְמָהּ\hdot

\halacha[2856]{הַשֻּׁתָּפִין שֶׁתָּֽרְמוּ זֶה אַחַר זֶה\hdot
