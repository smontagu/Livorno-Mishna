\perek{מָבוֹי}{שֶׁהוּא גָבוֹהַּ\hdot
לְמַעְלָה מֵעֶשְׂרִים אַמָּה יִמָּעֵט\hdot
רִבִּי יְהוּדָה אוֹמֵר אֵינוֹ צָרִיךְ\hdot
וְהָרַחַב מֵעֶשֶׂר אַמּוֹת יִמָּעֵט\hdot
וְאִם יֵשׁ לוֹ צוּרַת פֶּתַח\hdot
אַף עַל פִּי שֶׁהוּא רַחַב מֵעֶשֶׂר אַמּוֹת אֵין צָרִיךְ לְמָעֵט׃}

\halacha[109]{הֶכְשֵׁר מָבוֹי\hdot
בֵּית שַׁמָּאי אוֹמְרִים לֶחִי וְקוֹרָה\hdot
וּבֵית הִלֵּל אוֹמְרִים לֶחִי אוֹ קוֹרָה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר לְחָיַיִן\hdot
מִשּׁוּם רִבִּי יִשְׁמָעֵאל אָמַר תַּלְמִיד אֶחָד לִפְנֵי רִבִּי עֲקִיבָא\hdot
לֹא נֶחְלְקוּ בֵּית שַׁמָּאי וּבֵית הִלֵּל עַל מָבוֹי שֶׁהוּא פָחוּת מֵאַרְבַּע אַמּוֹת\hdot
שֶׁהוּא נִתָּר אוֹ בְלֶחִי אוֹ בְקוֹרָה\hdot
עַל מַה נֶּחֱלָקוּ\hdot
עַל רַחַב מֵאַרְבַּע אַמּוֹת וְעַד עֶשֶׂר\hdot
שֶׁבֵּית שַׁמָּאי אוֹמְרִים לֶחִי וְקוֹרָה\hdot
וּבֵית הִלֵּל אוֹמְרִים אוֹ לֶחִי אוֹ קוֹרָה\hdot
אָמַר רִבִּי עֲקִיבָא עַל זֶה וְעַל זֶה נֶחֱלָקוּ׃}

\halacha{הַקּוֹרָה שֶׁאָֽמְרוּ רְחָבָה כְּדֵי לְקַבֵּל אָרִיחַ\hdot
וְאָרִיחַ חֲצִי לְבֵנָה שֵֶׁל שְׁלֹשָׁה טְפָחִים\hdot
דַּיָּהּ לַקּוֹרָה שֶׁתְּהֵא רְחָבָה טֶפַח\hdot
כְּדֵי לְקַבֵּל אָרִיחַ לְאָרְכּוֹ׃}

\halacha{רְחָבָה כְּדֵי לְקַבֵּל אָרִיחַ וּבְרִיאָה כְּדֵי לְקַבֵּל אָרִיחַ\hdot
רִבִּי יְהוּדָה אוֹמֵר רְחָבָה אַף עַל פִּי שֶׁאֵינָהּ בְּרִיאָה׃}

\halacha[1132]{הָֽיְתָה שֶׁל קַשׁ אוֹ שֶׁל קָנִים\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא שֶׁל מַתֶּכֶת\hdot
עֲקוּמָה\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא פְשׁוּטָה\hdot
עֲגוּלָה\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא מְרֻבַּעַת\hdot
כָּל שֶׁיֵּשׁ בְּהֵקֵיפוֹ שְׁלֹשָׁה טְפָחִים\hdot
יֵשׁ בּוֹ רוֹחַב טֶפַח׃}

\halacha{לְחָיַיִן שֶׁאָֽמְרוּ\hdot
גָּבְהָן עֲשָׁרָה טְפָחִים\hdot
וְרָחְבָּן וְעָבְיָן כָּל שֶׁהוּא\hdot
רִבִּי יוֹסֵי אוֹמֵר רָחְבָּן שְׁלֹשָׁה טְפָחִים׃}

\halacha{בַּכֹּל עוֹשִׂין לְחָיַיִן\hdot
אֲפִילוּ בְדָבָר שֶׁיֵּשׁ בּוֹ רוּחַ חַיִּים\hdot
וְרִבִּי יוֹסֵי אוֹסֵר\hdot
וּמְטַמֵּא מִשּׁוּם גֹּלֶל\hdot
וְרִבִּי מֵאִיר מְטַהֵר\hdot
וְכוֹתְבִין עָלָין גִטֵּי נָשִׁים\hdot
וְרִבִּי יוֹסֵי הַגְּלִילִי פּוֹסֵל׃}

\halacha[2189]{שְׁיָירָה שֶׁחָֽנְתָה בַּבִּקְעָה וְהִקִּיפוּהָ בִּכְלֵי בְהֵמָה\hdot
מְטַלְטְלִין בְּתוֹכָהּ\hdot
וּבִלְבַד שֶׁיְּהֵא גָּדֵר גָּבוֹהַּ עֲשָׂרָה טְפָחִים\hdot
וְלֹא יְהוּ פְרָצוֹת יְתֵרוֹת עַל הַבִּנְיָן\hdot
כָּל פִּרְצָה שֶׁהִיא כְעֶשֶׂר אַמּוֹת מֻתֶּרֶת\hdot
מִפְּנֵי שֶׁהִיא כְפֶתַח\hdot
יָתֵר מִכָּאן אָסוּר׃}

\halacha[181]{מַקִּיפִין שְׁלֹשָׁה חֲבָלִים זֶה לְמַעְלָה מִזֶּה\hdot
וְזֶה לְמַעְלָה מִזֶּה\hdot
וּבִלְבַד שֶׁלֹּא יְהֵא בֵּין חֶבֶל לַחֲבֵירוֹ שְׁלֹשָׁה טְפָחִים\hdot
שִׁיעוּר חֲבָלִים וְעוֹבְיָן יָתֵר עַל טֶפַח\hdot
כְּדֵי שֶׁיְּהֵא הַכֹּל עֲשָׂרָה טְפָחִים׃}

\halacha[1642]{מַקִּיפִין בַּקָּנִים וּבִלְבַד שֶׁלֹּא יְהֵא בֵּין קָנֶה לַחֲבֵירוֹ שְׁלֹשָׁה טְפָחִים\hdot
בַּשְּׁיָירָא דִּבְּרוּ\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
וַחֲכָמִים אוֹמְרִים לֹא דִּבְּרוּ בַּשְּׁיָירָא אֶלָּא בַּהוֹוֶה\hdot
כָּל מְחִיצָה שֶׁאֵינָהּ שֵֶׁל שְׁתִי וְשֶׁל עֶרֶב אֵינָהּ מְחִיצָה\hdot
דִּבְרֵי רִבִּי יוֹסֵי בַּר יְהוּדָה\hdot
וַחֲכָמִים אוֹמְרִים מִשְּׁנֵי דְבָרִים\hdot
אַרְבָּעָה דְבָרִים פָּֽטְרוּ בַּמַּחֲנֶה\hdot
מְבִיאִין עֵצִים מִכָּל מָקוֹם\hdot
וּפְטוּרִים מֵרְחִיצַת יָדַיִם\hdot
וּמִדְּמָאי\hdot
וּמִלְּעָרֵב׃}

\perek[552]{עוֹשִׂין}{פַּסִּין לַבִּירָאוֹת\hdot
אַרְבָּעָה דְיוּמְדִין נִרְאִין כִּשְׁמֹנָה\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
רִבִּי מֵאִיר אוֹמֵר שְׁמֹנָה נִרְאִין כִּשְׁנֵים עָשָׂר\hdot
אַרְבָּעָה דְיוּמְדִין\hdot
וְאַרְבָּעָה פְשׁוּטִים\hdot
גָּבְהָן עֲשָׂרָה טְפָחִים\hdot
וְרָחְבָּן שִׁשָּׁה\hdot
וְעָבְיָן כָּל שֶׁהוּא\hdot
וּבֵינֵיהֶן כִּמְלֹא שְׁתֵּי רְבָקוֹת שֶׁל שָׁלֹשׁ שָׁלֹשׁ בָּקָר דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר שֶׁל אַרְבַּע אַרְבַּע\hdot
קְשׁוּרוֹת וְלֹא מוּתָרוֹת\hdot
אַחַת נִכְנֶסֶת וְאַחַת יוֹצְאָה׃}

\halacha{מוּתָר לְהַקְרִיב לַבְּאֵר\hdot
וּבִלְבַד שֶׁתְּהֵא פָרָה רֹאשָׁהּ וְרֻבָּהּ בִּפְנִים וְשׁוֹתָה\hdot
מוּתָר לְהַרְחִיק כָּל שֶׁהוּא\hdot
וּבִלְבַד שֶׁיַּרְבֶּה בַפַּסִּין׃}

\halacha[483]{רִבִּי יְהוּדָה אוֹמֵר עַד בֵּית סָאתַיִם\hdot
אָֽמְרוּ לוֹ לֹא אָֽמְרוּ בֵּית סָאתַיִם\hdot
אֶלָּא לַגִּנָּה וְלַקַּרְפָּף\hdot
אֲבָל אִם הָיָה דִיר\hdot
אוֹ סֹהַר\hdot
אוֹ מֻקְצֶה\hdot
אוֹ חָצֵר\hdot

\halacha[552]{רִבִּי יְהוּדָה אוֹמֵר אִם הָֽיְתָה דֶרֶךְ (הָרַבִּים) מַפְסִיקָתָן\hdot
יְסַלְּקֶנָּה לִצְדָדִין\hdot

אֶחָד בּוֹר הָרַבִּים וּבְאֵר הָרַבִּים\hdot

וְלַשְּׁאַר עוֹשִׂין חֲגוֹרָה גְבוֹהָה עֲשָׂרָה טְפָחִים׃}

\halacha[574]{וְעוֹד אָמַר רִבִּי יְהוּדָה בֶן בָּבָא\hdot
הַגִּנָּה וְהַקַּרְפָּף שֶׁהִיא שִׁבְעִים אַמָּה וְשִׁרַיִּים\hdot
עַל שִׁבְעִים אַמָּה וְשִׁירַיִּים\hdot
מֻקֶּפֶת גָּדֵר גְּבוֹהָה עֲשָׂרָה טְפָחִים\hdot
מְטַלְטֵל בְּתוֹכָהּ\hdot
וּבִלְבַד שֶׁיֵּשׁ בָּהּ שׁוֹמִירָה אוֹ בֵית דִּירָה אוֹ שֶׁתְּהֵא סְמוּכָה לָעִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר אֲפִילוּ אֵין בָּהּ אֶלָּא בּוֹר וְשִׁיחַ וּמְעָרָה\hdot
מְטַלְטְלִין בְּתוֹכָהּ\hdot
רִבִּי עֲקִיבָא אוֹמֵר אֲפִילוּ אֵין בָּהּ אֶחָד מִכָּל אֵלּוּ מְטַלְטְלִין בְּתוֹכָהּ\hdot
וּבִלְבַד שֶׁיְּהֵא בָּהּ שִׁבְעִים אַמָּה וְשִׁירַיִּים\hdot
עַל שִׁבְעִים אַמָּה וְשִׁירַיִּים\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אִם הָֽיְתָה אָרְכָּהּ יָתֵר עַל רָחְבָּהּ אֲפִילוּ אַמָּה אַחַת\hdot
אֵין מְטַלְטְלִין בְּתוֹכָהּ\hdot
רִבִּי יוֹסֵי אוֹמֵר אֲפִילוּ אָרְכָּהּ פִּי שְׁנַיִם בְּרָחְבָּהּ מְטַלְטְלִין בְּתוֹכָהּ׃}

\halacha{אָמַר רִבִּי אֶלְעָאי שָׁמַעְתִּי מֵרִבִּי אֱלִיעֶזֶר\hdot

\halacha[71]{מְעָֽרְבִין בַּדְּמָאי\hdot
וּבְמַעֲשֵׂר רִאשׁוֹן שֶׁנִטְּלָה תְרוּמָתוֹ\hdot
וּבְמַעֲשֵׂר שֵׁנִי וְהֶקְדֶּשׁ שֶׁנִּפְדּוּ\hdot
וְהַכֹּהֲנִים בְּחַלָּה (וּבִתְרוּמָה)\hdot
אֲבָל לֹא בְטֶבֶל\hdot
וְלֹא בְמַעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ\hdot
וְלֹא בְמַעֲשֵׂר שֵׁנִי וְהֶקְדֶּשׁ שֶׁלֹּא נִפְדּוּ\hdot
הַשּׁוֹלֵחַ עֵירוּבוֹ בְּיַד חֵרֵשׁ שׁוֹטֶה וְקָטָן\hdot
וּבְיַד מִי שֶׁאֵינוֹ מוֹדֶה בָעֵירוּב\hdot
אֵינוֹ עֵירוּב\hdot
וְאִם אָמַר לְאֶחָד לְקַבְּלוֹ מִמֶּנּוּ\hdot
הֲרֵי זֶה עֵירוּב׃}
\end{minipage}

\halacha[986]{נְתָנוֹ בָאִילָן לְמַעְלָה מֵעֲשָׂרָה טְפָחִים\hdot
