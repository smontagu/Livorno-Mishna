\tractatehead{מסכת בבא קמא}{baba kama}

\perek{אַרְבָּעָה}{אָבוֹת נְזִיקִין\hdot
הַשּׁוֹר\hdot
וְהַבּוֹר\hdot
וְהַמַּבְעֶה\hdot
וְהַהֶבְעֵר\hdot
לֹא הֲרֵי הַשּׁוֹר כַּהֲרֵי הַמַּבְעֶה\hdot
וְלֹא הֲרֵי הַמַּבְעֶה כַּהֲרֵי הַשּׁוֹר\hdot
וְלֹא זֶה וְזֶה שֶׁיֵּשׁ בָּהֶן רוּחַ חַיִּים\hdot
כַּהֲרֵי הָאֵשׁ שֶׁאֵין בּוֹ רוּחַ חַיִּים\hdot
וְלֹא זֶה וְזֶה שֶׁדַּרְכָּן לִילֵךְ וּלְהַזִּיק\hdot
כַּהֲרֵי הַבּוֹר שֶׁאֵין דַּרְכּוֹ לִילֵךְ וּלְהַזִּיק\hdot
הַצַּד הַשָּׁוֶה שֶׁבָּהֶן\hdot
שֶׁדַּרְכָּן לְהַזִּיק\hdot
וּשְׁמִירָתָן עָלֶיךָ\hdot
וּכְשֶׁהִזִּיק\hdot
חָב הַמַּזִּיק לְשַׁלֵּם תַּשְׁלוּמֵי נֶזֶק\hdot
בְּמֵיטַב הָאָרֶץ׃}

\halacha{כָּל שֶׁחַבְתִּי בִשְׁמִירָתוֹ\hdot
הִכְשַׁרְתִּי אֶת נִזְקוֹ\hdot
הִכְשַׁרְתִּי בְמִקְצַת נִזְקוֹ\hdot
חַבְתִּי בְתַשְׁלוּמֵי נִזְקוֹ\hdot
כְּהֶכְשֵׁר כָּל נִזְקוֹ\hdot
נְכָסִים שֶׁאֵין בָּהֶן מְעִילָה\hdot
נְכָסִים שֶׁל בְּנֵי בְרִית\hdot
נְכָסִים הַמְּיוּחָדִים (וּבְכָל מָקוֹם) חוּץ מֵרְשׁוּת הַמְּיוּחֶדֶת לַמַּזִּיק וּרְשׁוּת הַנִּזָּק וְהַמַּזִּיק\hdot
וּכְשֶׁהִזִּיק חָב הַמַּזִּיק לְשַׁלֵּם תַּשְׁלוּמֵי נֶזֶק\hdot
בְּמֵיטַב הָאָרֶץ׃}

\halacha{שׁוֹם כֶּסֶף\hdot
וְשָׁוֵה כֶסֶף\hdot
בִּפְנֵי בֵית דִּין וְעַל פִּי עֵדִים בְּנֵי חוֹרִין וּבְנֵי בְרִית\hdot
וְהַנָּשִׁים בִּכְלַל הַנֶּזֶק\hdot
וְהַנֶּזֶק וְהַמַּזִיק בַּתַּשְׁלוּמִין׃}

\halacha{חֲמִשָּׁה תַּמִּין\hdot
וַחֲמִשָּׁה מוּעָדִין\hdot
הַבְּהֵמָה אֵינָהּ מוּעֶדֶת לֹא לִגַּח וְלֹא לִגּוֹף\hdot
וְלֹא לָשׁוּךְ\hdot
וְלֹא לִרְבּוֹץ\hdot
וְלֹא לִבְעוֹט\hdot
הַשֵּׁן מוּעָד לֶאֱכוֹל אֶת הָרָאוּי לָהּ\hdot
הָרֶגֶל מוּעֶדֶת לִשְׁבּוֹר בְּדֶרֶךְ הִלּוּכָהּ\hdot
וְשׁוֹר הַמּוּעָד\hdot
וְשׁוֹר הַמַּזִיק בִּרְשׁוּת הַנִּזָּק\hdot
וְהָאָדָם\hdot
הַזְּאֵב וְהָאֲרִי וְהַדּוֹב וְהַנָּמֵר וְהַבַּרְדְּלָס וְהַנָּחָשׁ הֲרֵי אֵלּוּ מוּעָדִין\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר בִּזְמָן שֶׁהֵן בְּנֵי תַרְבּוּת אֵינָן מוּעָדִין\hdot
וְהַנָּחָשׁ מוּעָד לְעוֹלָם\hdot
מַה בֵּין תָּם לַמּוּעָד\hdot
(אֶלָּא) שֶׁהַתָּם מְשַׁלֵּם חֲצִי נֶזֶק מִגּוּפוֹ\hdot
וּמוּעָד מְשַׁלֵּם נֶזֶק שָׁלֵם מִן הָעֲלִיָּה׃}

\chead{יום יד}

\perek{כֵּיצַד}{הָרֶגֶל מוּעֶדֶת\hdot
לִשְׁבּוֹר בְּדֶרֶךְ הִלּוּכָהּ\hdot
הַבְּהֵמָה מוּעֶדֶת לְהַלֵּךְ כְּדַרְכָּהּ וִלִשְׁבּוֹר\hdot
הָֽיְתָה מְבָעֶטֶת\hdot
אוֹ שֶׁהָיוּ צְרוֹרוֹת מְנַתְּזִין מִתַּחַת רַגְלֶיהָ\hdot
וְשִׁבְּרָה אֶת הַכֵּלִים מְשַׁלֵּם חֲצִי נֶזֶק\hdot
דָֽרְסָה עַל הַכֶּלִי\hdot
וְשִׁבְּרַתּוּ\hdot
וְנָפַל עַל כֶּלי אַחֵר וִשְׁבְּרוֹ\hdot
עַל הָרִאשׁוֹן מְשַׁלֵּם נֶזֶק שַׁלֵם\hdot
וְעַל הָאַחֲרוֹן מְשַׁלֵּם חֲצִי נֶזֶק\hdot
הַתַּרְנְגוֹלִים מוּעָדִין לְהַלֵּךְ כְּדַרְכָּן וְלִשְׁבּוֹר\hdot
הָיָה דְלִיל קָשׁוּר בְּרַגְלָיו\hdot
אוֹ שֶׁהָיָה מְהַדֵּס וְשִׁבֵּר אֶת הַכֵּלִים\hdot
מְשַׁלֵּם חֲצִי נֶזֶק׃}

\halacha{כֵּיצַד הַשֵּׁן מוּעֶדֶת\hdot
לֶאֱכוֹל אֶת הָרָאוּי לָהּ\hdot
הַבְּהֵמָה מוּעֶדֶת לֶאֱכוֹל פֵּירוֹת וִירָקוֹת\hdot
אָֽכְלָה כְסוּת אוֹ כֵלִים\hdot
מְשַׁלֵּם חֲצִי נֶזֶק\hdot
בַּמֶּה דְּבָרִים אֲמוּרִים בִּרְשׁוּת הַנִּזָּק\hdot
אֲבָל בִּרְשׁוּת הָרַבִּים פָּטוּר\hdot
וְאִם נֶהֱנֵית\hdot
מְשַׁלֵּם מַה שֶּׁנֶּהֱנֵית\hdot
כֵּיצַד (מְשַׁלֵּם) מַה שֶּׁנֶּהֱנֵית\hdot
אָֽכְלָה מִתּוֹךְ הָֽרְחָבָה\hdot
מְשַׁלֵּם מַה שֶּׁנֶּהֱנֵית\hdot
מִצִּדֵּי הָֽרְחָבָה\hdot
מְשַׁלֵּם מַה שֶּׁהִזִּיקָה\hdot
מִפֶּתַּח הֶחָנוּת (מְשַׁלֵּם) מַה שֶּׁנֶּהֱנֵית\hdot
מִתּוֹךְ הֶחָנוּת (מְשַׁלֵּם) מַה שֶּׁהִזִּיקָה׃}

\halacha{הַכֶּלֶב וְהַגְּדִי שֶׁקָֽפְצוּ מֵרֹאשׁ הַגָּג\hdot
וְשִׁבּרוּ אֶת הַכֵּלִים\hdot
מְשַׁלֵּם נֶזֶק שָׁלֵם\hdot
מִפְּנֵי שֶׁהֵן מוּעָדִין\hdot
הַכֶּלֶב שֶׁנָּטַל הַחֲרָרָה וְהָלַךְ לַגָּדִישׁ\hdot
אָכַל הַחֲרָרָה וְהִדְלִיק הַגָּדִישׁ\hdot
עַל הַחֲרָרָה מְשַׁלֵּם נֶזֶק שָׁלֵם\hdot
וְעַל הַגָּדִישׁ מְשַׁלֵּם חֲצִי נֶזֶק׃}

\halacha{אֵיזֶה הוּא תָם\hdot
וְאֵיזֶה הוּא מוּעָד\hdot
מוּעָד כָּל שֶׁהֵעִידוּ בוֹ שְׁלֹשָׁה יָמִים\hdot
וְתָם מִשֶּׁיַּחֲזוֹר בּוֹ שְׁלֹשָׁה יָמִים\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
רִבִּי מֵאִיר אוֹמֵר מוּעָד שֶׁהֵעִידוּ בוֹ שָׁלֹשׁ פְּעָמִים\hdot
וְתָם כָּל שֶׁיְּהוּ הַתִּינוֹקוֹת מְמַשְׁמְשִׁין בּוֹ וְאֵינוֹ נוֹגֵחַ׃}

\halacha{שׁוֹר הַמַּזִּיק בִּרְשׁוּת הַנִּזָּק כֵּיצַד\hdot
נָגַח\hdot
נָגַף\hdot
נָשַׁךְ\hdot
רָבַץ\hdot
בָּעַט\hdot
בִּרְשׁוּת הָרַבִּים\hdot
מְשַׁלֵּם חֲצִי נֶזֶק\hdot
בִּרְשׁוּת הַנִּזָּק\hdot
רִבִּי טַרְפוֹן אוֹמֵר נֶזֶק שָׁלֵם\hdot
וַחֲכָמִים אוֹמְרִים חֲצִי נֶזֶק\hdot
אָמַר לָהֶם רִבִּי טַרְפוֹן וּמָה אִם בִּמְקוֹם שֶׁהֵקֵל עַל הַשֵּׁן וְעַל הָרֶגֶל בִּרְשׁוּת הָרַבִּים שֶׁהוּא פָּטוּר\hdot
הֶחְמִיר עֲלֵיהָם בִּרְשׁוּת הַנִּזָּק לְשַׁלֵּם נֶזֶק שָׁלֵם\hdot
מְקוֹם שֶׁהֶחְמִיר עַל הַקֶּרֶן בִּרְשׁוּת הָרַבִּים\hdot
לְשַׁלֵּם חֲצִי נֶזֶק\hdot
אֵינוֹ דִין שֶׁנַּחְמִיר עָלֶיהָ בִּרְשׁוּת הַנִּזָּק לְשַׁלֵּם נֶזֶק שָׁלֵם\hdot
אָֽמְרוּ לוֹ דַּיּוֹ לַבָּא מִן הַדִּין לִהְיוֹת כַּנִּדּוֹן\hdot
מַה בִּרְשׁוּת הָרַבִּים חֲצִי נֶזֶק\hdot
אַף בִּרְשׁוּת הַנִּזָּק חֲצִי נֶזֶק\hdot
אָמַר לָהֶם אֲנִי לֹא אָדוֹן קֶרֶן מִקֶּרֶן\hdot
אֲנִי אָדוֹן קֶרֶן מֵרֶגֶל\hdot
וּמָה אִם בִּמְקוֹם שֶׁהֵקֵל עַל הַשֵּׁן וְעַל הָרֶגֶל בִּרְשׁוּת הָרַבִּים\hdot
הֶחְמִיר בְּקֶרֶן\hdot
מְקוֹם שֶׁהֶחְמִיר עַל הַשֵּׁן וְעַל הָרֶגֶל בִּרְשׁוּת הַנִּזָּק\hdot
אֵינוֹ דִין שֶׁנַּחְמִיר בְּקֶרֶן\hdot
אָֽמְרוּ לוֹ דַּיּוֹ לַבָּא מִן הַדִּין לִהְיוֹת כַּנִּדּוֹן\hdot
מַה בִּרְשׁוּת הָרַבִּים חֲצִי נֶזֶק אַף בִּרְשׁוּת הַנִּזָּק חֲצִי נֶזֶק׃}

\halacha{אָדָם מוּעָד לְעוֹלָם בֵּין שׁוֹגֵג\hdot
בֵּין מֵזִיד\hdot
בֵּין עֵר\hdot
בֵּין יָשֵׁן\hdot
סִמָּא אֶת עֵין חֲבֵירוֹ וְשִׁבַּר אֶת הַכֵּלִים\hdot
מְשַׁלֵּם נֶזֶק שָׁלֵם׃}

\perek{הַמַּנִּיחַ}{אֶת הַכַּד בִּרְשׁוּת הָרַבִּים\hdot
וּבָא אַחֵר וְנִתְקַל בָּהּ\hdot
וְשִׁבְּרָהּ פָּטוּר\hdot
וְאִם הוּזָק בָּהּ בַּעַל הֶחָבִית חַּיָּב בְּנִזְקוֹ\hdot
נִשְׁבְּרָה כַדּוֹ בִּרְשׁוּת הָרַבִּים\hdot
וְהֻחְלַק אֶחָד בַּמַּיִם\hdot
אוֹ שֶׁלָּקָה בְחַרְסֶיהָ חַיָּב\hdot
רִבִּי יְהוּדָה אוֹמֵר בְּמִתְכַּוֵּין חַיָּב\hdot
בְּאֵינוֹ מִתְכַּוֵּין פָּטוּר׃}

\halacha{הַשּׁוֹפֵךְ מַיִם בִּרְשׁוּת הָרַבִּים וְהוּזָק בָּהֶן אַחֵר\hdot
חַיָּב בְּנִזְקוֹ\hdot
הַמַּצְנִיעַ אֶת הַקּוֹץ וְאֶת הַזְּכוּכִית\hdot
וְהַגּוֹדֵר אֶת גִּדְרוֹ בַקּוֹצִים\hdot
וְגָדֵר שֶׁנָּפַל לִרְשׁוּת הָרַבִּים\hdot
וְהֻזְּקוּ בָהֶן אַחֵרִים\hdot
חַיָּב בְנִזְקָן׃}

\halacha{הַמּוֹצִיא אֶת תִּבְנוֹ וְאֶת קַשּׁוֹ לִרְשׁוּת הָרַבִּים לַזְּבָלִים\hdot
וְהוּזָק בָּהֶן אַחֵר\hdot
חַיָּב בְּנִזְקוֹ\hdot
וְכָל הַקּוֹדֵם בָּהֶן זָכָה\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר כָּל הַמְּקַלְקְלִין בִּרְשׁוּת הָרַבִּים וְהִזִּיקוּ\hdot
חַיָּבִין לְשַּׁלֵּם\hdot
