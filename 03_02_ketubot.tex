\tractatehead{מסכת כתובות}{ketubot}

\perek{בְּתוּלָה}{נִשֵּׂאת לְיוֹם הָֽרְבִיעִי\hdot
וְאַלְמָנָה לְיוֹם חֲמִישִׁי\hdot
שֶׁפַּעֲמַיִם בְּשַׁבָּת בָּתֵּי דִינִין יוֹשְׁבִין בָּעֲרָיוֹת בַּיּוֹם הַשֵּׁנִי\hdot
וּבַיּוֹם הַחֲמִישִׁי\hdot
שֶׁאִם הָיָה לוֹ טַעֲנַת בְּתוּלִים הָיָה מַשְׁכִּים לְבֵית דִּין׃}

\halacha{בְּתוּלָה כְּתוּבָתָהּ מָאתַיִם\hdot
וְאַלְמָנָה מָנֶה\hdot
בְּתוּלָה אַלְמָנָה גְּרוּשָׁה וַחֲלוּצָה מִן הָאֵרוּסִין\hdot
כְּתוּבָתָן מָאתַיִם וְיֵשׁ לָהֶן טַעֲנַת בְּתוּלִים\hdot
הַגִּיּוֹרֶת וְהַשְּׁבוּיָה וְהַשִּׁפְחָה\hdot
שֶׁנִּפְדּוּ וְשֶׁנִּתְגַּיְּרוּ וְשֶׁנִּשְׁתַּחְרְרוּ\hdot
פְּחוּתוֹת מִבְּנוֹת שָׁלֹשׁ שָׁנִים וְיוֹם אֶחָד\hdot
כּתוּבָתָן מָאתַיִם\hdot
וְיֵשׁ לָהֶן טַעֲנַת בּתוּלִים׃}

\halacha{הַגָּדוֹל שֶׁבָּא עַל הַקְּטָנָה\hdot
וְהַקָּטָן שֶׁבָּא עַל הַגְּדוֹלָה\hdot
וּמֻכַּת עֵץ\hdot
כְּתוּבָתָן מָאתַיִם\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים מֻכַּת עֵץ כְּתוּבָתָהּ מָנֶה׃}

\halacha{בְּתוּלָה אַלְמָנָה גְּרוּשָׁה וַחֲלוּצָה מִן הַנִּשּׂוּאִין\hdot
כְּתוּבָתָן מָנֶה\hdot
וְאֵין לָהֶן טַעֲנַת בְּתוּלִים\hdot
הַגִּיּוֹרֶת וְהַשְּׁבוּיָה וְהַשִּׁפְחָה\hdot
שֶׁנִּפְדּוּ וְשֶׁנִּתְגַּיְּרוּ וְשֶׁנִּשְׁתַּחְרְרוּ\hdot
יְתֵירוֹת עַל בְּנוֹת שָׁלֹשׁ שָׁנִים וְיוֹם אֶחָד\hdot
כְּתוּבָתָן מָנֶה וְאֵין לָהֶן טַעֲנַת בְּתוּלִים׃}

\halacha{הָאוֹכֵל אֵצֶל חָמִיו בִּיהוּדָה שֶׁלֹּא בְעֵדִים\hdot
אֵינוֹ יָכוֹל לִטְעוֹן טַעֲנַת בְּתוּלִים מִפְּנֵי שֶׁמִּתְיַחֵד עִמָּהּ\hdot
אַחַת אַלְמָנַת יִשְׂרָאֵל וְאַחַת אַלְמָנַת כֹּהֵן כְּתוּבָתָהּ מָנֶה\hdot
בֵּית דִּין שֶׁל כֹּהֲנִים גּוֹבִין לַבְּתוּלָה אַרְבַּע מֵאוֹת זוּז\hdot
וְלֹא מִיחוּ בְיָדָם חֲכָמִים׃}

\halacha{הַנּוֹשֵׂא אֶת הָאִשָּׁה וְלֹא מָצָא לָהּ בְּתוּלִים\hdot
הִיא אוֹמֶרֶת מִשֶּׁאֵרַסְתָּנִי נֶאֱנַסְתִּי וְנִסְתַּחֲפָה שָׁדֶךָ\hdot
וְהַלָּה אוֹמֵר לֹא כִי אֶלָּא עַד שֶׁלֹּא אֵרַסְתִּיךְ\hdot
וְהָיָה מִקְחִי מֶקַח טָעוּת\hdot
רַבָּן גַּמְלִיאֵל וְרִבִּי אֱלִיעֶזֶר אוֹמְרִים נֶאֱמֶנֶת\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר לֹא מִפִּיהָ אָנוּ חַיִּין\hdot
אֶלָּא הֲרֵי זוֹ בְּחֶזְקַת בְּעוּלָה עַד שֶׁלֹּא תִתְאָרֵס וְהִטְעַתּוּ עַד שֶׁתָּבִיא רְאָיָה לִדְבָרֶיהָ׃}

\halacha{הִיא אוֹמֶרֶת מֻכַּת עֵץ אָנִי\hdot
וְהוּא אוֹמֵר לֹא כִי\hdot
אֶלָּא דְּרוּסַת אִישׁ אַתְּ\hdot
רַבָּן גַּמְלִיאֵל וְרִבִּי אֱלִיעֶזֶר אוֹמְרִים נֶאֱמֶנֶת\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר לֹא מִפִּיהָ אָנוּ חַיִּין\hdot
אֶלָּא הֲרֵי זוֹ בְּחֶזְקַת דְּרוּסַת אִישׁ עַד שֶׁתָּבִיא רְאָיָה לִדְבָרֶיהָ׃}

\halacha{רָאוּהָ מְדַבֶּרֶת עִם אֶחָד בַּשּׁוּק\hdot
אָֽמְרוּ לָהּ מַה טִּבּוֹ שֶׁל זֶה\hdot
אִישׁ פְּלוֹנִי וְכֹהֵן הוּא\hdot
רַבָּן גַּמְלִיאֵל וְרִבִּי אֱלִיעֶזֶר אוֹמְרִים נֶאֱמֶנֶת\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר לֹא מִפִּיהָ אָנוּ חַיִּין\hdot
אֶלָּא הֲרֵי זוֹ בְּחֶזְקַת בְּעוּלָה לְנָתִין וּלְמַמְזֵר\hdot
עַד שֶׁתָּבִיא רְאָיָה לִדְבָרֶיהָ׃}

\halacha{הָֽיְתָה מְעֻבֶּרֶת\hdot
וְאָֽמְרוּ לָהּ מַה טִּבּוֹ שֶׁל עֻבָּר זֶה\hdot
מֵאִישׁ פְּלוֹנִי וְכֹהֵן הוּא\hdot
רַבָּן גַּמְלִיאֵל וְרִבִּי אֱלִיעֶזֶר אוֹמְרִים נֶאֱמֶנֶת\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר לֹא מִפִּיהָ אָנוּ חַיִּין\hdot
אֶלָּא הֲרֵי זוֹ בְּחֶזְקַת מְעֻבֶּרֶת מִנָּתִין וּמַמְזֵר\hdot
עַד שֶׁתָּבִיא רְאָיָה לִדְבָרֶיהָ׃}

\halacha{אָמַר רִבִּי יוֹסֵי מַעֲשֶׂה בְּתִינוֹקֶת שֶיָּֽרְדָה לְמַלֹּאת מַיִם מִן הָעַיִן וְנֶאֱנָסָה\hdot
אָמַר רִבִּי יוֹחָנָן בֶּן נוּרִי אִם רוֹב אַנְשֵׁי הָעִיר מַשִּׂיאִים לַכְּהוּנָה הֲרֵי זוֹ תִנָּשֵׂא לַכְּהוּנָה׃}

\perek{הָאִשָּׁה}{שֶׁנִּתְאַרְמְלָה אוֹ שֶׁנִּתְגָּֽרְשָׁה\hdot
וְהִיא אוֹמֶרֶת בְּתוּלָה נְשָׂאתַנִי\hdot
וְהוּא אוֹמֵר לֹא כִי\hdot
אֶלָּא אַלְמָנָה נְשָׂאתִיךְ\hdot
אִם יֵשׁ עֵדִים שֶׁיָּצָאת בְּהִינוּמָא וְרֹאשָׁהּ פָּרוּעַ כְּתוּבָתָהּ מָאתַיִם\hdot
רִבִּי יוֹחָנָן בֶּן בְּרוֹקָא אוֹמֵר אַף חִלּוּק קְלָיוֹת רְאָיָה׃}

\halacha{וּמוֹדֶה רִבִּי יְהוֹשֻׁעַ בְּאוֹמֵר לַחֲבֵירוֹ שָׂדֶה זוֹ שֶׁל אָבִיךָ הָֽיְתָה וּלְקַחְתִּיהָ הֵימֶנּוּ\hdot
שֶׁהוּא נֶאֱמָן\hdot
שֶׁהַפֶּה שֶׁאָסַר הוּא הַפֶּה שֶׁהִתִּיר\hdot
וְאִם יֵשׁ עֵדִים שֶׁהָֽיְתָה שֶׁל אָבִיו\hdot
וְהוּא אוֹמֵר לְקַחְתִּיהָ הֵימֶנּוּ\hdot
אֵינוֹ נֶאֱמָן׃}

\halacha{הָעֵדִים שֶׁאָֽמְרוּ כְתַב יָדֵינוּ הוּא זֶה אֲבָל אֲנוּסִים הָיִינוּ\hdot
קְטַנִּים הָיִינוּ\hdot
פְּסוּלֵי עֵדוּת הָיִינוּ\hdot
הֲרֵי אֵלּוּ נֶאֱמָנִים\hdot
וְאִם יֵשׁ עֵדִים שֶׁהוּא כְתַב יָדָם\hdot
אוֹ שֶׁהָיָה כְתַב יָדָם יוֹצֵא מִמָּקוֹם אַחֵר\hdot
אֵינָן נֶאֱמָנִין׃}

\halacha{זֶה אוֹמֵר זֶה כְּתַב יָדִי\hdot
וְזֶה כְּתַב יָדוֹ שֶׁל חֲבֵרִי\hdot
וְזֶה אוֹמֵר זֶה כְּתַב יָדִי\hdot
וְזֶה כְּתַב יָדוֹ שֶׁל חֲבֵרִי\hdot
הֲרֵי אֵלּוּ נֶאֱמָנִין\hdot
זֶה אוֹמֵר זֶה כְּתַב יָדִי\hdot
וְזֶה אוֹמֵר זֶה כְּתַב יָדִי\hdot
צְרִיכִים לְצָרֵף עִמָּהֶם אַחֵר דִּבְרֵי רִבִּי\hdot
וַחֲכָמִים אוֹמְרִים אֵינָם צְרִיכִים לְצָרֵף עִמָּהֶם אַחֵר\hdot
אֶלָּא נֶאֱמָן אָדָם לוֹמַר זֶה כְּתַב יָדִי׃}

\halacha{הָאִשָּׁה שֶׁאָֽמְרָה אֵשֶׁת אִישׁ הָיִיתִי\hdot
וּגְרוּשָׁה אָנִי\hdot
נֶאֱמֶנֶת\hdot
שֶׁהַפֶּה שֶׁאָסַר הוּא הַפֶּה שֶׁהִתִּיר\hdot
וְאִם יֵשׁ עֵדִים שֶׁהָֽיְתָה אֵשֶׁת אִישׁ\hdot
וְהִיא אוֹמֶרֶת גְּרוּשָׁה אָנִי\hdot
אֵינָהּ נֶאֱמֶנֶת\hdot
אָֽמְרָה נִשְׁבֵּתִי וּטְהוֹרָה אָנִי\hdot
נֶאֱמֶנֶת\hdot
שֶׁהַפֶּה שָׁאָסַר הוּא הַפֶּה שֶׁהִתִּיר\hdot
וְאִם יֵשׁ עֵדִים שֶׁנִּשְׁבֵּית\hdot
וְהִיא אוֹמֶרֶת טְהוֹרָה אָנִי\hdot
אֵינָהּ נֶאֱמֶנֶת\hdot
וְאִם מִשֶּׁנִּשֵּׂאת בָּאוּ הָעֵדִים הֲרֵי זוֹ לֹא תֵצֵא׃}

\halacha{שְׁתֵּי נָשִׁים שֶׁנִּשְׁבּוּ\hdot
זֹאת אוֹמֶרֶת נִשְׁבֵּתִי וּטְהוֹרָה אָנִי\hdot
וְזֹאת אוֹמֶרֶת נִשְׁבֵּתִי וּטְהוֹרָה אָנִי\hdot
אֵינָן נֶאֱמָנוֹת\hdot
וּבִזְמַן שֶׁהֵן מְעִידוֹת זוֹ אֶת זוֹ\hdot
הֲרֵי אֵלּוּ נֶאֱמָנוֹת׃}

\halacha{וְכֵן שְׁנֵי אֲנָשִׁים\hdot
זֶה אוֹמֵר כֹּהֵן אָנִי\hdot
וְזֶה אוֹמֵר כֹּהֵן אָנִי\hdot
אֵינָן נֶאֱמָנִין\hdot
וּבִזְמַן שֶׁהֵן מְעִידִין זֶה אֶת זֶה\hdot
הֲרֵי אֵלּוּ נֶאֱמָנִין׃}

\halacha{רִבִּי יְהוּדָה אוֹמֵר אֵין מַעֲלִין לַכְּהוּנָה עַל פִּי עֵד אֶחָד\hdot
אָמַר רִבִּי אֶלְעָזָר אֵימָתַי בִּמְקוֹם שֶׁיֵּשׁ עוֹרְרִין\hdot
אֲבָל בִּמְקוֹם שֶׁאֵין עוֹרְרִין מַעֲלִין לַכְּהוּנָה עַל פִּי עֵד אֶחָד\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר מִשּׁוּם רִבִּי שִׁמְעוֹן בֶּן הַסְּגָן\hdot
מַעַלִין לַכְּהוּנָה עַל פִּי עֵד אֶחָד׃}

\halacha{הָאִשָּׁה שֶׁנֶּחְבְּשָׁה בִידֵי גוֹיִם עַל יְדֵי מָמוֹן\hdot
מוּתֶרֶת לְבַעְלָהּ\hdot
עַל יְדֵי נְפָשׁוֹת\hdot
אֲסוּרָה לְבַעְלָהּ\hdot
עִיר שֶׁנִּכְבְּשָׁה כַּרְכּוּם\hdot
כַָּל הַכֹּהֲנוֹת שֶׁנִּמְצְאוּ בְתוֹכָהּ\hdot
פְּסוּלוֹת\hdot
וְאִם יֵשׁ לָהֶן עֵדִים\hdot
אֲפִילוּ עֶבֶד\hdot
אֲפִילוּ שִׁפְחָה\hdot
הֲרֵי אֵלּוּ נֶאֱמָנִין\hdot

אָמַר רִבִּי יוֹסֵי לֹא פָסַק לָהּ שְׂעוֹרִים אֶלָּא רִבִּי יִשְׁמָעֵאל שֶׁהָיָה סָמוּךְ לֶאֱדוֹם\hdot
וְנוֹתֵן לָהּ חֲצִי קָב קָטְנִית\hdot

וְקָב גְּרוֹגְרוֹת אוֹ מָנֶה דְבֵילָה\hdot
וְאִם אֵין לוֹ\hdot
פּוֹסֵק לְעֻמָּתָן פֵּירוֹת מִמָּקוֹם אַחֵר\hdot
וְנוֹתֵן לָהּ מִטָּה וּמַפֵּץ וּמַחֲצֶלֶת\hdot
וְנוֹתֵן לָהּ כִּפָּה לְרֹאשָׁהּ\hdot
וַחֲגוֹר לְמָתְנֶיהָ\hdot

וְאֵינוֹ נוֹתֵן לָהּ לֹא חֲדָשִׁים בִּימוֹת הַחַמָּה\hdot
וְלֹא שְׁחָקִים בִּימוֹת הַגְּשָׁמִים\hdot
אֶלָּא נוֹתֵן לָהּ כֵּלִים שֶׁל חֲמִשִּׁים זוּז בִּימוֹת הַגְּשָׁמִים\hdot
וְהִיא מִתְכַּסָּה בְּבַלָּאוֹתֵיהֶן בִּימוֹת הַחַמָּה\hdot
וְהַשְּׁחָקִים שֶׁלָּהּ׃}

\halacha{נוֹתֵן לָהּ מְעָה כֶסֶף לְצָרְכֶיהָ\hdot
וְאוֹכֶלֶת עִמּוֹ מִלֵּילֵי שַׁבָּת לְלֵילֵי שַׁבָּת\hdot

מַעֲשֵׂה יָדֶיהָ שֶׁלָּהּ\hdot
וּמָה הִיא עוֹשָׂה לוֹ\hdot

אוֹ מִשְׁקַל עֶשֶׂר סְלָעִים עֶרֶב בִּיהוּדָה\hdot
שֶׁהֵן עֶשְׂרִים סְלָעִים בַּגָּלִיל\hdot

וּמוֹסִיפִין לָהּ עַל מְזוֹנוֹתֶיהָ\hdot

אֲבָל בַּמֻּכְבָּד הַכֹּל לְפִי כְבוֹדוֹ׃}

\perek{מְצִיאַת}{הָאִשָּׁה וּמַעֲשֵׂה יָדֶיהָ לְבַעְלָהּ\hdot

\halacha{פָּֽסְקָה לְהַכְנִיס לוֹ אֶלֶף דִּינָר\hdot
הוּא פוֹסֶק כְּנֶגְדָּן חֲמִשָּׁה עָשָׂר מָנֶה\hdot

שׁוֹם בְּמָנֶה הִיא נוֹתֶנֶת שְׁלֹשִׁים וְאֶחָד סֶלַע וְדִינָר\hdot
וּבְאַרְבַּע מֵאוֹת הִיא נוֹתֶנֶת חֲמֵשׁ מֵאוֹת\hdot

\halacha{פָּֽסְקָה לְהַכְנִיס לוֹ כְּסָפִים סֶלַע (כֶּסֶף) נַעֲשָׂה שִׁשָּׁה דִינָרִים\hdot

רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר הַכֹּל כְּמִנְהַג הַמְּדִינָה׃}

\halacha{הַמַּשִּׂיא אֶת בִּתּוֹ סְתָם\hdot

אִם יֵשׁ בְּכִיס מְפַרְנְסִין אוֹתָהּ לְפִי כְבוֹדָהּ׃}

\halacha{יְתוֹמָה שֶׁהִשִּׂיאוּהָ אִמָּהּ אוֹ אַחֶיהָ מִדַּעְתָּהּ\hdot

אֲבָל בַּקְּטַנָּה אֵין מַעֲשֵׁה קְטַנָּה כְּלוּם׃}

שְׁנַיִם יוֹצִיא וְיִתֵּן כְּתוּבָה\hdot

\halacha{הַמַּדִּיר אֶת אִשְׁתּוֹ שֶׁלֹּא תֵלֵךְ לְבֵית אָבִיהָ\hdot

\halacha{וְאֵלּוּ יוֹצְאוֹת שֶׁלֹּא בִכְתוּבָה\hdot

וְנוֹדֶרֶת וְאֵינָהּ מְקַיֶּמֶת\hdot
וְאֵיזוֹ הִיא דַת יְהוּדִית\hdot
יוֹצְאָה וְרֹאשָׁהּ פָּרוּעַ\hdot
וְטוֹוָה בַשּׁוּק\hdot

אַבָּא שָׁאוּל אוֹמֵר אַף הַמְּקַלֶּלֶת יוֹלְדָיו בְּפָנָיו\hdot

וּשְׁכֵנֶיהָ שׁוֹמְעִין קוֹלָהּ׃}

עַל מְנַת שֶׁאֵין בָּהּ מוּמִין וְנִמְצְאוּ בָהּ מוּמִין\hdot
אֵינָהּ מְקֻדֶּשֶׁת\hdot
כְּנָסָהּ סְתָם וְנִמְצְאוּ בָהּ מוּמִין\hdot
תֵּצֵא שֶׁלֹּא בִכְתוּבָה\hdot
כָּל הַמּוּמִין הַפּוֹסְלִים בַּכֹּהֲנִים\hdot
פּוֹסְלִין בַּנָשִׁים׃}

\halacha{הָיוּ בָהּ מוּמִין וְעוֹדָהּ בְּבֵית אָבִיהָ\hdot
הָאָב צָרִיךְ לְהָבִיא רְאָיָה שֶמִּשֶּׁנִּתְאָֽרְסָה נוֹלְדוּ בָהּ מוּמִין הַלָּלוּ\hdot
וְנִסְתַּחֲפָה שָׂדֵהוּ\hdot
נִכְנְסָה לִרְשׁוּת הַבַּעַל\hdot
הַבַּעַל צָרִיךְ לְהָבִיא רְאָיָה שֶׁעַד שֶׁלֹּא נִתְאָֽרְסָה\hdot
הָיוּ בָהּ מוּמִין אֵלּוּ\hdot
וְהָיָה מֶקְחוֹ מֶקַח טָעוּת\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחַכָמִים אוֹמְרִים בַּמֶּה דְבָרִים אֲמוּרִים בְּמוּמִין שֶׁבַּסֵּתֶר\hdot
אֲבָל בְּמוּמִין שֶׁבַּגָּלוּי\hdot
אֵינוֹ יָכוֹל לִטְעוֹן\hdot
וְאִם יֵשׁ מֶרְחָץ בְּאוֹתָהּ הָעִיר\hdot
אַף מוּמִין שֶׁבַּסֵּתֶר אֵינוֹ יָכוֹל לִטְעוֹן מִפְּנֵי שֶׁהוּא בוֹדֵק בִּקְרוֹבוֹתָיו׃}

\halacha{הָאִישׁ שֶׁנּוֹלְדוּ בוֹ מוּמִין\hdot
אֵין כּוֹפִין אוֹתוֹ לְהוֹצִיא\hdot
אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל בַּמֶּה דְבָרִים אֲמוּרִים בְּמוּמִין הַקְּטַנִּים\hdot
אֲבָל בְּמוּמִין הַגְּדוֹלִים כּוֹפִין אוֹתוֹ לְהוֹצִיא׃}
