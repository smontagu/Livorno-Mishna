\tractatehead{מסכת מגילה}{megilla}

\perek{מְגִילָה}{נִקְרֵאת בְּאַחַד עָשָׂר\hdot
בִּשְׁנֵים עָשָׂר\hdot
בִּשְׁלֹשָׁה עָשָׂר\hdot
בְּאַרְבָּעָה עָשָׂר\hdot
בַּחֲמִשָּׁה עָשָׂר\hdot
לֹא פָּחוּת וְלֹא יוֹתֵר\hdot
כְּרָכִים הַמּוּקָפִין חוֹמָה מִימוֹת יְהוֹשֻעַ בִּן נוּן\hdot
קוֹרִין בַּחֲמִשָּׁה עָשָׂר\hdot
כְּפָרִים וַעֲיָרוֹת גּדוֹלוֹת\hdot
קוֹרִין בְּאַרְבָּעָה עָשָׂר\hdot
אֶלָּא שֶׁהַכְּפָרִים מַקְדִּימִין לְיוֹם הַכְּנִיסָה׃}

\halacha{כֵּיצַד\hdot
חַל (יוֹם אַרְבָּעָה עָשָׂר) לִהְיוֹת בַּשֵּׁנִי\hdot
כְּפָרִים וַעֲיָרוֹת גְּדוֹלוֹת\hdot
קוֹרִין בּוֹ בַיּוֹם\hdot
וּמֻקָּפוֹת חוֹמָה לְמָחָר\hdot
חַל לִהְיוֹת בַּשְּׁלִישִׁי אוֹ בָֽרְבִיעִי\hdot
כְּפָרִים מַקְדִּימִין לְיוֹם הַכְּנִיסָה\hdot
וַעֲיָרוֹת גְּדוֹלוֹת קוֹרִין בּוֹ בַיּוֹם\hdot
וּמֻקָּפוֹת חוֹמָה לְמָחָר\hdot
חַל לִהְיוֹת בַּחֲמִישִׁי\hdot
כְּפָרִים וַעֲיָרוֹת גְּדוֹלוֹת\hdot
קוֹרִין בּוֹ בַיּוֹם\hdot
וּמֻקָּפוֹת חוֹמָה לְמָחָר\hdot
חַל לִהְיוֹת עֶרֶב שַׁבָּת\hdot
כְּפָרִים מַקְדִּימִין לְיוֹם הַכְּנִיסָה\hdot
וַעֲיָרוֹת גְּדוֹלוֹת וּמֻקָּפוֹת חוֹמָה\hdot
קוֹרִין בּוֹ בַיּוֹם\hdot
חַל לִהְיוֹת בְּשַׁבָּת\hdot
כְּפָרִים וַעֲיָרוֹת גּדוֹלוֹת\hdot
מַקְדִימִין וְקוֹרִין לְיוֹם הַכְּנִיסָה\hdot
וּמֻקָּפוֹת חוֹמָה לְמָחָר\hdot
חַל לִהְיוֹת אַחַר הַשַּׁבָּת\hdot
כְּפָרִים מַקְדִימִין לְיוֹם הַכְּנִיסָה\hdot
וַעֲיָרוֹת גְּדוֹלוֹת קוֹרִין בּוֹ בַיּוֹם\hdot
וּמֻקָּפוֹת חוֹמָה לְמָחָר׃}

\halacha{אֵי זוֹ הִיא עִיר גְּדוֹלָה כָּל שֶׁיֵּשׁ בָּהּ עֲשָׂרָה בַטְלָנִין\hdot
פָּחוּת מִכָּאן הֲרֵי זֶה כְּפָר\hdot
בְּאֵלּוּ אָֽמְרוּ מַקְדִימִין וְלֹא מְאַחֲרִין\hdot
אֲבָל זְמַן עֲצֵי כֹהֲנִים\hdot
וְתִשְׁעָה בְאָב\hdot
חֲגִיגָה וְהַקְהֵל\hdot
מְאַחֲרִין וְלֹא מַקְדִימִין\hdot
אַף עַל פִּי שֶׁאָֽמְרוּ מַקְדִּימִין וְלֹא מְאַחֲרִין\hdot
מֻתָּרִין בְּהֶסְפֵּד וּבְתַעֲנִית וּמַתָּנוֹת לָאֶבְיוֹנִים\hdot
אָמַר רִבִּי יְהוּדָה אֵימָתַי מָקוֹם שֶׁנִּכְנָסִין בַּשֵּׁנִי וּבַחֲמִישִׁי\hdot
אֲבָל מָקוֹם שֶׁאֵין נִכְנָסִין לֹא בַשֵּׁנִי וְלֹא בַחֲמִישִׁי\hdot
אֵין קוֹרִין אוֹתָהּ אֶלָּא בִזְמַנָּהּ׃}

\halacha{קָֽרְאוּ אֶת הַמְּגִילָה בַּאֲדָר הָרִאשׁוֹן וְנִתְעַבְּרָה הַשָּׁנָה\hdot
קוֹרִין אוֹתָהּ בַּאֲדָר הַשֵּׁנִי\hdot
אֵין בֵּין אֲדָר הָרִאשׁוֹן לַאֲדָר הַשֵּׁנִי\hdot
אֶלָּא קְרִיאַת הַמְּגִילָה\hdot
וּמַתָּנוֹת לָאֶבְיוֹנִים׃}

\halacha{אֵין בֵּין יוֹם טוֹב לְשַׁבָּת אֶלָּא אֹכֶל נֶפֶשׁ בִּלְבָד\hdot
אֵין בֵּין שַׁבָּת לְיוֹם הַכִּפּוּרִים אֶלָּא שֶׁזֶּה זְדוֹנוֹ בִידֵי אָדָם\hdot
וְזֶה זְדוֹנוֹ בְכָרֵת׃}

\halacha{אֵין בֵּין הַמֻּדָּר הַנָּאָה מֵחֲבֵירוֹ לַמֻּדָּר מִמֶּנּוּ מַאֲכָל\hdot
אֶלָּא דְרִיסַת הָרֶגֶל וְכֵלִים שֶׁאֵין עוֹשִׂין בָּהֶן אֹכֶל נֶפֶשׁ\hdot
אֵין בֵּין נְדָרִים לִנְדָבוֹת אֶלָּא שֶׁהַנְּדָרִים חַיָּב בְּאַחֲרָיוּתָן\hdot
וּנְדָבוֹת אֵינוֹ חַיָּב בְּאַחֲרָיוּתָן׃}

\halacha{אֵין בֵּין זָב הָרוֹאֶה שְׁתֵּי רְאִיּוֹת לָרוֹאֶה שָׁלֹשׁ\hdot
אֶלָּא קָרְבָּן\hdot
אֵין בֵּין מְצוֹרָע מֻסְגָּר לִמְצוֹרָע מֻחְלָט\hdot
אֶלָּא פְרִיעָה וּפְרִימָה\hdot
אֵין בֵּין טָהוֹר מִתוֹךְ הֶסְגֵר\hdot
לַטָּהוֹר מִתּוֹךְ הֶחְלֵט\hdot
אֶלָּא תִגְלַחַת וְצִפּוֹרִים׃}

\halacha{אֵין בֵּין סְפָרִים לִתְפִלִּין וּמְזוּזוֹת\hdot
אֶלָּא שֶׁהַסְּפָרִים נִכְתָּבִין בְּכָל לָשׁוֹן\hdot
וּתְפִילִין וּמְזוּזוֹת אֵינָן נִכְתָּבוֹת אֶלָּא אֲשׁוּרִית\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר אַף בַּסְּפָרִים לֹא הִתִּירוּ שֶׁיִּכָּֽתְבוּ\hdot
אֶלָּא יְוָנִית׃}

\halacha{אֵין בֵּין כֹּהֵן מָשׁוּחַ בְּשֶׁמֶן הַמִּשְׁחָה לִמְרֻבֶּה בְגָדִים\hdot
אֶלָּא פַר הַבָּא עַל כָּל הַמִּצְוֺת\hdot
אֵין בֵּין כֹּהֵן מְשַׁמֵּשׁ לְכֹהֵן שֶׁעָבַר\hdot
אֶלָּא פַר יוֹם הַכִּפּוּרִים\hdot
וַעֲשִׂירִית הָאֵיפָה׃}

\halacha{אֵין בֵּין בָּמָה גְדוֹלָה לְבָמָה קְטַנָּה\hdot
אֶלָּא פְסָחִים\hdot
זֶה הַכְּלָל כָּל שֶׁהוּא נִדָּר וְנִדָּב קָרֵב בַּבָּמָה\hdot
וְכָל שֶׁאֵינוֹ לֹא נִדָּר וְלֹא נִדָּב\hdot
אֵינוֹ קָרֵב בַּבָּמָה׃}

\halacha{אֵין בֵּין שִׁילֹה לִירוּשָׁלַיִם\hdot
אֶלָּא שֶׁבְּשִׁילֹה אוֹכְלִים קָדָשִׁים קַלִּים\hdot
וּמַעֲשֵׂר שֵׁנִי בְּכָל הָרוֹאֶה\hdot
וּבִירוּשָׁלַיִם לִפְנִים מִן הַחוֹמָה\hdot
וְכַאן וְכַאן קָדְשֵׁי קָדָשִׁים\hdot
נֶאֱכָלִים לִפְנִים מִן הַקְּלָעִים\hdot
קְדוּשַׁת שִׁילֹה יֵשׁ אַחֲרֶיהָ הֵתֵר\hdot
קְדוּשַׁת יְרוּשָׁלַיִם אֵין אַחֲרֶיהָ הֵתֵר׃}

\perek{הַקּוֹרֵא}{אֶת הַמְּגִלָּה לְמַפְרֵעַ לֹא יָצָא\hdot
קְרָאָהּ עַל פֶּה\hdot
קְרָאָהּ תַּרְגוּם\hdot
בְּכָל לָשׁוֹן לֹא יָצָא\hdot
אֲבָל קוֹרִין אוֹתָהּ לַלּוֹעֲזוֹת בְּלַעַז\hdot
וְהַלּוֹעֵז שֶׁשָּׁמַע אֲשׁוּרִית יָצָא׃}

\halacha{קְרָאָהּ סְרוּגִין וּמִתְנַמְנֵם יָצָא\hdot
הָיָה כוֹתְבָהּ\hdot
דּוֹרְשָׁהּ\hdot
וּמַגִּיהָהּ\hdot
אִם כִּיוֵן לִבּוֹ יָצָא\hdot
וְאִם לָאוּ לֹא יָצָא\hdot
הָֽיְתָה כְתוּבָה בְּסַם וּבְסִקְרָא וּבְקוֹמוֹס וּבְקַנְקַנְתּוֹם\hdot
עַל הַנְּיָר\hdot
וְעַל הַדִּפְתְּרָא\hdot
לֹא יָצָא\hdot
עַד שֶׁתְּהֵא כְתוּבָה אֲשׁוּרִית\hdot
עַל הַסֵּפֶר וּבַדְּיוֹ׃}

\halacha{בֶּן עִיר שֶׁהָלַךְ לַכְּרָךְ\hdot
וּבֶן כְּרָךְ שֶׁהָלַךְ לָעִיר\hdot
אִם עָתִיד לַחֲזוֹר לִמְקוֹמוֹ\hdot
קוֹרֵא כְמְקוֹמוֹ\hdot
וְאִם לָאוּ קוֹרֵא עִמָּהֶן\hdot
וּמֵהֵיכָן קוֹרֵא אָדָם אֶת הַמְּגִלָּה\hdot
וְיוֹצֵא בָּהּ יְדֵי חוֹבָתוֹ\hdot
רִבִּי מֵאִיר אוֹמֵר כֻּלָּהּ\hdot
רִבִּי יְהוּדָה אוֹמֵר מֵאִישׁ יְהוּדִי\hdot
רִבִּי יוֹסֵי אוֹמֵר מֵאַחַר הַדְּבָרִים הָאֵלֶּה׃}

\halacha{הַכֹּל כְּשֵׁרִין לִקְרוֹת אֶת הַמְּגִלָּה\hdot
חוּץ מֵחֵרֵשׁ שׁוֹטֶה וְקָטָן\hdot
רִבִּי יְהוּדָה מַכְשִׁיר בְּקָטָן\hdot
אֵין קוֹרִין אֶת הַמְּגִלָּה וְלֹא מָלִין\hdot
וְלֹא טוֹבְלִין\hdot
וְלֹא מַזִּין\hdot
וְכֵן שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם\hdot
לֹא תִטְבּוֹל עַד שֶׁתֵּנֵץ הַחַמָּה\hdot
וְכֻלָּן שֶׁעָשׂוּ מִשֶׁעָלָה עַמּוּד הַשַּׁחַר כָּשֵׁר׃}

\halacha{כָּל הַיּוֹם כָּשֵׁר לִקְרִיאַת הַמְּגִלָּה\hdot
וְלִקְרִיאַת הַהַלֵּל\hdot
וְלִתְקִיעַת שׁוֹפָר\hdot
וְלִנְטִילַת לוּלָב\hdot
וְלִתְפִלַּת הַמּוּסָפִין\hdot
וְלַמּוּסָפִין\hdot
וּלְוִדּוּי הַפָּרִים\hdot
וּלְוִדּוּי הַמַּעֲשֵׂר\hdot
וּלְוִדּוּי יוֹם הַכִּפּוּרִים\hdot
לִסְמִיכָה\hdot
לִשְׁחִיטָה\hdot
לְתִנוּפָה\hdot
לְהַגָּשָׁה\hdot
לִקְמִיצָה\hdot
וּלְהַקְטָרָה\hdot
לִמְלִיקָה\hdot
וּלְקַבָּלָה\hdot
וּלְהַזַּיָּה\hdot
וּלְהַשְׁקוֹת סוֹטָה\hdot
לַעֲרִיפַת הָעֶגְלָה וּלְטַהֲרַת הַמְּצוֹרָע׃}
