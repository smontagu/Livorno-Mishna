\tractatehead{מסכת שבת}{shabbat}

\perek[497]{יְצִיאוֹת}{הַשַּׁבָּת}{\hdot
שְׁתַּיִם שֶׁהֵן אַרְבַּע בִּפְנִים\hdot
וּשְׁתַּיִם שֶׁהֵן אַרְבַּע בַּחוּץ\hdot
כֵּיצַד הֶעָנִי עוֹמֵד בַּחוּץ וּבַעַל הַבַּיִת בִּפְנִים\hdot
פָּשַׁט הֶעָנִי אֶת יָדוֹ לִפְנִים\hdot
וְנָתַן לְתוֹךְ יָדוֹ שֶׁל בַּעַל הַבַּיִת\hdot
אוֹ שֶׁנָּטַל מִתּוֹכָהּ וְהוֹצִיא\hdot
הֶעָנִי חַיָּיב וּבַעַל הַבַּיִת פָּטוּר\hdot
פָּשַׁט בַּעַל הַבַּיִת אֶת יָדוֹ לַחוּץ\hdot
וְנָתַן לְתוֹךְ יָדוֹ שֶׁל עָנִי\hdot
אוֹ שֶׁנָּטַל מִתּוֹכָהּ וְהִכְנִיס\hdot
בַּעַל הַבַּיִת חַיָּיב וְהֶעָנִי פָּטוּר\hdot
פָּשַׁט הֶעָנִי אֶת יָדוֹ לִפְנִים\hdot
וְנָטַל בַּעַל הַבַּיִת מִתּוֹכָהּ\hdot
אוֹ שֶׁנָּתַן לְתוֹכָהּ וְהוֹצִיא\hdot
שְׁנֵיהֶם פְּטוּרִים\hdot
פָּשַׁט בַּעַל הַבַּיִת אֶת יָדוֹ לַחוּץ\hdot
וְנָטַל הֶעָנִי מִתּוֹכָהּ\hdot
אוֹ שֶנָּתַן לְתוֹכָהּ וְהִכְנִיס\hdot
שְׁנֵיהֶם פְּטוּרִין׃}

\halacha[430]{לֹא יֵשֵׁב אָדָם לִפְנֵי הַסַּפָּר סָמוּךְ לַמִּנְחָה\hdot
עַד שֶׁיִּתְפַּלֵּל\hdot
לֹא יִכָּנֵס אָדָם לַמֶּרְחָץ\hdot
וְלֹא לַבּוּרְסָקִי\hdot
וְלֹא לֶאֱכוֹל וְלֹא לָדוּן\hdot
וְאִם הִתְחִילוּ אֵין מַפְסִיקִין\hdot
מַפְסִיקִין לִקְרִיאַת שְׁמַע\hdot
וְאֵין מַפְסִיקִין לַתְּפִלָּה׃}

\halacha[607]{לֹא יֵצֵא הַחַיָּיט בְּמַחֲטוֹ סָמוּךְ לַחֲשֵׁכָה\hdot
שֶׁמֶּא יִשְׁכַּח וְיֵצֵא\hdot
וְלֹא הַלַּבְלָר בְּקָלְמוֹסוֹ\hdot
וְלֹא יְפַלֶּה אֶת כֵּלָיו\hdot
וְלֹא יִקְרָא לְאוֹר הַנֵּר\hdot
בֶּאֱמֶת אָֽמְרוּ\hdot
הֶחָזָן רוֹאֶה הֵיכָן תִּינוֹקוֹת קוֹרְאִים\hdot
אֲבָל הוּא לֹא יִקְרָא\hdot
כַּיּוֹצֵא בוֹ לֹא יֹאכַל הַזָּב עִם הַזָּבָה\hdot
מִפְּנֵי הֶרְגֵּל עֲבֵרָה׃}

\halacha{וְאֵלּוּ מִן הַהֲלָכוֹת שֶׁאָֽמְרוּ בַּעֲלִיַּת חֲנַנְיָה (בֶּן) חִזְקִיָּה בֶן גּוּרְיוֹן\hdot
כְּשֶׁעָלוּ לְבַקְּרוֹ\hdot
נִמְנוּ\hdot
וְרַבּוּ בֵּית שַׁמָּאי עַל בֵּית הִלֵּל\hdot
וּשְׁמוֹנָה עָשָׂר דְּבָרִים גָּֽזְרוּ בּוֹ בַיּוֹם׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין שׁוֹרִין דְּיוֹ וְסַמֲּמָנִים וְכַרְשִׁינִים\hdot
אֶלָּא כְּדֵי שֶׁיִּשּׁוֹרוּ מִבְּעוֹד יוֹם\hdot
וּבֵית הִלֵּל מַתִּירִין׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין נוֹתְנִין אוֹנִין שֶׁל פִּשְׁתָּן לְתוֹךְ הַתַּנּוּר אֶלָּא כְּדֵי שֶׁיַּהֲבִילוּ מִבְּעוֹד יוֹם\hdot
וְלֹא אֶת הַצֶּמֶר לַיּוֹרָה אֶלָּא כְּדֵי שֶׁיִּקְלוֹט הָעַיִן\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
בֵּית שַׁמָּאי אוֹמְרִים אֵין פּוֹרְשִׂין מְצוּדוֹת חַיָּה וְעוֹפוֹת וְדָגִים\hdot
אֶלָּא כְּדֵי שׁיִּצּוֹדוּ מִבְּעוֹד יוֹם\hdot
וּבֵית הִלֵּל מַתִּירִין׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין מוֹכְרִין לַנָּכְרִי וְאֵין טוֹעֲנִין עִמּוֹ\hdot
וְאֵין מִגְבִּיהִין עָלָיו\hdot
אֶלָּא כְּדֵי שֶׁיַּגִּיעַ לְמָקוֹם קָרוֹב וּבֵית הִלֵּל מַתִּירִין׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין נוֹתְנִין עוֹרוֹת לְעַבְדָּן\hdot
וְלֹא כֵלִים לְכוֹבֵס נָכְרִי\hdot
אֶלָּא כְּדֵי שֶׁיֵּעָשׂוּ מִבְּעוֹד יוֹם\hdot
וּבְכֻלָּן בֵּית הִלֵּל מַתִּירִין עִם הַשָּׁמֶשׁ׃}

\halacha{אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל נוֹהֲגִין הָיוּ בֵית אַבָּא\hdot
שֶׁהָיוּ נוֹתְנִין כְּלֵי לָבָן לַכּוֹבֵס נָכְרִי שְׁלֹשָׁה יָמִים קוֹדֶם לַשַּׁבָּת\hdot
וְשָׁוִין אֵלּוּ וְאֵלּוּ\hdot
שֶׁטּוֹעֲנִין קוֹרוֹת בֵּית הַבָּד וְעִגּוּלֵי הַגַּת׃}

\halacha{אֵין צוֹלִין בָּשָׁר\hdot
בָּצֵל\hdot
וּבֵיצָה\hdot
אֶלָּא כְּדֵי שֶׁיִּצּוֹלוּ מִבְּעוֹד יוֹם\hdot
אֵין נוֹתְנִין פַּת לַתַּנּוּר עִם חֲשֵׁכָה\hdot
וְלֹא חֲרָרָה עַל גַּבֵּי גֶחָלִים אֶלָּא כְּדֵי שֶׁיִּקָּֽרְמוּ פָנֶיהָ מִבְּעוֹד יוֹם\hdot
רִבִּי אֱלִיעֶזֶר (בֶּן יַעֲקׁב) אוֹמֵר כְּדֵי שֶׁיִּקְרוֹם הַתַּחְתּוֹן שֶׁלָּהּ׃}

\halacha{מְשַׁלְשְׁלִין אֶת הַפֶּסַח בַּתַּנּוּר עִם חֲשֵׁיכָה\hdot
וּמַאֲחִיזִין אֶת הָאוּר בִּמְדוּרַת בֵּית הַמּוֹקֵד\hdot
וּבַגְּבוּלִין כְּדֵי שֶׁתֶּאֱחוֹז הָאוּר בְּרֻבָּן\hdot
רִבִּי יְהוּדָה אוֹמֵר (אַף) בַּפֶּחָמִין כָּל שֶׁהוּא׃}

\perek[102]{בַּמֶּה}{מַדְלִיקִין}{\hdot
וּבַמֶּה אֵין מַדְלִיקִין\hdot
אֵין מַדְלִיקִין לֹא בְלֶכֶשׁ\hdot
וְלֹא בְחוֹסֶן\hdot
וְלֹא בְכָלָךְ\hdot
וְלֹא בִפְתִילַת הָאִידָן\hdot
וְלֹא בִפְתִילַת הַמִּדְבָּר\hdot
וְלֹא בִירוֹקָה שֶׁעַל פְּנֵי הַמַּיִם\hdot
לֹא בַזֶּפֶת\hdot
וְלֹא בַשַּׁעֲוָה\hdot
וְלֹא בְשֶׁמֶן קִיק\hdot
וְלֹא בְשֶׁמֶן שְׂרֵפָה\hdot
וְלֹא בָּאַלְיָה\hdot
וְלֹא בַחֵלֶב\hdot
נָחוּם הַמָּדִי אוֹמֵר\hdot
מַדְלִיקִין בְּחֵלֶב מְבֻשָּׁל\hdot
וַחֲכָמִים אוֹמְרִים אֶחָד מְבֻשָּׁל\hdot
וְאֶחָד שֶׁאֵינוֹ מְבֻשָּׁל\hdot
אֵין מַדְלִיקִין בּוֹ׃}

\halacha[725]{אֵין מַדְלִיקִין בְּשֶׁמֶן שְׂרֵפָה בְּיוֹם טוֹב\hdot
רִבִּי יִשְׁמָעֵאל אוֹמֵר אֵין מַדְלִיקִין בְּעִטְרָן מִפְּנֵי כְבוֹד הַשַּׁבָּת\hdot
וַחֲכָמִים מַתִּירִין בְּכָל הַשְּׁמָנִים\hdot
בְּשֶׁמֶן שֻׁמְשְׁמִין\hdot
בְּשֶׁמֶן אֱגוֹזִים\hdot
בְּשֶׁמֶן צְנוֹנוֹת\hdot
בְּשֶׁמֶן דָּגִים\hdot
בְּשֶׁמֶן פַּקֻּעוֹת\hdot
בְעִטְרָן וּבְנֶפְטְ\hdot
רִבִּי טַרְפוֹן אוֹמֵר אֵין מַדְלִיקִין אֶלָּא בְשֶׁמֶן זַיִת בִּלְבַד׃}

\halacha[391]{כָּל הַיּוֹצֵא מִן הָעֵץ אֵין מַדְלִיקִין בּוֹ אֶלָּא פִשְׁתָּן\hdot
וְכָל הַיּוֹצֵא מִן הָעֵץ אֵינוֹ מִטַּמֵּא טוּמְאַת אֹהָלִים אֶלָּא פִשְׁתָּן\hdot
פְּתִילַת הַבֶּגֶד שֶׁקִּפְּלָהּ וְלֹא הִבְהֲבָהּ\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר טְמֵאָה וְאֵין מַדְלִיקִין בָּהּ\hdot
רִבִּי עֲקִיבָא אוֹמֵר טְהוֹרַָה וּמַדְלִיקִין בָּהּ׃}

\halacha[419]{לֹא יִקּוֹב אָדָם שְׁפוֹפֶרֶת שֶׁל בֵּיצָה\hdot
וִימַלְּאֶנָּה שֶׁמֶן וְיִתְּנֶנָּה עַל פִּי הַנֵּר\hdot
בִּשְׁבִיל שֵֶׁתְּהֵא מְנַטֶּפֶת\hdot
אֲפִילוּ הִיא שֶׁל חֶרֶס\hdot
וְרִבִּי יְהוּדָה מַתִּיר\hdot
אֲבָל אִם חִבְּרָהּ הַיּוֹצֵר מִתְּחִלָּה מוּתָר\hdot
מִפְּנֵי שֶׁהוּא כְּלִי אֶחָד\hdot
לֹא יְמַלֵּא אָדָם קְעָרָה שֶׁמֶן\hdot
וְיִתְּנֶנָּה בְּצַד הַנֵּר\hdot
וְיִתֵּן רֹאשׁ הַפְּתִילָה בְּתוֹכָהּ\hdot
בִּשְׁבִיל שֶׁתְּהֵא שׁוֹאֶבֶת\hdot
וְרִבִּי יְהוּדָה מַתִּיר׃}

\halacha[574]{הַמְּכַבֶּה אֶת הַנֵּר\hdot
מִפְּנֵי שֶׁהוּא מִתְיָרֵא\hdot
מִפְּנֵי גוֹיִם\hdot
מִפְּנֵי לִסְטִים\hdot
מִפְּנֵי רוּחַ רָעָה\hdot
וְאִם בִּשְׁבִיל הַחוֹלֶה שֶׁיִּישָׁן פָּטוּר\hdot
כְּחַס עַל הַנֵּר\hdot
כְחַס עַל הַשֶּׁמֶן\hdot
כְּחַס עַל הַפְּתִילָה חַיָּיב\hdot
וְרִבִּי יוֹסֵי פּוֹטֵר בְּכֻלָּן\hdot
חוּץ מִן הַפְּתִילָה מִפְּנֵי שֶׁהוּא עוֹשָׂהּ פֶּחָם׃}

\halacha{עַל שָׁלֹשׁ עֲבֵירוֹת נָשִׁים מֵתוֹת בְּשַׁעַת לִדְתָּן\hdot
עַל שֶׁאֵינָן זְהִירוֹת\hdot
בַּנִּדָּה\hdot
וּבַחַלָּה\hdot
וּבְהַדְלָקַת הַנֵּר׃}

\halacha[118]{שְׁלֹשָׁה דְבָרִים צָרִיךְ אָדָם לוֹמַר בְּתוֹךְ בֵּיתוֹ עֶרֶב שַׁבָּת עִם חֲשֵׁכָה\hdot
עִשַּׂרְתֶּן\hdot
עִרַבְתֶּן\hdot
הַדְלִיקוּ אֶת הַנֵּר\hdot
סָפֵק חֲשֵׁכָה סָפֵק אֵין חֲשֵׁכָה\hdot
אֵין מְעַשְּׂרִין אֶת הַוַּדָּאי\hdot
וְאֵין מַטְבִּילִין אֶת הַכֵּלִים\hdot
וְאֵין מַדְלִיקִין אֶת הַנֵּרוֹת\hdot
אֲבָל מְעַשְּׂרִין אֶת הַדְּמָאי\hdot
וּמְעָֽרְבִין\hdot
וְטוֹמְנִין אֶת הַחַמִּין׃}

\perek[695]{כִּירָה}{שֶׁהִסִּיקוּהָ}{בְּקַשׁ וּבְגַבָּבָא\hdot
נוֹתְנִין עָלֶיהָ תַּבְשִׁיל\hdot
בְּגֶפֶת וּבְעֵצִים\hdot

\perek{כָּל}{הַכֵּלִים}{נִיטָּלִים בְּשַׁבָּת\hdot
וְדַלְתוֹתֵיהֶן עִמָּהֶן\hdot
אַף עַל פִּי שֶׁנִּתְפָֽרְקוּ בְשַׁבָּת\hdot
שֶׁאֵינָן דּוֹמִין לְדַלְתוֹת הַבַּיִת\hdot
לְפִי שֶׁאֵינָן מִן הַמּוּכָן׃}

\halacha{נוֹטֵל אָדָם קָרְנָס לְפַצֵּעַ בּוֹ אֶת הָאֱגוֹזִים\hdot
וְקָרְדוֹם לַחְתּוֹךְ בּוֹ אֶת הַדְּבֵילָה\hdot
מְגֵירָה לִגְרוֹר בָּהּ אֶת הַגְּבִינָה\hdot
מַגְרֵיפָה לִגְרוֹף בָּהּ אֶת הַגְּרוֹגְרוֹת\hdot
אֶת הָרַחַת וְאֵֶת הַמַּזְלֵֵג לָתֵת עָלָיו (פַּת) לַקָּטָן\hdot
אֶת הַכּוֹשׁ וְאֶת הַכַּרְכָּר לִתְחוֹב בּוֹ\hdot
מַחַט שֶׁל יָד\hdot
לִטּוֹל בּוֹ אֶת הַקּוֹץ\hdot
וְשֶל סַקָּאִים לִפְתּוֹחַ בּוֹ אֶת הַדָּלֶת׃}

\halacha{קָנֶה שֶׁל זֵיתִים\hdot
אִם יֵשׁ קֶשֶׁר בְּרֹאשׁוֹ מְקַבֵּל טֻמְאָה\hdot
וְאִם לָאוּ אֵינוֹ מְקַבֵּל טֻמְאָה\hdot
בֵּין כַּךְ וּבֵין כַּךְ נִטַּל בְּשַׁבָּת׃}

\halacha{(רִבִּי יְהוּדָה אוֹמֵר) כָּל הַכֵּלִים נִטָּלִין\hdot
חוּץ מִן הַמֶּסֶר הַגָּדוֹל\hdot
וְיָתֵד שֶׁל מַחֲרֵישָׁה\hdot
כָּל הַכֵּלִים נִטָּלִין לְצוֹרֶךְ\hdot
וְשֶׁלֹּא לְצוֹרֶךְ\hdot
רִבִּי נְחֶמְיָה אוֹמֵר אֵין נִטָּלִין אֶלָּא לְצוֹרֶךְ׃}

\halacha[460]{כָּל הַכֵּלִים הַנִּטָּלִין בְּשַׁבָּת\hdot
שִׁבְרֵיהֶן נִטָּלִין (עִמָּהֶן)\hdot
וּבִלְבַד שֵֶׁיְּהוּ עוֹשִׂין מֵעֵין מְלָאכָה\hdot
שִׁבְרֵי עֲרֵבָה לְכַסּוֹת בָּהֶן אֶת פִּי הֶחָבִית\hdot
שִׁבְרֵי זְכוּכִית לְכַסּוֹת בָּהֶן אֶת פִּי הַפַּךְ\hdot
רִבִּי יְהוּדָה אוֹמֵר וּבִלְבַד שֶׁיְּהוּ עוֹשׂין מֵעֵין מְלַאכְתָּן\hdot
שִׁבְרֵי עֲרֵיבָה לָצוּק לְתוֹכָן מַקְפָּה\hdot
וְשִׁבְרֵי זְכוּכִית לָצוּק לְתוֹכָן שָׁמֶן׃}

\halacha[857]{הָאֶבֶן שֶׁבַּקְּרוּיָה\hdot
אִם מְמַלְּאִין בָּהּ וְאֵינָהּ נוֹפֶלֶת\hdot
מְמַלְּאִין בָּהּ\hdot
וְאִם לָאוּ\hdot
אֵין מְמַלְּאִין בָּהּ\hdot
זְמוֹרָה שֶׁהִיא קְשׁוּרָה בְּטַפִּיחַ\hdot
מְמַלְּאִין בָּהּ בְּשַׁבָּת׃}

\halacha[190]{פְּקַק הַחַלּוֹן\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר בִּזְמָן שֶׁהוּא קָשׁוּר וְתָלוּי\hdot
פּוֹקְקִין בּוֹ\hdot
וְאִם לָאוּ\hdot
אֵין פּוֹקְקִין בּוֹ\hdot
וַחֲכָמִים אוֹמְרִים בֵּין כַּךְ וּבֵין כַּךְ פּוֹקְקִין בּוֹ׃}
