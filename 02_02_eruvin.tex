\tractatehead{מסכת ערובין}{eruvin}

\perek{מָבוֹי}{שֶׁהוּא גָבוֹהַּ\hdot
לְמַעְלָה מֵעֶשְׂרִים אַמָּה יִמָּעֵט\hdot
רִבִּי יְהוּדָה אוֹמֵר אֵינוֹ צָרִיךְ\hdot
וְהָרַחַב מֵעֶשֶׂר אַמּוֹת יִמָּעֵט\hdot
וְאִם יֵשׁ לוֹ צוּרַת פֶּתַח\hdot
אַף עַל פִּי שֶׁהוּא רַחַב מֵעֶשֶׂר אַמּוֹת אֵין צָרִיךְ לְמָעֵט׃}

\halacha[109]{הֶכְשֵׁר מָבוֹי\hdot
בֵּית שַׁמָּאי אוֹמְרִים לֶחִי וְקוֹרָה\hdot
וּבֵית הִלֵּל אוֹמְרִים לֶחִי אוֹ קוֹרָה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר לְחָיַיִן\hdot
מִשּׁוּם רִבִּי יִשְׁמָעֵאל אָמַר תַּלְמִיד אֶחָד לִפְנֵי רִבִּי עֲקִיבָא\hdot
לֹא נֶחְלְקוּ בֵּית שַׁמָּאי וּבֵית הִלֵּל עַל מָבוֹי שֶׁהוּא פָחוּת מֵאַרְבַּע אַמּוֹת\hdot
שֶׁהוּא נִתָּר אוֹ בְלֶחִי אוֹ בְקוֹרָה\hdot
עַל מַה נֶּחֱלָקוּ\hdot
עַל רַחַב מֵאַרְבַּע אַמּוֹת וְעַד עֶשֶׂר\hdot
שֶׁבֵּית שַׁמָּאי אוֹמְרִים לֶחִי וְקוֹרָה\hdot
וּבֵית הִלֵּל אוֹמְרִים אוֹ לֶחִי אוֹ קוֹרָה\hdot
אָמַר רִבִּי עֲקִיבָא עַל זֶה וְעַל זֶה נֶחֱלָקוּ׃}

\halacha{הַקּוֹרָה שֶׁאָֽמְרוּ רְחָבָה כְּדֵי לְקַבֵּל אָרִיחַ\hdot
וְאָרִיחַ חֲצִי לְבֵנָה שֵֶׁל שְׁלֹשָׁה טְפָחִים\hdot
דַּיָּהּ לַקּוֹרָה שֶׁתְּהֵא רְחָבָה טֶפַח\hdot
כְּדֵי לְקַבֵּל אָרִיחַ לְאָרְכּוֹ׃}

\halacha{רְחָבָה כְּדֵי לְקַבֵּל אָרִיחַ וּבְרִיאָה כְּדֵי לְקַבֵּל אָרִיחַ\hdot
רִבִּי יְהוּדָה אוֹמֵר רְחָבָה אַף עַל פִּי שֶׁאֵינָהּ בְּרִיאָה׃}

\halacha[1132]{הָֽיְתָה שֶׁל קַשׁ אוֹ שֶׁל קָנִים\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא שֶׁל מַתֶּכֶת\hdot
עֲקוּמָה\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא פְשׁוּטָה\hdot
עֲגוּלָה\hdot
רוֹאִין אוֹתָהּ כְּאִלּוּ הִיא מְרֻבַּעַת\hdot
כָּל שֶׁיֵּשׁ בְּהֵקֵיפוֹ שְׁלֹשָׁה טְפָחִים\hdot
יֵשׁ בּוֹ רוֹחַב טֶפַח׃}

\halacha{לְחָיַיִן שֶׁאָֽמְרוּ\hdot
גָּבְהָן עֲשָׁרָה טְפָחִים\hdot
וְרָחְבָּן וְעָבְיָן כָּל שֶׁהוּא\hdot
רִבִּי יוֹסֵי אוֹמֵר רָחְבָּן שְׁלֹשָׁה טְפָחִים׃}

\halacha{בַּכֹּל עוֹשִׂין לְחָיַיִן\hdot
אֲפִילוּ בְדָבָר שֶׁיֵּשׁ בּוֹ רוּחַ חַיִּים\hdot
וְרִבִּי יוֹסֵי אוֹסֵר\hdot
וּמְטַמֵּא מִשּׁוּם גֹּלֶל\hdot
וְרִבִּי מֵאִיר מְטַהֵר\hdot
וְכוֹתְבִין עָלָין גִטֵּי נָשִׁים\hdot
וְרִבִּי יוֹסֵי הַגְּלִילִי פּוֹסֵל׃}

\halacha[2189]{שְׁיָירָה שֶׁחָֽנְתָה בַּבִּקְעָה וְהִקִּיפוּהָ בִּכְלֵי בְהֵמָה\hdot
מְטַלְטְלִין בְּתוֹכָהּ\hdot
וּבִלְבַד שֶׁיְּהֵא גָּדֵר גָּבוֹהַּ עֲשָׂרָה טְפָחִים\hdot
וְלֹא יְהוּ פְרָצוֹת יְתֵרוֹת עַל הַבִּנְיָן\hdot
כָּל פִּרְצָה שֶׁהִיא כְעֶשֶׂר אַמּוֹת מֻתֶּרֶת\hdot
מִפְּנֵי שֶׁהִיא כְפֶתַח\hdot
יָתֵר מִכָּאן אָסוּר׃}
