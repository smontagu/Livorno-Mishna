\tractatehead{מסכת אבות}{abot}

\perek{מֹשֶׁה}{קִבֵּל תּוֹרָה מִסִּינַי\hdot
וּמְסָרָהּ לִיהוֹשֻׁעַ\hdot
וִיהוֹשֻׁעַ לִזְקֵנִים\hdot
וּזְקֵנִים לִנְבִיאִים\hdot
וּנְבִיאִים מְסָרוּהָ לְאַנְשֵׁי כְנֶסֶת הַגְּדוֹלָה\hdot
הֵם אָֽמְרוּ שׁלֹשָׁה דְבָרִים\hdot
הֲווֹ מְתוּנִים בַּדִּין\hdot
וְהַעֲמִידוּ תַּלְמִידִים הַרְבֶּה\hdot
וַעֲשׂוּ סְיַג לַתּוֹרָה׃}

\halacha{שִׁמְעוֹן הַצַּדִּיק הָיָה מִשִּׁיְּרֵי כְנֶסֶת הַגְּדוֹלָה\hdot
הוּא הָיָה אוֹמֵר עַל שְׁלֹשָׁה דְבָרִים הָעוֹלָם עוֹמֵד\hdot
עַל הַתּוֹרָה\hdot
וְעַל הָעֲבוֹדָה\hdot
וְעַל גְּמִילוּת חֲסָדִים׃}

\halacha{אַנְטִיגְנוֹס אִישׁ סוֹכוֹ קִבֵּל מִשִּׁמְעוֹן הַצַּדִּיק\hdot
הוּא הָיָה אוֹמֵר\hdot
אַל תִּהְיוּ כַעֲבָדִים\hdot
הַמְּשַׁמְּשִׁין אֶת הָרַב עַל מְנַת לְקַבֵּל פְּרָס\hdot
אֶלָּא הֲווֹ כַעֲבָדִים הַמְּשַׁמּשִׁין אֶת הָרַב\hdot
עַל מְנַת שֶׁלֹּא לְקַבֵּל פְּרָס\hdot
וִיהִי מוֹרָא שָׁמַיִם עֲלֵיכֶם׃}

\halacha{יוֹסֵי בֶן יוֹעֶזֶר אִישׁ צְרֵידָה וְיוֹסֵי בֶן יוֹחָנָן אִישׁ יְרוּשָׁלַיִם קִבְּלוּ מֵהֶם\hdot
יוֹסֵי בֶן יוֹעֶזֶר (אִישׁ צְרֵידָה) אוֹמֵר\hdot
יְהִי בֵיתְךָ בֵּית וַעַד לַחֲכָמִים\hdot
וַהֲוֵי מִתְאַבֵּק בַּעֲפַר רַגְלֵיהֶם\hdot
וַהֲוֵי שׁוֹתֶה בַצָּמָא אֶת דִּבְרֵיהֶם׃}

\halacha{יוֹסֵי בֶן יוֹחָנָן אִישׁ יְרוּשָׁלַיִם אוֹמֵר\hdot
יְהִי בֵיתְךָ פָּתוּחַ לָֽרְוָחָה\hdot
וְיִהְיוּ עֲנִיִּים בְּנֵי בֵיתֶךָ\hdot
וְאַל תַּרְבֶּה שִׂיחָה עִם הָאִשָּׁה בְּאִשְׁתּוֹ אָֽמְרוּ\hdot
קַל וָחֹמֶר בְּאֵשֶׁת חֲבֵירוֹ\hdot
מִכָּאן אָֽמְרוּ חֲכָמִים\hdot
כָּל זְמָן שֶׁאָדָם מַרְבֶּה שִׂיחָה עִל הָאִשָּׁה\hdot
גּוֹרֵם רָעָה לְעַצְמוֹ\hdot
וּבוֹטֵל מִדִּבְרֵי תוֹרָה\hdot
וְסוֹפוֹ יוֹרֵשׁ גֵּיהִנָּם׃}

\halacha{יְהוֹשֻׁעַ בֶּן פְּרַחְיָא וְנִתָּאי הָאַרְבְּלִי קִבְּלוּ מֵהֶם\hdot
יְהוֹשֻׁעַ בֶּן פְּרַחְיָה אוֹמֵר\hdot
עֲשֵׂה לְךָ רָב\hdot
וּקְנֵה לְךָ חָבֵר\hdot
וַהֲוֵי דָּן אֶת כָּל הָאָדָם לְכַף זְכוּת׃}

\halacha{נִתָּאי הָאַרְבְּלִי אוֹמֵר\hdot
הַרְחֵק מִשָּׁכֵן רָע\hdot
וְאַל תִּתְחַבֵּר לָרָשָׁע\hdot
וְאַל תִּתְיָאֵשׁ מִן הָפּוּרְעָנוּת׃}

\halacha{יְהוּדָה בֶן טַבָּאי וְשִׁמְעוֹן בֶּן שָׁטָח קִבְּלוּ מֵהֶם\hdot
יְהוּדָה בֶן טַבָּאי אוֹמֵר\hdot
אַל תַּעַשׂ עַצְמָךְ כְּעוֹרְכֵי הַדַּיָּינִין\hdot
וּכְשֶׁיִּהְיוּ בַעֲלֵי דִינִים עוֹמְדִים לְפָנֶיךָ יִהְיוּ בְעֵינֶיךָ כָּֽרְשָׁעִים\hdot
וּכְשֶׁנִּפְטָרִים מִלְּפָנֶיךָ\hdot
יִהְיוּ בְעֵינֶיךָ כְּזַכָּאִין\hdot
לִכְשֶׁקִּבְּלוּ עֲלֵיהֶן אֶת הַדִּין׃}

\halacha{שִׁמְעוֹן בֶּן שָׁטָח אוֹמֵר\hdot
הֲוֵי מַרְבֶּה לַחֲקוֹר אֶת הָעֵדִים\hdot
וַהֲוֵי זָהִיר בִּדְבָרֶיךָ\hdot
שֶׁמֶא מִתּוֹכָם יִלְמְדוּ לְשַׁקֵּר׃}

\halacha{שְׁמַעְיָה וְאַבְטַלְיוֹן קִבְּלוּ מֵהֶם\hdot
שְׁמַעְיָה אוֹמֵר\hdot
אֱהוֹב אֶת הַמְּלָאכָה\hdot
וּשְׂנֹא אֶת הָרַבָּנוֹת\hdot
וְאַל תִתְוַדַּע לָרָשׁוּת׃}

\halacha{אַבְטַלְיוֹן אוֹמֵר\hdot
חֲכָמִים הִזָּהֲרוּ בְדִבְרֵיכֶם\hdot
שֶׁמֶא תָחוֹבוּ חוֹבַת גָּלוּת\hdot
וְתִגְלוּ לִמְקוֹם מַיִם הָרָעִים\hdot
וְיִשְׁתּוּ הַתַּלְמִידִים הַבָּאִים אַחֲרֵיכֶם וְיָמוּתוּ\hdot
וְנִמְצָא שֵׁם שָׁמַיִם מִתְחַלֵּל׃}

\halacha{הִלֵּל וְשַׁמָּאי קִבְּלוּ מֵהֶם\hdot
הִלֵּל אוֹמֵר\hdot
הֱוֵי מִתַּלְמִידָיו שֶׁל אַהֲרֹן\hdot
אוֹהֵב שָׁלוֹם וְרוֹדֵף שָׁלוֹם\hdot
אוֹהֵב אֶת הַבִּרְיוֹת וּמְקָֽרְבָן לַתּוֹרָה׃}

\halacha{הוּא הָיָה אוֹמֵר\hdot
נְגַד שְׁמָא\hdot
אָבַד שְׁמֵיהּ\hdot
וּדְלָא מוֹסִיף\hdot
יֵסִיף\hdot
וּדְלָא יָלִיף קְטָלָא חַיּב\hdot
וּדְיִשְׁתַּמֵּשׁ בְּתָגָא חָלַף׃}

\halacha{הוּא הָיָה אוֹמֵר אִם אֵין אֲנִי לִי\hdot
מִי לִי\hdot
וּכְשֶׁאֲנִי לְעַצְמִי מָה אָנִי\hdot
וְאִם לֹא עַכְשָׁיו אֵימָתַי׃}

\halacha{שַׁמָּאי אוֹמֵר עֲשֵׂה תוֹרָֽתְךָ קֶבַע\hdot
אֱמוֹר מְעַט וַעֲשֵׂה הַרְבֶּה\hdot
וַהֲוֵי מְקַבֵּל אֶת כָּל הָאָדָם\hdot
בְּסֵבֶר פָּנִים יָפוֹת׃}

\halacha{רַבָּן גַּמְלִיאֵל אוֹמֵר עֲשֵׂה לְךָ רָב\hdot
וְהִסְתַּלֵּק מִן הַסָּפֵק\hdot
וְאַל תַּרְבֶּה לְעַשֵּׂר אוֹמָדוֹת׃}

\halacha{שִׁמְעוֹן בְּנוֹ אוֹמֵר\hdot
כָּל יָמַי גֻּדַּלְתִּי בֵּין הַחֲכָמִים\hdot
וְלֹא מָצָאתִי לַגּוּף טוֹב אֶלָּא שְׁתִיקָה\hdot
וְלֹא הַמֶּדְרָשׁ הוּא הָעִקָּר אֶלָּא הַמַּעֲשֶׂה\hdot
וְכָל הַמַּרְבֶּה דְבָרִים מֵבִיא חֶטְא׃}

\halacha{רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר\hdot
עַל שְׁלֹשָׁה דְבָרִים הָעוֹלָם עוֹמֵד\hdot
עַל הַדִּין\hdot
וְעַל הָאֱמֶת\hdot
וְעַל הַשָּׁלוֹם\hdot
שֶׁנֶּאֱמֵר אֱמֶת וּמִשְׁפַּט שָׁלוֹם שִׁפְטוּ בְּשַׁעֲרֵיכֶם׃}

\perek{רִבִּי}{אוֹמֵר\hdot
אֵיזוֹהִי דֶּרֶךְ יְשָׁרָה שֶׁיָבוּר לוֹ הָאָדָם\hdot
כָּל שֶׁהִיא תִּפְאֶרֶת לְעוֹשֶׂיהָ\hdot
וְתִפְאֶרֶת לוֹ מִן הָאָדָם\hdot
וַהֲוֵי זָהִיר בְּמִצְוָה קַלָּה כְּבַחֲמוּרָה\hdot
שֶׁאֵין אַתָּה יוֹדֵעַ מַתָּן שְׂכָרָן שֶׁל מִצְוֺת\hdot
וֶהֱוֵי מְחַשֵּׁב הֶפְסֵד מִצְוָה\hdot
כְּנֶגֶד שְׂכָרָהּ\hdot
וּשְׂכַר עֲבֵירָה כְּנֶגֶד הֶפְסֵדָהּ\hdot
וְהִסְתַּכֵּל בִּשְׁלֹשָׁה דְבָרִים וְאֵין אַתָּה בָא לִידֵי עֲבֵירָה\hdot
דַּע מָה לְמַעְלָה מִמָּךְ\hdot
עַיִן רוֹאָה וְאֹזֶן שׁוֹמַעַת\hdot
וְכָל מַעֲשֶׂיךָ בַּסֵּפֶר נִכְתָּבִין׃}

\halacha{רַבָּן גַּמְלִיאֵל בְּנוֹ שֶׁל רִבִּי יְהוּדָה הַנָּשִׂיא אוֹמֵר\hdot
יָפֶה תַלְמוּד תּוֹרָה עִם דֶּרֶךְ אֶרֶץ\hdot
שֶׁיְּגִיעַת שְׁנֵיהֶם מְשַׁכַּחַת עֲוֺן\hdot
וְכָל תּוֹרָה שׁאֵין עִמָּהּ מְלָאכָה סוֹפָהּ בְּטֵלָה\hdot
וְגוֹרֶרֶת עֲוֺן\hdot
וְכָל הָעֲמֵלִים עִם הַצִּבּוּר\hdot
יִהְיוּ עֲמֵלִים עִמָּהֶם לְשֵׁם שָׁמַיִם\hdot
שֶׁזְּכוּת אֲבוֹתָם מְסַיְּעָתָם\hdot
וְצִדְקָתָם עוֹמֶדֶת לָעַד\hdot
וְאַתֶּם מַעֲלֶה אֲנִי עֲלֵיכֶם\hdot
שָׂכָר הַרְבֶּה כְּאִלּוּ עֲשִׂיתֶם׃}

\halacha{הֱווֹ זְהִירִין בָּֽרְשׁוּת\hdot
שֶׁאֵין מְקָֽרְבִין לוֹ לְאָדָם\hdot
אֶלָּא לְצֹרֶךְ עַצְמָן\hdot
נִרְאִין כְּאוֹהֲבִין בְּשַׁעַת הֲנָאָתָן\hdot
וְאֵין עוֹמְדִין לוֹ לְאָדָם בְּשַׁעַת דּוֹחֲקוֹ׃}

\halacha{הוּא הָיָה אוֹמֵר\hdot
עֲשֵׂה רְצוֹנוֹ כִּרְצוֹנָךְ\hdot
כְּדֵי שֶׁיַּעֲשֶׂה רְצוֹנָךְ כִּרְצוֹנוֹ\hdot
בַּטֵּל רְצוֹנָךְ מִפְּנֵי רְצוֹנוֹ\hdot
כְּדֵי שֶׁיְּבַטֵּל רְצוֹן אֲחֵרִים מִפְּנֵי רְצוֹנָךְ\hdot
הִלֵּל אוֹמֵר אַל תִּפְרוֹשׁ מִן הַצִּבּוּר\hdot
וְאַל תַאֲמֵן בְּעַצְמָךְ עַד יוֹם מוֹתָךְ\hdot
וְאַל תָּדִין אֶת חֲבֵרָךְ עַד שֶׁתַּגִּיעַ לִמְקוֹמוֹ\hdot
וְאַל תֹּאמַר דָּבָר שֶׁאִי אֶפְשָׁר לִשְׁמוֹעַ שֶׁסּוֹפוֹ לְהִשָּׁמַע\hdot
וְאַל תֹּאמַר לִכְשֶׁאֶפְנֶה אֶשְׁנֶה\hdot
שֶׁמֶא לֹא תִפָּנֶה׃}

\halacha{הוּא הָיָה אוֹמֵר אֵין בּוּר יְרֵא חֵטְא\hdot
וְלֹא עַם הָאָרֶץ חָסִיד\hdot
וְלֹא הַבַּיְשָׁן לָמֵד\hdot
וְלֹא הַקַּפְדָּן מְלַמֵּד\hdot
וְלֹא כָל הַמַּרְבֶּה בִּסְחוֹרָה מַחְכִּים\hdot
וּבִמְקוֹם שֶׁאֵין אֲנָשִׁים\hdot
הִשְׁתַּדֵּל לִהְיוֹת אִישׁ׃}

\halacha{אַף הוּא רָאָה גֻּלְגֹּלֶת אֶחָת\hdot
שֶׁצָּפָה עַל פְּנֵי הַמַּיִם\hdot
אָמָר לָהּ (עַל) דְּאַטְפַת אַטְפוּךְ\hdot
וְסוֹף מְטִיפַיִךְ יְטוּפוּן׃}

\halacha{הוּא הָיָה אוֹמֵר\hdot
מַרְבֶּה בָשָׂר מַרְבֶּה רִמָּה\hdot
מַרְבֶּה נְכָסִים מַרְבֶּה דְאָגָה\hdot
מַרְבֶּה נָשִׁים מַרְבֶּה כְשָׁפִים\hdot

הַמְּכַבֵּד אֶת הַבִּרְיוֹת\hdot
שֶׁנֶּאֱמַר כִּי מְכַבְּדַי אֲכַבֵּד וּבוֹזַי יֵקָֽלּוּ׃}

\halacha{בֶּן עֲזָאי אוֹמֵר\hdot
הֱוֵי רָץ לְמִצְוָה קַלָּה (כְּבַחֲמוּרָה\hdot)
וּבוֹרֵחַ מִן הָעֲבֵירָה\hdot
שֶׁמִּצְוָה גּוֹרֶרֶת מִצְוָה\hdot
וַעֲבֵירָה גּוֹרֶרֶת עֲבֵירָה\hdot
שֶׁשְּׂכַר מִצְוָה מִצְוָה\hdot
וּשְׂכַר עֲבֵירָה עֲבֵירָה׃}

\halacha{הוּא הָיָה אוֹמֵר אַל תְּהִי בַז לְכָל אָדָם\hdot
וְאַל תְּהִי מַפְלִיג לְכָל דָּבָר\hdot
שֶׁאֵין לָךְ אָדָם שֶׁאֵין לוֹ שָׁעָה\hdot
וְאֵין לָךְ דָּבָר שֶׁאֵין לוֹ מָקוֹם׃}

\halacha{רִבִּי לְוִיטַס אִישׁ יַבְנֶה אוֹמֵר\hdot
מְאֹד מְאֹד הֱוֵי שְׁפַל רוּחַ (בִּפְנֵי כָל אָדָם\hdot)
שֶׁתִּקְוַת אֱנוֹשׁ רִמָּה\hdot
רִבִּי יוֹחָנָן בֶּן בְּרוֹקָא אוֹמֵר\hdot
(כָּל) הַמְּחַלֵּל שֵׁם שָׁמַיִם בַּסֵּתֶר\hdot
נִפְרָעִין מִמֶּנּוּ בַּגָּלוּי\hdot
אֶחָד שׁוֹגֵג ואֵחָד מֵזִיד בְּחִלּוּל הַשֵּׁם׃}

\halacha{רִבִּי יִשְׁמָעֵאל (בְּנוֹ) אוֹמֵר הַלּוֹמֵד עַל מְנַת לְלַמֵּד\hdot
מַסְפִּיקִין בְּיָדוֹ לִלְמוֹד וּלְלַמֵּד\hdot
וְהַלּוֹמֵד עַל מְנַת לַעֲשׂוֹת\hdot
מַסְפִּיקִין בְּיָדוֹ לִלְמוֹד וּלְלַמֵּד (לִשְׁמוֹר) וְלַעֲשׂוֹת\hdot
רִבִּי צָדוֹק אוֹמֵר אַל תַּעֲשֵׂם עֲטָרָה לְהִתְגַּדֵּל בָּהֶם\hdot
וְלֹא קַרְדוֹם לַחְפּוֹר בָּהֶם\hdot
וְכַךְ הָיָה הִלֵּל אוֹמֵר\hdot
וּדְאִשְׁתַּמֵּשׁ בְּתָגָא חָלַף\hdot
(הָא לָמַדְתָּ\hdot)
כָּל הַנֶּהֱנֶה מִדִּבְרֵי תוֹרָה נוֹטֵל חַיָּיו מִן הָעוֹלָם׃}

\halacha{רִבִּי יוֹסֵי אוֹמֵר כָּל הַמְּכַבֵּד אֶת הַתּוֹרָה\hdot
גּוּפוֹ מְכֻבָּד עַל הָבִּרְיוֹת\hdot
וְכָל הַמְּחַלֵּל אֶת הַתּוֹרָה\hdot
גּוּפוֹ מְחֻלָּל עַל הַבִּרְיוֹת׃}

\halacha{רִבִּי יִשְׁמָעֵאל בְּנוֹ שֶׁל רִבִּי יוֹסֵי אוֹמֵר הַחוֹשֵׂךְ עַצְמוֹ מִן הַדִּין\hdot
פּוֹרֵק מִמֶּנּוּ אֵיבָה וְגָזֵל וּשְׁבוּעַת שָֽׁוְא\hdot
וְהַגָּס לִבּוֹ בְּהוֹרָאָה\hdot
שׁוֹטֶה רָשָׁע וְגַס רוּחַ׃}

\halacha{הוּא הָיָה אוֹמֵר\hdot
אַל תְּהִי דָן יְחִידִי\hdot
שֶׁאֵין דָּן יְחִידִי אֶלָּא אֶחָד\hdot
וְאַל תֹּאמַר קַבְּלוּ דַעְתִּי\hdot
שֶׁהֵן רַשָּׁאִין וְלֹא אַתָּה׃}

\halacha{רִבִּי יוֹנָתָן אוֹמֵר\hdot
כָּל הַמְּקַיֵּם אֶת הַתּוֹרָה מֵעֹנִי\hdot
סוֹפוֹ לְקַיְּמָהּ מֵעֹשֶׁר\hdot
וְכָל הַמְּבַטֵּל אֶת הַתּוֹרָה מֵעֹשֶׁר\hdot
סוֹפוֹ לְבַטְּלָהּ מֵעֹנִי׃}

\halacha{רִבִּי מֵאִיר אוֹמֵר הֲוֵי מְמַעֵט בְּעֵסֶק\hdot
וַעֲסוֹק בַּתּוֹרָה\hdot
וֶהֱוֵי שְׁפַל רוּחַ בִּפְנֵי כָל אָדָם\hdot
אִם בִּטַּלְתָ מִן הַתּוֹרָה יֵשׁ לָךְ בְּטֵלִים הַרְבֶּה כְּנֶגְדָּךְ\hdot
וְאִם עָמַלְתָּ בַתּוֹרָה יֵשׁ (לוֹ) שָׂכָר הַרְבֶּה לִתֵּן לָךְ׃}

\halacha{רִבִּי אֱלִיעֶזֶר (בֶּן יַעֲקֹב) אוֹמֵר\hdot
הָעוֹשֶׂה מִצְוָה אֶחָת\hdot
קוֹנֶה לוֹ פְּרַקְלִיט אֶחָד\hdot
וְהָעוֹבֵר עֲבֵירָה אֶחָת קוֹנֶה לוֹ קַטֵּיגוֹר אֶחָד\hdot
תְּשׁוּבָה וּמַעֲשִׂים טוֹבִים כִּתְרִיס בִּפְנֵי הַפּוּרְעָנוּת\hdot
רִבִּי יוֹחָנָן הַסַּנְדְּלָר אוֹמֵר כָּל כְּנֵסִיָּה שֶׁהִיא לְשֵׁם שָׁמַיִם סוֹפָהּ לְהִתְקַיֵּם\hdot
וּשֶׁאֵינָהּ לְשֵׁם שָׁמַיִם אֵין סוֹפָהּ לְהִתְקַיֵּם׃}

\halacha{רִבִּי אֶלְעָזָר בֶּן שָׁמוּעַ אוֹמֵר\hdot
יְהִי כְבוֹד תַּלְמִידָךְ חָבִיב עָלַיִךְ כְּשֶׁלָּךְ\hdot
וּכְבוֹד חֲבֵרָךְ כְּמוֹרָא רַבָּךְ\hdot
וּמוֹרָא רַבָּךְ כְּמוֹרָא שָׁמַיִם׃}

\halacha{רִבִּי יְהוּדָה אוֹמֵר הֱוֵי זָהִיר בַּתַּלְמוּד\hdot
שֶׁשִּׁגְגַת הַתַּלְמוּד עוֹלֶה זָדוֹן\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר שְׁלֹשָׁה כְתָרִים הֵם\hdot
כֶּתֶר תּוֹרָה\hdot
וְכֶתֶר כְּהוּנָה\hdot
וְכֶתֶר מַלְכוּת\hdot
וְכֶתֶר שֵׁם טוֹב עוֹלֶה עַל גַּבֵּיהֶן׃}

\halacha{רִבִּי נְהוֹרָאי אוֹמֵר\hdot
הֱוֵי גוֹלֶה לִמְקוֹם תּוֹרָה\hdot
וְאַל תֹּאמַר שֶׁהִיא תָבוֹא אַחֲרֶיךָ שֶׁחֲבֵרֶךָ יְקַיְּמוּהָ בְּיָדֶךָ\hdot
וְאֶל בִּינָֽתְךָ אַל תִּשָּׁעֵן׃}

\halacha{רִבִּי יַנָּאי אוֹמֵר\hdot
אֵין בְּיָדֵינוּ לֹא מִשַּׁלְוַת הָֽרְשָׁעִים\hdot
וְאָף לֹא מִיִּסּוּרֵי הַצַּדִּיקִים\hdot
רִבִּי מַתְיָא בֶן חָרָשׁ אוֹמֵר\hdot
הֱוֵי מַקְדִּים בִּשְׁלוֹם כָּל אָדָם\hdot
וֶהֱוֵי זָנָב לָאֲרָיוֹת\hdot
וְאַל תְּהִי רֹאשׁ לַשּׁוּעָלִים׃}

\halacha{רִבִּי יַעֲקֹב אוֹמֵר הָעוֹלָם הַזֶּה דּוֹמֶה לִפְרוֹזְדוֹר בִּפְנֵי הָעוֹלָם הַבָּא\hdot
הַתְקֵן עַצְמָךְ בִּפְרוֹזְדוֹר\hdot
כְּדֵי שֶׁתִּכָּנֵס לִטְרַקְלִין׃}

\halacha{הוּא הָיָה אוֹמֵר\hdot
יָפָה שָׁעָה אַחַת בִּתְשׁוּבָה וּמַעֲשִׂים טוֹבִים בָּעוֹלָם הַזֶּה\hdot
מִכּל חַיֵּי הָעוֹלָם הַבָּא\hdot
וְיָפָה שָׁעָה אַחַת שֶׁל קוֹרַת רוּחַ בָּעוֹלָם הַבָּא\hdot
מִכָּל חַיֵּי הָעוֹלָם הַזֶּה׃}

\halacha{רִבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר\hdot
אַל תְּרַצֶּה אֶת חֲבֵרָךְ בְּשַׁעַת כַּעְסוֹ\hdot
וְאַל תְּנַחֲמֶנּוּ בְּשָׁעָה שֶׁמֵּתוֹ מֻטַּל לְפָנָיו\hdot
וְאַל תִּשְׁאַל לוֹ בְּשַׁעַת נִדְרוֹ\hdot
וְאַל תִּשְׁתַּדֵּל לִראוֹתוֹ בְּשַׁעַת קַלְקָלָתוֹ׃}

\halacha{שׁמוּאֵל הַקָּטָן אוֹמֵר\hdot
בִּנְפוֹל אוֹיִבְךָ אַל תִּשְׂמַח\hdot
וּבִכָּֽשְׁלוֹ אַל יָגֵל לִבֶּךָ\hdot
פֶּן יִרְאֶה יְיָ וְרַע בְּעֵינָיו וְהֵשִׁיב מֵעָלָיו אַפּוֹ׃}
