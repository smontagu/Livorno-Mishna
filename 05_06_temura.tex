\tractatehead{מסכת תמורה}{temura}

\perek{הַכֹּל}{מְמִירִין\hdot
אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים\hdot
לֹא שֶׁאָדָם רַשָּׁאי לְהָמִיר\hdot
אֶלָּא שֶׁאִם הֵמִיר מוּמָר\hdot
וְסוֹפֵג אֶת הָאַרְבָּעִים\hdot
הַכֹּהֲנִים מְמִירִים אֶת שֶׁלָּהֶם\hdot
וְיִשְׂרָאֵל מְמִירִים אֶת שֶׁלָּהֶן\hdot
אֵין הַכֹּהֲנִים מְמִירִים לֹא בַחַטָּאת וְלֹא בָאָשָׁם וְלֹא בַבְּכוֹר\hdot
אָמַר רִבִּי יוֹחָנָן בֶּן נוּרִי וְכִי מִפְּנֵי מָה אֵין מְמִירִים בַּבְּכוֹר\hdot
אָמַר לוֹ רִבִּי עֲקִיבָא חַטָּאת וְאָשָׁם מַתָּנָה לַכֹּהֵן\hdot
וְהַבְּכוֹר מַתָּנָה לַכֹּהֵן\hdot
מַה חַטָּׂאת וְאָשָׁם אֵין מְמִירִים בּוֹ\hdot
אַף הַבְּכוֹר לֹא יְמִירֶנּוּ בוֹ\hdot
אָמַר לוֹ רִבִּי יוֹחָנָן בֶּן נוּרִי מַה לִּי אֵינוֹ מֵמִיר בְּחַטָּאת וְאָשָׁם\hdot
שֶׁאֵין זָכִין בָּהֶן בְּחַיֵּיהֶן\hdot
תֹּאמַר בַּבְּכוֹר שֶׁזָּכִין בּוֹ בְּחַיָּיו\hdot
אָמַר לוֹ רִבִּי עֲקִיבָא וַהֲלֹא כְּבָר נֶאֱמַר\hdot
וְהָיָה הוּא וּתְמוּרָתוֹ יִהְיֶה קֹדֶשׁ\hdot
הֵיכָן קְדוּשָׁה חָלָה עָלָיו בְּבֵית הַבְּעָלִים\hdot
אַף תְּמוּרָה בְּבֵית הַבְּעָלִים׃}

\halacha{מְמִירִין מִן הַבָּקָר עַל הַצֹּאן\hdot
וּמִן הַצֹּאן עַל הַבָּקָר\hdot
מִן הַכְּבָשִׂים עַל הָעִזִּים\hdot
וּמִן הָעִזִּים עַל הַכְּבָשִׁים\hdot
מִן הַזְּכָרִים עַל הַנְּקֵבוֹת\hdot
וּמִן הַנְּקֵבוֹת עַל הַזְּכָרִים\hdot
מִן הַתְּמִימִים עַל בַּעֲלֵי מוּמִין\hdot
וּמִבַּעֲלֵי מוּמִין עַל הַתְּמִימִים\hdot
שֶׁנֶּאֱמַר לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אוֹתוֹ טוֹב בְּרָע\hdot
אוֹ רַע בְּטוֹב\hdot
אֵיזֶהוּ טוֹב בְּרָע\hdot
בַּעֲלֵי מוּמִין שֶׁקָּדַם הֶקְדֵּשָׁן אֶת מוּמָן\hdot
מְמִירִים אֶחָד בִּשְׁנַיִם\hdot
וּשְׁנַיִם בְּאֶחָד\hdot
אֶחָד בְּמֵאָה\hdot
וּמֵאָה בְאֶחָד\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אֵין מְמִירִין אֶלָּא אֶחָד בְּאֶחָד\hdot
שֶׁנֶּאֱמַר וְהָיָה הוּא וּתְמוּרָתוֹ\hdot
מַה הוּא מְיוּחָד\hdot
אַף תְּמוּרָתוֹ מְיוּחֶדֶת׃}

\halacha{אֵין מְמִירִים אֵבָרִים בְּעוֹבָרִים\hdot
וְלֹא עוֹבָרִים בְּאֵבָרִים\hdot
וְלֹא אֵבָרִים וְעוֹבָרִים בִּשְׁלֵמִים\hdot
וְלֹא שְׁלֵמִים בָּהֵן\hdot
רִבִּי יוֹסֵי אוֹמֵר מְמִירִים אֵבָרִים בִּשְׁלֵמִים\hdot
וְלֹא שְׁלֵמִים בְּאֵבָרִין\hdot
אָמַר רִבִּי יוֹסֵי וַהֲלֹא בַּמֻּקְדָּשִׁים\hdot
הָאוֹמֵר רַגְלָהּ זוֹ עוֹלָה כֻּלָּהּ עוֹלָה\hdot
אַף כְּשֶׁיֹּאמַר רַגְלָהּ שֶׁל זוֹ תַחַת זוֹ\hdot
תִּהְיֶה כֻלָּהּ תְּמוּרָה תַּחְתֶּיהָ׃}

\halacha{אֵין הַמְּדֻמָּע מְדַמֵּעַ אֶלָּא לְפִי חֶשְׁבּוֹן\hdot
אֵין הַמְּחֻמָץ מְחַמֵּץ אֶלָּא לְפִי חֶשְׁבּוֹן\hdot
אֵין הַמַּיִם שְׁאוּבִים פּוֹסְלִים אֶת הַמִּקְוֶה\hdot
אֶלָּא לְפִי חֶשְׁבּוֹן׃}

\halacha{אֵין מֵי חַטָּאת נַעֲשִׂים מֵי חַטָּאת\hdot
אֶלָּא עִם מַתַּן אֵפֶר\hdot
אֵין בֵּית הַפֶּרֶס עוֹשֶׂה בֵית הַפֶּרֶס\hdot
וְלֹא תְרוּמָה אַחַר תְּרוּמָה\hdot
וְלֹא תְמוּרָה עוֹשָׂה תְמוּרָה\hdot
וְלֹא הַוָּלֶד עוֹשֶׂה תְמוּרָה\hdot
רִבִּי יְהוּדָה אוֹמֵר הַוָּלֶד עוֹשֶׂה תְמוּרָה\hdot
אָֽמְרוּ לוֹ הֶקְדֵּשׁ עוֹשֶׂה תְמוּרָה\hdot
לֹא הַוָּלֶד\hdot
וְלֹא תְמוּרָה\hdot
עוֹשִׂין תְּמוּרָה׃}

\halacha{הָעוֹפוֹת וְהַמְּנָחוֹת אֵינָן עוֹשִׂין תְּמוּרָה\hdot
שֶׁלֹּא נֶאֱמַר אֶלָּא בַבְּהֵמָה\hdot
הַצִּבּוּר וְהַשּׁוּתָפִים אֵינָן עוֹשִׂין תְּמוּרָה\hdot
שֶׁנֶּאֱמַר לֹא יָמִיר אוֹתוֹ\hdot
יָחִיד עוֹשֶׂה תְמוּרָה\hdot
לֹא הַצִּבּוּר וְלֹא הַשּׁוּתָפִים עוֹשִׂים תְּמוּרָה\hdot
קָרְבְּנוֹת בֶּדֶק הַבַּיִת אֵינָן עוֹשִׂין תְּמוּרָה\hdot
אָמַר רִבִּי שִׁמְעוֹן וַהֲלֹא הַמַּעֲשֵׂר בִּכְלָל הָיָה\hdot
וְלָמָּה יָצָא לְהַקִּישׁ אֵלָיו\hdot
מַה מַּעֲשֵׂר קָרְבַּן יָחִיד\hdot
יָֽצְאוּ קָרְבְּנוֹת צִבּוּר\hdot
מַה מַּעֲשֵׂר קָרְבַּן מִזְבֵּחַ\hdot
יָֽצְאוּ קָרְבְּנוֹת בֶּדֶק הַבָּיִת׃}

\perek{יֵשׁ}{בְּקָרְבְּנוֹת הַיָּחִיד\hdot
מַה שֶּׁאֵין בְּקָרְבְּנוֹת הַצִּבּוּר\hdot
וְיֵשׁ בְּקָרְבְּנוֹת הַצִּבּוּר\hdot
מַה שֶּׁאֵין בְּקָרְבְּנוֹת הַיָּחִיד\hdot
שֶׁקָּרְבְּנוֹת הַיָּחִיד עוֹשִׂים תְּמוּרָה\hdot
וְקָרְבְּנוֹת הַצִּבּוּר אֵינָם עוֹשִׂים תְּמוּרָה\hdot
קָרְבְּנוֹת הַיָּחִיד נוֹהֲגִים בַּזְּכָרִים וּבַנְּקֵבוֹת\hdot
וְקָרְבְּנוֹת הַצִּבּוּר אֵין נוֹהֲגִין אֶלָּא בַּזְּכָרִים\hdot
קָרְבְּנוֹת הַיָּחִיד חַיָּבִין בְּאַחֲרָיוּתָן\hdot
וּבְאַחֲרָיוּת נִסְכֵּיהֶם\hdot
וְקָרְבְּנוֹת הַצִּבּוּר אֵינָן חַיָּבִין\hdot
לֹא בְאַחֲרָיוּתָן וְלֹא בַאַחֲרָיוּת נִסְכֵּיהֶן\hdot
אֲבָל חַיָּבִין בְּאַחֲרָיוּת נִסְכֵּיהֶם כְּשֶׁקָּרַב הַזֶּבַח\hdot
יֵשׁ בְּקָרְבְּנוֹת הַצִּבּוּר\hdot
מַה שֶּׁאֵין בְּקָרְבְּנוֹת הַיָּחִיד\hdot
שֶׁקָּרְבְּנוֹת הַצִּבּוּר דּוֹחִין אֶת הַשַּׁבָּת\hdot
וְאֶת הַטֻּמְאָה\hdot
וְקָרְבְּנוֹת הַיָּחִיד אֵינָן דּוֹחִים לֹא אֶת הַשַּׁבָּת וְלֹא אֶת הַטֻּמְאָה\hdot
אָמַר רִבִּי מֵאִיר וַהֲלֹא חֲבִיתֵי כֹהֵן גָּדוֹל וּפַר יוֹם הַכִּפּוּרִים קָרְבַּן יָחִיד\hdot
וְדוֹחִין אֶת הַשַּׁבָּת וְאֶת הַטֻּמְאָה\hdot
אֶלָּא שֶׁזְּמַנָּן קָבוּעַ׃}

\halacha{חַטָּאת הַיָּחִיד שֶׁכִּפְּרוּ הַבְּעָלִים מֵתוֹת\hdot
וְשֶׁל צִבּוּר אֵינָן מֵתוֹת\hdot
רִבִּי יְהוּדָה אוֹמֵר יָמוּתוּ\hdot
אָמַר רִבִּי שִׁמְעוֹן מַה מָּצִינוּ\hdot
בְּוָלֶד חַטָּאת וּבִתְמוּרַת חַטָּאת\hdot
וּבְחַטָּאת שֶׁמֵּתוּ בְעָלֶיהָ\hdot
בַּיָּחִיד דְּבָרִים אֲמוּרִים\hdot
אֲבָל לֹא בַצִּבּוּר\hdot
אַף שֶׁכִּפְּרוּ הַבְּעָלִים\hdot
וְשֶׁעָֽבְרָה שְׁנָתָן\hdot
בַּיָּחִיד דְּבָרִים אֲמוּרִים\hdot
אֲבָל לֹא בַצִּבּוּר׃}

\halacha{חֹמֶר בַּקֳּדָשִׁים מִבַּתְּמוּרָה\hdot
וּבַתְּמוּרָה מִבַּקֳּדָשִׁים\hdot
שֶׁהַקֳּדָשִׁים עוֹשִׂים תְּמוּרָה\hdot
וְאֵין תְּמוּרָה עוֹשָׂה תְמוּרָה\hdot
הַצִּבּוּר וְהַשּׁוּתָפִין מַקְדִּישִׁים\hdot
אֲבָל לֹא מְמִירִים\hdot
מַקְדִּישִׁים אֲבָרִים וְעוֹבָרִים\hdot
אֲבָל לֹא מְמִירִים\hdot
חֹמֶר בַּתְּמוּרָה שֶׁהַקְּדוּשָׁה חָלָה עַל בַּעֲלַת מוּם קָבוּעַ\hdot
וְאֵינָהּ יוֹצְאָה לַחוֹלִין לְהִגָּזֵז וּלְהֵעָבֵד\hdot
רִבִּי יוֹסֵי בַּר יְהוּדָה אוֹמֵר עָשָׂה שׁוֹגֵג כַּמֵּזִיד בַּתְּמוּרָה\hdot
וְלֹא עָשָׂה שׁוֹגֵג כַּמֵּזִיד בַּמֻּקְדָּשִׁים\hdot
רִבִּי אֶלְעָזָר אוֹמֵר הַכִּלְאַיִם וְהַטְּרֵפָה וְיוֹצֵא דֹפֶן טוֹמְטוֹם וְאַנְדְרוֹגִינוֹס\hdot
לֹא קָדָשִׁים וְלֹא מַקְדִּישִׁים׃}

\perek{אֵלּוּ}{קָדָשִׁים שֶׁוַלְדוֹתֵיהֶן וּתְמוּרוֹתֵיהֶן כַּיּוֹצֵא בָהֶן\hdot
וָלֶד שְׁלָמִים וּתְמוּרָתָן\hdot
וַלְדָּן וּוָלֶד וַלְדָּן עַד סוֹף הָעוֹלָם\hdot
הֲרֵי אֵלּוּ כַּשְּׁלָמִים\hdot
וּטְעוּנִים סְמִיכָה וּנְסָכִים וּתְנוּפַת חָזֶה וָשׁוֹק\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר וָלֶד שְׁלָמִים לֹא יִקָּרֵב שְׁלָמִים\hdot
וַחֲכָמִים אוֹמְרִים יִקָּרֵב\hdot
אָמַר רִבִּי שִׁמְעוֹן לֹא נֶחְלְקוּ עַל וָלֶד וָלֶד שְׁלָמִים\hdot
וְעַל וָלֶד וָלֶד תְּמוּרָה שֶׁלֹּא יִקָּרֵב\hdot
וְעַל מַה נֶחֱלָקוּ\hdot
עַל הַוָּלֶד\hdot
שֶׁרִבִּי אֱלִיעֶזֶר  אוֹמֵר לֹא יִקָּרֵב\hdot
וַחֲכָמִים אוֹמְרִים יִקָּרֵב\hdot
הֵעִיד רִבִּי יְהוֹשֻׁעַ וְרִבִּי פַּפְיָיס עַל וָלֶד שְׁלָמִים שֶׁיִּקָּרֵב שְׁלָמִים\hdot
אָמַר רִבִּי פַּפְיָיס אֲנִי מֵעִיד שֶׂהָֽיְתָה לָנוּ פָרָה זִבְחֵי שְׁלָמִים\hdot
