\chead{יום יח}

\perek{כָּל}{הַזְּבָחִים שֶׁנִּזְבְּחוּ שֶׁלֹּא לִשְׁמָן כְּשֵׁרִים\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִן הַפֶּסַח וּמִן הַחַטָּאת\hdot
הַפֶּסַח בִּזְמַנּוֹ\hdot

הַפֶּסַח בִּזְמַנּוֹ\hdot
וְהַחַטָּאת וְהָאָשָׁם בְּכָל זְמָן\hdot
אָמַר רִבִּי אֱלִיעֶזֶר הַחַטָּאת בָּאָה עַל חֵטְא\hdot
וְהָאָשָׁם בָּא עַל חֵטְא\hdot

\halacha{יוֹסֵי בֶּן חוֹנִי אוֹמֵר הַנִּשְׁחָטִים לְשֵׁם פֶּסַח וּלְשֵׁם חַטָּאת פְּסוּלִין\hdot
שִׁמְעוֹן אֲחִי עֲזַרְיָה אוֹמֵר שְׁחָטָן לְשֵׁם גָּבוֹהַּ מֵהֶם כְּשֵׁרִין\hdot
לְשֵׁם נָמוּךְ מֵהֶם פְּסוּלִין\hdot
כֵּיצַד קָדְשֵׁי קֳדָשִׁים שֶׁשְּׁחָטָן לְשֵׁם קֳדָשִׁים קַלִּים\hdot
פּסוּלִין\hdot
קֳדָשִׁים קַלִּים שֶׁשְּׁחָטָן לְשֵׁם קָדְשֵׁי קֳדָשִׁים\hdot
כְּשֵׁרִין\hdot
הַבְּכוֹר וְהַמַּעֲשֵׂר שֶׁשְּׁחָטָן לְשֵׁם שְׁלָמִים כְּשֵׁרִין\hdot
וּשְׁלָמִים שֶׁשְּׁחָטָן לְשֵׁם בְּכוֹר וּלְשֵׁם מַעֲשֵׂר פְּסוּלִין׃}

\halacha{הַפֶּסַח שֶׁשְּׁחָטוֹ בְּשַׁחְרִית בְּאַרְבָּעָה עָשָׂר שֶׁלֹּא לִשְׁמוֹ\hdot
רִבִּי יְהוֹשֻׁעַ מַכְשִׁיר כְּאִלּוּ נִשְׁחַט בִּשְׁלֹשָׁה עָשָׂר\hdot
בֶּן בְּתֵירָא פּוֹסֵל כְּאִלּוּ נִשְׁחַט בֵּין הָאַרְבָּעִים\hdot
אָמַר שִׁמְעוֹן בֶּן עֲזָאי מְקֻבָּל אֲנִי מִפִּי שִׁבְעִים וּשְׁנַיִם זְקֵנִים\hdot
בְּיוֹם שֶׁהוֹשִׁיבוּ רִבִּי אֶלְעָזָר בֶּן עֲזַרְיָה בִּישִׁיבָה שֶׁכָּל הַזְּבָחִים הַנֶּאֱכָלִים שֶׁנִּזְבְּחוּ שֶׁלֹּא לִשְׁמָן כְּשֵׁרִין\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִן הַפֶּסַח וּמִן הַחַטָּאת\hdot
וְלֹא הוֹסִיף בֶּן עֲזָאי אֶלָּא הָעוֹלָה\hdot
וְלֹא הוֹדוּ לוֹ חֲכָמִים׃}

\halacha{הַפֶּסַח וְהַחַטָּאת שֶׁשְּׁחָטָן שֶׁלֹּא לִשְׁמָן\hdot

לְשֵׁם שְׁלָמִים וּלְשֵׁם פֶּסַח\hdot

בִּשְׁחִיטָה\hdot
וּבְקִבּוּל\hdot
וּבְהִלּוּךְ\hdot
וּבִזְרִיקָה\hdot
רִבִּי שִׁמְעוֹן מַכְשִׁיר בְּהִלּוּךְ\hdot
שֶׁהָיָה רִבִּי שִׁמְעוֹן אוֹמֵר אֵי אֶפְשָׁר שֶׁלֹּא בִשְׁחִיטָה\hdot
וְשֶׁלֹּא בְקַבָּלָה\hdot
וְשֶׁלֹּא בִזְרִיקָה\hdot
אֲבָל אֶפְשָׁר שֶׁלֹּא בְהִלּוּךְ\hdot
שֶׁשּׁוֹחֵט בְּצַד הַמִּזְבֵּחַ וְזוֹרֵק\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר הַמְּהַלֵּךְ בְּמָקוֹם שֶׁהוּא צָרִיךְ לְהַלֵּךְ הַמַּחֲשָׁבָה פוֹסֶלֶת\hdot
וּבִמְקוֹם שֶׁאֵינוֹ צָרְיךְ לְהַלֵּךְ\hdot
אֵין הַמַּחֲשָׁבָה פוֹסֶלֶת׃}

\perek{כָּל}{הַזְּבָחִים שֶׁקִּבֵּל דָּמָן זָר\hdot

אוֹ כַּזַּיִת מִבְּשָׂרוֹ בַחוּץ\hdot
אוֹ לֶאֱכוֹל כַּזַּיִת מֵעוֹר הָאַלְיָה בַחוּץ\hdot
פָּסוּל וְאֵין בּוֹ כָרֵת\hdot
לִזְרוֹק דָּמוֹ לְמָחָר\hdot

\halacha{זֶה הַכְּלָל\hdot

חוּץ לִמְקוֹמוֹ\hdot

וּבִלְבָד שֶׁיִּקְרַב הַמַּתִּיר כְּמִצְוָתוֹ׃}

אוֹ שֶׁשָּׁחַט וְקִבֵּל וְהָלַךְ וְזָרַק חוּץ לִזְמַנּוֹ\hdot

יַחֲזִיר לִכְלִי קֹדֶשׁ\hdot
נִשְׁפַּךְ מִן הַכֶּלִי עַל הָרִצְפָה וַאֲסָפוֹ\hdot
כָּשֵׁר\hdot
נְתָנוֹ עַל גַּבֵּי הַכֶּבֶשׁ שֶׁלֹּא כְנֶגֶד הַיְּסוֹד\hdot
נָתַן אֶת הַנִּתָּנִין לְמַטָּה לְמַעְלָה\hdot
וְאֶת הַנִּתָּנִן לְמַעְלָה לְמַטָּה\hdot
אֶת הַנִּתָּנִין בִּפְנִים בַּחוּץ\hdot
וְאֶת הַנִּתָּנִין בַּחוּץ בִּפְנִים\hdot
אִם יֵשׁ דַּם הַנֶּפֶשׁ\hdot
יַחֲזוֹר הַכָּשֵׁר וִיקַבֵּל׃}

\halacha{הַשּׁוֹחֵט אֶת הַזֶּבַח\hdot
לֶאֱכוֹל דָּבָר שֶׁאֵין דַּרְכּוֹ לֶאֱכוֹל\hdot
וּלְהַקְטִיר דָּבָר שֶׁאֵין דַּרְכּוֹ לְהַקְטִיר\hdot
כָּשֵׁר\hdot
רִבִּי אֱלִיעֶזֶר פּוֹסֵל\hdot
לֶאֱכוֹל דָּבָר שֶׁדַּרְכּוֹ לֶאֱכוֹל\hdot
וּלְהַקְטִיר דָּבָר שֶׁדַּרְכּוֹ לְהַקְטִיר\hdot
פָּחוּת מִכַּזַּיִת כָּשֵׁר\hdot
לֶאֱכוֹל כַּחֲצִי זַיִּת\hdot
וּלְהַקְטִיר כַּחֲצִי זַיִת כָּשֵׁר\hdot
שֶׁאֵין אֲכִילָה וְהַקְטָרָה מִצְטָֽרְפִין׃}

\halacha{הַשּׁוֹחֵט אֶת הַזֶּבַח לֶאֱכוֹל כַּזַּיִת מִן הָעוֹר\hdot
מִן הָרֹטֶב\hdot
מִן הַקִּפָּה\hdot
מִן הָאָלָל\hdot
מִן הָעֲצָמוֹת\hdot
מִן הַגִּידִים\hdot
מִן הַטְּלָפַיִם\hdot
מִן הַקְּרָנַיִם\hdot
חוּץ לִזְמַנּוֹ\hdot
אוֹ חוּץ לִמְקוֹמוֹ כָּשֵׁר\hdot
וְאֵין חַיָּבִין עֲלֵיהֶם\hdot
מִשּׁוּם פִּגּוּל וְנוֹתָר וְטָמֵא׃}

\halacha{הַשּׁוֹחֵט אֶת הַמֻּקְדָּשִׁין\hdot
לֶאֱכוֹל שָׁלִיל אוֹ שִׁלְיָא בַחוּץ\hdot
לֹא פִגֵּל\hdot
הַמּוֹלֵק תּוֹרִים בִּפְנִים\hdot
לֶאֱכוֹל בֵּיצֵיהֶם בַּחוּץ\hdot
לֹא פִגֵּל\hdot
חֲלֵב הַמֻּקְדָּשִׁין וּבֵיצֵי תּוֹרִים\hdot
אֵין חַיָּבִין עֲלֵיהֶן מִשּׁוּם פִּגּוּל וְנוֹתָר וְטָמֵא׃}

\halacha{שְׁחָטוֹ עַל מְנַת לְהַנִּיחַ אֶת דָּמוֹ וְאֶת אֵמוּרָיו לְמָחָר\hdot
אוֹ לְהוֹצִיאָן לַחוּץ\hdot
רִבִּי יְהוּדָה פוֹסֵל\hdot
וַחֲכָמִים מַכְשִׁירִין\hdot
שְׁחָטוֹ עַל מְנַת לִתְּנוֹ עַל גַּבֵּי הַכֶּבֶשׁ שֶׁלֹּא כְנֶגֶד הַיְּסוֹד\hdot
לִתֵּן אֶת הַנִּתָּנִין לְמַטָּן לְמַעְלָן\hdot
וְאֶת הַנִּתָּנִין לְמַעְלָן לְמַטָּן\hdot
וְאֶת הַנִּתָּנִין בִּפְנִים בַּחוּץ\hdot
וְאֶת הַנִּתָּנִין בַּחוּץ בִּפְנִים\hdot
שֶׁיֹּאכְלוּהוּ טְמֵאִים\hdot
שֶׁיַּקְרִיבוּהוּ טְמֵאִים\hdot
שֶׁיֹּאכְלוּהוּ עֲרֵלִים\hdot
שֶׁיַּקְרִיבוּהוּ עֲרֵלִים\hdot
לְשַׁבֵּר עַצְמוֹת הַפֶּסַח\hdot
וְלֶאֱכוֹל מִמֶּנּוּ נָא\hdot
לְעָרֵב דָּמוֹ בְּדַם פְּסוּלִין כָּשֵׁר\hdot
שֶׁאֵין הַמַּחֲשָׁבָה פוֹסֶלֶת\hdot
אֶלָּא חוּץ לִזְמַנּוֹ וְחוּץ לִמְקוֹמוֹ\hdot
וְהַפֶּסַח וְהַחַטָּאת שֶׁלֹּא לִשְׁמָן׃}

\perek{בֵּית}{שַׁמָּאי אוֹמְרִים כָּל הַנִּתָּנִין עַל מִזְבֵּחַ הַחִיצוֹן\hdot
שֶׁאִם נְתָנָן מַתָּנָה אַחַת כִּפֵּר\hdot
וּבְחַטָּאת שְׁתֵּי מַתָּנוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים אַף חַטָּאת שֶׁנִּתְּנָה מַתָּנָה אַחַת כִּפֵּר\hdot
לְפִיכַךְ אִם נָתַן הָרִאשׁוֹנָה כְּתִקּוּנָהּ\hdot
וְאֶת הַשְּׁנִיָּה חוּץ לִזְמַנָּהּ כִּפֵּר\hdot
נָתַן אֶת הָרִאשׁוֹנָה חוּץ לִזְמַנָּהּ\hdot
וְאֶת הַשְּׁנִיָּה חוּץ לִמְקוֹמָהּ\hdot

לְפִיכַךְ אִם נָתַן כֻּלָּן כְּתִקּוּנָן וְאַחַת שֶׁלֹּא כְתִקּוּנָהּ\hdot
