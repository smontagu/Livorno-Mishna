\tractatehead{מסכת קידושין}{kidushin}

\perek{הָאִשָּׁה}{נִקְנֵית בִּשְׁלֹשָׁה דְרָכִים\hdot
וְקוֹנָה אֶת עַצְמָהּ בִּשְׁתֵּי דְרָכִים\hdot
נִקְנֵית בְּכֶסֶף בִּשְׁטָר וּבְבִיאָה\hdot
בְּכֶסֶף בֵּית שַׁמָּאי אוֹמְרִים בְּדִינָר וּבְשָׁוֶה דִינָר\hdot
וּבֵית הִלֵּל אוֹמְרִים בִּפְרוּטָה וּבְשָׁוֶה פְרוּטָה\hdot
וְכַמָּה הִיא פְרוּטָה\hdot
אֶחָד מִשְּׁמֹנָה בְּאִיסַר הָאִטַּלְקִי\hdot
וְקוֹנָה אֶת עַצְמָהּ\hdot
בְּגֵט\hdot
וּבְמִיתַת הַבַּעַל\hdot
הַיְּבָמָה נִקְנֵית בְּבִיאָה\hdot
וְקוֹנָה אֶת עַצְמָהּ בַּחֲלִיצָה\hdot
וּבְמִיתַת הַיָּבָם׃}

\halacha{עֶבֶד עִבְרִי נִקְנֶה בְּכֶסֶף וּבִשְׁטָר\hdot
וְקוֹנֶה אֶת עַצְמוֹ בְּשָׁנִים וּבְיוֹבֵל\hdot
וּבְגִרְעוֹן כֶּסֶף\hdot
יְתֵירָה עָלָיו הָאָמָה הָעִבְרִיָּה\hdot
שֶׁקּוֹנָה אֶת עַצְמָהּ בְּסִימָנִין\hdot
הַנִּרְצָע נִקְנֶה בִרְצִיעָה\hdot
וְקוֹנֶה אֶת עַצְמוֹ בַּיּוֹבל\hdot
וּבְמִיתַת הָאָדוֹן׃}

\halacha{עֶבֶד כְּנַעֲנִי נִקְנֶה בְּכֶסֶף וּבִשְׁטָר וּבַחֲזָקָה\hdot
וְקוֹנֶה אֶת עַצְמוֹ בְכֶסֶף\hdot
עַל יְדֵי אַחֵרִים\hdot
וּבִשְׁטָר עַל יְדֵי עַצְמוֹ\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים בְּכֶסֶף עַל יְדֵי עַצְמוֹ\hdot
וּבִשְׁטָר עַל יְדֵי אַחֵרִים\hdot
וּבִלְבַד שֶׁיְּהֵא הַכֶּסֶף מִשֶּׁל אַחֵרִים׃}
