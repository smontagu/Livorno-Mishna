\tractatehead{מסכת מנחות}{menahot}

\perek{כָּל}{הַמְּנָחוֹת שֶׁנִּקְמְצוּ שֶׁלֹּא לִשְׁמָן\hdot
כְּשֵׁרוֹת\hdot
אֶלָּא שֶׁלֹּא עָלוּ לַבְּעָלִים לְשֵׁם חוֹבָה\hdot
חוּץ מִמִּנְחַת חוֹטֵא וּמִנְחַת קְנָאוֹת\hdot
מִנְחַת חוֹטֵא וּמִנְחַת קְנָאוֹת שֶׁקְּמָצָן שֶׁלֹּא לִשְׁמָן\hdot
נָתַן בַּכֶּלִי וְהָלַךְ וְהִקְטִיר שֶׁלֹּא לִשְׁמָן\hdot
אוֹ לִשְׁמָן וְשֶׁלֹּא לִשְׁמָן\hdot
אוֹ שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן פְּסוּלוֹת\hdot
כֵּיצַד לִשְׁמָן\hdot
וְשֶׁלֹּא לִשְׁמָן\hdot
לְשֵׁם מִנְחַת חוֹטֵא\hdot
וּלְשֵׁם מִנְחַת נְדָבָה\hdot
אוֹ שֶׁלֹּא לִשְׁמָן וְלִשְׁמָן\hdot
לְשֵׁם מִנְחַת נְדָבָה\hdot
וּלְשֵׁם מִנְחַת חוֹטֵא׃}

\halacha{אֶחָד מִנְחַת חוֹטֵא\hdot
וְאֶחָד כָּל הַמְּנָחוֹת שֶׁקְּמָצָן זָר\hdot
אוֹנֵן\hdot
טְבוּל יוֹם\hdot
מְחֻסַּר בְּגָדִים\hdot
מְחֻסַּר כִּפּוּרִים\hdot
שֶׁלֹּא רְחוּץ יָדַיִם וְרַגְלַיִם\hdot
עָרֵל טָמֵא יוֹשֵׁב\hdot
עוֹמֵד עַל גַּבֵּי כֵלִים\hdot
עַל גַּבֵּי בְהֵמָה עַל גַּבֵּי רַגְלֵי חֲבֵרוֹ פָּסַל\hdot
קָמַץ בַּשְּׂמֹאל פָּסַל\hdot
בֵּן בְּתֵירָא אוֹמֵר יַחֲזִיר וְיַחֲזוֹר וְיִקְמוֹץ בְּיָמִין\hdot
קָמַץ וְעָלָה בְיָדוֹ צְרוֹר\hdot
אוֹ גַרְגִּיר מֶלַח\hdot
אוֹ קֹרֶט שֶׁל לְבוֹנָה פָּסַל\hdot
מִפְּנֵי שְׁאָֽמְרוּ הַקֹּמֶץ הַיָּתֵר וְהֶחָסֵר פָּסוּל\hdot
אֵיזֶהוּ הַיָּתֵר שֶׁקְּמָצוֹ מְבוֹרָץ\hdot
וְחָסֵר שֶׁקְּמָצוֹ בְּרָאשֵׁי אֶצְבְּעוֹתָיו\hdot
כֵּיצַד הוּא עוֹשֶׂה\hdot
פּוֹשֵׁט אֶת אֶצְבְּעוֹתָיו עַל פַּס יָדוֹ׃}
