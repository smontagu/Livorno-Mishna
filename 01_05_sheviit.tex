\tractatehead{מסכת שביעית}{sheviit}

\perek{עַד}{אֵימָתַי חוֹרְשִׁין בִּשְׂדֵה הָאִילָן עֶרֶב שְׁבִיעִית\hdot
בֵּית שַׁמָּאי אוֹמְרִים כָּל זְמַן שֶׁהוּא יָפֶה לַפֶּרִי\hdot
וּבֵית הִלֵּל אוֹמְרִים עַד הָעֲצֶרֶת\hdot
וּקְרוֹבִין דִּבְרֵי אֵלּוּ\hdot
לִהְיוֹת כְּדִבְרֵי אֵלּוּ׃}

\halacha{אֵיזֶהוּ שְׂדֵה הָאִילָן\hdot
כָּל שְׁלֹשָׁה אִילָנוֹת לְבֵית סֵאָה\hdot
אִם רְאוּיִין לַעֲשׂוֹת כִּכַּר דְּבֵלָה\hdot
שֶׁל שִׁשִּׁים מָנֶה בָּאִיטַלְקִי\hdot
חוֹרְשִׁים כָּל בֵּית סֵאָה בִּשְׁבִילָן\hdot
פָּחוֹת מִכָּאן\hdot
אֵין חוֹרְשִׁין לָהֶם אֶלָּא מְלֹא הָאוֹרֶה\hdot
וְסַלּוֹ חוּצָה לוֹ׃}

\halacha{אֶחָד אִילָן סְרָק\hdot
וְאֶחָד אִילָן מַאֲכָל\hdot
רוֹאִין אוֹתָן כְּאִלּוּ הֵם תְּאֵנִים\hdot
אִם רְאוּיִין לַעֲשׂוֹת כִּכַּר דְּבֵלָה\hdot
שֶׁל שִׁשִּׁים מָנֶה בָּאִיטַלְקִי\hdot
חוֹרְשִׁין כָּל בֵּית סֵאָה בִּשְׁבִילָן\hdot
פָּחוֹת מִכָּאן\hdot
אֵין חוֹרְשִׁין לָהֶן אֶלָּא לְצָרְכָן׃}

\halacha[135]{הָיָה אֶחָד עוֹשֶׂה כִּכַּר דְּבֵלָה וּשְׁנַיִם אֵין עוֹשִׂין\hdot
אוֹ שְׁנַיִם עוֹשִׂין וְאֶחָד אֵינוֹ עוֹשֶׂה\hdot
אֵין חוֹרְשִׁין לָהֶן אֶלָּא לְצָרְכָן\hdot
עַד שֶׁיִּהְיוּ מִשְּׁלֹשָׁה וְעַד תִּשְׁעָה\hdot
הָיוּ עֲשָׂרָה\hdot
מֵעֲשָׂרָה וּלְמַעְלָה\hdot
בֵּין עוֹשִׂין בֵּין שֶׁאֵינָן עוֹשִׂין\hdot
חוֹרְשִׁין כָּל בֵּית סֵאָה בִּשְׁבִילָן\hdot
שֶׁנֶּאֱמַר בֶּחָרִישׁ וּבַקָּצִיר תִּשְׁבּוֹת\hdot
אֵין צָרִיךְ לוֹמַר חָרִישׁ וְקָצִיר שֶׁל שְׁבִיעִית\hdot
אֶלָּא חָרִישׁ שֶׁל עֶרֶב שְׁבִיעִית שֶׁהוּא נִכְנָס בַּשְּׁבִיעִית\hdot
וְקָצִיר שֶׁל שְׁבִיעִית שֶׁהוּא יוֹצֵא לְמוֹצָאֵי שְׁבִיעִית\hdot
רִבִּי יִשְׁמָעֵאל אוֹמֵר מַה חָרִישׁ רְשׁוּת אַף קָצִיר רְשׁוּת\hdot
יָצָא קְצִיר הָעוֹמֶר׃}

\halacha{שְׁלֹשָׁה אִילָנוֹת שֶׁל שְׁלֹשָׁה אֲנָשִׁים הֲרֵי אֵלּוּ מִצְטָֽרְפִין\hdot
וְחוֹרְשִׁין כָּל בֵּית סֵאָה בִּשְׁבִילָן\hdot
וְכַמָּה יְהֵא בֵינֵהֶם\hdot
רַבָּן (שִׁמְעוֹן בֶּן) גַּמְלִיאֵל אוֹמֵר כְּדֵי שֶׁיְּהֵא הַבַּקָּר עוֹבֵר בְּכֵלָיו׃}

\halacha{עֶשֶׁר נְטִיעוֹת מְפוּזָרוּת בְּתוֹךְ בֵּית סֵאָה\hdot
חוֹרְשִׁין כָּל בֵּית סֵאָה בְּשְׁבִילָן\hdot
עַד רֹאשׁ הַשָּׁנָה\hdot
הָיוּ עֲשׂוּיוֹת שׁוּרָה וּמוּקָפוֹת עֲטָרָה\hdot
אֵין חוֹרְשִׁין לָהֶם אֶלָּא לְצָרְכָן׃}

\halacha[182]{הַנְּטִיעוֹת וְהַדְּלוּעִים מִצְטָֽרְפִין לְתוֹךְ בֵּית סֵאָה\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר כָּל עֲשָׂרָה דְּלוּעִים לְבֵית סֵאָה\hdot
חוֹרְשִׁין כָּל בֵּית סֵאָה עַד רֹאשׁ הַשָּׁנָה׃}

\halacha[4805]{עַד אֵימָתַי נִקְרְאוּ נְטִיעוֹת\hdot
רִבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר עַד שֶׁיָּחוֹלוּ\hdot
רִבִּי יְהוֹשֻעַ אוֹמֵר בַּת שֶׁבַע שָׁנִים\hdot
רִבִּי עֲקִיבָא אוֹמֵר נְטִיעָה כִּשְׁמָהּ\hdot
אִילָן שֶנִּגְמָם וְהוֹצִיא חֲלִיפִין\hdot
מִטֶּפַח וּלְמַטָּה כִּנְטִיעָה\hdot
מִטֶּפַח וּלְמַעְלָה כְּאִילָן\hdot
דִּבְרֵי רִבִּי שִׁמְעוֹן׃}

\perek[1087]{עַד}{אֵימָתַי חוֹרְשִׁין בִּשְׂדֵה לָבָן עֶרֶב שְׁבִיעִית\hdot
עַד שֶׁתִּכְלֶה הַלֵּיחָה\hdot
כָּל זְמַן שֶׁבְּנֵי אָדָם חוֹרְשִׁין לִטַּע בַּמִּקְשָׁאוֹת וּבַמִּדְלָעוֹת\hdot
אָמַר רִבִּי שִׁמְעוֹן נָתַתָּ תּוֹרַת כָּל אֶחָד וְאֶחָד בְּיָדוֹ\hdot
אֶלָּא בִּשְׁדֵה הַלָּבָן עַד הַפֶּסַח\hdot
וּבִשְׁדֵה הָאִילָן עַד עֲצֶרֶת׃}

\halacha[1817]{מְזַבְּלִין וּמְעַדְּרִין בַּמִּקְשָׁאוֹת וּבַמִּדְלָעוֹת עַד רֹאשׁ הַשָּׁנָה\hdot
וְכֵן בְּבֵית הַשְּׁלָחִים\hdot
מְיַבְּלִין מְפָֽרְקִין מְאַבְּקִין מְעַשְּׁנִין עַד רֹאשׁ הַשָּׁנָה\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אַף נוֹטֵל הוּא אֶת הֶעָלֶה מִן הָאֶשְׁכּוֹל בַּשְּׁבִיעִית׃}

\halacha{מְסַקְּלִין עַד רֹאשׁ הַשָּׁנָה\hdot
מְקַרְסְמִין מְזָֽרְדִין מְפַסְּלִין עַד רֹאשׁ הַשָּׁנָה\hdot
רִבִּי יְהוֹשֻׁעַ אוֹמֵר כְּזִרוּדָהּ וּכְפִסּוּלָהּ שֶׁל חֲמִישִׁית כַּךְ שֶׁל שִׁשִּׁית\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר כָּל זְמַן שֶׁאֲנִי רַשָּׁאי בַּעֲבוֹדַת הָאִילָן רַשָּׁאי אֲנִי בְּפִסּוּלוֹ׃}

\halacha[497]{מְזַהֲמִין אֶת הַנְּטִיעוֹת וְכוֹרְכִין אוֹתָן וְקוֹטְמִין אוֹתָן\hdot
וְעוֹשִׂין לָהֶם בָּתִּים\hdot
וּמַשְׁקִין אוֹתָן עַד רֹאשׁ הַשָּׁנָה\hdot
רִבִּי אֶלְעָזָר בַּר צָדוֹק אוֹמֵר אַף מַשְׁקֶה הוּא אֶת הַנּוֹף בַּשְּׁבִיעִית\hdot
אֲבָל לֹא אֶת הָעִקָּר׃}

\halacha{סָכִין אֶת הַפַּגִּים וּמְנַקְּבִים אוֹתָם עַד רֹאשׁ הַשָּׁנָה\hdot
פַּגֵּי עֶרֶב שְׁבִיעִית שֶׁנִּכְנְסוּ לַשְּׁבִיעִית\hdot
וְשֶׁל שְׁבִיעִית שֶׁיָּֽצְאוּ לְמוֹצָאֵי שְׁבִיעִית\hdot
לֹא סָכִין וְלֹא מְנַקְּבִין אוֹתָן\hdot
רִבִּי יְהוּדָה אוֹמֵר מָקוֹם שֶׁנָּהֲגוּ לָסוּךְ אֵינָן סָכִין מִפְּנֵי שֶׁהִיא עֲבוֹדָה\hdot
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לָסוּךְ סָכִין\hdot
רִבִּי שִׁמְעוֹן מַתִּיר בָּאִילָן\hdot
מִפְּנֵי שֶׁהוּא רַשָּׁאי בַּעֲבוֹדַת הָאִילָן׃}

\halacha[3635]{אֵין נוֹטְעִין\hdot
וְאֵין מַבְרִיכִין\hdot
וְאֵין מַרְכִּיבִין\hdot
עֶרֶב שְׁבִיעִית\hdot
פָחוֹת מִשְּׁלֹשִׁים יוֹם לִפְנֵי רֹאשׁ הַשָּׁנָה\hdot
וְאִם נָטַע אוֹ הִבְרִיךְ אוֹ הִרְכִּיב יַעֲקוֹר\hdot
רִבִּי יְהוּדָה אוֹמֵר כָּל הַרְכָּבָה שֶׁאֵינָהּ קוֹלֶטֶת לִשְׁלֹשָׁה יָמִים\hdot
שׁוּב אֵינָהּ קוֹלֶטֶת\hdot
רִבִּי יוֹסֵי וְרִבִּי שִׁמְעוֹן אוֹמְרִים לִשְׁתֵּי שַׁבָּתוֹת׃}

\halacha[248]{הָאוֹרֶז וְהַדּוֹחַן וְהַפְּרָגִין וְהַשּׁוּמְשְׁמִין שֶׁהִשְׁרִישׁוּ לִפְנֵי רֹאשׁ הַשָּׁנָה\hdot
מִתְעַשְּׂרִין לְשֶׁעָבַר וּמֻתָּרִין בַּשְּׁבִיעִית\hdot
וְאִם לָאוּ אֲסוּרִין בַּשְּׁבִיעִית\hdot
וּמִתְעַשְּׂרִין לְשָׁנָה הַבָּאָה׃}

\halacha[6559]{רִבִּי שִׁמְעוֹן שְׁזוֹרִי אוֹמֵר פּוֹל הַמִּצְרִי שֶׁזְּרָעוֹ לְזֶרַע בַּתְּחִלָּה כַּיּוֹצֵא בָהֶן\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אֲפוּנִין הַגַּמְלוֹנִין כַּיּוֹצֵא בָהֶן\hdot
רִבִּי אֶלְעָזָר אוֹמֵר אֲפוּנִין הַגַּמְלוֹנִין מִשֶּׁתַּרְמְלוּ לִפְנֵי רֹאשׁ הַשָּׁנָה׃}

\halacha[5667]{הַבְּצָלִים הַסָּרִיסִים וּפוֹל הַמִּצְרִי\hdot
שֶׁמָּנַע מֵהֶם מַיִם שְׁלֹשִׁים יוֹם לִפְנֵי רֹאשׁ הַשָּׁנָה\hdot
מִתְעַשְּׂרִין לְשֶׁעָבַר וּמֻתָּרִים בַּשְּׁבִיעִית\hdot
וְאִם לָאוּ אֲסוּרִין בַּשְּׁבִיעִית\hdot
וּמִתְעַשְּׂרִין לְשָׁנָה הַבָּאָה\hdot
וְשֶׁל בַּעַל שֶׁמָּנַע מֵהֶם מַיִם שְׁתֵּי עוֹנוֹת דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים שָׁלֹשׁ׃}

\halacha[1694]{הַדְּלוֹעִין שֶׁקִּיְּימָן לְזֶרַע\hdot
אִם הִקְשׁוּ לִפְנֵי רֹאשׁ הַשָּנָה וְנִפְסְלוּ מֵאֹכֶל אָדָם\hdot
מֻתָּר לְקַיְּימָן בַּשְּׁבִיעִית\hdot
וְאִם לָאוּ אָסוּר לְקַיְּימָן בַּשְּׁבִיעִית\hdot
הַתְּמָרוֹת שֶׁלָּהֶם אֲסוּרוֹת בַּשְּׁבִיעִית\hdot
וּמַרְבִּיצִין בֶּעָפָר לָבָן\hdot
דִּבְרֵי רִבִּי שִׁמְעוֹן\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹסֵר\hdot
מְמָֽרְסִין בָּאוֹרֶז בַּשְּׁבִיעִית\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר אֲבָל אֵין מְכַסְּחִין׃}

\perek{מֵאֵימָתַי}{מוֹצִיאִין זְבָלִים לָאַשְׁפָּתוֹת\hdot
מִשֶּׁיַּפְסִיקוּ עוֹבְרֵי עֲבֵירָה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר מִשֶּׁיִּיבָשׁ הַמָּתוֹק\hdot
רִבִּי יוֹסֵי אוֹמֵר מִשֶּׁיִּקְשׁוֹר׃}

\halacha{עַד כַּמָּה מְזַבְּלִין\hdot
