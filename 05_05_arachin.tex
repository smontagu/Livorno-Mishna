\tractatehead{מסכת ערכין}{arachin}

\perek{הַכֹּל}{מַעֲרִיכִין וְנֶעֱרָכִין\hdot
נוֹדְרִים וְנִדָּרִים\hdot
כֹּהֲנִים וּלְוִיִּם וְיִשְׂרָאֵלִים נָשִׁים וַעֲבָדִים\hdot
טוֹמְטוֹם וְאַנְדְרוֹגִינוֹס\hdot
נוֹדְרִים וְנִדָּרִים\hdot
וּמַעֲרִיכִין\hdot
אֲבָל לֹא נֶעֱרָכִין\hdot
שֶׁאֵינוֹ נֶעֱרָךְ אֶלָּא זָכָר וַדָּאי\hdot
וּנְקֵבָה וַדָּאִית\hdot
חֵרֵשׁ שׁוֹטֶה וְקָטָן נִדָּרִין וְנֶעֱרָכִין\hdot
אֲבָל לֹא נוֹדְרִין וְלֹא מַעֲרִיכִין\hdot
מִפְּנֵי שֶׁאֵין בָּהֶם דַּעַת\hdot
פָּחוּת מִבֶּן חֹדֶשׁ נִדָּר אֲבָל לֹא נֶעֱרָךְ׃}

\halacha{הַנָּכְרִי רִבִּי מֵאִיר אוֹמֵר נֶעֱרָךְ אֲבָל לֹא מַעֲרִיךְ\hdot
רִבִּי יְהוּדָה אוֹמֵר מַעֲרִיךְ אֲבָל לֹא נֶעֱרָךְ\hdot
זֶה וְזֶה מוֹדִים שֶׁנּוֹדְרִין וְנִדָּרִין׃}

\halacha{הַגּוֹסֵס וְהַיּוֹצֵא לִיהָרֵג\hdot
לֹא נִדָּר וְלֹא נֶעֱרָךְ\hdot
רִבִּי חֲנַנְיָא בֶן עֲקִיבָא אוֹמֵר נֶעֱרָךְ\hdot
מִפְּנֵי שֶׁדָּמָיו קְצוּבִים\hdot
אֲבָל אֵינוֹ נִדָּר מִפְּנֵי שֶׁאֵין דָּמָיו קְצוּבִין\hdot
רִבִּי יוֹסֵי אוֹמֵר נוֹדֵר וּמַעֲרִיךְ וּמַקְדִּישׁ\hdot
וְאִם הִזִּיק חַיָּב בְּתַשְׁלוּמִין׃}

\halacha{הָאִשָּׁה שֶׁהִיא יוֹצְאָה לִיהָרֵג\hdot
אֵין מַמְתִּינִין לָהּ עַד שֶׁתֵּלֵד\hdot
יָֽשְׁבָה עַל הַמַּשְׁבֵּר מַמְתִּינִין לָהּ עַד שֶׁתֵּלֵד\hdot
הָאִשָּׁה שֶׁנֶּהֶרְגָה נֶהֱנִין בִּשְׂעָרָהּ\hdot
בְּהֵמָה שֶׁנֶּהֶרְגָה אֲסוּרָה בַהֲנַיַּה׃}

\perek{אֵין}{נֶעֱרָכִין פָּחוּת מִסֶּלַע\hdot
וְלֹא יוֹתֵר עַל חֲמִשִּׁים סֶלַע\hdot
כֵּיצַד\hdot
נָתַן סֶלַע וְהֶעֱשִׁיר\hdot
אֵינוֹ נוֹתֵן כְּלוּם\hdot
פָּחוּת מִסֶּלַע וְהֶעֱשִׁיר\hdot
נוֹתֵן חֲמִשִּׁים סֶלַע\hdot
הָיוּ בְיָדָיו חֲמֵשׁ סְלָעִים\hdot
רִבִּי מֵאִיר אוֹמֵר אֵינוֹ נוֹתֵן אֶלָּא אֶחַת\hdot
וַחֲכָמִים אוֹמְרִים נוֹתֵן אֶת כֻּלָּם\hdot
אֵין בַּעֲרִיכִין פָּחוּת מִסֶּלַע\hdot
וְלֹא יוֹתֵר עַל חֲמִשִּׁים סֶלַע\hdot
אֵין פֶּתַח בַּטּוֹעָה פָּחוּת מִשִּׁבְעַה וְלֹא יוֹתֵר עַל שִׁבְעָה עָשָׂר\hdot
אֵין בַּנְּגָעִים פָּחוּת מִשָּׁבוּעַ אֶחָד\hdot
וְלֹא יוֹתֵר עַל שְׁלֹשָׁה שָׁבוּעוֹת׃}

\halacha{אֵין פּוֹחֲתִין מֵאַרְבָּעָה חֳדָשִׁים הַמְּעֻבָּרִים בַּשָּׁנָה\hdot
וְלֹא נִרְאֶה יוֹתֵר עַל שְׁמוֹנָה\hdot
שְׁתֵּי הַלֶּחֶם נֶאֱכָלוֹת אֵין פָּחוּת מִשְּׁנַיִם וְלֹא יוֹתֵר עַל שְׁלֹשָׁה\hdot
לֶחֶם הַפָּנִים נֶאֱכָל\hdot
אֵין פָּחוּת מִתִּשְׁעָה\hdot
וְלֹא יוֹתֵר עַל אַחַד עָשָׂר\hdot
קָטָן נִמּוֹל אֵין פָּחוּת מִשְּׁמוֹנָה\hdot
וְלֹא יוֹתֵר עַל שְׁנֵים עָשָׂר׃}

\halacha{אֵין פּוֹחֲתִין מֵעֶשְׂרִים וְאַחַת תְּקִיעוֹת בַּמִּקְדָּשׁ\hdot
וְלֹא מוֹסִיפִין עַל אַרְבָּעָה וּשְׁמוֹנֶה\hdot
אֵין פּוֹחֲתִין מִשְּׁנֵי נְבָלִין\hdot
וְלֹא מוֹסִיפִין עַל שִׁשָּׁה\hdot
אֵין פּוֹחֲתִין מִשְּׁנֵי חֲלִילִין\hdot
וְלֹא מוֹסִיפִין עַל שְׁנֵים עָשָׂר\hdot
וּבִשְׁנֵים עָשָׂר יוֹם בַּשָּׁנָה\hdot
הֶחָלִיל מַכֶּה לִפְנֵי הַמִּזְבֵּחַ\hdot
בִּשְׁחִיטַת פֶּסַח רִאשׁוֹן\hdot
וּבִשְׁחִיטַת פֶּסַח שֵׁנִי\hdot
וּבְיוֹם טוֹב רִאשׁוֹן שֶׁל פֶּסַח\hdot
וּבְיוֹם טוֹב שֶׁל עֲצֶרֶת\hdot
וּבִשְׁמוֹנַת יְמֵי הֶחָג\hdot
וְלֹא הָיָה מַכֶּה בַּאֲבוּב שֶׁל נְחֹשֶׁת\hdot
אֶלָּא בַּאֲבוּב שֶׁל קָנֶה\hdot
מִפְּנֵי שֶׁקּוֹלוֹ עָרֵב\hdot
וְלֹא הָיָה מַחֲלִיק אֶלָּא בַאֲבוּב יְחִידִי\hdot
מִפְּנֵי שֶׁהוּא מַחֲלִיק יָפֶה׃}

\halacha{וְעַבְדֵּי כֹהֲנִים הָיוּ\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יוֹסֵי אוֹמֵר מִשְׁפְּחוֹת בֵּית הַפְּגָרִים\hdot
וּבֵית צִפַּרְיָה וּמַאֲמָאוֹם הָיוּ מַשִּׂיאִין לַכְּהוּנָה\hdot
רִבִּי חֲנִינָא בֶן אַנְטִיגְנוֹס אוֹמֵר לְוִיִּם הָיוּ׃}

\halacha{אֵין פּוֹחֲתִין מִשִּׁשָּׁה טְלָאִים הַמְּבֻקָּרִין בְּלִשְׁכַּת הַטְּלָאִים\hdot
כְּדֵי לְשַׁבָּת וּשְׁנֵי יָמִים טוֹבִים שֶׁל רֹאשׁ הַשָּׁנָה\hdot
וּמוֹסִיפִין עַד לְעוֹלָם\hdot
אֵין פּוֹחֲתִין מִשְּׁתֵּי חֲצוֹצְרוֹת\hdot
וּמוֹסִיפִין עַד לְעוֹלָם\hdot
אֵין פּוֹחֲתִין מִתִּשְׁעָה כִנּוֹרוֹת\hdot
וּמוֹסִיפִין עַד לְעוֹלָם\hdot
וְהַצִּלְצָל לְבָד׃}

\halacha{אֵין פּוֹחֲתִין מִשְּׁנֵים עָשָׂר לְוִיִּם עוֹמְדִים עַל הַדּוּכָן\hdot
וּמוֹסִיפִין עַד לְעוֹלָם\hdot
אֵין קָטָן נִכְנָס לָעֲזָרָה וְלַעֲבוֹדָה\hdot
אֶלָּא בְּשָׁעָה שֶׁהַלְוִיִּם עוֹמְדִים בַּשִּׁיר\hdot
וְלֹא הָיוּ אוֹמְרִים בְּנֵבֶל וְכִנּוֹר\hdot
אֶלָּא בַּפֶּה\hdot
כְּדֵי לִתֵּן תֶּבֶל בַּנְּעִימָה\hdot
רִבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר אֵין עוֹלִין לְמִנְיָן\hdot
וְאֵין עוֹמְדִים עַל הַדּוּכָן\hdot
אֶלָּא בָּאָרֶץ הָיוּ עוֹמְדִים\hdot
וְרָאשֵׁיהֶן מִבֵּין רַגְלֵי הַלְוִיִּם\hdot
וְצוֹעֲרֵי הַלְוִיִּם הָיוּ נִקְרָאִים׃}

\perek{יֵשׁ}{בָּעֲרָכִין לְהָקֵל וּלְהַחְמִיר\hdot
בִּשְׂדֵה אֲחוּזָה לְהָקֵל וּלְהַחְמִיר\hdot
בְּשׁוֹר הַמּוּעָד שֶׁהֵמִית אֶת הָעֶבֶד לְהָקֵל וּלְהַחְמִיר\hdot
בָּאוֹנֵס וּבַמְּפַתֶּה וּבְמוֹצִיא שֵׁם רָע לְהָקֵל וּלְהַחְמִיר\hdot
יֵשׁ בָּעֲרָכִין לְהָקֵל וּלְהַחְמִיר\hdot
כֵּיצַד\hdot
אֶחָד שֶׁהֶעֱרִיךְ אֶת הַנָּאֶה שֶׁבְּיִשְׂרָאֵל\hdot
וְאֶת הַכָּעוּר שֶׁבְּיִשְׂרָאֵל\hdot
נוֹתֵן חֲמִשִּׁים סֶלַע\hdot
וְאִם אָמַר הֲרֵי דָמָיו עָלַי\hdot
נוֹתֵן אֶת שִׁוּוּיוֹ׃}

\halacha{בִּשְׂדֵה אֲחוּזָה לְהָקֵל וּלְהַחְמִיר כֵּיצַד\hdot
אֶחָד הַמַּקְדִּישׁ בְּחוֹלַת הַמָּחוֹז\hdot
וְאֶחָד הַמַּקְדִּישׁ בְּפַרְדְּסוֹת סָבַסְטִי\hdot
נוֹתֵן בְּזֶרַע חֹמֶר שְׂעוֹרִים חֲמִשִּׁים שֶׁקֶל כֶּסֶף\hdot
וּבִשְׂדֵה מִקְנָה נוֹתֵן אֶת שִׁוּוּיוֹ\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אֶחָד שְׂדֵה אֲחוּזָה\hdot
וְאֶחָד שְׂדֵה מִקְנָה\hdot
מַה בֵּין שְׂדֵה אֲחוּזָה לִשְׂדֵה מִקְנָה\hdot
אֶלָּא שֶׁבִּשְׂדֵה אֲחוּזָה\hdot
נוֹתֵן חֹמֶשׁ\hdot
וּבִשְׂדֵה מִקְנָה\hdot
אֵינוֹ נוֹתֵן חֹמֶשׁ׃}
