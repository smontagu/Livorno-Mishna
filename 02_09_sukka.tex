\tractatehead{מסכת סוכה}{sukka}

\perek{סֻכָּה}{שֶׁהִיא גְבוֹהָה לְמַעְלָה מֵעֶשְׂרִים אַמָּה פְּסוּלָה\hdot
רִבִּי יְהוּדָה מַכְשִׁיר\hdot
וְשֶׁאֵינָהּ גְּבוֹהָה עֲשָׂרָה טְפָחִים\hdot
וְשֶׁאֵין לָהָּ שָׁלֹשׁ דְּפַנּוֹת וְשֶׁחַמָּתָהּ מְרֻבָּה מִצִּלָּתָהּ\hdot
פְּסוּלָה\hdot
סֻכָּה יְשָׁנָה\hdot
בֵּית שַׁמָּאי פּוֹסְלִין\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot
וְאֵיזוֹ הִיא סֻכָּה יְשָׁנָה\hdot
כָּל שֶׁעֲשָׂאָהּ קֹדֶם לַחַג שְׁלֹשִׁים יוֹם\hdot
אֲבָל אִם עֲשָׂאָהּ לְשֵׁם חַג\hdot
אֲפִילוּ מִתְּחִלַּת הַשָּׁנָה כְּשֵׁרָה׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ תַּחַת הָאִילָן כְּאִלּוּ עֲשָׂאָהּ בְּתוֹךְ הַבַּיִת\hdot
סֻכָּה עַל גַּבֵּי סֻכָּה\hdot
הָעֶלְיוֹנָה כְּשֵׁרָה\hdot
וְהַתַּחְתּוֹנָה פְּסוּלָה\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
אִם אֵין דְּיוּרִין בָּעֶלְיוֹנָה\hdot
הַתַּחְתּוֹנָה כְּשֵׁרָה׃}

\halacha{פֵּרַס עָלֶיהָ סָדִין מִפְּנֵי הַחַמָּה\hdot
אוֹ תַחְתֶּיהָ מִפְּנֵי הַנֶּשֶׁר\hdot
אוֹ שֶׁפֵּרַס עַל גַּבֵּי הַקִּנּוֹף פְּסוּלָה\hdot
אֲבָל פּוֹרֵס הוּא\hdot
עַל גַּבֵּי נַקְלִיטֵי הַמִּטָּה׃}

\halacha{הִדְלָה עָלֶיהָ אֶת הַגֶּפֶן\hdot
וְאֶת הַדָּלַעַת\hdot
וְאֶת הַקִּיסוֹם\hdot
וְסִכֵּךְ עַל גַּבָּהּ פְּסוּלָה\hdot
וְאִם הָיָה סִכּוּךְ הַרְבֶּה מֵהֶן\hdot
אוֹ שֶׁקִצְּצָן\hdot
כְּשֵׁרָה\hdot
זֶה הַכְּלָל\hdot
כָּל דָּבָר שֶׁהוּא מְקַבֵּל טֻמְאָה\hdot
וְאֵין גִּדּוּלוֹ מִן הָאָרֶץ\hdot
אֵין מְסַכְּכִין בּוֹ\hdot
וְכָל דָּבָר שֶׁאֵינוֹ מְקַבֵּל טֻמְאָה\hdot
וְגִדּוּלוֹ מִן הָאָרֶץ\hdot
מְסַכְּכִין בּוֹ׃}

\halacha{חֲבִילֵי קַשׁ\hdot
וַחֲבִילֵי עֵצִים\hdot
וַחֲבִילֵי זְרָדִין אֵין מְסַכְּכִין בָּהֶן\hdot
וְכֻלָּן שֶׁהִתִּירָן\hdot
כְּשֵׁרוֹת\hdot
וְכֻלָּן כְּשֵׁרוֹת לַדְּפָנוֹת׃}

\halacha{מְסַכְּכִין בִּנְסָרִים\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
וְרִבִּי מֵאִיר אוֹסֵר\hdot
נָתַן עָלֶיהָ נֶסֶר שֶׁהוּא רַחַב אַרְבָּעָה טְפָחִים\hdot
כְּשֵׁרָה\hdot
וּבִלְבַד שֶׁלֹּא יִישָׁן תַּחְתָּיו׃}

\halacha{תְּקָרָה שֶׁאֵין עָלֶיהָ מַעֲזִיבָה\hdot
רִבִּי יְהוּדָה אוֹמֵר בֵּית שַׁמָּאי אוֹמְרִים מְפַקְפֵּק וְנוֹטֵל אַחַת מִבִּנְתַּיִם\hdot
וּבֵית הִלֵּל אוֹמְרִים מְפַקְפֵּק אוֹ נוֹטֵל אַחַת מִבִּנְתַּיִם\hdot
רִבִּי מֵאִיר אוֹמֵר נוֹטֵל אַחַת מִבִּנְתַּיִם וְאֵינוֹ מְפַקְפֵּק׃}

\halacha{הַמְּקָרֶה סֻכָּתוֹ בִּשְׁפוֹדִין\hdot
אוֹ בָאֲרוּכוֹת הַמִּטָּה\hdot
אִם יֵשׁ רֵיוַח בֵּינֵיהֶן כְּמוֹתָן\hdot
כְּשֵׁרָה\hdot
הַחוֹטֵט בַּגָּדִישׁ לַעֲשׂוֹת בּוֹ סֻכָּה אֵינָהּ סֻכָּה׃}

\halacha{הַמְּשַׁלְשֵׁל דְּפָנוֹת מִלְּמַעְלָה לְמַטָּה\hdot
אִם גָּבוֹהַּ מִן הָאָרֶץ שְׁלֹשָׁה טְפָחִים פְּסוּלָה\hdot
מִלְּמַטָּה לְמַעְלָה\hdot
אִם גָּבוֹהַּ מִן הָאָרֶץ עֲשָׂרָה טְפָחִים כְּשֵׁרָה\hdot
רִבִּי יוֹסֵי אוֹמֵר כְּשֵׁם שֶׁמִּלְּמַטָּה לְמַעְלָה עֲשָׂרָה טְפָחִים\hdot
כַּךְ מִלְּמַעְלָה לְמַטָּה עֲשָׂרָה טְפָחִים\hdot
הִרְחִיק אֶת הַסִּכּוּךְ מִן הַדְּפָָנוֹת שְׁלְֹשָׁה טְפָחִים\hdot
פְּסוּלָה׃}

\halacha{בַּיִת שֶֶׁנִּפְחַת וְסִכֵּךְ עַל גַּבָּיו\hdot
אִם יֵשׁ מִן הַכֹּתֶל לַסִּכּוּךְ אַרְבַּע אַמּוֹת פְּסוּלָה\hdot
וְכֵן חָצֵר שֶׁהִיא מֻקֶּפֶת אַכְסַדְרָה\hdot
סֻּכָּה גְדוֹלָה שֶׁהִקִּיפוּהָ בְּדָבָר שֶׁאֵין מְסַכְּכִין בּוֹ\hdot
אִם יֵשׁ תַּחְתָּין אַרְבַּע אַמּוֹת פְּסוּלָה׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ כְּמִין צָרִיף\hdot
אוֹ שֶׁסְּמָכָהּ לַכֹּתֶל\hdot
רִבִּי אֱלִיעֶזֶר פּוֹסֵל מִפְּנֵי שֶׁאֵין לָהּ גָּג\hdot
וַחֲכָמִים מַכְשִׁירִין\hdot
מַחֲצֶלֶת קָנִים גְּדוֹלָה\hdot
עֲשָׂאָהּ לִשְׁכִיבָה\hdot
מְקַבֶּלֶת טֻמְאָה\hdot
וְאֵין מְסַכְּכִין בָּהּ\hdot
לְסִכּוּךְ מְסַכְּכִין בָּהּ וְאֵינָהּ מְקַבֶּלֶת טֻמְאָה\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אַחַת קְטַנָּה וְאַחַת גְּדוֹלָה\hdot
עֲשָׂאָהּ לִשְׁכִיבָה מְקַבֶּלֶת טֻמְאָה\hdot
וְאֵין מְסַכְּכִין בָּהּ\hdot
לְסִכּוּךְ מְסַכְּכִין בָּהּ וְאֵינָהּ מְקַבֶּלֶת טֻמְאָה׃}

\perek{הַיָּשֵׁן}{תַּחַת הַמִּטָּה בַּסֻּכָּה\hdot
לֹא יָצָא יְדֵי חוֹבָתוֹ\hdot
אָמַר רִבִּי יְהוּדָה נוֹהֲגִין הָיִינוּ\hdot
שֶׁהָיִינוּ יְשֵׁנִים תַּחַת הַמִּטָּה בִּפְנֵי הַזְּקֵנִים\hdot
וְלֹא אָֽמְרוּ לָנוּ דָבָר\hdot
אָמַר רִבִּי שִׁמְעוֹן מַעֲשֶׂה בְּטָבִי עַבְדּוֹ שֶׁל רַבָּן גַּמְלִיאֵל שֶׁהָיָה יָשֵׁן תַּחַת הַמִּטָּה\hdot
וְאָמַר לָהֶן רַבָּן גַּמְלִיאֵל לַזְּקֵנִים\hdot
רְאִיתֶם טָבִי עַבְדִּי שֶׁהוּא תַּלְמִיד חָכָם\hdot
וְיוֹדֵעַ שֶׁעֲבָדִים פְּטוּרִים מִן הַסֻּכָּה\hdot
לְפִיכַךְ יָשֵׁן הוּא תַּחַת הַמִּטָּה\hdot
וּלְפִי דַרְכֵּנוּ לָמַדְנוּ שֶׁהַיָּשֵׁן תַּחַת הַמִּטָּה לֹא יָצָא יְדֵי חוֹבָתוֹ׃}

\halacha{הַסּוֹמֵךְ סֻכָּתוֹ בִכְרָעֵי הַמִּטָּה כְּשֵׁרָה\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם אֵינָהּ יְכוֹלָה לַעֲמוֹד בִּפְנֵי עַצְמָהּ\hdot
פְּסוּלָה\hdot
סֻכָּה הַמְּדֻבְלֶלֶת וְשֶׁצִּלָּתָהּ מְרֻבָּה מֵחַמָּתָהּ כְּשֵׁרָה\hdot
הַמְּעוּבָה כְּמִין בַּיִת\hdot
אַף עַל פִּי שֶׁאֵין הַכּוֹכָבִים נִרְאִים מִתּוֹכָהּ\hdot
כְּשֵׁרָה׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ בְּרֹאשׁ הָעֲגָלָה\hdot
אוֹ בְרֹאשׁ הַסְּפִינָה\hdot
כְּשֵׁרָה\hdot
וְעוֹלִין לָהּ בְּיוֹם טוֹב\hdot
בְּרֹאשׁ הָאִילָן\hdot
אוֹ עַל גַּבֵּי גָמָל\hdot
כְּשֵׁרָה\hdot
וְאֵין עוֹלִין לָהּ בְּיוֹם טוֹב\hdot
שְׁתַּיִם בָּאִילָן\hdot
וְאַחַת בִּידֵי אָדָם\hdot
אוֹ שְׁתַּיִם בִּידֵי אָדָם\hdot
וְאַחַת בָּאִילָן\hdot
כְּשֵׁרָה\hdot
וְאֵין עוֹלִין לָהּ בְּיוֹם טוֹב\hdot
שְׁלֹשָׁה בִידֵי אָדָם\hdot
וְאַחַת בָּאִילָן\hdot
כְּשֵׁרָה\hdot
וְעוֹלִין לָהּ בְּיוֹם טוֹב\hdot
זֶה הַכְּלָל\hdot
כָּל שֶׁיִּנָּטֵל הָאִילָן וִיכוֹלָה לַעֲמוֹד בִּפְנֵי עַצְמָהּ\hdot
כְּשֵׁרָה\hdot
וְעוֹלִין לָהּ בְּיוֹם טוֹב׃}

\halacha{הָעוֹשֶׂה סֻכָּתוֹ בֵּין הָאִילָנוֹת וְהָאִילָנוֹת דְּפַנּוֹת לָהּ\hdot
כְּשֵׁרָה\hdot
שְׁלוּחֵי מִצְוָה\hdot
פְּטוּרִין מִן הַסֻּכָּה\hdot
חוֹלִין וּמְשַׁמְּשֵׁיהֶן\hdot
פּטוּרִין מִן הַסֻּכָּה\hdot
אוֹכְלִין וְשׁוֹתִין עַרָאי חוּץ לַסֻּכָּה׃}

\halacha{מַעֲשֶׂה וְהֵבִיאוּ לוֹ לְרַבָּן יוֹחָנָן בֶּן זַכָּאי לִטְעוֹם הַתַּבְשִׁיל\hdot
וּלְרַבָּן גַּמְלִיאֵל שְׁתֵּי כוֹתָבוֹת וּדְלִי שֶׁל מַיִם\hdot
וְאָֽמְרוּ הַעֲלוּם לַסֻּכָּה\hdot
וּכְשֶׁנָּֽתְנוּ לוֹ לְרִבִּי צָדוֹק אֹכֶל פָּחוּת מִכַּבֵּיצָה\hdot
נְטָלוֹ בַמַּפָּה וַאֲכָלוֹ חוּץ לַסֻּכָּה וְלֹא בֵירַךְ אַחֲרָיו׃}

\halacha{רִבִּי אֱלִיעֶזֶר אוֹמֵר אַרְבַּע עֶשְׂרֵה סְעוּדוֹת חַיָּיב אָדָם לֶאֱכוֹל בַּסֻּכָּה\hdot
אַחַת בַּיּוֹם וְאַחַת בַּלַּיְלָה\hdot
וַחֲכָמִים אוֹמְרִים אֵין לַדָּבָר קִצְבָּה\hdot
חוּץ מִלֵּילֵי יוֹם טוֹב רִאשׁוֹן שֶׁל חָג בִּלְבַד\hdot
וְעוֹד אָמַר רִבִּי אֱלִיעֶזֶר מִי שֶׁלֹּא אָכַל בְּלֵילֵי יוֹם טוֹב הָרִאשׁוֹן יַשְׁלִים בְּלֵילֵי יוֹם טוֹב הָאַחֲרוֹן\hdot
וַחֲכָמִים אוֹמְרִים אֵין לַדָּבָר תַּשְׁלוּמִין\hdot
עַל זֶה נֶאֱמַר מְעֻוָּת לֹא יוּכַל לִתְקוֹן וְחֶסְרוֹן לֹא יוּכַל לְהִמַָּנוֹת׃}

\halacha{מִי שֶׁהָיָה רֹאשׁוֹ וְרֻבּוֹ בַסֻּכָּה וְשֻׁלְחָנוֹ בְתוֹךְ הַבַּיִת\hdot
בֵּית שַׁמָּאי פּוֹסְלִין\hdot
וּבֵית הִלֵּל מַכְשִׁירִין\hdot

עֲרָבָה שִׁבְעָה\hdot
וּשְׁאַר כָּל הַיָּמִים שִׁשָּׁה׃}

\halacha{מִצְוַת לוּלָב כֵּיצַד\hdot
יוֹם טוֹב הָרִאשׁוֹן שֶׁל חָג שֶׁחַל לִהְיוֹת בְּשַׁבָּת\hdot
(כָּל הָעָם) מוֹלִיכִין אֶת לוּלָבֵיהֶם לְהַר הַבַּיִת מֵעֶרֶב שַׁבָּת\hdot
וְהַחֲזָנִים מְקַבְּלִין מֵהֶן\hdot
וְסוֹדְרִין אוֹתָן עַל גַּבֵּי אִצְטְבָא\hdot
וְהַזְּקֵנִים מַנִּיחִין אֶת שֶׁלָּהֶן בַּלִּשְׁכָּה\hdot
וּמְלַמְּדִים אוֹתָם לוֹמַר כָּל מִי שֶׁמַּגִּיעַ לוּלָבִי בְיָדוֹ הֲרֵי הוּא לוֹ בְמַתָּנָה\hdot
לְמָחָר מַשְׁכִּימִין וּבָאִין\hdot
וְהַחֲזָנִים זוֹרְקִים אוֹתָם לִפְנֵיהֶם\hdot
וְהֵן מַחֲטִיפִין וּמַכִּין אִישׁ אֶת חֲבֵירוֹ\hdot
וּכְשֶׁרָאוּ בֵית דִּין שֶׁבָּאוּ לִידֵי סַכָּנָה\hdot
הִתְקִינוּ שֶׁיְּהֵא כָּל אֶחָד וְאֶחָד נוֹטֵל בְּבֵיתוֹ׃}

\halacha{מִצְוַת עֲרָבָה כֵּיצַד\hdot
מָקוֹם הָיָה לְמַטָּה מִירוּשָׁלַיִם וְנִקְרָא מוֹצָא\hdot
יוֹרְדִין לְשָׁם וּמְלַקְּטִין מִשָּׁם\hdot
מֻרְבִיּוֹת שֶׁל עֲרָבָה\hdot
וּבָאִין וְזוֹקְפִין אוֹתָן בְּצִדֵּי הַמִּזְבֵּחַ וְרָאשֵׁיהֶן כְּפוּפִין עַל גַּבֵּי הַמִּזְבֵּחַ\hdot
תָּֽקְעוּ וְהֵרִיעוּ וְתָֽקְעוּ\hdot
בְּכָל יוֹם מַקִּיפִין אֶת הַמִּזְבֵּחַ פַּעַם אֶחָת\hdot
וְאוֹמְרִים אָנָּא יְיָ הוֹשִׁיעָה נָּא אָנָּא יְיָ הַצְלִיחָה נָּא\hdot
רִבִּי יְהוּדָה אוֹמֵר אֲנִי וָהוּ הוֹשִׁיעָה נָּא\hdot
וְאוֹתוֹ הַיּוֹם מַקִּיפִין אֶת הַמִּזְבֵּחַ שִׁבְעָה פְעָמִים\hdot
בְּשַׁעַת פְּטִירָתָן מַה הֵן אוֹמְרִים\hdot
יוֹפִי לָךְ מִזְבֵּחַ יוֹפִי לָךְ מִזְבֵּחַ\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר לְיָהּ וְלָךְ מִזְבֵּחַ (לְיָהּ וְלָךְ מִזְבֵּחַ)׃}

\halacha{כְּמַעֲשֵׂהוּ בַחוֹל כַּךְ מַעֲשֵׂהוּ בְשַׁבָּת\hdot
אֶלָּא שֵׁהָיוּ מְלַקְּטִין אוֹתָן מֵעֶרֶב שַׁבָּת וּמַנִּיחִים אוֹתָן בְּגִיגִיּוֹת שֶׁל זָהָב כְּדֵי שֶׁלֹּא יִכְמוֹשׁ\hdot
רִבִּי יוֹחָנָן בֶּן בְּרוֹקָא אוֹמֵר חֲרִיּוֹת שֶׁל דְּקָל הָיוּ מְבִיאִין\hdot
וְחוֹבְטִין אוֹתָן בַּקַּרְקַע בְּצִדֵּי הַמִּזְבֵּחַ\hdot
וְאוֹתוֹ הַיּוֹם נִקְרָא יוֹם חִבּוּט חֲרִיּוֹת׃}

\halacha{מִיַּד הַתִּינוֹקוֹת שׁוֹמְטִין אֶת לוּלָבֵיהֶן\hdot
וְאוֹכְלִין אֶתרוֹגֵיהֶן׃}

\halacha{הַהַלֵּל וְהַשִּׂמְחָה שְׁמוֹנָה\hdot
כֵּיצַד\hdot
מְלַמֵּד שֶׁחַיָּיב אָדָם בַּהַלֵּל וּבַשִּׂמְחָה וּבִכְבוֹד יוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג כִּשְׁאַר כָּל יְמוֹת הֶחָג\hdot
סֻכָּה שִׁבְעָה\hdot
כֵּיצַד\hdot
גָּמַר מִלֶּאֱכוֹל לֹא יַתִּיר סֻכָּתוֹ\hdot
אֲבָל מוֹרִיד אֶת הַכֵּלִים מִן הַמִּנְחָה וּלְמַעְלָה\hdot
מִפְּנֵי כְבוֹד יוֹם טוֹב הָאַחֲרוֹן שֶׁל חָג׃}

\halacha{נִסּוּךְ הַמַּיִם (שִׁבְעָה)\hdot
כֵּיצַד\hdot
צְלוֹחִית שֶׁל זָהָב מַחֲזֶקֶת שְׁלֹשָׁה לוֹגִים\hdot
הָיָה מְמַלֵּא מִן הַשִּׁלּוֹחַ\hdot
הִגִּיעוּ לְשַׁעַר הַמַּיִם\hdot
תָּֽקְעוּ וְהֵרִיעוּ וְתָֽקְעוּ\hdot
עָלָה בַכֶּבֶשׁ וּפָנָה לִשְׂמֹאלוֹ\hdot
וּשְׁנֵי סְפָלִים שֶׁל כֶּסֶף הָיוּ שָׁם\hdot
רִבִּי יְהוּדָה אוֹמֵר שֶׁל סִיד הָיוּ\hdot
אֶלָּא שֶׁהָיוּ מֻשְׁחָרִין פְּנֵיהֶם מִפְּנֵי הַיַּיִן\hdot
וּמְנֻקָּבִין כְּמִין שְׁנֵי חֳטָמִין דַקִּין\hdot
אֶחָד מְעֻבֶּה\hdot
וְאֶחָד דַּק\hdot
כְּדֵי שֶׁיִּהְיוּ שְׁנֵיהֶם כָּלִין בְּבַת אַחַת\hdot
מַעֲרָבִי שֶׁל מַיִם\hdot
מִזְרָחִי שֶׁל יַיִן\hdot
עִירָה שֶׁל מַיִם לְתוֹךְ שֶׁל יַיִן\hdot
וְשֶׁל יַיִן לְתוֹךְ שֶׁל מַיִם יָצָא\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
בַּלּוֹג הָיָה מְנַסֵּךְ כָּל שְׁמוֹנָה\hdot
וְלַמְּנַסֵּךְ אוֹמְרִים לוֹ הַגְבֵּהַּ יָדֶךָ\hdot
שֶׁפַּעַם אַחַת נִסֵּךְ אֶחָד עַל גַּבֵּי רַגְלָיו\hdot
וּרְגָמוּהוּ כָּל הָעָם בְּאֶתְרוֹגֵיהֶן׃}

\halacha{כְּמַעֲשֵׂהוּ בַחוֹל כַּךְ מַעֲשֵׁהוּ בְשַׁבָּת\hdot
אֶלָּא שֶׁהָיָה מְמַלֵּא מֵעֶרֶב שַׁבָּת חָבִית שֶׁל זָהָב שֶׁאֵינָהּ מְקֻדֶּשֶׁת\hdot
מִן הַשִּׁלּוֹחַ\hdot

\halacha{מִבְּלָאֵי מִכְנְסֵי כֹהֲנִים וּמֵהֶמְיָנֵיהֶן\hdot

\halacha{חֲסִידִים וְאַנְשֵׁי מַעֲשֶׂה\hdot
הָיוּ מְרַקְּדִים לִפְנֵיהֶם בַּאֲבוּקוֹת שֶׁל אוֹר שֶׁבִּידֵיהֶן\hdot
וְאוֹמְרִים לִפְנֵיהֶן דִּבְרֵי שִׁירוֹת וְתִשְׁבָּחוֹת\hdot
וְהַלְּוִיִּם בְּכִנּוֹרוֹת וּבִנְבָלִים וּבִמְצִלְתַּיִם וּבַחֲצוֹצְרוֹת\hdot
וּבִכְלֵי שִׁיר בְּלֹא מִסְפַּר\hdot
עַל חֲמִשָּׁה עָשָׂר מַעֲלוֹת הַיּוֹרְדוֹת מֵעֲזָרַת יִשְׂרָאֵל לַעֲזָרַת נָשִׁים\hdot
כְּנֶגֶד חֲמִשָּׁה עָשָׂר שִׁיר הַמַּעֲלוֹת שֶׁבַּתְּהִלִּים\hdot
שֶׁעֲלֵיהֶן לִוִיִּם עוֹמְדִים בִּכְלֵי שִׁיר וְאוֹמְרִים שִׁירָה\hdot
עָֽמְדוּ שְׁנֵי הַכֹּהֲנִים בְּשַׁעַר הָעֶלְיוֹן\hdot
שֶׁיּוֹרֵד מֵעֲזָרַת יִשְׂרָאֵל לַעֲזָרַת נָשִׁים\hdot
וּשְׁתֵּי חֲצוֹצְרוֹת בִּידֵיהֶן\hdot
קָרָא הַגֶּבֶר\hdot
תָּֽקְעוּ וְהֵרִיעוּ וְתָֽקְעוּ\hdot
הִגִּיעוּ לְמַעֲלָה עֲשִׂירִית\hdot
תָּֽקְעוּ וְהֵרִיעוּ וְתָֽקְעוּ\hdot
הִגִּיעוּ לָעֲזָרָה תָּֽקְעוּ וְהֵרִיעוּ וְתָֽקְעוּ\hdot
הָיוּ תוֹקְעִין וְהוֹלְכִין עַד שֶׁמַּגִּיעִין לַשַּׁעַר הַיּוֹצֵא לַמִּזְרָח\hdot
הִגִּיעוּ לַשַּׁעַר הַיּוֹצֵא מִמִּזְרָח\hdot
הָֽפְכוּ פְנֵיהֶן לְמַעֲרָב וְאָֽמְרוּ\hdot
אֲבוֹתֵינוּ שֶׁהָיוּ בַמָּקוֹם הַזֶּה\hdot
אֲחוֹרֵיהֶם אֶל הֵיכָל יְיָ וּפְנֵיהֶם קֵדְמָה\hdot
וְהֵמָּה מִשְׁתַּחֲוִיתֶם קֵדְמָה לַשָּׁמֶשׁ\hdot
וְאָנוּ לְיָהּ עֵינֵינוּ\hdot
רִבִּי יְהוּדָה אוֹמֵר הָיוּ שׁוֹנִין וְאוֹמְרִין אָנוּ לְיָהּ\hdot
וּלְיָהּ עֵינֵינוּ׃}

שָׁלֹשׁ לְהַבְטִיל הָעָם מִמְּלָאכָה\hdot
וְשָׁלֹשׁ לְהַבְדִּיל בֵּין קֹדֶשׁ לְחוֹל\hdot
עֶרֶב שַׁבָּת שֶׁבְּתוֹךְ הֶחָג הָיוּ שָׁם שְׁמוֹנֶה וְאַרְבָּעִים\hdot
שָׁלֹשׁ לִפְתִיחַת שְׁעָרִים\hdot
שָׁלֹשׁ לַשַּׁעַר הָעֶלְיוֹן\hdot
וְשָׁלֹשׁ לַשַּׁעַר הַתַּחְתּוֹן\hdot
וְשָׁלֹשׁ לְמִלּוּי הַמַּיִם\hdot
וְשָׁלֹשׁ עַל גַּבֵּי מִזְבֵּחַ\hdot
תֵּשַׁע לְתָמִיד שֶׁל שַׁחַר\hdot
וְתֵשַׁע לְתָמִיד שֶׁל בֵּין הָעַרְבַּיִם\hdot
וְתֵשַׁע לְמוּסָפִין\hdot
שָׁלֹשׁ לְהַבְטִיל אֶת הָעָם מִמְּלָאכָה\hdot
וְשָׁלֹשׁ לְהַבְדִּיל בֵּין קֹדֶשׁ לְחוֹל׃}

\halacha{יוֹם טוֹב הָרִאשׁוֹן שֶׁל חָג הָיוּ שָׁם שְׁלֹשָׁה עָשָׂר פָּרִים וְאֵלִים שְׁנַיִם וּשְׂעִיר אֶחָד\hdot
נִשְׁתַּיְּרוּ שָׁם אַרְבָּעָה עָשָׂר כְּבָשִׂים לִשְׁמוֹנָה מִשְׁמָרוֹת\hdot
בַּיּוֹם הָרִאשׁוֹן שִׁשָּׁה מַקְרִיבִין שְׁנַיִם שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בַּשֵּׁנִי\hdot
חֲמִשָּׁה מַקְרִיבִין שְׁנַיִם שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בַּשְּׁלִישִׁי\hdot
אַרִבָּעָה מַקְרִיבִין שְׁנַיִם שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בָּרְבִיעִי\hdot
שְׁלֹשָׁה מַקְרִיבִין שְׁנַיִם שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בַּחֲמִישִׁי\hdot
שְׁנַיִם מַקְרִיבִין שְׁנַיִם שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בַּשִּׁשִּׁי\hdot
אֶחָד מַקְרִיב שְׁנַיִם\hdot
וְהַשְּׁאָר אֶחָד אֶחָד\hdot
בַּשְּׁבִיעִי כֻּלָּן שָׁוִין\hdot
בַּשְּׁמִינִי חָֽזְרוּ לְפַיֵּס כְּבָֽרְגָלִים\hdot
אָֽמְרוּ מִי שֶׁהִקְרִיב פָּרִים הַיּוֹם לֹא יַקְרִיב לְמָחָר\hdot
אֶלָּא חוֹזְרִין חָלִילָה׃}

\halacha{בִּשְׁלֹשָׁה פְרָקִים בַּשָּׁנָה הָיוּ כָל הַמִּשְׁמָרוֹת שָׁווֹת\hdot
בְּאֵימוּרֵי הָֽרְגָלִים\hdot
וּבְחִלּוּק לֶחֶם הַפָּנִים\hdot
בַּעֲצֶרֶת אוֹמְרִים לוֹ\hdot
הֵילָךְ מַצָּה הֵילָךְ חָמֵץ\hdot
מִשְׁמָר שֶׁזְּמַנּוֹ קָבוּעַ הוּא מַקְרִיב תְּמִידִין נְדָרִים וּנְדָבוֹת\hdot
וּשְׁאַר קָרְבְּנוֹת צִבּוּר\hdot
וּמַקְרִיב אֶת הַכֹּל\hdot
יוֹם טוֹב הַסָּמוּךְ לְשַׁבָּת בֵּין מִלְּפָנֶיהָ בֵּין לְאַחֲרֶיהָ\hdot
הָיוּ כָל הַמִּשְׁמָרוֹת שָׁווֹת בְּחִלּוּק לֶחֶם הַפָּנִים׃}
