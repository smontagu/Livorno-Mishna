\tractatehead{מסכת מעילה}{meila}

\perek{קָדְשֵׁי}{קָדָשִׁים שֶׁשְּׁחָטָן בַּדָּרוֹם\hdot
מוֹעֲלִין בָּהֵן\hdot
שְׁחָטָן בַּדָּרוֹם וְקִבֵּל דָּמָן בַּצָּפוֹן\hdot
בַּצָּפוֹן וְקִבֵּל דָּמָן בַּדָּרוֹם\hdot
שְׁחָטָן בַּיּוֹם וְזָרַק בַּלַּיְלָה\hdot
בַּלַּיְלָה וְזָרַק בַּיּוֹם\hdot
אוֹ שְׁשְּׁחָטָן חוּץ לִזְמָנָן וְחוּץ לִמְקוֹמָן\hdot
מוֹעֲלִין בָּהֵן\hdot
כְּלָל אָמַר רִבִּי יְהוֹשֻׁעַ כָּל שֶׁהָיָה לָהּ שַׁעַת הֵיתֶר לַכֹּהֲנִים אֵין מוֹעֲלִין בָּהּ\hdot
וְשֶׁלֹּא הָיָה לָהּ שַׁעַת הֵיתֶר לַכֹּהֲנִים מוֹעֲלִין בָּהּ\hdot
אֵיזוֹ הִיא שֶׁהָיָה לָהּ שַׁעַת הֵיתֶר לַכֹּהֲנִים\hdot
שֶׁלָּנָה וְשֶׁנִּטְמָאָה וְשֶׁיָּצָאָה\hdot
וְאֵיזוֹ הִיא שֶׁלֹּא הָיָה לָהּ שַׁעַת הֵיתֶר לַכֹּהֲנִים\hdot
שֶׁנִּשְׁחֲטָה חוּץ לִזְמָנָהּ וְחוּץ לִמְקוֹמָהּ\hdot
וְשֶׁקִּבְּלוּ פְּסוּלִין\hdot
וְזָֽרְקוּ אֶת דָּמָהּ׃}

\halacha{בְּשַׂר קָדְשֵׁי קָדָשִׁים\hdot
שֶׁיָּצָא לִפְנֵי זְרִיקַת דָּמִים\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר מוֹעְלִין בּוֹ\hdot
וְאֵין חַיָּבִין עָלָיו מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְרִבִּי עֲקִיבָא אוֹמֵר אֵין מוֹעֲלִין בּוֹ\hdot
אֲבָל חַיָּבִין עָלָיו מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
אָמַר רִבִּי עֲקִיבָא וַהֲרֵי הַמַּפרִישׁ חַטָּאתוֹ וְנֶאֶבָדָה\hdot
וְהִפְרִישׁ אַחֶרֶת תַּחְתֶּיהָ\hdot
וְאַחַר כַּךְ נִמְצְאָה הָרִאשׁוֹנָה\hdot
הֲרֵי שְׁתֵּיהֶן עומְדוֹת\hdot
לֹא כְשֵׁם שֶׁדָּמָהּ פּוֹטֵר אֶת בְּשָׂרָהּ כַּךְ הוּא פוֹטֵר אֶת בְּשַׂר חֲבֵרְתָּהּ\hdot
וְאִם פָּטַר דָּמָהּ אֶת בְּשַׂר חֲבֵרְתָּהּ מִן הַמְּעִילָה\hdot
דִּין הוּא שֶׁיִּפְטוֹר אֶת בְּשָׂרָהּ׃}

\halacha{אֵימוּרֵי קָדָשִׁים קַלִּים\hdot
שֶׁיָּֽצְאוּ לִפְנֵי זְרִיקַת דָּמִים\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר אֵין מוֹעְלִין בָּהֶן\hdot
וְאֵין חַיָּבִין עֲלֵיהֶן\hdot
מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
רִבִּי עֲקִיבָא אוֹמֵר מוֹעֲלִין בָּהֶן\hdot
וְחַיָּבִין מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא׃}

\halacha{מַעֲשֶׁה דָמִים בְּקָדְשֵׁי קָדָשִׁים לְהָקֵל וּלְהַחְמִיר\hdot
וּבְקָדָשִׁים קַלִּים כֻּלָּן לְהַחְמִיר\hdot
כֵּיצַד קָדְשֵׁי קָדָשִׁים לִפְנֵי זְרִיקַת דָּמִים\hdot
מוֹעֲלִין בָּאֵימוּרִין וּבַבָּשָׂר\hdot
לְאָחָר זְרִיקַת דָּמים\hdot
מוֹעֲלִין בָּאֵימוּרִין וְאֵין מוֹעֲלִין בַּבָּשָׂר\hdot
עָל זֶה וְעַל זֶה חַיָּבִין מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וּבְקָדָשִׁים קַלִּים כֻּלָּן לְהַחְמִיר\hdot
כֵּיצַד\hdot
קָדָשִׁים קַלִּים לִפְנֵי זְרִיקַת דָּמִים\hdot
אֵין מוֹעֲלִין לֹא בָאֵימוּרִין וְלֹא בַבָּשָׂר\hdot
לְאַחַר זְרִיקַת דָּמִים מוֹעֲלִין בָּאֵימוּרִין\hdot
וְאֵין מוֹעֲלִין בַּבָּשָׂר\hdot
עַל זֶה וְעַל זֶה חַיָּבִים מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
נִמְצָא מַעֲשֶׁה דָמִים בְּקָדְשֵׁי קָדָשִׁים לְהָקֵל וּלְהַחְמִיר\hdot
וּבְקָדָשִׁים קַלִּים כֻּלּוֹ לְהַחְמִיר׃}

\perek{חַטָּאת}{הָעוֹף מוֹעֲלִין בָּהּ מִשֶּׁהֻקְדָּשָׁה\hdot
נִמְלָקָה\hdot
הֻכְשָׁרָה לְהִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים וּבַלִּינָה\hdot
הֻזָּה דָמָהּ\hdot
חַיָּבִין עָלֶיהָ מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְאֵין בָּהּ מְעִילָה׃}

\halacha{עוֹלַת הָעוֹף מוֹעֲלִין בָּהּ מִשֶּׁהֻקְדָּשָׁה נִמְלָקָה הֻכְשָׁרָה לְהִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים וּבַלִּינָה\hdot
מִצָּה דָמָהּ\hdot
חַיָּבִין עָלֶיהָ מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וּמוֹעֲלִין בָּהּ עַד שֶׁתֵּצֵא לְבֵית הַדֶּשֶׁן׃}

\halacha{פָּרִים הַנִּשְׂרָפִים וּשְׂעִירִים הַנִּשְׂרָפִים מוֹעֲלִין בָּהֵן מִשֶּׁהֻקְדָּשׁוּ\hdot
נִשְׁחָטוּ הֻכְשָׁרוּ לְהִפָּסֵל בִּטְבוּל יוֹם וּבִמְחֻסַּר כִּפּוּרִים וּבַלִּינָה\hdot
הֻזָּה דָמָן\hdot
חַיָּבִין עֲלֵיהֶן מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וּמוֹעֲלִין בָּהֵן בְּבֵית הַדֶּשֶׁן\hdot
עַד שֶׁיֻּתָּךְ הַבָּשָׂר׃}

\halacha{הָעוֹלָה מוֹעֲלִין בָּהּ מִשֶּׁהֻקְדָּשָׁה\hdot
נִשְׁחָטָה הֻכְשָׁרָה לְהִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים וּבְלִינָה\hdot
נִזְרַק דָּמָהּ\hdot
חַיָּבִין עָלֶיהָ  מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְאֵין מוֹעֲלִין בְּעוֹרָהּ\hdot
אֲבָל מוֹעֲלִין בַּבָּשָׂר\hdot
עד שֶׁיֵּצֵא לְבֵית הַדֶּשֶׁן׃}

\halacha{חַטָּאת וְאָשָׁם וְזִבְחֵי שַׁלְמֵי צִבּוּר מוֹעֲלִין בָּהֵן מִשֶּׁהֻקְדָּשׁוּ\hdot
נִשְׁחָטוּ הֻכְשָׁרוּ לְהִפָּסֵל בִּטְבוּל יוֹם וּבִמְחֻסַּר כִּפּוּרִים וּבְלִינָה\hdot
נִזְרַק דָּמָן חֲיָבִין עֲלֵיהֶן מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
אֵין מוֹעֲלִין בַּבָּשָׂר\hdot
אֲבָל מוֹעֲלִין בָּאֵמוּרִים\hdot
עָד שֶׁיֵּצְאוּ לְבֵית הַדֶּשֶׁן׃}

\halacha{שְׁתֵּי הַלֶּחֶם מוֹעֲלִין בָּהֵן מִשֶּׁהֻקְדָּשׁוּ\hdot
קָֽרְמוּ בַתַּנּוּר\hdot
הֻכְשְׁרוּ לְהִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים וְלִשְׁחוֹט עֲלֵיהֶן אַת הַזֶּבַח\hdot
נִזְרַק דָּמָן שֶׁל כְּבָשִׁים\hdot
חַיָּבִין עֲלֵיהֶן מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְאֵין בָּהֵן מְעִילָה׃}

\halacha{לֶחֶם הַפָּנִים מוֹעֲלִין בּוֹ מִשֶּׁהֻקְדַּשׁ\hdot
קָרַם בַּתַּנּוּר\hdot
הֻכְשַׁר לְהִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים\hdot
וְּלְהִסָּדֵר עַל גַּבֵּי הַשֻּׁלְחָן\hdot
קָֽרְבוּ הַבָּזִיכִין\hdot
חַיָּבִין עָלָיו מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְאֵין בּוֹ מְעִילָה׃}

\halacha{הַמְּנָחוֹת מוֹעֲלִין בָּהֵן מִשֶּׁהֻקְדָּשׁוּ\hdot
קִדִּשׁוּ בַכֵּלִים\hdot
הֻכְשׁרוּ לִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפוּרִים וּבַלִּינָה\hdot
קָרַב הַקֹּמֶץ\hdot
חַיָּבִין עָלָיו מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
וְאֵין מוֹעֲלִין בַּשִּׁירַיִם\hdot
אֲבָל מוֹעֲלִין בַּקֹּמֶץ עַד שֶׁיֵּצֵא לְבֵית הַדֶּשֶׁן׃}

\halacha{הַקֹּמֶץ וְהַלְּבוֹנָה וְהַקְּטֹרֶת\hdot
וּמִנְחַת כֹּהֲנִים\hdot
וּמִנְחַת כֹּהֵן מָשִׁיחַ\hdot
וּמִנְחַת נְסָכִין\hdot
מוֹעֲלִין בָּהֵן מִשֶּׁהֻקְדָּשׁוּ\hdot
קִדְּשׁוּ בַכֶּלִי\hdot
הֻכְשְׁרוּ לִפָּסֵל בִּטְבוּל יוֹם\hdot
וּבִמְחֻסַּר כִּפּוּרִים וּבַלִּינָה\hdot
וְחַיָּבִין עֲלֵיהֶן מִשּׁוּם נוֹתָר\hdot
מִשּׁוּם טָמֵא\hdot
וּפִגּוּל אֵין בָּהֵן\hdot
זֶה הַכְּלָל כָּל שֶׁיֵּשׁ לוֹ מַתִּירִין\hdot
אֵין חַיָּבִין עָלָיו מִשּׁוּם פִּגּוּל נוֹתָר וְטָמֵא\hdot
עַד שֶׁיִּקָּֽרְבוּ מַתִּירָיו\hdot
וְכָל שֶׁאֵין לוֹ מַתִּירִין\hdot
כֵּיוַן שֶׁקִּדֵּשׁ בַּכֶּלִי\hdot
חַיָּבִין עָלָיו מִשּׁוּם נוֹתָר\hdot
וּמִשּׁוּם טָמֵא\hdot
וּפִגּוּל אֵין בּוֹ׃}

\chead{יום כב}

\perek{וָלֶד}{חַטָּאת וּתְמוּרַת חַטָּאת וְחַטָּאת שֶׁמֵּתוּ בְעָלֶיה יָמוּתוּ\hdot
שֶׁעָֽבְרָה שְׁנָתָהּ\hdot
וְשֶׁנֶּאֱבָדָה\hdot
וְנִמְצְאָה בַּעֲלַת מוּם\hdot
אִם מִשֶּׁכִּפְּרוּ הַבְּעָלִים תָּמוּת\hdot
וְאֵינָהּ עוֹשָׂה תְמוּרָה\hdot
וְלֹא נֶהֱנִין וְלֹא  מוֹעֲלִין\hdot
וְאִם עַד שֶׁלֹּא כִפְּרוּ הַבְּעָלִים\hdot
תִּרְעֶה עַד שֶׁתִּסְתָּאֵב\hdot
וְתִמָּכֵר וְיָבִיא בְדָמֶיהָ אַחֶרֶת\hdot
וְעוֹשָׂה תְמוּרָה\hdot
וּמוֹעֲלִין בָּהּ׃}

\halacha{הַמַּפְרִישׁ מָעוֹת לִנְזִירוּתוֹ\hdot
