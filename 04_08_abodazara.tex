\tractatehead{מסכת עבודה זרה}{abodazara}

\perek{לִפְנֵי}{אֵידֵיהֶן שֶׁל עוֹבְדֵי אֱלִילִים\hdot
שְׁלֹשָׁה יָמִים אָסוּר לָשֵׂאת וְלָתֵת עִמָּהֶם\hdot
לְהַשְׁאִילָן וְלִשׁאוֹל מֵהֶן\hdot
לְהַלְווֹתָן וּלְלַוּוֹת מֵהֶן\hdot
לְפוֹרְעָן וּלְהִפָּרַע מֵהֶן\hdot
רִבִּי יְהוּדָה אוֹמֵר נִפְרָעִין מֵהֶן מִפְּנֵי שֶׁהוּא מֵיצֵר לוֹ\hdot
אָֽמְרוּ לוֹ אַף עַל פִּי שֶׁמֵּיצֵר הוּא עַכְשָׁיו\hdot
שָׂמֵחַ הוּא לְאַחַר זְמָן׃}

\halacha{רִבִּי יִשְׁמָעֵאל אוֹמֵר שְׁלֹשָׁה יָמִים לִפְנֵיהֶם\hdot
וּשְׁלֹשָׁה יָמִים לְאַחֲרֵיהֶם\hdot
אָסוּר\hdot
וַחֲכָמִים אוֹמְרִים לִפְנֵי אֵידֵיהֶן אָסוּר\hdot
לְאַחַר אֵידֵיהֶן מֻתָּר׃}

\halacha{וְאֵלּוּ אֵידֵיהֶן שֶׁל עֲבוֹדַת אֱלִילִים\hdot
קָלֶנְדָא\hdot
וְסַטַּרְנוֹרָא\hdot
וְקַרְטִיסִים\hdot
וְיוֹם גְּנֻסְיָא שֶׁל מְלָכִים\hdot
וְיוֹם הַלֵּידָה\hdot
וְיוֹם הַמִּיתָה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים כָּל מִיתָה שֶׁיֵּשׁ בָּהּ שְׂרֵיפָה יֵשׁ בָּהּ עֲבוֹדַת אֱלִילִים\hdot
שֶׁאֵין בָּהּ שְׂרֵיפָה אֵין בָּהּ עֲבוֹדַת אֱלִילִים\hdot
יוֹם תִּגְלַחַת זְקָנוֹ וּבְלוֹרִיתוֹ\hdot
יוֹם שֶׁעָלָה בוֹ מִן הַיָּם\hdot
וְיוֹם שֶׁיָּצָא בוֹ מִבֵּית הָאֲסוּרִין\hdot
(וְגוֹי שֶׁעָשָׂה מִשְׁתֶּה לִבְנוֹ\hdot)
אֵינוֹ אָסוּר אֶלָּא אוֹתוֹ הַיּוֹם וְאוֹתוֹ הָאִישׁ בִּלְבָד׃}

\halacha{עִיר שֶׁיֵּשׁ בָּהּ עֲבוֹדַת אֱלִילִים\hdot
חוּצָה לָהּ מֻתָּר\hdot
הָיָה חוּצָה לָהּ עֲבוֹדַת אֱלִילִים תּוֹכָהּ מֻתָּר\hdot
מָהוּ לִילֵךְ לְשָׁם\hdot
בִּזְמָן שֶׁהַדֶּרֶךְ מְיוּחֶדֶת לְאוֹתוֹ מָקוֹם אָסוּר\hdot
וְאִם הָיָה יָכוֹל לְהַלֵּךְ בָּהּ לְמָקוֹם אַחֵר מֻתָּר\hdot
עִיר שֶׁיֵּשׁ בָּהּ עֲבוֹדַת אֱלִילִים וְהָיוּ בָּהּ חֲנוּיוֹת מְעֻטָּרוֹת וְשֶׁאֵינָן מְעֻטָּרוֹת\hdot
זֶה הָיָה מַעֲשֶׂה בְּבֵית שְׁאָן\hdot
וְאָֽמְרוּ חֲכָמִים\hdot
הַמְּעֻטָּרוֹת אֲסוּרוֹת\hdot
וְשֶׁאֵינָן מְעֻטָּרוֹת מֻתָּרוֹת׃}

\halacha{אֵלּוּ דְבָרִים אֲסוּרִים לִמְכּוֹר לְעובְדֵי אֱלִילִים\hdot
אִיצְטְרוֹבְלִין\hdot
וּבְנוֹת שׁוּחַ\hdot
וּפְטוֹטְרוֹתֵיהֶם\hdot
וּלְבוֹנָה\hdot
וְתַרְנְגוֹל הַלָּבָן\hdot
רִבִּי יְהוּדָה אוֹמֵר מֻתָּר לִמְכּוֹר לוֹ תַרְנְגוֹל לָבָן בֵּין הַתַּרְנְגוֹלִים\hdot
וּבִזְמָן שֶׁהוּא בִפְנֵי עַצְמוֹ\hdot
קוֹטֵעַ אֶת אֶצְבָּעוֹ וּמוֹכְרוֹ לוֹ\hdot
לְפִי שֶׁאֵין מַקְרִיבִין חָסֵר לַעֲבוֹדַת אֱלִילִים\hdot
וּשְׁאַר כָּל הַדְּבָרִים\hdot
סְתָמָן מֻתָּר וּפֵירוּשָׁן אָסוּר\hdot
רִבִּי מֵאִיר אוֹמֵר אַף דְּקַל טַב\hdot
וְחָצָב\hdot
וְנַקְלִיבָם\hdot
אָסוּר לִמְכּוֹר לְעוֹבְדֵי אֱלִילִים׃}

\halacha{מָקוֹם שֶׁנָּהֲגוּ לִמְכּוֹר בְּהֵמָה דַקָּה לְעוֹבְדֵי אֱלִילִים מוֹכְרִין\hdot
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לִמְכּוֹר אֵין מוֹכְרִין\hdot
וּבְכָל מָקוֹם אֵין מוֹכְרִין לָהֶם בְּהֵמָה גַסָּה\hdot
עֲגָלִים\hdot
וְסַיָּחִים\hdot
שְׁלֵמִים וּשְׁבוּרִין\hdot
רִבִּי יְהוּדָה מַתִּיר בַּשְּׁבוּרָה\hdot
וּבֶן בְּתֵירָא מַתִּיר בַּסּוּס׃}

\halacha{אֵין מוֹכְרִין לָהֶם דּוֹבִין וַאֲרָיוֹת וְכָל דָּבָר שֶׁיֵּשׁ בּוֹ נֶזֶק לָרַבִּים\hdot
אֵין בּוֹנִין עִמָּהֶם בָּסִילְקִי\hdot
וְגַרְדוֹם\hdot
וְאִיצְטַדְיָא\hdot
וּבִימָה\hdot
אֲבָל בּוֹנִים עִמָּהֶם בִּימוֹסִיאוֹת וּמֶרְחֲצָאוֹת\hdot
הִגִּיעוּ לַכִּיפָה שֶׁמַּעֲמִידִין בָּהּ עֲבוֹדַת אֱלִילִים\hdot
אָסוּר לִבְנוֹתָהּ׃}

\halacha{(וְאֵין עוֹשִׂין תַּכְשִׁיטִים לַעֲבוֹדַת אֱלִילִים קוֹטְלָאוֹת וּנְזָמִים וְטַבָּעוֹת\hdot
רִבִּי אֱלִיעֶזֶר אוֹמֵר בִּשְׂכָר מוּתָר\hdot)
אֵין מוֹכְרִין לָהֶם בִּמְחֻבָּר לַקַּרְקַע\hdot
אֲבָל מוֹכֵר הוּא מִשֶּׁיְּקַצֵּץ\hdot
רִבִּי יְהוּדָה אוֹמֵר מוֹכֵר הוּא לוֹ עַל מְנַת לָקוֹץ\hdot
אֵין מַשְׂכִּירִין לָהֶם בָּתִּים בְּאֶרֶץ יִשְׂרָאֵל\hdot
וְאֵין צָרִיךְ לוֹמַר שָׂדוֹת\hdot
וּבְסוּרְיָא מַשְׂכִּירִין לָהֶם בָּתִּים אֲבָל לֹא שָׂדוֹת\hdot
וּבְחוּצָה לָאָרֶץ מוֹכְרִין לָהֶם בָּתִּים וּמַשְׂכִּירִין שָׂדוֹת\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יוֹסֵי אוֹמֵר בְּאֶרֶץ יִשְׂרָאֵל מַשְׂכִּירִין לָהֶם בָּתִּים אֲבָל לֹא שָׂדוֹת\hdot
וּבְסוּרְיָא מוֹכְרִין בָּתִּים וּמַשְׂכִּירִין שָׂדוֹת\hdot
וּבְחוּצָה לָאָרֶץ מוֹכְרִין אֵלּוּ וְאֵלּוּ׃}

\halacha{אַף בִּמְקוֹם שֶׁאָֽמְרוּ לְהַשְׂכִּיר\hdot
לֹא לְבֵית דִּירָה אָֽמְרוּ\hdot
מִפְּנֵי שֶׁהוּא מַכְנִיס לְתוֹכוֹ עֲבוֹדַת אֱלִילִים\hdot
שֶׁנֶּאֱמַר וְלֹא תָבִיא תוֹעֵבָה אֶל בֵּיתֶךָ\hdot
וּבְכָל מָקוֹם לֹא יַשְׂכִּיר לוֹ אֶת הַמֶּרְחָץ\hdot
מִפְּנֵי שֶׁהוּא נִקְרָא עַל שְׁמוֹ׃}

\perek{אֵין}{מַעֲמִידִין בְּהֵמָה בְּפֻנְדְקָאוּת שֶׁל עוֹבְדֵי אֱלִילִים\hdot
מִפְּנֵי שֶׁחֲשׁוּדִים עַל הָֽרְבִיעָה\hdot
וְלֹא תִתְיַחֵד אִשָּׁה עִמָּהֶן\hdot
מִפְּנֵי שֶׁחֲשׁוּדִין עַל הָעֲרָיוֹת\hdot
ולְאֹ יִתְיַחֵד אָדָם עִמָּהֶן\hdot
מִפְּנֵי שֶׁחֲשׁוּדִין עַל שְׁפִיכוּת דָּמִים\hdot
בַּת יִשְׂרָאֵל לֹא תְיַלֵּד אֶת הָעוֹבֶדֶת אֱלִילִים\hdot
מִפְּנֵי שֶׁמְּיַלֶּדֶת בֵּן לַעֲבוֹדַת אֱלִילִים\hdot
אֲבָל נָכְרִית מְיַלֶּדֶת בַּת יִשְׂרָאֵל\hdot
בַּת יִשְׂרָאֵל לֹא תָנִיק בְּנָהּ שֶׁל עוֹבֶדֶת אֱלִילִים\hdot
אֲבָל עוֹבֶדֶת אֱלִילִים מְנִיקָה בְּנָהּ שֶׁל בַּת יִשְׂרָאֵל בִּרְשׁוּתָהּ\hdot
מִתְרַפְּאִין מֵהֶן רִפּוּי מָמוֹן\hdot
אֲבָל לֹא רִפּוּי נְפָשׁוֹת\hdot
וְאֵין מִסְתַּפְּרִין מֵהֶן בְּכָל מָקוֹם\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים בִּרְשׁוּת הָרַבִּים מוּתָר\hdot
אֲבָל לֹא בֵּינוֹ לְבֵינוֹ׃}

\halacha{אֵלּוּ דְבָרִים שֶׁל עוֹבְדֵי אֱלִילִים אֲסוּרִין וְאִסּוּרָן אִסּוּר הֲנָאָה\hdot
הַיַּיִן וְהַחֹמֶץ שֶׁל עוֹבְדֵי אֱלִילִים שֶׁהָיָה מִתְּחִלָּתוֹ יַיִן\hdot
וְחֵרֶס הַדְרִיָּנִי\hdot
וְעוֹרוֹת לְבוּבִין\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר בִּזְמָן שֶׁהַקֶּרַע שֶׁלּוֹ עָגוֹל אָסוּר\hdot
מָשׁוּךְ מוּתָר\hdot
בָּשָׂר הַנִּכְנָס לַעֲבוֹדַת אֱלִילִים מוּתָר\hdot
וְהַיּוֹצֵא אָסוּר\hdot
מִפְּנֵי שֶׁהוּא זִבְחֵי מֵתִים\hdot
דִּבְרֵי רִבִּי עֲקִיבָא\hdot
הַהוֹלְכִין לַתַּרְפּוּת אָסוּר לָשֵׂאת וְלָתֵת עִמָּהֶן וְהַבָּאִין מוּתָרִין׃}

\halacha{נוֹדוֹת הָעוֹבְדֵי אֱלִילִים\hdot
וְקַנְקַנֵּיהֶן וְיַיִן שֶׁל יֵשְׂרָאֵל כָּנוּס בָּהֶן\hdot
אֲסוּרִין\hdot
וְאִסּוּרָן אִסּוּר הֲנָאָה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים אֵין אִסּוּרָן אִסּוּר הֲנָאָה\hdot
הַחַרְצַנִּים וְהַזַּגִּין שֶׁל עוֹבְדֵי אֱלִילִים\hdot
אֲסוּרִין וְאִסּוּרָן אִסּוּר הֲנָאָה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים לַחִין אֲסוּרִין יְבֵשִׁין מוּתָרִין\hdot
הַמֻּרְיָס וּגְבִינַת בֵּית אֲנַיְקִי שֶׁל עוֹבְדֵי אֱלִילִים אֲסוּרִין\hdot
(וְאִסּוּרָן) אִסּוּר הֲנָאָה\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
וַחֲכָמִים אוֹמְרִים אֵין אִסּוּרָן אִסּוּר הֲנָאָה׃}

\halacha{אָמַר רִבִּי יְהוּדָה שָׁאַל רִבִּי יִשְׁמָעֵאל אֶת רִבִּי יְהוֹשֻׁעַ כְּשֶׁהָיוּ מְהַלְּכִין בַּדֶּרֶךְ\hdot
אָמַר לוֹ מִפְּנֵי מָה אָֽסְרוּ גְבִינוֹת עוֹבְדֵי אֱלִילִים\hdot
אָמַר לוֹ מִפְּנֵי שֶׁמַּעֲמִידִין אוֹתָהּ בְּקֵיבָה שֶׁל נְבֵילָה\hdot
אָמַר לוֹ וַהֲלֹא קֵיבַת עוֹלָה חֲמוּרָה מֵקִּיבַת נְבֵלָה\hdot
וְאָֽמְרוּ כֹּהֵן שֶׁדַּעְתּוֹ יָפָה שׂוֹרְפָה חַיָּה וְלֹא הוֹדוּ לוֹ\hdot
אֲבָל אָֽמְרוּ לֹא נֶהֱנִין וְלֹא מוֹעֲלִין\hdot
