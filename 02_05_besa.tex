\tractatehead{מסכת יום טוב}{besa}

\perek{בֵּיצָה}{שֶׁנּוֹלְדָה בְיוֹם טוֹב\hdot
בֵּית שַׁמָּאי אוֹמְרִים תֵּאָכֵל\hdot
וּבֵית הִלֵּל אוֹמְרִים לֹא תֵאָכֵל\hdot
בֵּית שַׁמָּאי אוֹמְרִים שְׂאוֹר בְּכַזַּיִת\hdot
וְחָמֵץ בְּכַכּוֹתֶבֶת\hdot
וּבֵית הִלֵּל אוֹמְרִים זֶה וְזֶה בְּכַזַּיִת׃}

\halacha{הַשּׁוֹחֵט חַיָּה וְעוֹף בְּיוֹם טוֹב\hdot
בֵּית שַׁמָּאי אוֹמְרִים יַחְפּוֹר בַּדָּקָר וִיכַסֶּה\hdot
וּבֵית הִלֵּל אוֹמְרִים לֹא יִשְׁחוֹט אֶלָּא אִם כֵּן הָיָה לוֹ עָפָר מוּכָן מִבְּעוֹד יוֹם\hdot
וּמוֹדִים שֶׁאִם שָׁחַט שֶׁיַּחְפּוֹר בַּדָּקָר וִיכַסֶּה\hdot
שֶׁאֵפֶר כִּירָה מוּכָן הוּא׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין מוֹלִיכִין אֶת הַסֻּלָּם מִשּׁוֹבָךְ לַשּׁוֹבָךְ\hdot
אֲבָל מַטֵּהוּ מֵחַלּוֹן לַחַלּוֹן\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
בֵּית שַׁמָּאי אוֹמְרִים לֹא יִטּוֹל אֶלָּא אִם כֵּן נִעְנֵעַ מִבְּעוֹד יוֹם\hdot
וּבֵית הִלֵּל אוֹמְרִים עוֹמֵד וְאוֹמֵר\hdot
זֶה וְזֶה אֲנִי נוֹטֵל׃}

\halacha{זִמֵּן שְׁחוֹרִים וּמָצָא לְבָנִים\hdot
לְבָנִים וּמָצָא שְׁחוֹרִים\hdot
שְׁנַיִם וּמָצָא שְׁלֹשָׁה\hdot
אֲסוּרִין\hdot
שְׁלֹשָׁה וּמָצָא שְׁנַיִם\hdot
מוּתָרִים\hdot
בְּתוֹךְ הַקֵּן וּמָצָא לִפְנֵי הַקֵּן\hdot
אֲסוּרִים\hdot
וְאִם אֵין שָׁם אֶלָּא הֵם\hdot
הֲרֵי אֵלּוּ מוּתָרִים׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין מְסַלְּקִין אֶת הַתְּרִיסִין בְּיוֹם טוֹב\hdot
וּבֵית הִלֵּל מַתִּירִין אַף לְהַחֲזִיר\hdot
בֵּית שַׁמָּאי אוֹמְרִים אֵין נוֹטְלִין אֶת הָעֶלִי לְקַצֵּב עָלָיו בָּשָׂר וּבֵית הִלֵּל מַתִּירִין\hdot
בֵּית שַׁמָּאי אוֹמְרִים אֵין נוֹתְנִין אֶת הָעוֹר לִפְנֵי הַדָּרְסָן\hdot
וְלֹא יַגְבִּיהֶנּוּ אֶלָּא אִם כֵּן יֵשׁ עִמּוֹ כַּזַּיִת בָּשָׂר\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
בֵּית שַׁמָּאי אוֹמְרִים אֵין מוֹצִיאִין לֹא אֶת הַקָּטָן\hdot
וְלֹא אֶת הַלּוּלָב\hdot
וְלֹא אֶת סֵפֶר תּוֹרָה לִרְשׁוּת הָרַבִּים\hdot
וּבֵית הִלֵּל מַתִּירִין׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין מוֹלִיכִין חַלָּה וּמַתָּנוֹת לַכֹּהֵן בְּיוֹם טוֹב\hdot
בֵּין שֶׁהוּרָמוּ מֵאֶמֶשׁ\hdot
בֵּין שֶׁהוּרָמוּ מֵהַיּוֹם\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
אָֽמְרוּ לָהֶם בֵּית שַׁמָּאי גְּזֵרָה שָׁוָה\hdot
חַלָּה וּמַתָּנוֹת\hdot
מַתָּנָה לַכֹּהֵן\hdot
וּתְרוּמָה מַתָּנָה לַכֹּהֵן\hdot
כְּשֵׁם שֶׁאֵין מוֹלִיכִין אֶת הַתְּרוּמָה\hdot
כַּךְ אֵין מוֹלִיכִין אֶת הַמַּתָּנוֹת\hdot
אָֽמְרוּ לָהֶם בֵּית הִלֵּל\hdot
לֹא\hdot
אִם אֲמַרְתֶּם בַּתְּרוּמָה שֶׁאֵינוֹ זַכָּאי בַּהֲרָמָתָהּ\hdot
תֹּאמְרוּ בַמַּתָּנוֹת שֶׁזַּכָּאי בַּהֲרָמָתָן׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים\hdot
תְּבָלִין נִדּוֹכִין בְּמָדוֹךְ שֶׁל עֵץ\hdot
וְהַמֶּלַח בַּפַּךְ וּבְעֵץ הַפָּרוּר\hdot
וּבֵית הִלֵּל אוֹמְרִים תְּבָלִין נִדּוֹכִין כְּדַרְכָּן בְּמָדוֹךְ שֶׁל אֶבֶן וְהַמֶּלַח בְּמָדוֹךְ שֶׁל עֵץ׃}

\halacha{הַבּוֹרֵר קָטְנִית בְּיוֹם טוֹב\hdot
בֵּית שַׁמָּאי אוֹמְרִים בּוֹרֵר אֹכֶל וְאֹכֵל\hdot
וּבֵית הִלֵּל אוֹמְרִים בּוֹרֵר כְּדַרְכּוֹ בְחֵיקוֹ\hdot
בְּקָנוֹן וּבְתַמְחוּי\hdot
אֲבָל לֹא בְטַבְלָא\hdot
וְלֹא בְנָפָה\hdot
וְלֹא בִכְבָרָה\hdot
רַבָּן גַּמְלִיאֵל אוֹמֵר אַף מַדִּיחַ וְשׁוֹלֶה׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים אֵין מְשַׁלְּחִין בְּיוֹם טוֹב אֶלָּאּ מָנוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים מְשַׁלְּחִין בְּהֵמָה חַיָּה וְעוֹף בֵּין חַיִּין בֵּין שְׁחוּטִין\hdot
מְשַׁלְּחִין יֵינוֹת שְׁמָנִים וּסְלָתוֹת וְקָטְנִיּוֹת\hdot
אֲבָל לֹא תְבוּאָה\hdot
וְרִבִּי שִׁמְעוֹן מַתִּיר בַּתְּבוּאָה׃}

\halacha{מְשַׁלְּחִין כֵּלִים בֵּין תְּפוּרִין בֵּין שֶׁאֵינָן תְּפוּרִין\hdot
וְאַף עַל פִּי שֶׁיֵּשׁ בָּהֶן כִּלְאַיִם וְהֵן לְצוֹרֶךְ הַמּוֹעֵד\hdot
אֲבָל לֹא סַנְדָּל הַמְּסֻמָּר\hdot
וְלֹא מַנְעָל שֶׁאֵינוֹ תָפוּר\hdot
רִבִּי יְהוּדָה אוֹמֵר אַף לֹא מַנְעָל לָבָן\hdot
מִפְּנֵי שֶׁצָּרִיךְ אוּמָן\hdot
זֶה הַכְּלָל\hdot
כָּל שֶׁנֵּאוֹתִין בּוֹ בְּיוֹם טוֹב מְשַׁלְּחִין אוֹתוֹ׃}

\perek{יוֹם}{טוֹב שֶׁחַל לִהְיוֹת עֶרֶב שַׁבָּת\hdot
לֹא יְבַשֵּׁל אָדָם בַּתְּחִלָּה מִיּוֹם טוֹב לְשַׁבָּת\hdot
אֲבָל מְבַשֵּׁל הוּא לְיוֹם טוֹב\hdot
וְאִם הוֹתִיר הוֹתִיר לְשַׁבָּת\hdot
וְעוֹשֶׂה תַבְשִׁיל מֵעֶרֶב יוֹם טוֹב\hdot
וְסוֹמֵךְ עָלָיו לְשַׁבָּת\hdot
בֵּית שַׁמָּאי אוֹמְרִים שְׁנֵי תַבְשִׁילִין\hdot
וּבֵית הִלֵּל אוֹמְרִים תַּבְשִׁיל אֶחָד\hdot
וְשָׁוִין בְּדָג וּבֵיצָה שֶׁעָלָיו\hdot
שֶׁהֵן שְׁנֵי תַבְשִׁילִין\hdot
אֲכָלוֹ\hdot
אוֹ שֶׁנֶּאֱבַד\hdot
לֹא יְבַשֵּׁל עָלָיו בַּתְּחִלָּה\hdot
וְאִם שִׁיֵּיר מִמֶּנּוּ כָּל שֶׁהוּא\hdot
סוֹמֵךְ עָלָיו לְשַׁבָּת׃}

\halacha{חַל לִהְיוֹת אַחַר שַׁבָּת\hdot
בֵּית שַׁמָּאי אוֹמְרִים מַטְבִּילִין אֶת הַכֹּל מִלִּפְנֵי שַׁבָּת\hdot
וּבֵית הִלֵּל אוֹמְרִים כֵּלִים מִלִּפְנֵי הַשַּׁבָּת וְאָדָם בְּשַׁבָּת׃}

\halacha{וְשָׁוִין שֶׁמַּשִּׁיקִין אֶת הַמַּיִם בִּכְלֵי אֶבֶן לְטַהֲרָן\hdot
אֲבָל לֹא מַטְבִּילִין\hdot
וּמַטְבִּילִין מִגַּב לַגַּב\hdot
וּמֵחֲבוּרָה לַחֲבוּרָה׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים מְבִיאִין שְׁלָמִים וְאֵין סוֹמְכִין עֲלֵיהֶם\hdot
אֲבָל לֹא עוֹלוֹת\hdot
וּבֵית הִלֵּל אוֹמְרִים מְבִיאִין שְׁלָמִים וְעוֹלוֹת וְסוֹמְכִין עֲלֵיהֶן׃}

\halacha{בֵּית שַׁמָּאי אוֹמְרִים לֹא יֵחַם אָדָם חַמִּין לְרַגְלָיו אֶלָּא אִם כֵּן רְאוּיִין לִשְׁתִּיָּה\hdot
וּבֵית הִלֵּל מַתִּירִין\hdot
עוֹשׂהֶ אָדָם מְדוּרָה וּמִתְחַמֵּם כְּנֶגְדָּהּ׃}

\halacha{שְׁלֹשָׁה דְבָרִים רַבָּן גַּמְלִיאֶל מַחְמִיר כְּדִבְרֵי בֵית שַׁמָּאי\hdot
אֵין טוֹמְנִין אֶת הַחַמִּין מִיּוֹם טוֹב לְשַׁבָּת\hdot
וְאֵין זוֹקְפִין אֶת הַמְּנוֹרָה בְּיוֹם טוֹב\hdot
וְאֵין אוֹפִין פִּתִּין גְּרִיצִין אֶלָּא רְקִיקִין\hdot
אָמַר רַבָּן גַּמְלִיאֵל מִימֵיהֶן שֶׁל בֵּית אַבָּא לֹא הָיוּ אוֹפִין פִּתִּין גְּרִיצִין\hdot
אֶלָּא רְקִיקִין\hdot
אָֽמְרוּ לוֹ מַה נַּעֲשֶׂה לְבֵית אָבִיךָ שֶׁהָיוּ מַחְמִירִין עַל עַצְמָן וּמְקִילִין לְכָל יִשְׂרָאֵל לִהְיוֹת אוֹפִין פִּתִּין גְּרִיצִין וְחוֹרִי׃}

\halacha{אַף הוּא אָמַר שְׁלֹשׁה דְבָרִים לְהָקֵל\hdot
מְכַבְּדִים בֵּין הַמִּטּוֹת\hdot
וּמַנִּיחִין אֶת הַמֻּגְמָר בְּיוֹם טוֹב\hdot
וְעוֹשִׂין גְּדִי מְקֻלָּס בְּלֵילֵי פְסָחִים\hdot
וַחֲכָמִים אוֹסְרִין׃}

\halacha{שְׁלֹשָׁה דְבָרִים רִבִּי אֶלְעָזָר בֶּן עֲזַרְיָה מַתִּיר\hdot
וַחֲכָמִים אוֹסְרִין\hdot
פָּרָתוֹ יוֹצְאָה בִרְצוּעָה שֶׁבֵּין קַרְנֶיהָ\hdot
וּמְקָֽרְדִין אֶת הַבְּהֵמָה בְּיוֹם טוֹב\hdot
וְשׁוֹחֲקִין אֶת הַפִּלְפְּלִין בְּרֵחַיִם שֶׁלָּהֶם\hdot
רִבִּי יְהוּדָה אוֹמֵר אֵין מְקָֽרְדִין אֶת הַבְּהֵמָה בְּיוֹם טוֹב מִפְּנֵי שֶׁעוֹשֶׂה חַבּוּרָה\hdot
אֲבָל מְקַרְצְפִין\hdot
וַחֲכָמִים אוֹמְרִים אֵין מְקָֽרְדִין אַף לֹא מְקַרְצְפִין׃}

\halacha{הָרֵחַיִם שֶׁל פִּלְפְּלִין טְמֵאָה מִשּׁוּם שְׁלֹשָׁה כֵלִים\hdot
מִשּׁוּם כְּלִי קִבּוּל\hdot
וּמִשּׁוּם כְּלִי מַתֶּכֶת\hdot
וּמִשּׁוּם כְּלִי כְבָרָה׃}

\halacha{עֲגָלָה שֶׁל קָטָן טְמֵאָה מִדְרָס\hdot
וְנִטֶּלֶת בְּשַׁבָּת\hdot
וְאֵינָהּ נִגְרֶרֶת אֶלָּא עַל גַּבֵּי כֵלִים\hdot
רִבִּי יְהוּדָה אוֹמֵר כָּל הַכֵּלִים אֵין נִגְרָרִים חוּץ מִן הָעֲגָלָה מִפְּנֵי שֶׁהִיא כוֹבָשֶׁת׃}

\perek{אֵין}{צָדִין דָּגִים מִן הַבִּיבָרִים בְּיוֹם טוֹב\hdot
וְאֵין נוֹתְנִין לִפְנֵיהֶם מְזוֹנוֹת\hdot
אֲבָל צָדִין חַיָּה וְעוֹף מִן הַבִּיבָרִין וְנוֹתְנִים לִפְנֵיהֶם מְזוֹנוֹת\hdot
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר לֹא כָל הַבִּיבָרִין שָׁוִין\hdot
זֶה הַכְּלָל כָּל הַמְּחֻסָּר צֵידָה אָסוּר\hdot
וְשֶׁאֵינוֹ מְחֻסָּר צֵידָה מוּתָר׃}

\halacha{מְצוּדוֹת חַיָּה וְעוֹף וְדָגִים שֶׁעֲשָׂאָן מֵעֶרֶב יוֹם טוֹב\hdot
לֹא יִטּוֹל מֵהֶן בְּיוֹם טוֹב אֶלָּא אִם כֵּן יוֹדֵעַ שֶׁנִּצָּדוּ מֵעֶרֶב יוֹם טוֹב\hdot
וּמַעֲשֶׂה בְּגוֹי אֶחָד שֶׁהֵבִיא דָגִים לְרַבָּן גַּמְלִיאֵל וְאָמַר מוּתָרִין הֵן\hdot
אֶלָּא שֶׁאֵין רְצוֹנִי לְקַבֵּל הֵימֶנּוּ׃}

\halacha{בְּהֵמָה מְסֻכֶּנֶת\hdot
לֹא יִשְׁחוֹט\hdot
אֶלָּא אִם כֵּן יֵשׁ שָׁהוּת בַּיּוֹם לֶאֱכוֹל מִמֶּנּוּ כַּזַּיִת צָלִי\hdot
רִבִּי עֲקִיבָא אוֹמֵר אֲפִילוּ כַּזַּיִת חַי מִבֵּית טְבִיחָתָהּ\hdot
שְׁחָטָהּ בַּשָּׂדֶה\hdot
לֹא יְבִיאֶנָּה בַמּוֹט וּבַמּוֹטָָה\hdot
אֲבָל מֵבִיא בְיָדוֹ אֵבָרִים אֵבָרִים׃}

\halacha{בְּכוֹר שֶׁנָּפַל לַבּוֹר\hdot
רִבִּי יְהוּדָה אוּמֵר\hdot
יֵרֵד מֻמְחֶה וְיִרְאֶה\hdot
אִם יֵשׁ בּוֹ מוּם\hdot
יַעֲלֶה וְיִשְׁחוֹט\hdot
וְאִם לָאוּ\hdot
לֹא יִשְׁחוֹט\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר\hdot
כָּל שֶׁאֵין מוּמוֹ נִכָּר מִבְּעוֹד יוֹם\hdot
אֵין זֶה מִן הַמּוּכָן׃}

\halacha{בְּהֵמָה שֶׁמֵּתָה לֹא יְזִיזֶנָּה מִמְּקוֹמָהּ\hdot
וּמַעֲשֶׂה וְשָׁאֲלוּ אֶת רִבִּי טַרְפוֹן עָלֶיהָ וְעַל הַחַלָּה שֶׁנִּטְמְאָה\hdot
וְנִכְנַס לְבֵית הַמֶּדְרָשׁ וְשָׁאַל\hdot
וְאָֽמְרוּ לוֹ לֹא יְזִיזֵם מִמְּקוֹמָם׃}

\halacha{אֵין נִמְנִין עַל הַבְּהֵמָה לְכַתְּחִלָּה בְּיוֹם טוֹב\hdot
אֲבָל נִמְנִין עָלֶיהָ מֵעֶרֶב יוֹם טוֹב\hdot
וְשׁוֹחֲטִין וּמְחַלְּקִין בֵּינֵיהֶן\hdot
רִבִּי יְהוּדָה אוֹמֵר\hdot
שׁוֹקֵל אָדָם בָּשָׂר כְּנֶגֶד הַכֶּלִי\hdot
אוֹ כְּנֶגֶד הַקּוּפִיץ\hdot
וַחֲכָמִים אוֹמְרִים אֵין מַשְׁגִּיחִין בְּכַף מֹאזְנַיִם כָּל עִקָּר׃}

\halacha{אֵין מַשְׁחִיזִין אֶת הַסַּכִּין בְּיוֹם טוֹב\hdot
אֲבָל מַשִּׂיאָהּ עַל גַּבֵּי חֲבֵרְתָּהּ\hdot
לֹא יֹאמַר אָדָם לַטַּבָּח שְׁקוֹל לִי בְּדִינָר בָּשָׂר\hdot
אֲבָל שׁוֹחֵט וּמְחַלְּקִין בֵּינֵיהֶן׃}

\halacha{אוֹמֵר אָדָם לַחֲבֵירוֹ מַלֵּא לִי כְּלִי זֶה\hdot
אֲבָל לֹא בְמִדָּה\hdot
רִבִּי יְהוּדָה אוֹמֵר אִם הָיָה כְּלִי שֶׁל מִדָּה לֹא יְמַלְּאֶנּוּ\hdot
מַעֲשֶׂה בְּאַבָּא שָׁאוּל בֶּן בָּטְנִית שֶׁהָיָה מְמַלֵּא מִדּוֹתָיו מֵעֶרֶב יוֹם טוֹב\hdot
וְנוֹתְנָן לַלְּקוּחוֹת בְּיוֹם טוֹב\hdot
אַבָּא שָׁאוּל אוֹמֵר אַף בְּמוֹעֵד עוֹשֶׂה כֵן\hdot
(מִפְּנֵי בְרוּרֵי הַמִּדּוֹת\hdot)
וַחֲכָמִים אוֹמְרִים אַף בַּחוֹל עוֹשֶׂה כֵן\hdot
מִפְּנֵי מִצּוּי הַמִּדּוֹת\hdot
הוֹלֵךְ אָדָם אֵצֶל חַנְוָנִי הָרָגִיל אֶצְלוֹ וְאוֹמֵר לוֹ\hdot
תֵּן לִי בֵיצִים וֶאֱגוֹזִים בְּמִנְיַן\hdot
שֶׁכֵּן דֶּרֶךְ בַּעַל הַבַּיִת\hdot
לִהְיוֹת מוֹנֶה בּתוֹךְ בֵּיתוֹ׃}

\perek{הַמֵּבִיא}{כַדֵּי יַיִן מִמָּקוֹם לְמָקוֹם\hdot
לֹא יְבִיאֵם בְּסַל וּבְקֻפָּה\hdot
אֲבָל מֵבִיא הוּא עַל כְּתֵפוֹ\hdot
אוֹ לְפָנָיו\hdot
וְכֵן הַמּוֹלִיךְ אֶת הַתֶּבֶן\hdot
לֹא יַפְשִׁיל אֶת הַקֻּפָּה לַאֲחוֹרָיו\hdot
אֲבָל מְבִיאָהּ הוּא בְיָדוֹ\hdot
וּמַתְחִילִין בַּעֲרֵמַת הַתֶּבֶן\hdot
אֲבָל לֹא בָּעֵצִים שֶׁבַּמֻּקְצָה׃}

\halacha{אֵין נוֹטְלִין עֵצִים מִן הַסֻּכָּה\hdot
אֶלָּא מִן הַסָּמוּךְ לָהּ\hdot
מְבִיאִין עֵצִים מִן הַשָּׂדֶה מִן הַמְּכֻנָּס\hdot
וּמִן הַקַּרְפָּף אֲפִילוּ מִן הַמְּפֻזָּר\hdot
אֵיזֶהוּ קַרְפָּף כָּל שֶׁסָּמוּךְ לָעִיר\hdot
דִּבְרֵי רִבִּי יְהוּדָה\hdot
רִבִּי יוֹסֵי אוֹמֵר כָּל שֶׁנִּכְנָסִין לוֹ בּפוֹתַחַת\hdot
וַאֲפִילוּ בְתוֹךְ תְּחוּם שַׁבָּת׃}

\halacha{אֵין מְבַקְּעִין עֵצִים לֹא מִן הַקּוֹרוֹת\hdot
וְלֹא מִן הַקּוֹרָה שֶׁנִשְׁבְּרָה בְיוֹם טוֹב\hdot
וְאֵין מְבַקְּעִין לֹא בַקַּרְדּוֹם וְלֹא בַמְּגֵרָה וְלֹא בַמַּגָּל אֶלָּא בַקּוּפִיץ\hdot
בַּיִת שֶׁהוּא מָלֵא פֵירוֹת\hdot
סָתוּם וְנִפְחַת\hdot
נוֹטֵל מִמָּקוֹם הַפַּחַת\hdot
רִבִּי מֵאִיר אוֹמֵר אַף פּוֹחֵת לְכַתְּחִלָּה וְנוֹטֵל׃}
