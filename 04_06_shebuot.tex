\tractatehead{מסכת שבועות}{shebuot}

\perek{שְׁבוּעוֹת}{שְׁתַּיִם שֶׁהֵן אַרְבַּע\hdot
יְדִיעוֹת הַטֻּמְאָה שְׁתַּיִם שֶׁהֵן אַרְבַּע\hdot
יְצִיאוֹת הַשַּׁבָּת שְׁתַּיִם שֶׁהֵן אַרְבַּע\hdot
מַרְאוֹת נְגָעִים שְׁנַיִם שֶׁהֵן אַרְבָּעָה׃}

\halacha{כָּל שֶׁיֵּשׁ בָּהּ יְדִיעָה בַתְּחִלָּה\hdot
וִידִיעָה בַסּוֹף וְהֶעְלֵם בִּנְתַּיִם\hdot
הֲרֵי זֶה בְּעוֹלֶה וְיוֹרֵד\hdot
יֵשׁ בָּהּ יְדִיעָה בַתְּחִלָּה\hdot
וְאֵין בָּהּ יְדִיעָה בַסּוֹף\hdot
שָׂעִיר הַנַּעֲשָׂה בִּפְנִים וְיוֹם הַכִּפּוּרִים תּוֹלֶה\hdot
עַד שֶׁיִּוָּדַע לוֹ וְיָבִיא בְעוֹלֶה וְיוֹרֵד׃}

\halacha{אֵין בָּהּ יְדִיעָה בַתְּחִלָּה\hdot
אֲבָל יֵשׁ בָּהּ יְדִיעָה בַסּוֹף\hdot
שָׂעִיר הַנַּעֲשָׂה בַחוּץ וְיוֹם הַכִּפּוּרִים מְכַפֵּר\hdot
שֶׁנֶּאֱמַר מִלְּבַד חַטַּאת הַכִּפֻּרִים\hdot
עַל מַה שֶּׁזֶּה מְכַפֵּר\hdot
זֶה מְכַפֵּר\hdot
מַה הַפְּנִימִי אֵינוֹ מְכַפֵּר\hdot
אֶלָּא עַל דָּבָר שֶׁיֵּשׁ בָּהּ יְדִיעָה\hdot
אַף הַחִיצוֹן אֵינוֹ מְכַפֵּר\hdot
אֶלָּא עַל דָּבָר שֶׁיֵּשׁ בָּהּ יְדִיעָה׃}

\halacha{וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף\hdot
שְׂעִירֵי הָֽרְגָלִים\hdot
וּשְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִין דִּבְרֵי רִבִּי יְהוּדָה\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר שְׂעִירֵי הָֽרְגָלִים מְכַפְּרִין\hdot
אֲבָל לֹא שְׂעִירֵי רָאשֵׁי חֳדָשִׁים\hdot
וְעַל מָה שְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִין\hdot
עַל הַטָּהוֹר שֶׁאָכַל אֶת הַטָּמֵא\hdot
רִבִּי מֵאִיר אוֹמֵר כָּל הַשְׂעִירִין כַּפָּרָתָן שָׁוָה\hdot
עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו\hdot
הָיָה רִבִּי שִׁמְעוֹן אוֹמֵר שְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִין עַל הַטָּהוֹר שֶׁאָכַל אֶת הַטָּמֵא\hdot
וְשֶׁל רְגָלִים מְכַפְּרִין עַל שֶׁאֵין בָּהּ יְדִיעָה\hdot
לֹא בַתְּחִלָּה וְלֹא בַסּוֹף\hdot
וְשֶׁל יוֹם הַכִּפּוּרִים מְכַפֵּר\hdot
עַל שֶׁאֵין בָּהּ יְדִיעָה בַתְּחִלָּה אֲבָל יֵשׁ בָּהּ יְדִיעָה בַסּוֹף\hdot
אָֽמְרוּ לוֹ מָהוּ שֶׁיִּקָּרְבוּ זֶה בָזֶה\hdot
אָמַר לָהֶן יִקָּרֵבוּ\hdot
אָֽמְרוּ לוֹ הוֹאִיל וְאֵין כַּפָּרָתָן שָׁוָה\hdot
הֵיאַךְ קְרֵבִין זֶה בָזֶה\hdot
אָמַר לָהֶן כֻּלָּן בָּאִין לְכַפֵּר עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו׃}

\halacha{רִבִּי שִׁמְעוֹן בֶּן יְהוּדָה אוֹמֵר מִשְּׁמוֹ שְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִין עַל הַטָּהוֹר שֶׁאָכַל אֶת הַטָּמֵא\hdot
מוֹסִיף עֲלֵיהֶן שֶׁל רְגָלִים\hdot
שֶׁמְכַפְּרִין עַל הַטָּהוֹר שֶׁאָכַל אֶת הַטָּמֵא\hdot
וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף\hdot
מוֹסִיף עֲלֵיהֶן שֶׁל יוֹם הַכִּפּוּרִים\hdot
שֶׁהֵן מְכַפְּרִין עַל הַטָּהוֹר שֶׁאָכַל אֶת הַטָּמֵא\hdot
וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף\hdot
עַל שֶׁאֵין בָּהּ יְדִיעָה בַתְּחִלָּה\hdot
אֲבָל יֵשׁ בָּהּ יְדִיעָה בַסּוֹף\hdot
אָֽמְרוּ לוֹ (אוֹמֵר הָיָה רִבִּי) מָהוּ שֶׁיִּקָּֽרְבוּ זֶה בָזֶה\hdot
אָמַר לָהֶן\hdot
הֵן\hdot
אָֽמְרוּ לוֹ אִם כֵּן יִהְיוּ שֶׁל יוֹם הַכִּפּוּרִים קְרֵבִים בְּרָאשֵׁי חֳדָשִׁים\hdot
אֲבָל הֵיאַךְ שֶׁל רָאשֵׁי חֳדָשִׁים קְרֵבִים בְּיוֹם הַכִּפּוּרִים\hdot
לְכַפֵּר כַּפָּרָה שֶׁאֵינָהּ שֶׁלָּהֵן\hdot
אָמַר לָהֶן כֻּלָּן בָּאִין לְכַפֵּר עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו׃}

\halacha{וְעַל זְדוֹן טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו שָׂעִיר הַנַּעֲשָׂה בִּפְנִים וְיוֹם הַכִּפּוּרִים מְכַפְּרִין\hdot
וְעַל שְׁאַר עֲבֵרוֹת שֶׁבַּתּוֹרָה\hdot
הַקַּלּוֹת וְהַחֲמוּרוֹת\hdot
הַזְּדוֹנוֹת וְהַשְּׁגָגוֹת\hdot
הוֹדַע וְלֹא הוֹדַע\hdot
עֲשֵׂה וְלֹא תַעֲשֶׂה\hdot
כְּרִיתוֹת וּמִיתוֹת בֵּית דִּין\hdot
שָׂעִיר הַמִּשְׁתַּלֵּחַ מְכַפֵּר׃}

\halacha{אֶחָד יִשְׂרָאֵלִים\hdot
וְאֶחָד כֹּהֲנֵים\hdot
וְאֶחָד כֹּהֵן מָשׁוּחַ\hdot
מַה בֵּין יִשְׂרָאֵלִים לַכֹּהֲנִים וּלְכֹהֵן מָשׁוּחַ\hdot
אֶלָּא שֶׁדַּם הַפָּר מְכַפֵּר עַל הַכֹּהֲנִים וְעַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר כְּשֵׁם שֶׁדַּם הַשָּׂעִיר הַנַּעֲשָׁה בִּפְנִים מְכַפֵּר עַל יִשְׂרָאֵל\hdot
כַּךְ דַּם הַפָּר מְכַפֵּר עַל הַכֹּהֲנִים\hdot
כְּשֵׁם שֶׁוִּדּוּיוֹ שֶׁל שָׂעִיר הַמִּשְׁתַּלֵּחַ מְכַפֵּר עַל יִשְׂרָאֵל\hdot
כַּךְ וִדּוּיוֹ שֶׁל פָּר מְכַפֵּר עַל הַכֹּהֲנִים׃}

\perek{יְדִיעוֹת}{הַטֻּמְאָה שְׁתַּיִם שֶׁהֵן אַרְבַּע\hdot
נִטְמָא וְיָדַע\hdot
וְנֶעֶלְמָה מִמֶּנּוּ הַטֻּמְאָה\hdot
וְזָכוּר אֶת הַקֹּדֶשׁ\hdot
נֶעְלַם מִמֶּנּוּ הַקֹּדֶשׁ\hdot
וְזָכוּר אֶת הַטֻּמְאָה\hdot
נֶעֶלְמוּ מִמֶּנּוּ זֶה וְזֶה\hdot
וְאָכַל אֶת הַקֹּדֶשׁ וְלֹא יָדַע\hdot
וּמִשֶּׂאָכַל יָדַע\hdot
הֲרֵי זֶה בְּעוֹלֶה וְיוֹרֵד\hdot
נִטְמָא וְיָדַע\hdot
וְנֶעֶלְמָה מִמֶּנּוּ טֻמְאָה\hdot
וְזָכוּר אֶת הַמִּקְדָּשׁ\hdot
נֶעְלַם מִמֶּנּוּ מִקְדָּשׁ\hdot
וְזָכוּר אֶת הַטֻּמְאָה\hdot
נֶעְלַם מִמֶּנּוּ זֶה וְזֶה\hdot
וְנִכְנַס לַמִּקְדָּשׁ וְלֹא יָדַע\hdot
וּמִשֶּׁיָּצָא יָדַע\hdot
הֲרֵי זֶה בְּעוֹלֶה וְיוֹרֵד׃}

\halacha{אֶחָד הַנִּכְנָס לַעֲזָרָה\hdot
וְאֶחָד הַנִּכְנָס לְתוֹסֶפֶת הָעֲזָרָה\hdot
שֶׁאֵין מוֹסִיפִין עַל הָעִיר וְעַל הָעֲזָרוֹת\hdot
אֶלָּא בְמֶלֶךְ וְנָבִיא\hdot
וּבְאוּרִים וְתֻמִּים\hdot
וּבְסַנְהֶדְרִין שֶׁל שִׁבְעִים וְאֶחָד\hdot
וּבִשְׁתֵּי תוֹדוֹת\hdot
וּבִשִׁיר\hdot
וּבֵית דִּין מְהַלְּכִין\hdot
וּשְׁתֵּי תוֹדוֹת אַחֲרֵיהֶם\hdot
וְכָל יִשְׂרָאֵל אַחֲרֵיהֶם\hdot
הַפְּנִימִית נֶאֱכֶלֶת\hdot
וְהַחִיצוֹנָה נִשְׂרֶפֶת\hdot
וְכָל שֶׁלֹּא נַעֲשָׂה בְּכָל אֵלּוּ\hdot
הַנִּכְנָס לְשָׁם\hdot
אֵין חַיָּיבִין עָלֶיהָ׃}

\halacha{נִטְמָא בָעֲזָרָה וְנֶעֶלְמָה מִמֶּנּוּ טֻמְאָה\hdot
וְזָכוּר אֶת הַמִּקְדָּשׁ\hdot
נֶעְלָם מִמֶּנּוּ מִקְדָּשׁ\hdot
וְזָכוּר הַטֻּמְאָה\hdot
נֶעְלָם מִמֶּנּוּ זֶה וְזֶה\hdot
וְהִשְׁתַּחֲוָה אוֹ שֶׁשָּׁהָא בִּכְדֵי הִשְׁתַּחֲוָאָה\hdot
בָּא לוֹ בָּאֲרֻכָּה חַיָּב\hdot
בַּקְּצָרָה פָּטוּר\hdot
זוֹ הִיא מִצְוַת עֲשֵׂה שֶׁבַּמִּקְדָּשׁ שֶׁאֵין חַיָּבִין עָלֶיהָ׃}

\halacha{וְאֵיזוֹ הִיא מִצְוַת עֲשֵׂה שֶׁבַּנִּדָּה שֶׁחַיָּבִין עָלֶיהָ\hdot
הָיָה מְשַׁמֵּשׁ עִם הַטְּהוֹרָה\hdot
וְאָֽמְרָה לוֹ נִטְמֵאתִי וּפֵירַשׁ מִיָּד חַיָּב\hdot
מִפְּנֵי שֶׁיְּצִיאָתוֹ הֲנָאָה לוֹ כְּבִיאָתוֹ׃}

\halacha{רִבִּי אֱלִיעֶזֶר אוֹמֵר הַשֶּׁרֶץ וְנֶעְלַם מִמֶּנּוּ\hdot
עַל הֶעְלֵם שֶׁרֶץ חַיָּב\hdot
וְאֵינוֹ חַיָּב עַל הֶעְלֵם מִקְדָּשׁ\hdot
רִבִּי עֲקִיבָא אוֹמֵר וְנֶעְלַם מִמֶּנּוּ וְהוּא טָמֵא\hdot
עַל הֶעְלֵם טֻמְאָה חַיָּב\hdot
וְאֵינוֹ חַיָּב עַל הֶעְלֵם מִקְדָּשׁ\hdot
רִבִּי יִשְׁמָעֵאל אוֹמֵר וְנֶעְלַם וְנֶעְלַם שְׁתֵּי פְעָמִים\hdot
לְחַיֵּב עַל הֶעְלֵם טֻמְאָה\hdot
וְעַל הֶעְלֵם מִקְדָּשׁ׃}

\perek{שְׁבוּעוֹת}{שְׁתַּיִם שֶׁהֵן אַרְבַּע\hdot
שְׁבוּעָה שֶׁאוֹכַל וְשֶׁלֹּא אוֹכַל\hdot
שֶׁאָכַלְתִּי וְשֶׁלֹּא אָכַלְתִּי\hdot
שְׁבוּעָה שֶׁלֹּא אוֹכַל\hdot
וְאָכַל כָּל שֶׁהוּא חַיָּב דִּבְרֵי רִבִּי עֲקִיבָא\hdot
אָֽמְרוּ לוֹ לְרִבִּי עֲקִיבָא וְכִי הֵיכָן מָצִינוּ בְּאוֹכֵל כָּל שֶׁהוּא שֶׁהוּא חַיָּב\hdot
שֶׁזֶּה חַיָּב\hdot

כָּֽפְרָה הָרִאשׁוֹנָה וְאַחַר כַּךְ כָּפְרָה הַשְּׁנִיָּה\hdot
שְׁתֵּיהֶן חַיָּבוֹת\hdot
מִפְּנֵי שֶׁהָעֵדוּת יְכוֹלָה לְהִתְקַיֵּם בִּשְׁתֵּיהֶן׃}

\halacha{מַשְׁבִּיעַ אֲנִי אֲלֵיכֶם\hdot
אִם לֹא תָבוֹאוּ וְתָעִידוּנִי\hdot
שֶׁיֵּשׁ לִי בִּיַד פְּלוֹנִי נֶזֶק וַחֲצִי נֶזֶק\hdot
תַּשְׁלוּמֵי כֶפֶל\hdot
תַּשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה\hdot
וְשֶׁאָנַס אִישׁ פְּלוֹנִי אֶת בִּתִּי וּפִתָּה אֶת בִּתִּי\hdot
וְשֶׁהִכָּנִי בְּנִי\hdot
וְשֶׁחִבֵּל בִּי חֲבֵרִי\hdot
וְשֶׁהִדְלִיק אֶת גָּדִישִׁי בְּיוֹם הַכִּפּוּרִים\hdot
הֲרֵי אֵלּוּ חַיָּבִין׃}
