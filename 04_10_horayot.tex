\tractatehead{מסכת הוריות}{horayot}

\perek{הוֹרוּ}{בֵית דִּין לַעֲבוֹר עַל אַחַת מִכָּל מִצְוֺת הָאֲמוּרוֹת בַּתּוֹרָה\hdot
וְהָלַךְ הַיָּחִיד וְעָשָׂה שׁוֹגֵג עַל פִּיהֶם\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה עִמָּהֶן בֵּין שֶׁעָשׂוּ וְעָשָׂה אַחֲרֵיהֶן\hdot
בֵּין שֶׁלֹּא עָשׂוֹ וְעָשָׂה פָּטוּר\hdot
מִפְּנֵי שֶׁתָּלָה בְּבֵית דִּין\hdot
הוֹרוּ בֵית דִּין וְיָדַע אֶחָד מֵהֶן שֶׁטָּעוּ\hdot
אוֹ תַלְמִיד וְהוּא רָאוּי לְהוֹרָאָה\hdot
וְהָלַךְ וְעָשָׂה עַל פִּיהֶן\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה עִמָּהֶן\hdot
בֵּין שֶׁעָשׂוּ וְעָשָׂה אַחֲרֵיהֶן\hdot
בֵּין שֶׁלֹּא עָשׂוֹ וְעָשָׂה\hdot
הֲרֵי זֶה חַיָּב\hdot
מִפְּנֵי שֶׁלֹּא תָלָה בְּבֵית דִּין\hdot
זֶה הַכְּלָל הַתּוֹלֶה בְעַצְמוֹ חַיָּיב\hdot
וְהַתּוֹלֶה בְּבֵית דִּין פָּטוּר׃}

\halacha{הוֹרוּ בֵית דִּין וְיָֽדְעוּ שֶׁטָּעוּ וְחָֽזְרוּ בָהֶן\hdot
בֵּין שֶׁהֵבִיאוּ כַפָּרָתָן וּבֵין שֶׁלֹּא הֵבִיאוּ כַפָּרָתָן\hdot
וְהָלַךְ וְעָשָׂה עַל פִּיהֶן\hdot
רִבִּי שִׁמְעוֹן פּוֹטֵר\hdot
וְרִבִּי אֱלִיעֶזֶר אוֹמֵר סָפֵק\hdot
אֵיזֶהוּ סָפֵק\hdot
יָשַׁב לוֹ בְּתוֹךְ בֵּיתוֹ חַיָּב\hdot
הָלַךְ לוֹ לִמְדִינַת הַיָּם פָּטוּר\hdot
אָמַר רִבִּי עֲקִיבָא מוֹדֶה אֲנִי בְזֶה שֶׁהוּא קָרוֹב לַפִּטּוּר מִן הַחוֹבָה\hdot
אָמַר לוֹ בֶן עַזָּאי מַה שְּׁנָא זֶה מִן הַיּוֹשֵׁב בְּבֵיתוֹ\hdot
שֶׁהַיּוֹשֵׁב בְּבֵיתוֹ אֶפְשַׁר הָיָה לוֹ שֶׁיִּשְׁמַע\hdot
וְזֶה לֹא הָיָה אֶפְשַׁר לוֹ שֶׁיִּשְׁמַע׃}

\halacha{הוֹרו בֵית דִּין לַעֲקוֹר אֶת כָּל הַגּוּף\hdot
אָֽמְרוּ אֵין נִדָּה בַתּוֹרָה\hdot
אֵין שַׁבָּת בַּתּוֹרָה\hdot
אֵין עֲבוֹדַת אֱלִילִים  בַּתּוֹרָה\hdot
הֲרֵי אֵלּוּ פְטוּרִין\hdot
הוֹרוּ לְבַטֵּל מִקְצַת וּלְקַיֵּם מִקְצַת\hdot
הֲרֵי אֵלּוּ חַיָּבִין\hdot
כֵּיצַד\hdot
אָֽמְרוּ יֵשׁ נִדָּה בַתּוֹרָה\hdot
אֲבָל הַבָּא עַל שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם פָּטוּר\hdot
יֵשׁ שַׁבָּת בַּתּוֹרָה\hdot
אֲבָל הַמּוֹצִיא מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הָרַבִּים פָּטוּר\hdot
יֵשׁ עֲבוֹדַת אֱלִילִים בַּתּוֹרָה\hdot
אֲבָל הַמִּשְׁתַּחֲוֶה פָּטוּר\hdot
הֲרֵי אֵלּו חַיָּבִין\hdot
שֶׁנֶּאֱמַר וְנֶעְלַם דָּבָר\hdot
דָּבָר וְלֹא כָל הַגּוּף׃}

\halacha{הוֹרוּ בֵית דִּין וְיָדַע אֶחָד מֵהֶן שֶׁטָּעוּ\hdot
וְאָמַר לָהֶן טוֹעִין אַתֶּם\hdot
אוֹ שֶׁלֹּא הָיָה מֻפְלֶא שֶׁל בֵּית דִּין שָׁם\hdot
אוֹ שֶׁהָיָה אֶחָד מֵהֶן גֵּר\hdot
אוֹ מַמְזֵר\hdot
אוֹ נְתִין\hdot
אוֹ זָקֵן שֶׁלֹּא רָאָה (לוֹ) בָנִים\hdot
הֲרֵי אֵלּוּ פְטוּרִין\hdot
שֶׁנֶּאֱמַר כָּאן עֵדָה\hdot
וְנֶאֱמַר לְהַלָּן עֵדָה\hdot
מַה עֵדָה הָאֲמוּרָה לְהַלָּן עַד שֶׁיִּהְיוּ כֻלָּם רְאוּיִם לְהוֹרָאָה\hdot
אַף עֵדָה הָאֲמוּרָה כָּאן עַד שֶׁיִּהְיוּ כֻלָּן רְאוּיִם לְהוֹרָאָה\hdot
הוֹרוּ בֵית דִּין שׁוֹגְגִין וְעָשׂוּ כָל הַקָּהָל שׁוֹגְגִין\hdot
מְבִיאִין פָּר\hdot
מְזִידִין וְעָשׂוּ שׁוֹגְגִין\hdot
מְבִיאִין כִּשְׂבָּה וּשְׂעִירָה\hdot
שׁוֹגְגִין וְעָשׂוּ מְזִידִין\hdot
הֲרֵי אֵלּוּ פְטוּרִין׃}

\halacha{הוֹרוּ בֵית דִּין וְעָשׂוּ כָל הַקָּהָל אוֹ רֻבָּן עַל פִּיהֶם\hdot
מְבִיאִין פָּר\hdot
וּבַעֲבוֹדַת אֱלִילִים מְבִיאִין פָר וְשָׂעִיר\hdot
דִּבְרֵי רִבִּי מֵאִיר\hdot
רִבִּי יְהוּדָה אוֹמֵר שְׁנֵים עָשָׂר שְׁבָטִים\hdot
מְבִיאִים שְׁנֵים עָשָׂר פָּרִים\hdot
וּבַעֲבוֹדַת אֱלִילִים מְבִיאִין שְׁנֵים עָשָׂר פָּרִים וּשְׁנֵים עָשָׂר שְׂעִירִים\hdot
רִבִּי שִׁמְעוֹן אוֹמֵר שְׁלֹשָׁה עָשָׂר פָּרִים\hdot
וּבַעֲבוֹדַת אֱלִילִים\hdot
שְׁלֹשָׁה עָשָׂר פָּרִים\hdot
וּשְׁלֹשָׁה עָשָׂר שְׂעִירִים\hdot
פָּר וְשָׂעִיר לְכָל שֵׁבֶט וְשֵׁבֶט\hdot
פָּר וְשָׂעִיר לְבֵית דִּין\hdot

}
