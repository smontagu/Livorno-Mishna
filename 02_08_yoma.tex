\tractatehead{מסכת יומא}{yoma}

\perek[611]{שִׁבְעַת}{יָמִים קוֹדֶם יוֹם הַכִּפּוּרִים\hdot
מַפְרִישִׁין כֹּהֵן גָּדוֹל מִבֵּיתוֹ לְלִשְׁכַּת פְּלַהְדְּרִין\hdot
וּמַתְקִינִין לוֹ כֹּהֵן אַחֵר תַּחְתָּיו\hdot
שֶׁמֶא יֶאֱרַע בּוֹ פִּסּוּל\hdot
רִבִּי יְהוּדָה אוֹמֵר אַף אִשָּׁה אַחֶרֶת מַתְקִינִין לוֹ\hdot
שֶׁמֶא תָמוּת אִשְׁתּוֹ\hdot
שֶׁנֶּאֱמַר וְכִפֵּר בַּעֲדוֹ וּבְעַד בֵּיתוֹ\hdot
בֵּיתוֹ זוֹ אִשְׁתּוֹ\hdot
אָֽמְרוּ לוֹ אִם כֵּן אֵין לַדָּבָר סוֹף׃}

\halacha{כָּל שִׁבְעַת הַיָּמִים הוּא זוֹרֵק אֶת הַדָּם\hdot
וּמַקְטִיר אֶת הַקְּטֹרֶת\hdot
וּמֵטִיב אֶת הַנֵּרוֹת\hdot
וּמַקְרִיב אֶת הָרֶגֶל\hdot
וּשְׁאַר כָּל הַיָּמִים אִם רָצָה לְהַקְרִיב\hdot
מַקְרִיב\hdot
שֶׁכֹּהֵן גָּדוֹל מַקְרִיב חֵלֶק בְּרֹאשׁ וְנוֹטֵל חֵלֶק בָּרֹאשׁ׃}
